def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority of each bin based on how well the item fits, favoring bins that have enough space but aren't too empty after packing the item.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = item / cap
            priority.append(fill_ratio)
        else:
            priority.append(-1)  # Assign a low priority for bins that can't fit the item
    return priority
