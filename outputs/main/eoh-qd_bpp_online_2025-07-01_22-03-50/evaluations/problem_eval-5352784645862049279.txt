def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on the remaining capacity of the bin and the item size, favoring bins that can accommodate the item with minimal remaining space and penalizing those that would result in significant wasted space.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority = 1.0 / (cap - item + 0.0001)  # Higher priority for smaller remaining space
    else:
      priority = -10000  # Very low priority for bins that cannot fit the item

    priorities.append(priority)
  return priority
