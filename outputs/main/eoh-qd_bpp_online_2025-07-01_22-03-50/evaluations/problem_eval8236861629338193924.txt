def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority score to each bin based on the remaining capacity, favoring bins that can accommodate the item with minimal wasted space, and penalizing bins that are too full or too empty.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(1 / (cap - item + 0.00001))
    else:
      priority.append(-1000)
  return priority
