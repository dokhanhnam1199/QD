```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fill ratio and remaining space, balancing bin utilization."""

    new_capacities = bins_remain_cap - item
    valid_bins = new_capacities >= 0
    priorities = np.full_like(bins_remain_cap, -np.inf)

    if not np.any(valid_bins):
        return priorities
    
    valid_capacities = bins_remain_cap[valid_bins]
    remaining_space = valid_capacities - item
    
    fill_ratio_score = item / valid_capacities
    remaining_space_penalty = 1 / (remaining_space + 0.000001)
    
    priorities[valid_bins] = fill_ratio_score + remaining_space_penalty
    
    return priorities
```
