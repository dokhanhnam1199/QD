import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate remaining capacity after adding the item.
    new_capacities = bins_remain_cap - item

    # Identify bins that can accommodate the item.
    valid_bins = new_capacities >= 0

    # Initialize priorities with a very low value (e.g., -np.inf) for invalid bins.
    priorities = np.full_like(bins_remain_cap, -np.inf)
    priorities[valid_bins] = 0  # Reset valid bins to 0 for further scoring.

    # For valid bins, calculate a score based on remaining capacity after packing.
    remaining_space = bins_remain_cap[valid_bins] - item

    # Scoring valid bins, considering multiple factors:
    # 1. Fill ratio: How much of the bin's capacity is used by the item.
    fill_ratio = item / bins_remain_cap[valid_bins]

    # 2. Remaining space: Less remaining space might be better (first-fit decreasing principle).  Add a small constant to avoid division by zero.
    space_score = 1 / (remaining_space + 0.000001)

    # 3. Bin utilization: Encourage packing into bins that are already relatively full
    utilization = 1 - (bins_remain_cap[valid_bins] / np.max(bins_remain_cap)) # Normalize and invert remaining capacity for utilization score

    # Combine the scores with weights. Experiment with different weights to fine-tune performance.
    priorities[valid_bins] = (0.4 * fill_ratio) + (0.5 * space_score) + (0.1 * utilization)


    return priorities
