```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines ratio of item size to bin capacity and remaining space, avoiding zero division.
    Prioritizes bins that fit the item well and leave minimal space.
    """
    new_capacities = bins_remain_cap - item
    valid_bins = new_capacities >= 0
    priorities = np.full_like(bins_remain_cap, -np.inf)
    priorities[valid_bins] = 0

    remaining_space = bins_remain_cap[valid_bins] - item
    
    # Calculate score if valid bins
    priorities[valid_bins] = (item / bins_remain_cap[valid_bins]) + (1 / (remaining_space + 0.000001))
    
    return priorities
```
