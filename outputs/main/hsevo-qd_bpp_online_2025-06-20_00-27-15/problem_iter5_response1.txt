```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 1e-6, invalid_priority: float = -np.inf) -> np.ndarray:
    """Prioritizes bins using fullness and remaining space with epsilon."""
    new_capacities = bins_remain_cap - item
    valid_bins = new_capacities >= 0
    priorities = np.full_like(bins_remain_cap, invalid_priority)
    priorities[valid_bins] = (item / bins_remain_cap[valid_bins]) + (1 / (new_capacities[valid_bins] + epsilon))
    return priorities
```
