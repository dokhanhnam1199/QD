import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins using fill ratio, space balance, and adaptive weights."""

    new_capacities = bins_remain_cap - item
    valid_bins = new_capacities >= 0
    priorities = np.full_like(bins_remain_cap, -np.inf)

    if not np.any(valid_bins):
        return priorities

    priorities[valid_bins] = 0
    remaining_space = bins_remain_cap[valid_bins] - item
    fill_ratio = item / bins_remain_cap[valid_bins]
    space_balance = np.where(remaining_space > item / 2, 0.5, 0.1)

    # Adaptive weight based on item size
    item_size_ratio = item / np.max(bins_remain_cap)
    fill_ratio_weight = 1.0 + item_size_ratio * 0.5
    space_balance_weight = 0.5 - item_size_ratio * 0.2

    priorities[valid_bins] = fill_ratio_weight * fill_ratio + space_balance_weight * space_balance
    return priorities
