import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Worst‑Fit priority function for the online Bin Packing Problem.

    The function gives higher scores to bins that have the largest remaining
    capacity while still being able to accommodate the incoming item. Bins
    that cannot fit the item receive a very low (negative infinite) priority,
    ensuring they are never selected.

    Parameters
    ----------
    item : float
        Size of the incoming item.
    bins_remain_cap : np.ndarray
        1‑D array where each entry represents the remaining capacity of a bin.

    Returns
    -------
    np.ndarray
        Priority score for each bin (same shape as ``bins_remain_cap``).
    """
    # Identify bins that can actually accommodate the item.
    feasible = bins_remain_cap >= item

    # Use the remaining capacity as the priority for feasible bins.
    # Infeasible bins receive -inf to make them unattractive.
    priorities = np.where(feasible, bins_remain_cap, -np.inf)

    return priorities
