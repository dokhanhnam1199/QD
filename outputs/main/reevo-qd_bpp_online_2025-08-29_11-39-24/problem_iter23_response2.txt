import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Returns:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Exact fit first
    exact_fit_bins = np.where(bins_remain_cap == item)[0]
    priorities[exact_fit_bins] = 1.0

    # If no exact fit, prioritize based on waste minimization
    if np.sum(priorities) == 0:
        feasible_bins = np.where(bins_remain_cap >= item)[0]
        if len(feasible_bins) > 0:
            waste = bins_remain_cap[feasible_bins] - item
            # Normalize waste to avoid very small values dominating
            normalized_waste = waste / (np.max(waste) + 1e-6)
            priorities[feasible_bins] = 1.0 / (normalized_waste + 1e-6)

            # Smooth scoring for moderate bin fullness (sigmoid)
            utilization = (bins_remain_cap[feasible_bins] - item) / bins_remain_cap[feasible_bins]
            sigmoid_scores = 1 / (1 + np.exp(-5 * (utilization - 0.7)))  # Center around 0.7
            priorities[feasible_bins] *= sigmoid_scores
        else:
            priorities[:] = -1.0  # No bin can fit the item

    return priorities
