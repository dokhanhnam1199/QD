import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm constructs a heuristic matrix by simulating ant colony optimization with adaptive parameter adjustments based on the iteration's best path length and pheromone updates favoring edges in elite solutions.}"""
    n = distance_matrix.shape[0]
    pheromone = np.ones((n, n))  # Initialize pheromone trails
    heuristics_matrix = np.zeros((n, n))
    alpha = 0.7  # Pheromone influence
    beta = 2   # Distance influence
    rho = 0.3  # Evaporation rate
    Q = 100    # Pheromone deposit constant
    num_ants = 30
    num_iterations = 60
    elite_ants = 5  # Number of top ants to reinforce pheromones

    for iteration in range(num_iterations):
        all_paths = []
        all_path_lengths = []

        for ant in range(num_ants):
            start_node = np.random.randint(n)
            current_node = start_node
            unvisited = set(range(n))
            unvisited.remove(current_node)
            path = [current_node]

            while unvisited:
                probabilities = (pheromone[current_node, :]**alpha) * ((1 / (distance_matrix[current_node, :] + 1e-6))**beta)
                
                # Mask visited nodes
                for visited_node in path:
                    probabilities[visited_node] = 0
                
                probabilities = probabilities / np.sum(probabilities) if np.sum(probabilities) > 0 else np.ones(n) / n

                next_node = np.random.choice(n, p=probabilities)
                path.append(next_node)
                unvisited.remove(next_node)
                current_node = next_node

            path.append(start_node)
            all_paths.append(path)

            path_length = 0
            for i in range(n):
                path_length += distance_matrix[path[i], path[(i + 1) % n]]
            all_path_lengths.append(path_length)

        # Find the best path in this iteration
        best_path_index = np.argmin(all_path_lengths)
        best_path = all_paths[best_path_index]
        best_path_length = all_path_lengths[best_path_index]
        
        # Sort ants by path length to find elite ants
        sorted_indices = np.argsort(all_path_lengths)
        elite_paths = [all_paths[i] for i in sorted_indices[:elite_ants]]

        # Update pheromones
        pheromone *= (1 - rho)  # Evaporation
        for path in elite_paths:
            path_length = all_path_lengths[all_paths.index(path)]
            for i in range(n):
                pheromone[path[i], path[(i + 1) % n]] += Q / path_length
                pheromone[path[(i + 1) % n], path[i]] += Q / path_length

    # Construct heuristic matrix based on pheromone levels
    heuristics_matrix = pheromone
    return heuristics_matrix
