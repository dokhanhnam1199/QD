import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm iteratively samples TSP solutions using a nearest neighbor heuristic, updating a matrix that reflects the frequency with which each edge appears in promising solutions.}"""
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    num_samples = 100
    
    for _ in range(num_samples):
        start_node = np.random.randint(n)
        current_node = start_node
        unvisited = set(range(n))
        unvisited.remove(start_node)
        path = [start_node]
        path_length = 0

        while unvisited:
            nearest_neighbor = -1
            min_distance = float('inf')
            for neighbor in unvisited:
                distance = distance_matrix[current_node, neighbor]
                if distance < min_distance:
                    min_distance = distance
                    nearest_neighbor = neighbor

            path.append(nearest_neighbor)
            path_length += min_distance
            unvisited.remove(nearest_neighbor)
            current_node = nearest_neighbor

        path.append(start_node)
        path_length += distance_matrix[current_node, start_node]

        # Update heuristics matrix based on path length (better paths contribute more)
        if path_length > 0:
            reward = 1 / path_length 
        else:
            reward = 0  # avoid division by zero
            
        for i in range(n):
            node1 = path[i]
            node2 = path[i+1]
            heuristics_matrix[node1, node2] += reward
            heuristics_matrix[node2, node1] += reward

    return heuristics_matrix
