import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm estimates edge desirability for TSP by iteratively constructing tours using a nearest neighbor approach with a stochastic element, improving them with 2-opt swaps with a limited number of iterations, and averaging the inclusion frequency of each edge across multiple runs.}"""
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros((n, n))
    num_trials = 50

    def calculate_tour_length(tour):
        length = 0
        for i in range(n):
            length += distance_matrix[tour[i], tour[(i + 1) % n]]
        return length

    def two_opt_swap(tour, i, k):
        new_tour = tour[:i] + tour[i:k+1][::-1] + tour[k+1:]
        return new_tour

    def two_opt(tour, max_iterations=10):
        best_tour = tour
        best_length = calculate_tour_length(tour)
        improved = True
        iterations = 0
        while improved and iterations < max_iterations:
            improved = False
            for i in range(1, n - 1):
                for k in range(i + 1, n):
                    new_tour = two_opt_swap(best_tour, i, k)
                    new_length = calculate_tour_length(new_tour)
                    if new_length < best_length:
                        best_tour = new_tour
                        best_length = new_length
                        improved = True
            iterations += 1
        return best_tour, best_length

    def nearest_neighbor_tour(start_node, stochasticity=0.2):
        tour = [start_node]
        unvisited_nodes = set(range(n))
        unvisited_nodes.remove(start_node)
        current_node = start_node

        while unvisited_nodes:
            # Introduce stochasticity: select from the top candidates
            distances = {node: distance_matrix[current_node, node] for node in unvisited_nodes}
            sorted_nodes = sorted(distances.items(), key=lambda item: item[1])
            k = min(int(len(sorted_nodes) * stochasticity) + 1, len(sorted_nodes))
            candidates = [node for node, dist in sorted_nodes[:k]]

            nearest_node = np.random.choice(candidates)
            tour.append(nearest_node)
            unvisited_nodes.remove(nearest_node)
            current_node = nearest_node
        tour.append(start_node)
        tour = tour[:-1]
        return tour

    for _ in range(num_trials):
        # Generate a tour using nearest neighbor heuristic with stochasticity
        start_node = np.random.randint(0, n)
        initial_tour = nearest_neighbor_tour(start_node)

        # Apply 2-opt to improve the tour
        best_tour, best_length = two_opt(initial_tour)

        # Update the heuristics matrix based on the edges in the improved tour
        for i in range(n):
            node1 = best_tour[i]
            node2 = best_tour[(i + 1) % n]
            heuristics_matrix[node1, node2] += 1
            heuristics_matrix[node2, node1] += 1

    heuristics_matrix /= num_trials
    return heuristics_matrix
