import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins using Best Fit and adds a slight preference for exact fits."""
    priorities = np.zeros_like(bins_remain_cap)
    suitable_bins_mask = bins_remain_cap >= item

    if np.any(suitable_bins_mask):
        suitable_capacities = bins_remain_cap[suitable_bins_mask]
        remaining_capacities_after_fit = suitable_capacities - item

        exact_fit_mask = remaining_capacities_after_fit == 0
        if np.any(exact_fit_mask):
            priorities[suitable_bins_mask][exact_fit_mask] = 1.0
        else:
            min_remaining_capacity = np.min(remaining_capacities_after_fit)
            best_fit_mask = (suitable_capacities - item == min_remaining_capacity)
            priorities[suitable_bins_mask][best_fit_mask] = 0.5

    return priorities
