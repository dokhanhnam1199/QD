def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    fit_values = np.where(bins_remain_cap >= item, bins_remain_cap - item, -np.inf)
    exp_fit_values = np.exp(fit_values - np.max(fit_values))
    priorities = exp_fit_values / np.sum(exp_fit_values)
    return priorities
