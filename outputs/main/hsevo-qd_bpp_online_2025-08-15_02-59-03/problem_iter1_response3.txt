```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Almost Full Fit strategy: prioritize bins that are nearly full and can accommodate the item.
    Higher priority is given to bins with less remaining capacity after placing the item.
    Bins that cannot fit the item are assigned a priority of -1.
    """
    priorities = np.full_like(bins_remain_cap, -1.0)
    for i, remaining_cap in enumerate(bins_remain_cap):
        if remaining_cap >= item:
            priorities[i] = remaining_cap - item
    return priorities
```
