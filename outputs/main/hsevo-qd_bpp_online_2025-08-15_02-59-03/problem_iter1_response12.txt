def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    bins_with_sufficient_capacity = bins_remain_cap >= item
    priorities = np.zeros_like(bins_remain_cap)
    if np.any(bins_with_sufficient_capacity):
        priorities[bins_with_sufficient_capacity] = bins_remain_cap[bins_with_sufficient_capacity] - item
    return priorities
