import numpy as np

def heuristics_v2(distance_matrix: np.ndarray, coordinates: np.ndarray, demands: np.ndarray, capacity: int) -> np.ndarray:
    """CVRP heuristic combining distance, demand, angle, depot proximity, and route balancing with adaptive sparsification."""
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((n, n))

    # Distance heuristic (inverse square)
    distance_heuristic = 1.0 / (distance_matrix + 1e-6)**2
    heuristic_matrix = distance_heuristic.copy()

    # Demand-based penalty/reward
    demand_penalty = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            if i == j:
                demand_penalty[i, j] = 0
                continue

            combined_demand = demands[i] + demands[j]
            if combined_demand > capacity and i != 0 and j != 0:
                demand_penalty[i,j] = 0.01  # Harsh penalty
            elif combined_demand <= (capacity / 4) and i != 0 and j != 0:
                demand_penalty[i,j] = 0.1
            else:
                demand_penalty[i,j] = 0.05 # Regular penalty

    heuristic_matrix = heuristic_matrix - demand_penalty # Subtract demand penalty
                
    #Proximity to depot
    depot_proximity = np.zeros((n,n))
    for i in range(n):
      for j in range(n):
        if i == j:
          depot_proximity[i, j] = 0
          continue

        if j == 0 and i != 0:
            depot_proximity[i,j] = 0.2  # Discount connection to depot
        if i == 0 and j != 0:
            depot_proximity[i,j] = 0.2 # Discount connection to depot
    heuristic_matrix = heuristic_matrix - depot_proximity

    # Angle heuristic (encourage straighter paths)
    angle_heuristic = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            if i == j: continue

            angle_sum = 0
            for k in range(n):
                if k == i or k == j: continue

                v1 = coordinates[i] - coordinates[k]
                v2 = coordinates[j] - coordinates[i]

                dot_product = np.dot(v1, v2)
                magnitudes = np.linalg.norm(v1) * np.linalg.norm(v2)

                if magnitudes > 0:
                    cosine_angle = dot_product / magnitudes
                    angle = np.arccos(np.clip(cosine_angle, -1.0, 1.0))
                    angle_sum += (np.pi - angle)

            if n > 2:
                angle_heuristic[i,j] = angle_sum / (n - 2)
            else:
                angle_heuristic[i,j] = 0

            heuristic_matrix[i, j] += 0.05 * angle_heuristic[i,j]


    # Depot attraction (nodes closer to the depot are encouraged)
    depot_attraction = np.zeros((n, n))
    for i in range(n):
      for j in range(n):
        if i == j: continue
      depot_attraction[i, j] = (1 / (distance_matrix[i, 0] + 1e-6)) + (1 / (distance_matrix[j, 0] + 1e-6))
      heuristic_matrix[i, j] += 0.025 * depot_attraction[i, j]

    # Route balancing: Encourage edges that connect nodes with disparate distances to the depot.
    route_balance = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j or i == 0 or j == 0:
                continue
            depot_distance_diff = abs(distance_matrix[i, 0] - distance_matrix[j, 0])
            route_balance[i, j] = depot_distance_diff / (np.max(distance_matrix) + 1e-6)  # Normalize
            heuristic_matrix[i, j] += 0.01 * route_balance[i, j] #Adding this value


    # Normalize heuristics for each node (Row-wise normalization)
    min_val = np.min(heuristic_matrix)
    max_val = np.max(heuristic_matrix)

    if max_val > min_val:
      heuristic_matrix = (heuristic_matrix - min_val) / (max_val - min_val)

    for i in range(n):
        row_sum = np.sum(heuristic_matrix[i,:])
        if row_sum > 0:
            heuristic_matrix[i,:] /= row_sum

    # Adaptive Sparsification
    threshold = np.mean(heuristic_matrix) * 0.5 # Dynamic threshold
    heuristic_matrix[heuristic_matrix < threshold] = 0

    return heuristic_matrix
