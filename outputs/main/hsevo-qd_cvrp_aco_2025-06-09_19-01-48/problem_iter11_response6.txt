import numpy as np

def heuristics_v2(distance_matrix: np.ndarray, coordinates: np.ndarray, demands: np.ndarray, capacity: int) -> np.ndarray:
    """CVRP heuristic combining distance, demand, angle, depot proximity, and capacity buffer with adaptive sparsification."""
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((n, n))

    # Distance heuristic (inverse square)
    distance_heuristic = 1.0 / (distance_matrix + 1e-6)**2
    heuristic_matrix = distance_heuristic.copy()

    # Demand-based penalty/reward with capacity buffer consideration
    demand_factor = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j:
                demand_factor[i, j] = 0
                continue

            combined_demand = demands[i] + demands[j]
            remaining_capacity = capacity - combined_demand
            
            if i == 0 or j == 0:
                demand_factor[i, j] = 1.0 # No demand penalty involving depot
                continue

            if combined_demand > capacity:
                demand_factor[i, j] = 0.01  # Very harsh penalty for exceeding capacity
            elif remaining_capacity > (capacity / 2):
                demand_factor[i, j] = 1.15 # Reward ample capacity
            elif remaining_capacity > (capacity / 4):
                 demand_factor[i,j] = 1.05 # Slight reward

            else:
                demand_factor[i, j] = 0.9 # Minor penalty for low remaining capcity
    heuristic_matrix = heuristic_matrix * demand_factor
                
    # Angle heuristic (encourage straighter paths)
    angle_heuristic = np.zeros((n,n))
    for i in range(n):
        for j in range(n):
            if i == j: continue

            angle_sum = 0
            for k in range(n):
                if k == i or k == j: continue

                v1 = coordinates[i] - coordinates[k]
                v2 = coordinates[j] - coordinates[i]

                dot_product = np.dot(v1, v2)
                magnitudes = np.linalg.norm(v1) * np.linalg.norm(v2)

                if magnitudes > 0:
                    cosine_angle = dot_product / magnitudes
                    angle = np.arccos(np.clip(cosine_angle, -1.0, 1.0))
                    angle_sum += (np.pi - angle)

            if n > 2:
                angle_heuristic[i,j] = angle_sum / (n - 2)
            else:
                angle_heuristic[i,j] = 0

            heuristic_matrix[i, j] += 0.1 * angle_heuristic[i,j]


    # Depot attraction (nodes closer to the depot are encouraged), but penalize direct depot connections
    depot_attraction = np.zeros((n, n))
    for i in range(n):
      for j in range(n):
        if i == j: continue
      depot_attraction[i, j] = (1 / (distance_matrix[i, 0] + 1e-6)) + (1 / (distance_matrix[j, 0] + 1e-6))
      heuristic_matrix[i, j] += 0.05 * depot_attraction[i, j]

    for i in range(1, n):
        heuristic_matrix[0, i] *= 0.5 # Discourage direct connections from depot
        heuristic_matrix[i, 0] *= 0.5 # Discourage direct connections to depot


    # Capacity Aware Depot Connection
    capacity_depot_connection = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j: continue
            if i == 0 and demands[j] > capacity / 2:
                capacity_depot_connection[i, j] = 0.1  # Heavily Penalize
            if j == 0 and demands[i] > capacity / 2:
                capacity_depot_connection[i, j] = 0.1 # Heavily Penalize
    heuristic_matrix = heuristic_matrix * (1-capacity_depot_connection)
    

    # Node criticality:  Nodes with high demand have high criticality to be close to depot.
    node_criticality = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i == j: continue
            if (demands[i] > (capacity / 2) or demands[j] > (capacity / 2)) and i != 0 and j != 0:
                node_criticality[i,j] = (1 / (distance_matrix[i, 0] + 1e-6)) + (1 / (distance_matrix[j, 0] + 1e-6))
                heuristic_matrix[i,j] += 0.15 * node_criticality[i,j]

    # Normalize heuristics for each node, but only if needed to avoid div by zero
    for i in range(n):
        row_sum = np.sum(heuristic_matrix[i,:])
        if row_sum > 0:
            heuristic_matrix[i,:] /= row_sum


    # Adaptive Sparsification based on quantiles.
    threshold = np.quantile(heuristic_matrix[heuristic_matrix > 0], 0.25) # Use 25th percentile as threshold
    heuristic_matrix[heuristic_matrix < threshold] = 0

    return heuristic_matrix
