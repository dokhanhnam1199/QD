import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on item size and remaining capacity.
    
    Prioritizes bins that can fit the item with minimal leftover space, using the ratio
    of item size to bin's remaining capacity. Bins that cannot fit the item have priority -inf.
    """
    valid_mask = bins_remain_cap >= item
    epsilon = 1e-9  # Prevent division by zero
    return np.where(
        valid_mask,
        item / (bins_remain_cap + epsilon),
        -np.inf
    )
