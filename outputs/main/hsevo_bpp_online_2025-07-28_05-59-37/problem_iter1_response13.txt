import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on a combination of fit efficiency and bin utilization.
    
    Bins that can fit the item are prioritized by the ratio of item size to bin's remaining capacity,
    favoring bins where the item fills a larger proportion of the available space. Additionally,
    a secondary term encourages using bins with smaller remaining capacities to promote compact packing.
    """
    can_fit = bins_remain_cap >= item
    fit_ratio = item / bins_remain_cap
    # Secondary term to prefer bins with smaller remaining capacity when fit ratios are similar
    capacity_weight = -bins_remain_cap
    priority = np.where(can_fit, fit_ratio + 0.01 * capacity_weight, -np.inf)
    return priority
