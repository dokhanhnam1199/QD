import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit heuristic.
    
    Prioritizes bins with the least remaining capacity that can still fit the item,
    equivalent to minimizing leftover space. Bins that cannot fit the item are assigned
    a very low priority (-inf).
    """
    leftover = bins_remain_cap - item
    valid = leftover >= 0
    # Prioritize bins with minimal leftover (Best Fit)
    priority = np.where(valid, -leftover, -np.inf)
    return priority
