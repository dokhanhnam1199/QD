import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins using a Best Fit heuristic with a bonus for exact fits.
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
    
    Returns:
        Array of priority scores for each bin.
    """
    valid = bins_remain_cap >= item
    # Prioritize bins that fit exactly with a high bonus
    exact_fit_bonus = np.where(valid & (bins_remain_cap == item), 1e6, 0.0)
    # Prioritize bins by minimizing remaining capacity after placement
    fit_priority = np.where(valid, item - bins_remain_cap, -np.inf)
    return exact_fit_bonus + fit_priority
