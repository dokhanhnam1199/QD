```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores based on minimizing leftover space after placement.
    
    Bins that can fit the item are prioritized by how tightly they can pack it,
    favoring bins with remaining capacity just enough to fit the item (Best Fit).
    """
    valid_bins = bins_remain_cap >= item
    leftover = bins_remain_cap - item
    # Use negative leftover as priority: higher values (closer to 0) indicate better fit
    priority = np.where(valid_bins, -leftover, -np.inf)
    return priority
```
