import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function combining Best Fit and Worst Fit heuristics based on item size."""
    remaining_space = bins_remain_cap - item
    valid = remaining_space >= 0
    factor = 1 - 2 * item  # Switches behavior based on item size threshold (0.5)
    priority = np.where(valid, remaining_space * factor, -np.inf)
    return priority
