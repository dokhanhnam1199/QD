import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins using a Best Fit heuristic.

    Prioritizes bins that can fit the item with the smallest remaining capacity,
    encouraging tight packing to minimize leftover space. Bins that cannot fit
    the item are assigned a very low priority.

    Args:
        item: Size of the item to pack.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin.
    """
    valid_mask = bins_remain_cap >= item
    return np.where(valid_mask, -bins_remain_cap, -np.inf)
