import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function based on Best Fit strategy with a twist of space efficiency.
    
    Bins that can fit the item are prioritized by how tightly they pack the item,
    favoring bins where the item fills the bin completely. Bins that cannot fit the
    item are assigned -infinity to ensure they are not selected.
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
    
    Returns:
        Array of priority scores for each bin.
    """
    can_fit = bins_remain_cap >= item
    # Prioritize bins that minimize remaining space (Best Fit)
    # Use negative of remaining space after packing to maximize tight fits
    priorities = np.where(can_fit, -(bins_remain_cap - item), -np.inf)
    return priorities
