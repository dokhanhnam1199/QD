def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on how close the item size is to a fraction of the remaining capacity, aiming for balanced bin utilization.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      # Calculate how close the item is to filling a certain fraction of the bin (e.g., 75%)
      target_fill = 0.75 * cap
      priority_score = -abs(item - target_fill) # Minimize the difference between the item size and the target fill
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
