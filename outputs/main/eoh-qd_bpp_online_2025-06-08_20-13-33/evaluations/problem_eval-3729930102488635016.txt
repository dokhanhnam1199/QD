def priority_v2(item, bins_remain_cap):
  """{Assign a score to each bin based on the ratio of item size to remaining capacity, favoring bins where the item fills a large portion of the remaining space without exceeding it.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(item / (cap + 0.1)) # Prioritize bins where item fills a larger proportion of remaining capacity
    else:
      priority.append(-1.0) # Disqualify bins that cannot fit the item
  return priority
