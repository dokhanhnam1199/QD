def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates a priority score for each bin based on the square of the remaining capacity if the item is placed in the bin, giving preference to bins that result in intermediate remaining capacities, and heavily penalizing bins where the item doesn't fit.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      remaining_cap = cap - item
      priority = remaining_cap**2
    else:
      priority = -float('inf')
    priorities.append(priority)
  return priority
