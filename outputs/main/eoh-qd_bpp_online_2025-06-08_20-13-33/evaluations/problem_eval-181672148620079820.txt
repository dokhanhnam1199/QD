def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on how close the item size is to a fraction of the remaining capacity, favoring bins where the item fills a specific percentage of the bin.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            target_fill_ratio = 0.75  # Aim for 75% fill
            fill_ratio = float(item) / cap
            priority_score = -abs(fill_ratio - target_fill_ratio) # Closer to target, higher priority
        else:
            priority_score = -1000 # Very low priority
        priority.append(priority_score)
    return priority
