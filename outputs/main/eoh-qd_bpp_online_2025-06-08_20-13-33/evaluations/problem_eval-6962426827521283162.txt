def priority_v2(item, bins_remain_cap):
  """{Assign a priority score to each bin based on how close its remaining capacity is to half of the item size if the item fits, otherwise assign a low priority.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      ideal_cap = item + (item / 2)
      priority.append(1.0 / (abs(cap - ideal_cap) + 0.1))
    else:
      priority.append(-1.0)

  return priority
