def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority score to each bin based on its remaining capacity, favoring bins that are close in size to the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      priority.append(1 / (abs(item - cap) + 0.1))
    else:
      priority.append(-1000)
  return priority
