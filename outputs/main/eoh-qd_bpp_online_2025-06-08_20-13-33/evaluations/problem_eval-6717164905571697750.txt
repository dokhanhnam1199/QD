def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on how close the item size is to the golden ratio proportion of the bin's remaining capacity, penalizing bins that are too small or lead to highly uneven filling.}"""
  priority = []
  golden_ratio = 1.618
  for cap in bins_remain_cap:
    if item <= cap:
      ideal_item_size = cap / golden_ratio
      priority.append(100 - abs(item - ideal_item_size))
    else:
      priority.append(-100)
  return priority
