def priority_v2(item, bins_remain_cap):
  """{Assign a priority score to each bin based on how full it will be if the item is placed in it, prioritizing bins that will be filled to a certain percentage after packing the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_percentage = (item / (cap + 1e-9)) # Calculate fill percentage after adding the item
      priority.append(abs(fill_percentage - 0.75)) # Closer to 75% fill is better
    else:
      priority.append(1.0) # Disqualify bins that cannot fit the item

  return priority
