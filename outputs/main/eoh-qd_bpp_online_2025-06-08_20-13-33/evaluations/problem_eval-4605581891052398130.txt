def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on the wasted space after placing the item, penalizing bins where the item doesn't fit or leaves a large space.}"""
  priorities = []
  for capacity in bins_remain_cap:
    if capacity >= item:
      wasted_space = capacity - item
      priority = -wasted_space # Smaller wasted space is better
    else:
      priority = -float('inf')  # Very low priority if the item doesn't fit

    priorities.append(priority)
  return priority
