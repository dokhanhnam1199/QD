def priority_v2(item, bins_remain_cap):
    """{This algorithm sorts bins based on how much the item fills them relative to their size, favoring bins that will be filled to a higher percentage.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = item / cap
            priority.append(fill_ratio * 100)
        else:
            priority.append(-100)
    return priority
