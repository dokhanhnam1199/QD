def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the utilization rate they would achieve if the item were placed, penalizing bins that cannot accommodate the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      utilization = item / (cap + 1e-9) # Adding a tiny value to avoid division by zero
      priority.append(utilization)
    else:
      priority.append(-100)
  return priority
