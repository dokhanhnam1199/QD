def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin by considering the ratio of item size to remaining capacity, penalized by how much space would be wasted if the item were placed in the bin.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = item / cap
      waste_penalty = (cap - item) / cap
      priority.append(fill_ratio - waste_penalty)
    else:
      priority.append(-1)
  return priority
