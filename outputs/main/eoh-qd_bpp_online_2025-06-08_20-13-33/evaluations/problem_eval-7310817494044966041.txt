def priority_v2(item, bins_remain_cap):
  """{Assign a priority score to each bin based on its remaining capacity relative to the item size, favoring bins with remaining capacity slightly larger than the item size.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(1.0 / (cap - item + 0.1)) # Prioritize bins with smaller differences between cap and item size
    else:
      priority.append(-1.0) # Disqualify bins that cannot fit the item

  return priority
