def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on how close the item size is to half the remaining capacity of the bin, penalizing bins where the item is significantly larger or smaller than half the remaining capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            ideal_size = cap / 2.0
            priority_score = 1 / (abs(item - ideal_size) + 0.01)
        else:
            priority_score = -1000
        priority.append(priority_score)
    return priority
