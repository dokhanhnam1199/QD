def priority_v2(item, bins_remain_cap):
    """{Assigns a priority to each bin by considering the percentage of bin utilization after packing the item and a bonus for bins that would be filled beyond a certain threshold.}"""
    priorities = []
    for cap in bins_remain_cap:
        if item <= cap:
            utilization_percentage = (item / 100) #How utilized the bin will be with the item.

            #Bonus for filling the bin beyond a threshold
            fill_bonus = 0
            if (100 - cap + item) >= 75:  #If bin has filled 75% after item will be placed.
                fill_bonus = 1
            priority = utilization_percentage + fill_bonus
        else:
            priority = -1
        priorities.append(priority)
    return priority
