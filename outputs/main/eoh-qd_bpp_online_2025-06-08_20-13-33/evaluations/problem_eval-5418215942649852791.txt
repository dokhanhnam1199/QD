def priority_v3(item, bins_remain_cap):
  """{This algorithm calculates a priority score for each bin based on the wasted space percentage if the item were placed in that bin.}"""
  priorities = []
  for capacity in bins_remain_cap:
    if capacity >= item:
      wasted_space_percentage = (capacity - item) / capacity if capacity > 0 else 0
      priority = 1 - wasted_space_percentage  # Higher priority for less wasted space
    else:
      priority = -1
    priorities.append(priority)
  return priority
