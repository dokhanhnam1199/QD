def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority of each bin based on the ratio of item size to remaining capacity, rewarding bins where the item fills a significant portion of the remaining space without overflowing.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(item / cap)
        else:
            priority.append(-1)
    return priority
