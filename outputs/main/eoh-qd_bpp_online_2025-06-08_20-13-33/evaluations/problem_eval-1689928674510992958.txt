def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates a priority score for each bin based on the ratio of remaining capacity to item size, favoring bins where the item fills a significant portion of the remaining capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = float(item) / cap
            priority_score = fill_ratio * cap  # Prioritize higher fill ratio, scaled by the remaining capacity
            priority.append(priority_score)
        else:
            priority.append(-1)
    return priority
