def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns bins priority based on the wasted space percentage if the item is placed in the bin, penalizing bins with high percentage of wasted space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      wasted_space = cap - item
      wasted_percentage = (wasted_space / cap) * 100
      priority.append(100 - wasted_percentage)
    else:
      priority.append(-1)
  return priority
