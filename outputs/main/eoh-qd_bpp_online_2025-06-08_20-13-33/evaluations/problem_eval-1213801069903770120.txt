def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority score to each bin based on the ratio of the item size to the bin's remaining capacity, favoring bins where the item fills a significant portion of the remaining space without exceeding it.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(item / cap)
    else:
      priority.append(-1000)
  return priority
