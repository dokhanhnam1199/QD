def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates a priority score for each bin based on the wasted space if the item is placed in the bin, normalizing by the bin's remaining capacity.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      wasted_space = cap - item
      priority_score = 1 - (wasted_space / cap)  # Prioritize bins with less wasted space relative to capacity
      priority.append(priority_score)
    else:
      priority.append(-1)  # Assign a low priority if the item doesn't fit
  return priority
