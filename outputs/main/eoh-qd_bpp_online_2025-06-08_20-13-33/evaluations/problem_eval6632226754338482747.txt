def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the ratio of the item size to the remaining capacity, scaled by the remaining capacity itself; bins with remaining capacity smaller than the item receive the lowest priority.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append((item / cap) * cap) 
    else:
      priority.append(-float('inf'))
  return priority
