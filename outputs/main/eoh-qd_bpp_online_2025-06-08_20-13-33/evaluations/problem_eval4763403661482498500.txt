def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on the ratio of item size to remaining bin capacity, favoring bins where the item fills a significant portion of the remaining space, but only if the bin is not filled over a certain threshold.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = item / cap
      if fill_ratio <= 0.95:
        priority = fill_ratio
      else:
        priority = -2 # Penalize bins that would be filled too much
    else:
      priority = -1 # Item doesn't fit
    priorities.append(priority)
  return priority
