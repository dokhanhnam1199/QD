def priority_v2(item, bins_remain_cap):
  """{The algorithm assigns a priority score to each bin based on its remaining capacity, favoring bins that can accommodate the item without excessive waste, and penalizing bins that are nearly full or too large.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      priority.append(1 / (waste + 0.1)) 
    else:
      priority.append(-1000)
  return priority
