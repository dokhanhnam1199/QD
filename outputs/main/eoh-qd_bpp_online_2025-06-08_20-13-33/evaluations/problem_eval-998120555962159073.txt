def priority_v2(item, bins_remain_cap):
    """
    {This algorithm prioritizes bins based on a combination of remaining capacity and how well the item fits, favoring bins with enough space and a close fit.}
    """
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            fit_ratio = item / cap
            priorities.append(fit_ratio)
        else:
            priorities.append(-1)
    return priority
