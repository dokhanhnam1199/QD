def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin by considering the square of the remaining capacity after placing the item divided by the current remaining capacity, favoring bins that result in a higher percentage of used capacity relative to the original capacity of the bin.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append((cap - item)**2 / cap if cap > 0 else 0)
    else:
      priority.append(-1)
  return priority
