importance of each factor based on a tunable parameter.}"""
    priorities = []
    alpha = 0.5 
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            fill_ratio = item / cap
            
            priority = alpha * (1 / (waste + 0.0001)) + (1 - alpha) * fill_ratio
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
