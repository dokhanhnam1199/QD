def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates a priority score for each bin based on the waste ratio if the item is placed into the bin, favoring bins that minimize waste.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      waste_ratio = waste / cap if cap > 0 else 0
      priority_score = 1 - waste_ratio # Higher score for lower waste
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
