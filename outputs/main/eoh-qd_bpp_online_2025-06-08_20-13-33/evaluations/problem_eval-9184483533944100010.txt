def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the wasted space if the item is placed, with a penalty for small remaining capacities and a bonus for filling the bin closer to its maximum capacity.}"""
  priority = []
  for cap in bins_remain_cap:
    if item <= cap:
      waste = cap - item
      
      # Penalty for small remaining capacities.
      small_cap_penalty = 0
      if cap < 15:
        small_cap_penalty = -0.2
      
      # Bonus for filling the bin closer to its maximum capacity
      fill_bonus = (100 - cap + item) / 100
      
      priority.append(1 / (waste + 0.01) + small_cap_penalty + fill_bonus)
    else:
      priority.append(-1000)
  return priority
