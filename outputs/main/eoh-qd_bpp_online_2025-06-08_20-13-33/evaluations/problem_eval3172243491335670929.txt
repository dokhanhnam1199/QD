def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a score based on how fully the item will fill the bin, rewarding bins that become mostly full after the item is placed.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = item / cap
      priority.append(fill_ratio)
    else:
      priority.append(-1.0)
  return priority
