def priority_v2(item, bins_remain_cap):
    """
    {This algorithm calculates the priority of each bin based on a combination of remaining capacity and the wasted space if the item is placed in the bin, favoring bins with closer remaining capacity to item size and less waste.}
    """
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            priority.append(cap - waste/2 ) # Modified priority
        else:
            priority.append(-1)
    return priority
