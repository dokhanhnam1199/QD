def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority score to each bin based on a combination of remaining capacity and a capacity utilization factor, rewarding bins that are a good fit for the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      utilization = item / cap
      priority.append(utilization / (waste + 0.1))
    else:
      priority.append(-1000)
  return priority
