def priority_v2(item, bins_remain_cap):
    """
    {This algorithm calculates a priority score for each bin based on the square of the remaining capacity after placing the item, inversely weighted by the initial remaining capacity.}
    """
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = (cap - item)**2 / (cap + 0.0001)
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
