```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function combining Best Fit for large items and Worst Fit for small items.
    
    Uses a threshold of 0.5 to determine item size class. For large items (>=0.5), prioritizes
    bins with smallest remaining capacity (Best Fit). For small items, prioritizes bins with
    largest remaining capacity (Worst Fit). Ensures proper handling of edge cases like
    item size = 0.5.
    """
    can_fit = bins_remain_cap >= item
    if item >= 0.5:
        # Best Fit: prioritize bins with smallest remaining capacity
        priority = np.where(can_fit, -bins_remain_cap, -np.inf)
    else:
        # Worst Fit: prioritize bins with largest remaining capacity
        priority = np.where(can_fit, bins_remain_cap, -np.inf)
    return priority
```
