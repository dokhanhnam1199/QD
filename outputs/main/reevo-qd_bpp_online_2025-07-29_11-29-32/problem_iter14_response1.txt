def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function using threshold-based heuristic: Best Fit for large items (>0.5), Worst Fit for small."""
    can_fit = bins_remain_cap >= item
    if item > 0.5:
        priority = item - bins_remain_cap  # Prioritize minimal residual for large items
    else:
        priority = bins_remain_cap  # Prioritize largest space for small items
    return np.where(can_fit, priority, -np.inf)
