def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function prioritizing bins with minimal slack to reduce fragmentation.
    
    Calculates priority scores by evaluating the remaining capacity after placing the item.
    Valid bins (with sufficient capacity) are scored by negative slack (item - remaining),
    ensuring bins with the tightest fit (smallest leftover space) receive the highest priority.
    Invalid bins are assigned -inf to exclude them from selection.
    """
    slack = bins_remain_cap - item
    valid = slack >= 0
    return np.where(valid, -slack, -np.inf)
