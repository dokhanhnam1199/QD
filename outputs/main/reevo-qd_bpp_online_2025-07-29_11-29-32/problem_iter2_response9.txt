```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores based on item-to-bin remaining capacity ratio.
    
    Prioritizes bins that can fit the item by maximizing the ratio of item size
    to bin's remaining capacity. This encourages tight packing and aligns with
    the Best Fit heuristic. Invalid bins (cannot fit the item) receive -inf.
    """
    valid = bins_remain_cap >= item
    priority = np.full_like(bins_remain_cap, -np.inf, dtype=np.float64)
    priority[valid] = item / bins_remain_cap[valid]
    return priority
```
