import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Improved priority function using Best Fit for large items and enhanced Worst Fit for small items with perfect fit preference."""
    can_fit = bins_remain_cap >= item
    slack = bins_remain_cap - item
    
    if item >= 0.5:
        priority = -slack  # Best Fit: prioritize minimal slack
    else:
        perfect_fit = slack <= 1e-5  # Consider near-perfect fits
        priority = np.where(perfect_fit, np.inf, bins_remain_cap)  # Prefer perfect fits, then Worst Fit
    
    return np.where(can_fit, priority, -np.inf)
