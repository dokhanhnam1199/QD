def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Implements Best Fit heuristic for online bin packing.

    Computes priority scores by minimizing residual space after placement.
    Valid for single item application where bins are fixed-sized.

    Args:
        item: Size of item to be packed (0 <= item <= 1)
        bins_remain_cap: Array of current bin capacities

    Returns:
        Priorities with same shape as bins_remain_cap. Higher values are better.
        Infeasible bins are marked with -inf priority.
    """
    can_fit = bins_remain_cap >= item
    return np.where(can_fit, -(bins_remain_cap - item), -np.inf)
