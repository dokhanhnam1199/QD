import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores using Best Fit for items â‰¥0.5 and Worst Fit for smaller, balancing bin utilization."""
    can_fit = bins_remain_cap >= item
    if item >= 0.5:
        return np.where(can_fit, -bins_remain_cap, -np.inf)
    else:
        return np.where(can_fit, bins_remain_cap, -np.inf)
