def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores based on item-to-capacity ratio for dynamic adaptive selection.

    Args:
        item: Size of the item to pack.
        bins_remain_cap: Array of each bin's remaining capacity.

    Returns:
        Array with priority scores where higher values indicate better bin fit (less leftover space).
    """
    return np.where(bins_remain_cap >= item, item / bins_remain_cap, -np.inf)
