import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function implementing Best Fit heuristic with floating point tolerance."""
    EPSILON = 1e-9
    can_fit = (bins_remain_cap + EPSILON) >= item
    residual = bins_remain_cap - item
    priority = np.where(can_fit, -residual, -np.inf)
    return priority
