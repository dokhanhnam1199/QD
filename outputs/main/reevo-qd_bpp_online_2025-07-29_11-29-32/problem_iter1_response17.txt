import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority based on negative of leftover space (Best Fit heuristic)."""
    leftover = bins_remain_cap - item
    valid = leftover >= 0
    # Assign negative leftover as priority (smallest leftover has highest priority)
    # Invalid bins receive a very low priority to avoid selection
    return np.where(valid, -leftover, -1e10)
