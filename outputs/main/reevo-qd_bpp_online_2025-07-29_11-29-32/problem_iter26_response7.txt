def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Implements Best Fit heuristic for bin selection in online Bin Packing."""
    EPSILON = 1e-9  # For numeric stability
    can_fit = bins_remain_cap >= item - EPSILON  # Allow for precision errors
    residual = bins_remain_cap - item
    # Prioritize bins with smallest residual space
    return np.where(can_fit, -residual, -np.inf)
