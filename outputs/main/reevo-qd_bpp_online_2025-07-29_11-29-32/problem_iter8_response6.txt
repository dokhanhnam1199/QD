def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins using a modified Best Fit heuristic.

    Bins that can fit the item are prioritized by the negative of the leftover space after placement.
    Infeasible bins are assigned -inf to ensure they are not selected. This encourages minimal
    leftover space and harshly penalizes invalid placements.

    Args:
        item: Size of the item to pack.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin.
    """
    can_fit = bins_remain_cap >= item
    priority = np.where(can_fit, -(bins_remain_cap - item), -np.inf)
    return priority
