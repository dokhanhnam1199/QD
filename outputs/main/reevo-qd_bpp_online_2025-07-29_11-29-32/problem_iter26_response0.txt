def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit heuristic.

    Bins that can fit the item are prioritized by the smallest residual capacity after placement.
    Bins that cannot fit the item are assigned -inf priority.

    Args:
        item: Size of item to place.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Priorities for each bin where higher value indicates better fit.
    """
    can_fit = bins_remain_cap >= item
    priority = np.where(can_fit, -(bins_remain_cap - item), -np.inf)
    return priority
