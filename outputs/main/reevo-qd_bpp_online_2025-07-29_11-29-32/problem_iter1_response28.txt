import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit heuristic.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores where higher values indicate better fit.
    """
    # Calculate remaining capacity after placing the item
    remaining = bins_remain_cap - item
    # Identify valid bins (can fit the item)
    valid = remaining >= 0
    # Priority is negative of remaining space (prioritizes smallest leftover)
    priority = np.where(valid, -remaining, -np.inf)
    return priority
