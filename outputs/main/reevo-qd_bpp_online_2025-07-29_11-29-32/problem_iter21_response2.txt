```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins using Best/Worst Fit heuristics based on item size threshold."""
    can_fit = bins_remain_cap >= item
    threshold = 0.5
    use_best = item > threshold
    
    best_fit_score = -(bins_remain_cap - item)  # Minimize slack for large items
    worst_fit_score = bins_remain_cap          # Preserve capacity for small items
    
    priority = np.where(use_best, best_fit_score, worst_fit_score)
    priority = np.where(can_fit, priority, -np.inf)
    
    return priority
```
