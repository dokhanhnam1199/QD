def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on the ratio of item size to bin's remaining capacity.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin. Higher values indicate better fit.
    """
    # Prioritize bins where the item occupies a larger fraction of the bin's remaining capacity.
    # This ensures tight fits, with higher priority given to bins where item/rem_cap is maximized.
    return np.where(bins_remain_cap >= item, item / bins_remain_cap, -np.inf)
