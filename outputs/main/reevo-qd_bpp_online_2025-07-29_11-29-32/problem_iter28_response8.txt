import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function implementing Best Fit strategy for online Bin Packing Problem.
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
    
    Returns:
        Array of priority scores for each bin, where higher score indicates higher preference.
    """
    if bins_remain_cap.size == 0:
        return np.array([], dtype=np.float64)
    
    C = np.max(bins_remain_cap)  # Determine bin capacity
    can_fit = bins_remain_cap >= item  # Feasibility filter
    filled_before = C - bins_remain_cap  # Current utilization of bins
    
    # Prioritize by current utilization for bins that can fit the item
    return np.where(can_fit, filled_before, -np.inf)
