import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Improved priority function combining Best Fit and Worst Fit linearly with item size."""
    can_fit = bins_remain_cap >= item
    slack = bins_remain_cap - item
    
    # Adaptive weight based on item size (higher weight â†’ favors Best Fit)
    weight = item
    priority = -slack * weight + bins_remain_cap * (1 - weight)
    
    return np.where(can_fit, priority, -np.inf)
