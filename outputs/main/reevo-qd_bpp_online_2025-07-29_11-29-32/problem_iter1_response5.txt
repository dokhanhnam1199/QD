import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins that can fit the item with minimal leftover space (Best Fit heuristic).
    
    Bins that cannot fit the item are assigned -infinity. The bin with the smallest
    remaining capacity >= item (i.e., minimal leftover) receives the highest priority.
    """
    # Calculate the leftover space if item is placed in each bin
    leftover = bins_remain_cap - item
    
    # Prioritize bins with minimal non-negative leftover (Best Fit)
    # Scores for valid bins are negative of leftover to favor smaller leftover
    return np.where(leftover >= 0, -leftover, -np.inf)
