import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritize bins with residual after placement closest to zero using quadratic penalty."""
    can_fit = bins_remain_cap >= item
    residual_after = bins_remain_cap - item
    
    # Target residual after placement is zero (Best Fit), penalize deviations quadratically
    priority = -residual_after**2
    
    # Mask infeasible bins and return scores
    return np.where(can_fit, priority, -np.inf)
