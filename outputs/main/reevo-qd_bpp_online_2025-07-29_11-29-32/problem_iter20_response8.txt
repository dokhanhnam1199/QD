import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Improved priority function using adaptive weights based on item size for Best/Worst Fit blending."""
    can_fit = bins_remain_cap >= item
    slack = bins_remain_cap - item

    # Dynamic blend between Best Fit (-slack) and Worst Fit (bins_remain_cap) using item size
    weight = 1.0 - np.exp(-item * 5)  # Exponential weighting for sharper transition at small item sizes
    priority = weight * (-slack) + (1 - weight) * bins_remain_cap

    return np.where(can_fit, priority, -np.inf)
