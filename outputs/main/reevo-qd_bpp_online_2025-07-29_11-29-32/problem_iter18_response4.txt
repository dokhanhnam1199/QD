def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit heuristic.

    Bins that can fit the item are prioritized by smallest residual capacity after placement.
    Infeasible bins receive -inf priority.

    Args:
        item: Size of the item to be added.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin.
    """
    can_fit = bins_remain_cap >= item
    residual = bins_remain_cap - item
    priority = np.where(can_fit, -residual, -np.inf)
    return priority
