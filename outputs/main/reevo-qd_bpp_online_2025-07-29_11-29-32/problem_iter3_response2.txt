```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores combining Best Fit for large and Worst Fit for small items.
    
    Uses a dynamic formula based on item size to balance slack minimization and space preservation.
    """
    can_fit = bins_remain_cap >= item
    priority = np.where(
        can_fit,
        bins_remain_cap * (1 - 2 * item) + (item ** 2),
        -np.inf
    )
    return priority
```
