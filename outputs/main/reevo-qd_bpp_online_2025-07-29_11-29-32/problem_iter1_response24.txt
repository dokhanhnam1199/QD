```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores based on minimizing leftover space after placement.
    
    Bins that can fit the item are prioritized by the inverse of resulting leftover space.
    Bins that cannot fit the item are assigned a very low priority.
    """
    can_fit = bins_remain_cap >= item
    leftover = bins_remain_cap - item
    # Use negative leftover as priority (smaller leftover = higher priority)
    # Add a small epsilon to avoid division by zero in edge cases
    priority = np.where(can_fit, -leftover + 1e-9, -np.inf)
    return priority
```
