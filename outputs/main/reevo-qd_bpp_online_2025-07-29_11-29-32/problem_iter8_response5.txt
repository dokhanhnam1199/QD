import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Improved priority function using hybrid Best Fit/Worst Fit based on item size."""
    can_fit = bins_remain_cap >= item
    # Use Best Fit for large items (> 0.5), Worst Fit for small items
    if item > 0.5:
        # Prioritize bins with smallest remaining capacity after placement (Best Fit)
        priority = np.where(can_fit, -(bins_remain_cap - item), -np.inf)
    else:
        # Prioritize bins with largest remaining capacity (Worst Fit)
        priority = np.where(can_fit, bins_remain_cap, -np.inf)
    return priority
