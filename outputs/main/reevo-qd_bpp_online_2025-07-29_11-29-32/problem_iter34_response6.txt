import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on remaining capacity after packing.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores where higher values indicate better fit:
        - Prioritize bins that can fit the item with minimal remaining capacity.
        - Invalid bins (cannot fit the item) are heavily penalized.
    """
    # Calculate priority as negative remaining capacity to select bins with smallest space left.
    return np.where(bins_remain_cap >= item, -bins_remain_cap, -np.inf)
