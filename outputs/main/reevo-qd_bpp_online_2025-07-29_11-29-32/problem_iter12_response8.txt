def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on efficient utilization of remaining capacity.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin. Higher values indicate better fit and more efficient utilization.
    """
    # Prioritize bins that can fit the item with the highest ratio of item size to bin's remaining capacity.
    # This metric emphasizes using the remaining capacity efficiently, favoring bins where the item fills a larger portion.
    return np.where(bins_remain_cap >= item, item / bins_remain_cap, -np.inf)
