```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function that prioritizes bins fitting the item exactly first, 
    followed by bins with minimal leftover space (Best Fit with exact match bonus)."""
    mask = bins_remain_cap >= item
    space_left = bins_remain_cap - item
    exact_fill = np.isclose(bins_remain_cap, item)  # Handle floating-point precision
    # Assign high priority to exact fits; otherwise use negative space_left for Best Fit
    priorities = np.where(mask, 1e6 * exact_fill - space_left, -np.inf)
    return priorities
```
