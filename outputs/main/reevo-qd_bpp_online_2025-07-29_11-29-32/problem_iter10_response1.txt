import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Improved priority function using a smooth transition between Best Fit and Worst Fit based on item size."""
    can_fit = bins_remain_cap >= item
    slack = bins_remain_cap - item
    alpha = np.clip(2 * item, 0, 1)  # Smooth blending factor: 0 (WF) for small items, 1 (BF) for large items
    priority = alpha * (-slack) + (1 - alpha) * bins_remain_cap
    return np.where(can_fit, priority, -np.inf)
