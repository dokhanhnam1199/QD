import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Adaptive BF-WF: weighted blend of slack minimization and capacity preservation."""
    can_fit = bins_remain_cap >= item
    slack = bins_remain_cap - item

    # Balance BF (-slack) and WF (bin remainder) using item size as weight
    alpha = item  # Weight for slack minimization (BF component)
    beta = 1 - item  # Weight for capacity preservation (WF component)
    priority = alpha * (-slack) + beta * bins_remain_cap

    return np.where(can_fit, priority, -np.inf)
