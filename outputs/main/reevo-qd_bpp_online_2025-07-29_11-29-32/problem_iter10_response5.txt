import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Improved adaptive priority with exact-fit prioritization and Best/Worst Fit."""
    can_fit = bins_remain_cap >= item
    slack = bins_remain_cap - item
    exact_fit = (slack == 0) & can_fit
    
    # Use Best Fit for large items (min slack) and Worst Fit for small items (max capacity)
    if item >= 0.5:
        temp_priority = -slack  # -slack ensures minimal slack gets highest priority
    else:
        temp_priority = bins_remain_cap  # Max remaining capacity
    
    # Prioritize exact fits above all else
    priority = np.where(exact_fit, np.inf, temp_priority)
    
    # Apply -inf to bins that can't fit the item
    return np.where(can_fit, priority, -np.inf)
