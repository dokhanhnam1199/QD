import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function using a hybrid strategy: Best Fit for large items (>=0.5), Worst Fit for small items."""
    valid_mask = bins_remain_cap >= item
    priority = np.full_like(bins_remain_cap, -np.inf, dtype=np.float64)
    
    if not np.any(valid_mask):
        return priority  # No valid bins to place the item
    
    if item >= 0.5:
        # Best Fit: prioritize bins with smallest remaining capacity
        priority[valid_mask] = -bins_remain_cap[valid_mask]
    else:
        # Worst Fit: prioritize bins with largest remaining capacity
        priority[valid_mask] = bins_remain_cap[valid_mask]
    
    return priority
