import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores based on the Best Fit heuristic.

    The priority for each bin is the negative of its remaining capacity if the item
    can fit, otherwise negative infinity. This encourages placing items in bins
    with the smallest remaining capacity that can still accommodate the item.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin.
    """
    priority = np.full_like(bins_remain_cap, -np.inf, dtype=np.float64)
    can_fit = bins_remain_cap >= item
    priority[can_fit] = -bins_remain_cap[can_fit]
    return priority
