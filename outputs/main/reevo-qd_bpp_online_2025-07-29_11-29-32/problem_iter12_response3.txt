def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Dynamic combination of Best Fit and Worst Fit using a streamlined polynomial expression."""
    can_fit = bins_remain_cap >= item
    priority = np.where(
        can_fit,
        bins_remain_cap * (1 - 2 * item),
        -np.inf
    )
    return priority
