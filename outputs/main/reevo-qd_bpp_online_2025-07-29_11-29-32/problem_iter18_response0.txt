def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores using threshold-based Best Fit for large and Worst Fit for small items.
    
    Directly switches between Best Fit (minimizing slack) when item > 0.5 and Worst Fit (maximizing remaining space)
    when item <= 0.5. Uses linear priority scoring and avoids complex penalty terms.
    """
    can_fit = bins_remain_cap >= item
    if item > 0.5:
        # Best Fit: prioritize bins with minimal residual space after packing
        priority = - (bins_remain_cap - item)
    else:
        # Worst Fit: prioritize bins with maximum remaining capacity
        priority = bins_remain_cap.copy()
    return np.where(can_fit, priority, -np.inf)
