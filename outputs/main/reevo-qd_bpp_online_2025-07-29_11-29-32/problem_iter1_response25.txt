```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function combining Best Fit for large items and Worst Fit for small items."""
    can_fit = bins_remain_cap >= item
    large_item_threshold = 0.5
    
    if item >= large_item_threshold:
        # Best Fit: prioritize bins with remaining capacity closest to item size
        fit_scores = item - bins_remain_cap
    else:
        # Worst Fit: prioritize bins with largest remaining capacity
        fit_scores = bins_remain_cap
    
    # Assign -inf to bins that can't fit the item
    priorities = np.where(can_fit, fit_scores, -np.inf)
    return priorities
```
