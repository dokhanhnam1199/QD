import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for each bin using adaptive BF/WF strategy."""
    if bins_remain_cap.size == 0:
        return np.array([])
    
    C = np.max(bins_remain_cap)
    can_fit = bins_remain_cap >= item - 1e-9
    
    if not np.any(can_fit):
        return np.full_like(bins_remain_cap, -np.inf)
    
    residual = bins_remain_cap - item
    item_ratio = item / (C + 1e-9)
    
    bf_component = C - residual
    wf_component = residual
    
    weight_bf = item_ratio
    weight_wf = 1.0 - weight_bf
    
    priority = weight_bf * bf_component + weight_wf * wf_component
    priority = np.where(can_fit, priority, -np.inf)
    
    return priority
