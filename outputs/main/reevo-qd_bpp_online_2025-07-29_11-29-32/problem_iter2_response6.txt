def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins using Best Fit heuristic.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores where higher values indicate better fit.
    """
    # Calculate remaining space after placing the item
    remaining = bins_remain_cap - item
    # Mask for bins that can accommodate the item
    valid = remaining >= 0
    # Prioritize bins with minimal leftover space (Best Fit)
    priority = np.where(valid, -remaining, -np.inf)
    return priority
