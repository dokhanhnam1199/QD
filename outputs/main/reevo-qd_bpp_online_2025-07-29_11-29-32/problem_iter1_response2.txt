```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins using a Best Fit heuristic.
    
    The priority for each bin is the negative of its remaining capacity after placing
    the item, if it fits. This encourages selecting bins that minimize leftover space.
    Bins that cannot fit the item are assigned a very low priority.
    """
    can_fit = bins_remain_cap >= item
    return np.where(
        can_fit,
        item - bins_remain_cap,  # Equivalent to -(remaining_cap - item)
        -np.inf
    )
```
