def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on minimizing leftover space.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin. Higher values indicate better fit.
    """
    # Prioritize bins that can fit the item with the least leftover space.
    # Valid bins have scores item - remaining_cap (<= 0), with higher (less negative) values being better.
    # Invalid bins are assigned -inf to ensure they are not selected.
    return np.where(bins_remain_cap >= item, item - bins_remain_cap, -np.inf)
