def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on inverse of leftover space.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin. Higher values indicate better fit.
    """
    # Calculate leftover space after packing the item
    leftover = bins_remain_cap - item
    # Prioritize bins with minimal leftover space using inverse scaling
    # Bins that can't fit the item are assigned -inf
    return np.where(bins_remain_cap >= item, 1.0 / (leftover + 1), -np.inf)
