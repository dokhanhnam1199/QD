import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins using a Best Fit strategy.

    The priority is highest for the bin with the smallest remaining capacity that can fit the item.
    Bins that cannot accommodate the item are assigned -infinity.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin.
    """
    eligible = bins_remain_cap >= item
    return np.where(eligible, item - bins_remain_cap, -np.inf)
