```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Improved priority function implementing Best Fit heuristic for online BPP."""
    if bins_remain_cap.size == 0:
        return np.array([], dtype=np.float64)  # Handle empty bins case
    
    # Determine which bins can accommodate the item
    can_fit = bins_remain_cap >= item
    
    # Calculate residual space after placing item
    residuals = bins_remain_cap - item
    
    # Best Fit: Prioritize bins with minimal residual space
    priorities = np.where(can_fit, -residuals, -np.inf)
    
    return priorities.astype(np.float64)  # Ensure float type for numerical stability
```
