def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority to each bin based on the ratio of the item size to the remaining capacity of the bin, penalized by the amount of wasted space if the item is placed in the bin.}"""
  priorities = []
  for capacity in bins_remain_cap:
    if capacity >= item:
      priority = (item / capacity) - 0.1 * (capacity - item)
    else:
      priority = -1000  # Assign a very low priority if the bin cannot fit the item
    priorities.append(priority)
  return priority
