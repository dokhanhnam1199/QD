def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on the ratio of remaining capacity to item size, penalized by the absolute difference between remaining capacity and item size.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority = (cap / item) - abs(cap - item)
    else:
      priority = -1
    priorities.append(priority)
  return priority
