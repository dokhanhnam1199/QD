def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority score for each bin based on the utilization rate after packing the item, prioritizing bins that lead to higher utilization without exceeding capacity.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = item / (cap+1e-9) # Prioritize higher utilization
        else:
            priority = -1 
        priorities.append(priority)
    return priority
