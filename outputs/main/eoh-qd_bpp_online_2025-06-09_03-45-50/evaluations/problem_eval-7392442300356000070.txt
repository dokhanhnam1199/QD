def priority_v2(item, bins_remain_cap):
  """{This algorithm penalizes bins with very small remaining capacity or requires to open a new bin, and favors bins with remaining capacity close to the item size.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      diff = abs(cap - item)
      priority_score = 1 / (1 + diff)
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
