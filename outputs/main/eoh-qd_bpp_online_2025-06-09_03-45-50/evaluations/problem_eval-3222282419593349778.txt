def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority based on the absolute difference between item size and remaining capacity, favoring bins where the item fills the bin most completely.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority_score = 1 - abs(cap - item)
            priority.append(priority_score)
        else:
            priority.append(-1)
    return priority
