def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority score for each bin based on the ratio of item size to remaining capacity, penalizing bins that are either too full or too empty.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = item / cap #Prioritize bins where the item fills a larger proportion of the bin.
        else:
            priority = -1 # Invalid bin.
        priorities.append(priority)
    return priority
