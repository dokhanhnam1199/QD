def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the smallest waste, ensuring the item fits and minimizing unused space in each bin, with a penalty for exceeding a waste threshold.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      priority = -waste
      if waste > 0.5:
        priority -= 50
    else:
      priority = -1000
    priorities.append(priority)
  return priority
