def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority scores based on the wasted space if an item is placed in a bin, giving higher priority to bins where the wasted space is minimized relative to the item size.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            wasted_space_ratio = (cap - item) / item
            priority.append(1 / (1 + wasted_space_ratio)) #Prioritize lower waste
        else:
            priority.append(-1)
    return priority
