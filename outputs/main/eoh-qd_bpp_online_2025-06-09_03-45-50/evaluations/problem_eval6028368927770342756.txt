def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns a score to each bin based on how well the item's size matches the bin's remaining capacity, using the absolute difference between the item's size and half of the remaining capacity, favoring matches close to half-full.}"""
    priority = []
    for cap in bins_remain_cap:
        if item <= cap:
            priority.append(-abs(item - cap / 2.0))
        else:
            priority.append(-float('inf'))
    return priority
