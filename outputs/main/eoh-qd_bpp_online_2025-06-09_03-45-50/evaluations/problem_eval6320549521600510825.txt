def priority_v3(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the absolute remaining capacity after placing the item, favoring bins with minimal remaining space and avoiding bins where the item won't fit.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap < item:
            priority.append(-1)
        else:
            remaining_cap = cap - item
            priority.append(1 - remaining_cap)
    return priority
