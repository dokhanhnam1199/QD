def priority_v2(item, bins_remain_cap):
    """{Assign a priority score to each bin based on the waste space if the item were placed in the bin, favoring bins with minimal waste and avoiding bins where the item doesn't fit.}"""
    priority = []
    for cap in bins_remain_cap:
        if item <= cap:
            waste = cap - item
            priority.append(1 - waste)  # Higher priority for less waste
        else:
            priority.append(-1)  # Item doesn't fit
    return priority
