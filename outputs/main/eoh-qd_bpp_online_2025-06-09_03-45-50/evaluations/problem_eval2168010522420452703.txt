def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on a combination of remaining capacity and waste minimization.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      priority = (1.0 / (waste + 0.1)) + (cap / sum(bins_remain_cap)) # Avoid zero division and add remaining capacity factor
    else:
      priority = -1  # Assign low priority to bins that cannot fit the item
    priorities.append(priority)
  return priority
