def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates priority scores based on the percentage of bin capacity the item would occupy, favoring bins where the item fills a significant portion without exceeding capacity.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = item / cap
      priority.append(fill_ratio)
    else:
      priority.append(-1)
  return priority
