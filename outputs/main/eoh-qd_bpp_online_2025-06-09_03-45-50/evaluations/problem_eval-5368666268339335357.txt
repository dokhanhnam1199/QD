def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the square of the remaining capacity after packing the item, effectively favoring bins that offer a tighter fit but still have sufficient space.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append((cap - item)**2)
        else:
            priority.append(-1)  # Invalid bin
    return priority
