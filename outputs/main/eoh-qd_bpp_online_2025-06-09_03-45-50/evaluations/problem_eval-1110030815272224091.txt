def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, penalizing both bins that are too full and those that are excessively empty.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = item / cap - (cap - item) / (cap + 1e-9)
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
