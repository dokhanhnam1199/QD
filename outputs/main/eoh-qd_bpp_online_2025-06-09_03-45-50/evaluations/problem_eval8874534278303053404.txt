def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity if the item fits, otherwise assigns a negative priority.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = item / cap # Ratio of item size to remaining capacity.
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
