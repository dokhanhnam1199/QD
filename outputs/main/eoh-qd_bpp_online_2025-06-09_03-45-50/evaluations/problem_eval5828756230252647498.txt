def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin by considering the remaining capacity, giving higher priority to bins that can accommodate the item with minimal wasted space, and penalizing bins that are too full or too empty.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      priorities.append(1 / (cap - item + 0.0001))
