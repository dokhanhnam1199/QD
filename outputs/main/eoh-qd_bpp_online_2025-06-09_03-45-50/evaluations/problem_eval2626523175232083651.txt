def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on how close the item size is to the remaining capacity of each bin, while penalizing bins with too much or too little remaining space relative to the item.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = (item / cap) - abs(1 - (item / cap))
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
