def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins with remaining capacity close to the item size, penalizing both very empty and nearly full bins to encourage balanced utilization.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority = 1.0 / abs(cap - item + 0.0001) # Higher priority if cap is close to item
      if cap > item + 0.5: #0.5 threshold
          priority -= 5 #penalty for being too empty
      if cap - item < 0.1: #0.1 threshold
          priority -= 5 #penalty for being nearly full
    else:
      priority = -1000 #very low priority, makes it undesirable
    priorities.append(priority)
  return priority
