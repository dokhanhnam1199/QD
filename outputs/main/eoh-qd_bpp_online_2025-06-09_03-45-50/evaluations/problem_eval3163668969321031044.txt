def priority_v3(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on remaining capacity fit, favoring bins that are closest to half-full after packing the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = abs(0.5 - (item) / (item + (cap - item))) #abs(0.5 - item / bin_cap)
      priority.append(1-fill_ratio)
    else:
      priority.append(-1)
  return priority
