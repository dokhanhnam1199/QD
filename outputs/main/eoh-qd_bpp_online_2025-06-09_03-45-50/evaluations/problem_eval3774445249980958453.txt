def priority_v2(item, bins_remain_cap):
    """
    {This algorithm calculates the priority score for each bin based on the remaining capacity and item size, favoring bins with enough space but not excessively large remaining capacity.}
    """
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = (cap - item) / cap  # Prioritize smaller waste
        else:
            priority = -1  # Invalid bin
        priorities.append(priority)
    return priority
