def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priorities based on the ratio of item size to the squared remaining capacity, penalizing bins with very small remaining capacity and prioritizing bins where the item fills a significant portion of the bin.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(item / (cap**2))
    else:
      priority.append(-1)
  return priority
