def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates a priority score for each bin based on the remaining capacity and a bonus for bins that are filled to a specific target level.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority = cap  # Base priority: the remaining capacity
      target_fill = 0.75 # Target fill level is 75%
      if abs((cap - item) / cap - (1 - target_fill)) < 0.1:  # check if the bin is near target fill level
        priority += 20  # Bonus for filling near target
      priorities.append(priority)
    else:
      priorities.append(-float('inf'))  # Assign negative infinity if item doesn't fit
  return priority
