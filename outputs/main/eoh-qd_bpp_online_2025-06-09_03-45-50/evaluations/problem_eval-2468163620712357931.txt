def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on a combination of remaining capacity fit and how full the bin would be after packing the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = (item) / (item + (cap - item)) #item / bin_cap
      priority.append(fill_ratio)
    else:
      priority.append(-1)
  return priority
