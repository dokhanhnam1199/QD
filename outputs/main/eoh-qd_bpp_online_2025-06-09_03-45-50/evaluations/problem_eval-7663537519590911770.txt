def priority_v2(item, bins_remain_cap):
  """{Assign a priority score to each bin based on how much the item would fill the bin and how close it would get to full, favoring bins that would be filled to a higher percentage.}"""
  priority = []
  for cap in bins_remain_cap:
    if item <= cap:
      fill_percentage = item / cap
      remaining_percentage = (cap - item) / cap if cap - item > 0 else 0
      priority_score = fill_percentage + (1 - remaining_percentage) 
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
