def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority scores based on the ratio of item size to remaining capacity, with a bonus for bins that can accommodate the item within a certain capacity threshold and a penalty for those that cannot.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            capacity_ratio = item / cap
            if cap <= 2 * item: # Bonus if the bin is not too much larger than item
                priority = capacity_ratio + 0.5
            else:
                priority = capacity_ratio
        else:
            priority = -1000
        priorities.append(priority)
    return priority
