def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the waste and a bonus for bins already containing items, aiming to balance fitting items well and distributing items across bins.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      if cap < 1:
          cap = 1 #avoid division by zero
      occupancy_bonus = (0.5 if cap != bins_remain_cap[0] else 0) #giving bonus to the bin if it is not an empty bin
      priority_score = -waste + occupancy_bonus
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
