def priority_v3(item, bins_remain_cap):
  """{This algorithm prioritizes bins with a closer fit to the item size while still considering remaining capacity, penalizing bins that become too empty after packing.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fit_ratio = item / cap
      remaining_ratio = (cap - item) / cap

      # Emphasize fit and penalize large remaining capacity.
      score = 0.9 * fit_ratio + 0.1 * (1 - remaining_ratio) 
      priority.append(score)
    else:
      priority.append(-1)  # Invalid bin
  return priority
