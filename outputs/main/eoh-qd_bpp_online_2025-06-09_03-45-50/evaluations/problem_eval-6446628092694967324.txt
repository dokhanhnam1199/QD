def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates a priority score for each bin based on how well the item fills the bin, penalizing bins with too little or too much remaining capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = item / cap
            priority.append(fill_ratio - abs(fill_ratio - 0.8)**2) # Prefer bins filled around 80%
        else:
            priority.append(-1000)  # Very low priority if item doesn't fit

    return priority
