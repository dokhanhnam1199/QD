def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on minimizing wasted space while also penalizing bins that are already close to full.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      priority.append(1 / (waste + 1) - (1 - (cap / 100 if type(cap) is int else cap))) # Assuming bin capacity is 100; 100 is a constant
    else:
      priority.append(-1)
  return priority
