def priority_v2(item, bins_remain_cap):
  """{Assign a priority score based on the wasted space after placing the item, prioritizing bins with less wasted space and penalizing bins that cannot accommodate the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if item <= cap:
      wasted_space = cap - item
      priority_score = 1 / (wasted_space + 1) 
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
