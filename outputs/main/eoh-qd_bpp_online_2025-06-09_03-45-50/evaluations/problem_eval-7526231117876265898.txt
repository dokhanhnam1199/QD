def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins by combining the waste ratio if the item is placed in that bin and the current utilization of the bin, penalizing both high waste and low utilization.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste_ratio = (cap - item) / cap
      utilization_ratio = 1 - (cap / 100)  # Assuming bin size is 100
      score = 1 - (0.6 * waste_ratio + 0.4 * utilization_ratio)
      priority.append(score)
    else:
      priority.append(-1)
  return priority
