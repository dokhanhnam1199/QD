def priority_v2(item, bins_remain_cap):
    """{Assign a priority score to each bin based on the ratio of item size to remaining capacity, normalized by the remaining capacity itself, favoring bins that closely fit the item but also have smaller capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap < item:
            priority.append(-1)
        else:
            priority.append((item / cap) / cap)
    return priority
