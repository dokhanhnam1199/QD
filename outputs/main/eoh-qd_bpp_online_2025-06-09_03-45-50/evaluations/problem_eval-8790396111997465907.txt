def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority scores based on how fully an item can fill a bin, penalizing both insufficient space and excessive wasted space.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = item / cap
            priority = fill_ratio - abs(fill_ratio - 0.8)  # Target fill ratio around 80%
        else:
            priority = -1
        priorities.append(priority)
    return priority
