def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority to each bin based on the ratio of item size to remaining capacity, penalizing bins that are too full or too empty.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = item / cap
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
