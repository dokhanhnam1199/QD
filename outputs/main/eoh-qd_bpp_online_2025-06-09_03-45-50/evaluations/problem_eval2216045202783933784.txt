def priority_v3(item, bins_remain_cap):
  """{Assign a priority score to each bin based on the ratio of remaining capacity to item size, favoring bins where the item fits well with minimal wasted space.}"""
  priority = []
  for cap in bins_remain_cap:
    if item <= cap:
      priority_score = cap / item
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
