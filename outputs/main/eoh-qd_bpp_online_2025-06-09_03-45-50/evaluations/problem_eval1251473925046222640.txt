def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on the ratio of item size to remaining capacity, penalized by the square of the wasted space.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      priority = (item / cap) - (waste**2 / (sum(bins_remain_cap)+0.1)) # Ratio of item to capacity, minus squared waste
    else:
      priority = -1
    priorities.append(priority)
  return priority
