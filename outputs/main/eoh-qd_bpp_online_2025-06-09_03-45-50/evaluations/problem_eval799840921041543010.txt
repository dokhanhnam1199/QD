def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on the fraction of the bin's remaining capacity that the item would occupy, favoring bins where the item fills a substantial portion but not all of the remaining space.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = float(item) / cap
      priority = fill_ratio if 0.5 <= fill_ratio <= 1.0 else -1
    else:
      priority = -1
    priorities.append(priority)
  return priority
