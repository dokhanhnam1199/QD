def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on the "square root of the remaining capacity" if the item fits, otherwise assigns a negative penalty, favoring bins with moderate remaining capacity.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append((cap**0.5))
    else:
      priority.append(-100)
  return priority
