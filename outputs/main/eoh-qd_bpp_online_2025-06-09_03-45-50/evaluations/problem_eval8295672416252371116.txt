def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the percentage of bin capacity the item will occupy, favoring bins where the item fills a significant portion without exceeding capacity.}"""
  priorities = []
  for capacity in bins_remain_cap:
    if capacity >= item:
      priority = item / capacity
    else:
      priority = -1
    priorities.append(priority)
  return priority
