def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the absolute difference between the remaining capacity and the item size, favoring bins where the item fills the bin most completely, and penalizing bins that are too small or result in large wasted space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(-abs(cap - item))
    else:
      priority.append(-float('inf'))
  return priority
