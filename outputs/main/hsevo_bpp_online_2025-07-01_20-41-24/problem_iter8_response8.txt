import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization, fill ratio, and a near-full bonus."""
    valid_bins = bins_remain_cap > 0
    if not np.any(valid_bins):
        return np.full_like(bins_remain_cap, -np.inf)

    waste = bins_remain_cap - item
    waste[waste < 0] = np.inf

    priority = np.where(waste >= 0, 1, -np.inf)

    fill_ratio = (bins_remain_cap - waste) / bins_remain_cap
    fill_ratio[np.isinf(fill_ratio)] = 0
    priority += fill_ratio * 10

    almost_full_bonus = np.where((fill_ratio >= 0.9), np.exp(fill_ratio * 5), 0)  # Exponential bonus
    priority += almost_full_bonus
    priority[bins_remain_cap < item] = -np.inf

    return priority
