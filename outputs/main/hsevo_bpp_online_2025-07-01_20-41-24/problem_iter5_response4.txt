```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins: Perfect fit > fill ratio > remaining space. Handles infeasible bins."""
    priorities = np.full(bins_remain_cap.shape, -np.inf)
    eligible_bins = bins_remain_cap >= item

    if not np.any(eligible_bins):
        space_needed = item - bins_remain_cap
        priorities = -space_needed
        return priorities
    
    priorities[eligible_bins] = 0

    perfect_fit = bins_remain_cap[eligible_bins] == item
    if np.any(perfect_fit):
        priorities[eligible_bins][perfect_fit] = np.inf

    remaining_space = bins_remain_cap[eligible_bins] - item
    fill_ratio = item / bins_remain_cap[eligible_bins]
    
    priorities[eligible_bins] = fill_ratio - (remaining_space * 0.01)

    return priorities
```
