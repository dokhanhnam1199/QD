import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins using a combination of fit and fullness."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    for i, cap in enumerate(bins_remain_cap):
        if item <= cap:
            remaining_space = cap - item
            priorities[i] = (cap*item) / (remaining_space + 1e-9) + (cap / np.sum(bins_remain_cap))
        else:
            priorities[i] = -1e9
    return priorities
