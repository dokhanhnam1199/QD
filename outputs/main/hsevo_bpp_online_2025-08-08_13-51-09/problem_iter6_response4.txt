def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin using a refined logic

    considers the "tightness" of the fit.

    if not np.any(suitable_bins_mask):
        return priorities
