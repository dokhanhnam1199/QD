import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Enhanced heuristic combining inverse distance, node degree penalty, 
    and a sparsification to focus the search.
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix)
    avg_distance = np.mean(distance_matrix[distance_matrix != 0])
    small_constant = 1e-9

    node_degrees = np.sum(distance_matrix < avg_distance * 1.5, axis=1)
    degree_penalty_exponent = 1.5

    for i in range(n):
        for j in range(n):
            if i != j:
                degree_penalty = (node_degrees[i] + node_degrees[j]) ** degree_penalty_exponent
                heuristics[i, j] = (1 / (distance_matrix[i, j] + small_constant)) / (1 + (degree_penalty * (distance_matrix[i, j]/avg_distance)))
            else:
                heuristics[i, j] = 0

    threshold = np.quantile(heuristics[heuristics > 0], 0.25)
    heuristics[heuristics < threshold] = 0

    return heuristics
