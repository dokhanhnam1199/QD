import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Combines inverse distance, degree penalty, sparsification, and a node potential factor.
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix)
    avg_distance = np.mean(distance_matrix[distance_matrix != 0])
    small_constant = 1e-6
    avg_distance_weight = 0.5

    # Node degrees as before, but scaled
    node_degrees = np.sum(distance_matrix < (avg_distance * 1.5), axis=1)
    normalized_degrees = node_degrees / np.max(node_degrees)  # Scale to [0, 1]

    # Calculate a "node potential" based on average distance to other nodes
    node_potential = np.mean(distance_matrix, axis=1)
    normalized_potential = (node_potential - np.min(node_potential)) / (np.max(node_potential) - np.min(node_potential) + small_constant) # Scale to [0, 1]

    for i in range(n):
        for j in range(n):
            if i != j:
                degree_penalty = (node_degrees[i] + node_degrees[j])**1.5
                # Inverse distance weighted by degree penalty
                heuristics[i, j] = (1 / (distance_matrix[i, j] + small_constant)) / (1 + (degree_penalty * (distance_matrix[i, j]/avg_distance) * avg_distance_weight))

                # Local neighborhood density penalty
                local_neighbors = np.sum(distance_matrix[i, :] < (distance_matrix[i, j] * 1.2)) + np.sum(distance_matrix[j, :] < (distance_matrix[j, i] * 1.2))
                heuristics[i,j] /= (1 + (local_neighbors * 0.1))

                # Node potential influence (nodes with lower potential are preferred)
                heuristics[i, j] *= (1 - (normalized_potential[i] + normalized_potential[j])/2)
            else:
                heuristics[i, j] = 0

    # Sparsification based on percentile
    for i in range(n):
        row = heuristics[i, :]
        threshold = np.percentile(row[row > 0], 75)
        heuristics[i, row < threshold] = 0

    # Distance thresholding to penalize long edges
    distance_threshold = np.percentile(distance_matrix[distance_matrix != 0], 25)
    for i in range(n):
        for j in range(n):
          if distance_matrix[i,j] > distance_threshold and heuristics[i,j] > 0:
             heuristics[i,j] *= 0.5

    return heuristics
