```python
import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Combines inverse distance with node degree and sparsification.
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix)
    avg_distance = np.mean(distance_matrix[distance_matrix != 0])
    small_constant = 1e-6
    avg_distance_weight = 0.1

    node_degrees = np.sum(distance_matrix < avg_distance, axis=1)

    # Sparsification based on percentile
    sparsification_percentile = 75
    threshold_distance = np.percentile(distance_matrix[distance_matrix != 0], sparsification_percentile)
    
    for i in range(n):
        for j in range(n):
            if i != j:
                if distance_matrix[i, j] > threshold_distance:
                    heuristics[i, j] = 0  # Sparsify
                else:
                    degree_penalty = (node_degrees[i] + node_degrees[j])
                    heuristics[i, j] = (1 / (distance_matrix[i, j] + small_constant)) / (1 + (degree_penalty * (distance_matrix[i, j]/avg_distance) * avg_distance_weight))
            else:
                heuristics[i, j] = 0

    return heuristics
```
