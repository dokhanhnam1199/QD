import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines fit score, utilization factor, and a simple tie-breaker for online Bin Packing Problem.
    """
    # Calculate remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Fit score: prioritize bins where the item fits well
    fit_score = np.where(bins_remain_cap >= item, -(np.abs(remaining_cap_after_addition)), -np.inf)
    
    # Utilization factor: prefer bins that are not too empty or too full after adding the item
    post_utilization_factor = -np.abs((bins_remain_cap - item) / (bins_remain_cap + 1e-6) - 0.5)
    
    # Tie-breaker based on bin index with a small coefficient
    tie_breaker = np.log(len(bins_remain_cap) / (1 + np.arange(len(bins_remain_cap)))) * 1e-3
    
    # Combine fit score, utilization factor, and tie-breaker
    priority_scores = fit_score + post_utilization_factor + tie_breaker
    
    return priority_scores
