```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for adding an item to bins.
    Prioritizes bins with just enough capacity, using negative remaining capacity and adding noise to break ties.
    """
    # Calculate priority scores for bins that can fit the item
    mask = bins_remain_cap >= item
    priority_scores = np.full_like(bins_remain_cap, -np.inf)
    # Use negative remaining capacity after addition and add a small random noise
    priority_scores[mask] = -(bins_remain_cap[mask] - item) + np.random.uniform(0, 1e-6, size=np.sum(mask))
    
    return priority_scores
```
