```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority score for each bin based on remaining capacity and a small random noise to break ties.
    """
    # Calculate priority based on remaining capacity after adding the item
    priority = np.where(bins_remain_cap >= item, -(bins_remain_cap - item) + np.random.uniform(0, 1e-6, size=len(bins_remain_cap)), -np.inf)
    
    return priority
```
