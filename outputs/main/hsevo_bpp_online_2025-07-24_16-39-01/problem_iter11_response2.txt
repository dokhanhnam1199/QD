import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines effective prioritization of bins with just enough capacity and robust tie-breaking.
    Uses a combination of negative remaining capacity and bin utilization ratio for prioritization.
    """
    # Calculate the remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Check if the item can fit into the bin
    valid_bins = (bins_remain_cap >= item).astype(float)
    
    # Prioritize bins that are just enough for the item
    priority = valid_bins * (-(remaining_cap_after_addition) + (1 / (bins_remain_cap + 1e-9)))
    
    # Add a small random noise for tie-breaking
    num_bins = len(bins_remain_cap)
    priority += valid_bins * (np.log(num_bins / (1 + np.arange(num_bins))) + np.random.rand(*priority.shape) * 1e-6)
    
    return priority
