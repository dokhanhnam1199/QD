import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Heuristic combining weighted utilization, remaining capacity, and a nuanced tie-breaker.
    """
    epsilon = 1e-6  # Avoid division by zero
    tie_breaker_coef = 1e-3  # Coefficient for tie-breaker
    
    # Weighted utilization factor
    utilization_factor = (bins_remain_cap - item) / np.maximum(bins_remain_cap, epsilon)
    norm_remain_cap = bins_remain_cap / np.max(bins_remain_cap)
    weighted_utilization = np.where(bins_remain_cap >= item, -np.abs(utilization_factor) * 0.7 + norm_remain_cap * 0.3, -np.inf)
    
    # Nuanced tie-breaker based on bin index and remaining capacity
    num_bins = len(bins_remain_cap)
    tie_breaker = (bins_remain_cap >= item) * ((np.log(num_bins / (1 + np.arange(num_bins))) * 0.5 + 
                                                bins_remain_cap / (1 + np.sum(bins_remain_cap)) * 0.5) * tie_breaker_coef)
    
    # Combine weighted utilization and tie-breaker for priority score
    priority = weighted_utilization + tie_breaker
    
    return priority
