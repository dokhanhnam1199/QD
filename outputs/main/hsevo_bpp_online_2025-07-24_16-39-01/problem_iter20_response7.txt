import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines fit score and a tie-breaker based on bin index for online Bin Packing Problem.
    """
    # Calculate remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Fit score: prioritize bins where the item fits well
    fit_score = np.where(bins_remain_cap >= item, -(np.abs(remaining_cap_after_addition)), -np.inf)
    
    # Tie-breaker based on bin index with a small coefficient
    tie_breaker = np.log(len(bins_remain_cap) / (1 + np.arange(len(bins_remain_cap)))) * 1e-3
    
    # Add a small random noise for further tie-breaking
    random_noise = np.random.uniform(0, 1e-6, size=len(bins_remain_cap))
    
    # Combine fit score, tie-breaker, and random noise
    priority_scores = fit_score + tie_breaker + random_noise
    
    return priority_scores
