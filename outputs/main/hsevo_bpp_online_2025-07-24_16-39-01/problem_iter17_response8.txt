import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines prioritizing bins with just enough capacity and a robust tie-breaking strategy.
    """
    # Prioritize bins with just enough capacity and consider utilization factor
    priority_scores = np.where(bins_remain_cap >= item, -(bins_remain_cap - item) + 0.1 * (item / bins_remain_cap), -np.inf)
    
    # Add a tie-breaker based on bin index
    num_bins = len(bins_remain_cap)
    priority_scores += np.log(num_bins / (1 + np.arange(num_bins))) * 1e-3
    
    # Add a small random noise for further tie-breaking
    priority_scores += np.random.rand(*priority_scores.shape) * 1e-6
    
    return priority_scores
