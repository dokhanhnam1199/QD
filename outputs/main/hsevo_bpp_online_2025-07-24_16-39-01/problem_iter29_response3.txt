def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines nuanced utilization factor, fit score, and a simple tie-breaker for effective prioritization.
    """
    epsilon = 1e-6  # Avoid division by zero
    # Calculate utilization factor and fit score
    utilization_factor = (bins_remain_cap - item) / np.maximum(bins_remain_cap, epsilon)
    relative_size = item / np.maximum(bins_remain_cap, epsilon)
    fit_score = np.where(bins_remain_cap >= item, 1 - relative_size, 0)
    
    # Priority score for bins that can fit the item with nuanced weighting
    priority = np.where(bins_remain_cap >= item, 
                        -np.abs(utilization_factor) * 0.95 + fit_score * 0.5,  # Adjusted weights
                        -np.inf)
    
    # Add a simple tie-breaker based on bin index with adjusted coefficient
    num_bins = len(bins_remain_cap)
    priority += (bins_remain_cap >= item) * (np.log(num_bins / (1 + np.arange(num_bins))) * 0.005)
    
    # Favor bins that are almost full to minimize waste with adjusted condition
    almost_full_bins = np.logical_and(bins_remain_cap >= item, bins_remain_cap - item < item * 0.05)
    priority += almost_full_bins * (1 - utilization_factor) * 0.15  # Adjusted weight
    
    return priority
