import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes feasible bins, rewards near-full, penalizes excess space."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    if not np.any(feasible_bins):
        priorities = -bins_remain_cap
        return priorities

    remaining_space = bins_remain_cap - item
    
    # Prioritize almost full bins and remaining space
    priorities[feasible_bins] = 1 / (remaining_space[feasible_bins] + 1e-9)
    
    # Heavily prioritize bins that become almost full
    almost_full = np.logical_and(feasible_bins, remaining_space < 0.1)
    priorities[almost_full] += 5.0
    
    # Add a fixed reward for using feasible bins
    priorities[feasible_bins] += 1.0 

    # Penalize leaving excessive space (avoid fragmentation)
    excessive_space = remaining_space > 0.5  #tuneable parameter
    priorities[feasible_bins & excessive_space] -= 0.5 #tuneable parameter

    return priorities
