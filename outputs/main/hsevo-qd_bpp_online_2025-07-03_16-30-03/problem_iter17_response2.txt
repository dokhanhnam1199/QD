import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes near-full bins and penalizes excessive space.
    Combines ratio-based and space-based approaches for efficiency.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    if not np.any(feasible_bins):
        priorities = -bins_remain_cap
        return priorities

    remaining_space = bins_remain_cap - item
    
    # Initial priority based on ratio of item size to remaining capacity
    ratios = item / bins_remain_cap
    priorities[feasible_bins] = -np.log(ratios[feasible_bins])

    # Prioritize almost full bins
    almost_full = np.logical_and(feasible_bins, remaining_space < 0.1)
    priorities[almost_full] += 5.0

    # Penalize excessive space
    excessive_space = np.logical_and(feasible_bins, remaining_space > 0.5)
    priorities[excessive_space] -= 0.5 * remaining_space[excessive_space]
    
    return priorities
