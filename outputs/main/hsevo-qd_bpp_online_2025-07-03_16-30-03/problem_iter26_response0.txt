import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on remaining space, feasibility, and fullness."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    if not np.any(feasible_bins):
        priorities = -bins_remain_cap
        return priorities

    remaining_space = bins_remain_cap - item

    # Prioritize almost full bins.
    almost_full = np.logical_and(feasible_bins, remaining_space < 0.1)
    priorities[almost_full] += 10.0

    # Prioritize bins with smaller remaining space.
    priorities[feasible_bins] += 1.0 / (remaining_space[feasible_bins] + 1e-9)

    # Penalize bins that are too empty.
    priorities[feasible_bins] -= (bins_remain_cap[feasible_bins] > 0.7) * 0.5

    # Add a small bonus if the item fits snugly.
    snug_fit = np.logical_and(feasible_bins, remaining_space < 0.3)
    priorities[snug_fit] += 2.0

    return priorities
