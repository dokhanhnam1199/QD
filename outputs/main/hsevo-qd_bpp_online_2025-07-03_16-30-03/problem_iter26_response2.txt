import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines remaining space, feasibility, near-fullness, and excessive space penalty."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    if not np.any(feasible_bins):
        priorities = -bins_remain_cap
        return priorities

    remaining_space = bins_remain_cap - item

    # Prioritize based on inverse of remaining space, plus a bonus for feasibility.
    priorities[feasible_bins] = 1 / (remaining_space[feasible_bins] + 1e-9)
    priorities[feasible_bins] += 1.0 # Base reward for being a feasible bin

    # Heavily prioritize bins that are almost full after packing.
    almost_full = np.logical_and(feasible_bins, remaining_space < 0.1)
    priorities[almost_full] += 5.0

    # Penalize bins with excessive remaining space to avoid fragmentation.
    max_cap = np.max(bins_remain_cap)
    priorities[feasible_bins] -= (remaining_space[feasible_bins] / max_cap) * 0.5

    return priorities
