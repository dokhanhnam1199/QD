import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines remaining space, feasibility, near-fullness, and excessive space penalty."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    # If no bin can fit the item, prioritize bins with the most space
    if not np.any(feasible_bins):
        priorities = -bins_remain_cap
        return priorities

    remaining_space = bins_remain_cap - item
    
    # Prioritize based on inverse remaining space
    priorities[feasible_bins] = 1 / (remaining_space[feasible_bins] + 1e-9)
    
    # Prioritize bins that become almost full
    almost_full = np.logical_and(feasible_bins, remaining_space < 0.1)
    priorities[almost_full] += 5.0

    # Penalty for excessive remaining space
    priorities[feasible_bins] -= (remaining_space[feasible_bins] / np.max(bins_remain_cap)) * 0.5

    return priorities
