import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Improved priority function for online bin packing."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    if not np.any(feasible_bins):
        priorities = -bins_remain_cap
        return priorities

    remaining_space = bins_remain_cap - item
    priorities[feasible_bins] = 1 / (remaining_space[feasible_bins] + 1e-9)
    priorities[feasible_bins] += np.where(remaining_space[feasible_bins] >= 0, 1.0, 0)

    almost_full = np.logical_and(feasible_bins, remaining_space < 0.1)
    priorities[almost_full] += 5.0

    # Add a term to prioritize bins that are already relatively full
    # regardless of the item size, but only if the item fits.
    priorities[feasible_bins] += (bins_remain_cap[feasible_bins] / np.max(bins_remain_cap)) * 0.5

    return priorities
