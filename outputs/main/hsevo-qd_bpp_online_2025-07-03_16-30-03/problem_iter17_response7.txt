import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on remaining space and a near-fullness bonus."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    if not np.any(feasible_bins):
        priorities = -bins_remain_cap # Try using emptiest bins if no feasible bin.
        return priorities

    remaining_space = bins_remain_cap - item
    # Prioritize bins with smaller remaining space, avoid fragmentation.
    priorities[feasible_bins] = 1.0 / (remaining_space[feasible_bins] + 1e-9)

    # Heavily prioritize almost full bins.
    almost_full = np.logical_and(feasible_bins, remaining_space < 0.1)
    priorities[almost_full] += 10.0

    # Penalize bins that are too empty after placing the item
    too_empty = np.logical_and(feasible_bins, remaining_space > 0.5)
    priorities[too_empty] -= 1.0
    return priorities
