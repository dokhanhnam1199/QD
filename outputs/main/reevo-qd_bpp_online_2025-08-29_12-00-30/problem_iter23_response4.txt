```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin. Args:
    item: Size of item to be added to the bin.
    bins_remain_cap: Array of capacities for each bin.
    Returns: Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities

    distances = bins_remain_cap[valid_bins] - item
    exact_fit = (distances == 0).astype(float) * 10  # High priority for exact fits

    # Sigmoid for fit - more emphasis on tighter fits
    sigmoid_fit = 1 / (1 + np.exp(-distances * 5))

    fullness = bins_remain_cap[valid_bins] / np.max(bins_remain_cap)
    # Prioritize bins around 70% full
    fullness_score = np.abs(fullness - 0.7)
    fullness_priority = np.exp(-fullness_score * 5) # Exponential decay

    # Item-size aware exploration: Larger items get more exploration
    exploration = np.random.rand(len(valid_bins)) * (0.1 / (item + 1))

    priorities[valid_bins] = (0.4 * sigmoid_fit + 0.4 * fullness_priority + exact_fit + 0.2 * exploration)
    return priorities
```
