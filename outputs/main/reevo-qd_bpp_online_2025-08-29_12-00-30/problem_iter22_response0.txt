import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Returns:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    valid_bins = bins_remain_cap >= item

    if not np.any(valid_bins):
        return priorities

    # Calculate the difference between the item size and the remaining capacity
    diffs = bins_remain_cap[valid_bins] - item

    # Prioritize bins with smaller differences (tighter fits)
    fit_score = 1.0 / (diffs + 1e-9)

    # Calculate bin fullness
    fullness = 1.0 - (bins_remain_cap[valid_bins] / np.max(bins_remain_cap))

    # Prioritize bins that are approximately half full
    fullness_score = np.abs(fullness - 0.5)
    fullness_score = np.where(fullness_score == 0, 1e-9, fullness_score)
    fullness_priority = 1.0 / fullness_score

    # Add a small amount of randomness for exploration
    epsilon = 0.01  # Exploration rate
    exploration_bonus = np.random.rand(len(valid_bins)) * epsilon

    # Combine the scores with weights
    priorities[valid_bins] = 0.5 * fit_score + 0.3 * fullness_priority + 0.2 * exploration_bonus

    return priorities
