import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Implements the Exact Fit First strategy for the Online Bin Packing Problem.
    Assigns a higher priority to bins that can exactly fit the current item.
    Bins that can fit the item but not exactly have a lower priority.
    Bins that cannot fit the item have zero priority.
    """
    priorities = np.zeros_like(bins_remain_cap)
    exact_fit_mask = (bins_remain_cap == item)
    fits_but_not_exact_mask = (bins_remain_cap > item) & ~exact_fit_mask

    priorities[exact_fit_mask] = 1.0
    priorities[fits_but_not_exact_mask] = 0.5

    return priorities
