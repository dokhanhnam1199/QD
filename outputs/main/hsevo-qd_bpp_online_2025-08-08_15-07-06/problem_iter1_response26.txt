def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    suitable_bins = bins_remain_cap >= item
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    
    if not np.any(suitable_bins):
        return priorities
    
    suitable_remain_cap = bins_remain_cap[suitable_bins]
    
    best_fit_priorities = 1 / (1 + np.exp((suitable_remain_cap - item) - (suitable_remain_cap.mean())))
    
    priorities[suitable_bins] = best_fit_priorities
    
    return priorities
