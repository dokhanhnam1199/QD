import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Compute negative leftover with exact‑fit bonus and tie‑break by index."""
    gap = bins_remain_cap - item
    feasible = gap >= 0
    idx = np.arange(len(bins_remain_cap))
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    bonus = np.where(gap == 0, 1e-6, 0.0)
    priorities[feasible] = -gap[feasible] + bonus[feasible] - 1e-12 * idx[feasible]
    return priorities
