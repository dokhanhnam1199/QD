import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit squared penalty, worst-fit capacity weight, inverse residual tie-breaker, and large exact-fit bonus for deterministic online BPP priority."""
    feasible = bins_remain_cap >= item
    residual = bins_remain_cap - item
    base = np.where(feasible, -residual**2 + 0.1 * bins_remain_cap, -np.inf)
    tie = np.where(feasible, 1.0/(1.0+residual), 0.0)
    exact = np.isclose(residual, 0.0)
    priorities = base + tie + np.where(exact, 1e6, 0.0)
    return priorities
