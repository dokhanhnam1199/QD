import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, temperature: float = 1.0) -> np.ndarray:
    """Returns priority scores for each bin using a softmax fit strategy."""
    caps = bins_remain_cap.astype(float)
    mask = caps >= item
    priorities = np.zeros_like(caps, dtype=float)
    if not np.any(mask):
        return priorities
    residual = caps[mask] - item
    exponent = np.exp(-residual / max(temperature, 1e-12))
    total = exponent.sum()
    if total > 0:
        priorities[mask] = exponent / total
    return priorities
