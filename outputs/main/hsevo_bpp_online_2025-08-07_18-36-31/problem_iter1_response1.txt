import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    leftover = bins_remain_cap - item
    feas = leftover >= 0
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=np.float64)
    priorities[feas] = -leftover[feas]
    return priorities
