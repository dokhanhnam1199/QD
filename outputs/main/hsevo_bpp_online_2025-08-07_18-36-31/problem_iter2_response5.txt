import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Deterministic priority based on negative leftover, with a tiny exact‑fit bonus.
    """
    # Feasible bins: remaining capacity >= item size
    feasible = bins_remain_cap >= item

    # Initialize all priorities to -∞ (infeasible)
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=float)

    # Negative leftover: larger for tighter fits (item - remaining_cap)
    priorities[feasible] = item - bins_remain_cap[feasible]

    # Small exact‑fit bonus to break ties for perfect fits
    exact_fit = feasible & (bins_remain_cap == item)
    priorities[exact_fit] += 1e-6

    return priorities
