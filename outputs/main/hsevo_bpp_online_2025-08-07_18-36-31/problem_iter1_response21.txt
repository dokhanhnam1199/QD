import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority for best fit placement of an item into bins.

    Args:
        item: Size of the item to be added to a bin.
        bins_remain_cap: Array of remaining capacities for each bin.

    Return:
        Array of the same shape as bins_remain_cap with priority scores.
    """
    priorities = np.full(bins_remain_cap.shape, -np.inf, dtype=float)
    fit = bins_remain_cap >= item
    priorities[fit] = item - bins_remain_cap[fit]
    return priorities
