import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Monotonic inverse-leftover score with exactâ€‘fit boost.
    """
    eps = 1e-9
    fit_mask = bins_remain_cap >= item
    base = np.where(fit_mask, 1.0 / (bins_remain_cap - item + eps), -np.inf)
    exact_bonus = np.where(fit_mask & (bins_remain_cap == item), 1e7, 0.0)
    return base + exact_bonus
