import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    leftover = bins_remain_cap - item
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    fit_mask = leftover >= 0
    priorities[fit_mask] = 1.0 / (1.0 + leftover[fit_mask])
    return priorities
