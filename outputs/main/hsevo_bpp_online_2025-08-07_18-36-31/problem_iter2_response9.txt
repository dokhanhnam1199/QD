import numpy as np

# Choose bin with smallest leftover, give a boost to exact fits.
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Score bins: high for minimal waste, extra bonus for exact fits."""
    leftover = bins_remain_cap - item
    feasible = leftover >= 0
    scores = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    scores[feasible] = -leftover[feasible]                 # less waste â†’ higher score
    exact_fit = leftover == 0
    if np.any(exact_fit):
        bonus = bins_remain_cap.max() + 1.0                # dominates other scores
        scores[exact_fit] += bonus
    return scores
