import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    eps = 1e-9
    k = 12.0
    residual = bins_remain_cap - item
    fit = residual >= 0
    base = np.where(fit, 1.0 / (1.0 + np.exp(k * residual)), -np.inf)
    exact = np.abs(residual) < eps
    return base + exact.astype(float)
