import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Best‑fit with large exact‑fit bonus, inverse‑residual term, and index tie‑breaker.
    """
    gap = bins_remain_cap - item
    feasible = gap >= 0
    idx = np.arange(len(bins_remain_cap))
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    priorities[feasible] = -gap[feasible]
    exact = feasible & (gap == 0)
    priorities[exact] += 1e6
    priorities[feasible] += 1.0 / (1.0 + gap[feasible])
    priorities[feasible] -= 1e-12 * idx[feasible]
    return priorities
