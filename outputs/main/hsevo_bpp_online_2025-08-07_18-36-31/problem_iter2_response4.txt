import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Score bins: highest for smallest leftover, boost exact fits, deterministic tieâ€‘break."""
    mask = bins_remain_cap >= item
    residual = bins_remain_cap - item
    priority = -residual
    priority[np.isclose(bins_remain_cap, item)] += 1.0
    priority += -np.arange(bins_remain_cap.shape[0], dtype=float) * 1e-12
    priority = np.where(mask, priority, -np.inf)
    return priority
