import numpy as np

# Priority: minimize leftover space, large boost for exact fits
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Score bins: higher for smaller leftover after item, with big bonus for exact fits."""
    fit = bins_remain_cap >= item
    leftover = bins_remain_cap - item
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    exact_bonus = 1e6
    eps = 1e-9
    priorities[fit] = -leftover[fit] + np.where(leftover[fit] <= eps, exact_bonus, 0.0)
    return priorities
