import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Best-fit with strong exact-fit boost and inverse-residual tweak.
    """
    mask = bins_remain_cap >= item
    residual = bins_remain_cap - item
    # combine best-fit, exact-fit, inverse residual, deterministic tie-break
    priority = -residual
    priority[np.isclose(bins_remain_cap, item)] += 1e6
    priority += 1.0 / (1.0 + residual) * 1e-3
    priority -= np.arange(bins_remain_cap.size) * 1e-12
    priority = np.where(mask, priority, -np.inf)
    return priority
