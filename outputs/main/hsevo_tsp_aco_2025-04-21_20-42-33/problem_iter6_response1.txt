import numpy as np
import scipy.sparse
from scipy.sparse.csgraph import dijkstra

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """TSP heuristic: Combines inverse distance, adaptive degree bias, shortest paths,
    and node potential with sparsification and temperature-controlled exploration."""
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros_like(distance_matrix, dtype=float)

    # Inverse distance
    inverse_distance = 1 / (distance_matrix + 1e-9)

    # Shortest path estimate (Dijkstra)
    graph = scipy.sparse.csr_matrix(distance_matrix)
    shortest_paths = dijkstra(graph, directed=False, indices=range(n))

    # Adaptive degree bias
    degree_penalty = np.ones((n, n))
    degrees = np.sum(inverse_distance, axis=1)
    avg_degree = np.mean(degrees)

    for i in range(n):
        for j in range(n):
            if i != j:
                penalty_i = max(0.1, avg_degree / (degrees[i] + 1e-9))  # Ensure a minimum penalty
                penalty_j = max(0.1, avg_degree / (degrees[j] + 1e-9))  # Ensure a minimum penalty
                degree_penalty[i, j] = penalty_i * penalty_j
            else:
                degree_penalty[i, j] = 0

    # Node Potential (based on total distance to other nodes)
    node_potential = np.sum(distance_matrix, axis=1)
    potential_matrix = np.zeros_like(distance_matrix)
    for i in range(n):
        for j in range(n):
            if i != j:
                potential_matrix[i, j] = (node_potential[i] + node_potential[j])  # Sum potentials
            else:
                potential_matrix[i, j] = 0

    # Temperature parameter for exploration
    temperature = 1.0  # Adjust to control stochasticity

    # Combine heuristics (weighted combination)
    heuristic_matrix = (
        inverse_distance
        * (degree_penalty**temperature)
        / ((shortest_paths + 1e-9)**temperature)
        / ((potential_matrix + 1e-9)**temperature)
    )

    # Sparsification (adaptive threshold)
    threshold = np.percentile(heuristic_matrix[heuristic_matrix > 0], 30)  # Adjust percentile
    heuristic_matrix[heuristic_matrix < threshold] = 0

    # Normalize (after sparsification)
    max_heuristic = np.max(heuristic_matrix)
    if max_heuristic > 0:
        heuristic_matrix /= max_heuristic

    return heuristic_matrix
