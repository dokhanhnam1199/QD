```python
import numpy as np
import scipy.sparse
from scipy.sparse.csgraph import dijkstra

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """TSP heuristic: Combines inverse distance, adaptive degree bias,
    shortest paths, informed randomness, and component normalization with sparsification.
    """
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros_like(distance_matrix, dtype=float)

    # Inverse distance
    inverse_distance = 1 / (distance_matrix + 1e-9)

    # Shortest path estimate (Dijkstra)
    graph = scipy.sparse.csr_matrix(distance_matrix)
    shortest_paths = dijkstra(graph, directed=False, indices=range(n))

    # Adaptive degree bias
    degree_penalty = np.ones((n, n))
    degrees = np.sum(inverse_distance, axis=1)
    avg_degree = np.mean(degrees)

    for i in range(n):
        for j in range(n):
            if i != j:
                penalty_i = max(0.1, avg_degree / (degrees[i] + 1e-9)) # Min penalty of 0.1
                penalty_j = max(0.1, avg_degree / (degrees[j] + 1e-9)) # Min penalty of 0.1
                degree_penalty[i, j] = penalty_i * penalty_j
            else:
                degree_penalty[i, j] = 0

    # Combine heuristics (before sparsification)
    heuristic_matrix = inverse_distance * degree_penalty / (shortest_paths + 1e-9)

    # Sparsification (before normalization)
    threshold = np.percentile(heuristic_matrix[heuristic_matrix > 0], 30) # Increased threshold
    heuristic_matrix[heuristic_matrix < threshold] = 0

    # Informed Randomness
    randomness_strength = 0.05 # Reduced randomness strength
    random_matrix = np.random.rand(n, n) * randomness_strength
    heuristic_matrix += random_matrix

    # Component Normalization
    inverse_distance_norm = inverse_distance / (np.max(inverse_distance) + 1e-9)
    degree_penalty_norm = degree_penalty / (np.max(degree_penalty) + 1e-9)
    shortest_paths_norm = 1 / (shortest_paths + 1e-9)
    shortest_paths_norm = shortest_paths_norm / (np.max(shortest_paths_norm) + 1e-9)
    heuristic_matrix_norm = heuristic_matrix / (np.max(heuristic_matrix) + 1e-9)


    # Recombine normalized components
    heuristic_matrix = 0.4 * inverse_distance_norm + 0.3 * degree_penalty_norm + 0.2 * shortest_paths_norm + 0.1 * heuristic_matrix_norm

    # Final Normalization
    max_heuristic = np.max(heuristic_matrix)
    if max_heuristic > 0:
        heuristic_matrix /= max_heuristic

    return heuristic_matrix
```
