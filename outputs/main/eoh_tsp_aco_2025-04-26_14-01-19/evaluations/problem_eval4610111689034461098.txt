import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm samples multiple TSP solutions using a shortest edge selection heuristic with randomized starting nodes and a weighted probability based on edge distance, and then averages the edge occurrences across these solutions to estimate edge importance.}"""
    n = distance_matrix.shape[0]
    num_samples = 100
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)

    for _ in range(num_samples):
        start_node = np.random.randint(n)
        unvisited_nodes = set(range(n))
        current_node = start_node
        unvisited_nodes.remove(current_node)
        path = [current_node]

        while unvisited_nodes:
            # Calculate probabilities based on inverse distance
            probabilities = np.zeros(n)
            for neighbor in unvisited_nodes:
                probabilities[neighbor] = 1 / (distance_matrix[current_node, neighbor] + 1e-9)  # Add a small constant to avoid division by zero
            probabilities /= np.sum(probabilities)

            # Choose the next node based on the probabilities
            next_node = np.random.choice(list(unvisited_nodes), p=probabilities[list(unvisited_nodes)])
            
            path.append(next_node)
            current_node = next_node
            unvisited_nodes.remove(current_node)

        path.append(start_node)

        for i in range(n):
            node1 = path[i]
            node2 = path[i+1]
            heuristics_matrix[node1, node2] += 1
            heuristics_matrix[node2, node1] += 1

    heuristics_matrix /= num_samples
    return heuristics_matrix
