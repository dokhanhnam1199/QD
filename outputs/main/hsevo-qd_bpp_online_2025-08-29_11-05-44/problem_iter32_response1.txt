```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste decay (like v0) and fit score (like v6) with weighted sum."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities
    waste = bins_remain_cap[possible_bins] - item
    fit_score = 1.0 / (waste + 1e-6)
    waste_decay = np.exp(-waste / item)
    priorities[possible_bins] = 0.5 * waste_decay * fit_score + 0.5 * fit_score
    return priorities
```
