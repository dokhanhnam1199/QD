import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best fit with exponentially decayed waste to prioritize bins."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities

    fit_scores = bins_remain_cap[valid_bins] - item
    waste = np.sum(bins_remain_cap) - np.sum(bins_remain_cap[valid_bins])
    waste_decay = -np.log(waste + 1) if waste > 0 else 0
    preferences = (fit_scores * 0.8) + (waste_decay * 0.2)
    priorities[valid_bins] = preferences
    return priorities
