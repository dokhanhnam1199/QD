import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """A refined priority function for online bin packing."""
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities

    fit_scores = -np.abs(bins_remain_cap[possible_bins] - item)
    waste = np.sum(bins_remain_cap) - np.sum(bins_remain_cap[possible_bins])
    age_bonus = np.arange(len(bins_remain_cap))[possible_bins] * 0.002

    weights = {'fit': 1.0, 'waste': 0.1, 'age': 0.05}
    waste_decay_rate = 0.015

    preferences = (weights['fit'] * fit_scores) + \
                  (weights['waste'] * np.exp(-waste_decay_rate * waste)) + \
                  (weights['age'] * age_bonus)

    priorities[possible_bins] = preferences
    return priorities
