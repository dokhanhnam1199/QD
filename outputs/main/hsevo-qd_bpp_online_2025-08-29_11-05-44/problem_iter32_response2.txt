import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit, waste decay, and age for bin priority."""
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)

    fit_scores = -np.abs(bins_remain_cap[possible_bins] - item)
    waste = np.sum(bins_remain_cap) - np.sum(bins_remain_cap[possible_bins])
    waste_penalty = np.exp(-0.005 * waste)
    bin_ages = np.arange(len(bins_remain_cap))
    age_factor = 0.001
    age_scores = bin_ages[possible_bins] * age_factor

    weights = {'fit': 0.6, 'waste': 0.3, 'age': 0.1}
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = (weights['fit'] * fit_scores +
                                 weights['waste'] * waste_penalty +
                                 weights['age'] * age_scores)

    return priorities
