import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit, waste, and age with weighted scoring and exponential waste decay."""
    weights = {'fit': 0.6, 'waste': 0.3, 'age': 0.1}
    decay_rate = 0.01
    age_factor = 0.001
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item

    if not np.any(possible_bins):
        return priorities

    fit_scores = -np.abs(bins_remain_cap[possible_bins] - item)
    waste_influence = np.exp(-decay_rate * np.sum(bins_remain_cap))
    age_bonus = np.arange(n_bins)[possible_bins] * age_factor

    priorities[possible_bins] = (weights['fit'] * fit_scores +
                                weights['waste'] * waste_influence +
                                weights['age'] * age_bonus)

    return priorities
