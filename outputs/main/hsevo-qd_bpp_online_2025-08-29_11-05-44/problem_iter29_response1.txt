```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit, waste decay, and age bonus for balanced bin selection."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item

    if not np.any(possible_bins):
        return priorities

    fit_score = 1.0 / (bins_remain_cap[possible_bins] - item + 0.0001)
    waste_decay = np.exp(-0.05 * (1 - (bins_remain_cap[possible_bins] - item) / bins_remain_cap[possible_bins]))
    age_bonus = np.arange(n_bins) / n_bins

    priorities[possible_bins] = 0.5 * fit_score * waste_decay + 0.5 * age_bonus[possible_bins]
    return priorities
```
