import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines worst-fit with waste decay and age bonus for improved bin selection."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    worst_fit_capacities = bins_remain_cap[valid_bins] - item
    max_waste = np.max(worst_fit_capacities)
    waste_decay = -np.log(np.sum(bins_remain_cap) + 1)
    age_bonus = np.arange(len(bins_remain_cap))[valid_bins] * 0.005
    preferences = (max_waste * 0.8) + (waste_decay * 0.1) + age_bonus
    priorities[valid_bins] = preferences
    return priorities
