```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines the efficiency of filtering valid bins with a priority 
    based on the inverse of remaining waste, adding a small constant
    for numerical stability.
    """
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if np.any(possible_bins):
        waste = bins_remain_cap[possible_bins] - item
        priorities[possible_bins] = 1.0 / (waste + 1e-6)
    return priorities
```
