import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines features from heuristics 6 and 9: prioritizes bins with enough capacity,
    boosts the priority of the best-fit bin.
    """
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if np.any(possible_bins):
        priorities[possible_bins] = 1.0
        best_bin_index = np.argmin(bins_remain_cap[possible_bins] - item)
        priorities[possible_bins][best_bin_index] = 2.0
    return priorities
