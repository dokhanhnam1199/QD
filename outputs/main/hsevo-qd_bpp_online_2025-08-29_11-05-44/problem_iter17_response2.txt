import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fit and exponentially decayed waste."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities

    fit_scores = bins_remain_cap[possible_bins] - item
    waste = np.sum(bins_remain_cap) - np.sum(bins_remain_cap[possible_bins])
    waste_decay = np.exp(-waste / np.sum(bins_remain_cap)) if np.sum(bins_remain_cap) > 0 else 0

    preferences = (fit_scores * 0.9) + (waste_decay * 0.1)
    priorities[possible_bins] = preferences
    return priorities
