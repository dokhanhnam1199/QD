import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit, waste decay, and age bonus with tuned weights."""
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities
    remaining_capacities = bins_remain_cap[possible_bins]
    fit_scores = -np.abs(remaining_capacities - item)
    waste_decay = np.exp(-0.005 * np.sum(bins_remain_cap))
    age_bonus = np.arange(len(bins_remain_cap))[possible_bins] * 0.001
    priorities[possible_bins] = (0.6 * fit_scores + 0.3 * waste_decay + 0.1 * age_bonus)
    return priorities
