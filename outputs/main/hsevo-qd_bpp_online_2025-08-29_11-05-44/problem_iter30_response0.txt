```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit score with waste decay, age bonus, and a bin fullness penalty."""
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities

    fit_scores = -np.abs(bins_remain_cap[possible_bins] - item)
    waste = np.sum(bins_remain_cap) - np.sum(bins_remain_cap[possible_bins])
    age_bonus = np.arange(len(bins_remain_cap))[possible_bins] * 0.001
    fullness_penalty = 1.0 - (bins_remain_cap[possible_bins] / np.max(bins_remain_cap))

    weight_fit = 0.5
    weight_waste = 0.3
    weight_age = 0.1
    weight_fullness = 0.1
    
    preferences = (weight_fit * fit_scores) + (weight_waste * np.exp(-0.01 * waste)) + (weight_age * age_bonus) + (weight_fullness * fullness_penalty)

    priorities[possible_bins] = preferences
    return priorities
```
