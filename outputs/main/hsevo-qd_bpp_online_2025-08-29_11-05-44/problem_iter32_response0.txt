```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines inverse waste with a small boost for immediate fit and bin age."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item

    if not np.any(possible_bins):
        return priorities

    waste = bins_remain_cap[possible_bins] - item
    priorities[possible_bins] = 1.0 / (1.0 + waste)
    fit_score = np.where(bins_remain_cap >= item, 1.0, 0.0)
    priorities = priorities + 0.1 * fit_score
    
    bin_ages = np.arange(n_bins)
    age_score = np.exp(-0.005 * bin_ages)
    priorities = priorities + 0.05 * age_score

    return priorities
```
