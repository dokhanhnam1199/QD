import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, bin_ages: np.ndarray = None) -> np.ndarray:
    """Combines best-fit preference with waste decay and age bonus."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item

    if not np.any(possible_bins):
        return priorities

    fit_score = -np.abs(item - bins_remain_cap[possible_bins])
    waste = bins_remain_cap[possible_bins] - item
    preference = fit_score * np.exp(-0.1 * waste)
    
    if bin_ages is not None:
        preference += 0.05 * bin_ages[possible_bins]

    priorities[possible_bins] = preference

    best_bin_index = np.argmax(priorities[possible_bins])
    priorities[possible_bins][best_bin_index] *= 2.0

    return priorities
