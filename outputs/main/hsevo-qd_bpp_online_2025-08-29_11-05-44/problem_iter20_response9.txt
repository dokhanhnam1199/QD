```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit, waste, and age for bin selection. Prioritizes good fit and minimizes waste."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities
    waste = bins_remain_cap[possible_bins] - item
    fit_score = 1.0 / (waste + 1e-6)
    age_penalty = np.log(np.sum(possible_bins) + 1) / 10
    priorities[possible_bins] = fit_score - age_penalty
    return priorities
```
