import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit score, exponential waste decay, and age bonus."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities
    fit_scores = -np.abs(bins_remain_cap[possible_bins] - item)
    waste = np.sum(bins_remain_cap) - np.sum(bins_remain_cap[possible_bins])
    waste_decay = np.exp(-0.002 * waste)
    age_bonus = np.arange(len(bins_remain_cap))[possible_bins] * 0.001
    priorities[possible_bins] = (0.5 * fit_scores + 0.3 * waste_decay + 0.2 * age_bonus)
    return priorities
