import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, small_boost: float = 3.032658373456192e-05) -> np.ndarray:
    """Combines inverse waste with a small boost for recently used bins."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    waste = bins_remain_cap[valid_bins] - item
    priorities[valid_bins] = 1.0 / (waste + small_boost)
    return priorities
