import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bin fit with exponential waste decay for fragmentation control."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities

    fit_scores = bins_remain_cap[valid_bins] - item
    waste = np.sum(bins_remain_cap)
    decay_factor = np.exp(-0.005 * waste)
    priorities[valid_bins] = (1.0 / (fit_scores + 1e-6)) * decay_factor
    return priorities
