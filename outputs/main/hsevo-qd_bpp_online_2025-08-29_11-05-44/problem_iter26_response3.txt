import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit score with exponential waste decay for stable bin selection."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities
    remaining_capacities = bins_remain_cap[possible_bins]
    fit_score = 1.0 / (remaining_capacities - item + 1e-6)
    waste = remaining_capacities - item
    priorities[possible_bins] = fit_score * np.exp(-waste / item)
    return priorities
