```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit, waste (with exponential decay), and age for bin prioritization."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities
    waste = bins_remain_cap[possible_bins] - item
    fit_scores = 1.0 / (1.0 + waste)
    age_bonus = np.arange(n_bins)
    age_bonus = age_bonus / np.max(age_bonus)
    priorities[possible_bins] = 0.6 * fit_scores + 0.3 * np.exp(-0.01 * waste) + 0.1 * age_bonus
    return priorities
```
