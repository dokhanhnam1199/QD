import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit, decaying waste, and age for bin prioritization."""
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)

    fit_scores = bins_remain_cap[possible_bins] - item
    waste = np.sum(bins_remain_cap)
    decay_factor = np.exp(-0.005 * waste)
    age_bonus = np.arange(len(bins_remain_cap))[possible_bins] * 0.01
    priorities = (fit_scores * decay_factor) + age_bonus
    
    result = np.zeros_like(bins_remain_cap)
    result[possible_bins] = priorities
    return result
