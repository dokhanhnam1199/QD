import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit score with waste decay and boosts the best-fit bin."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities

    remaining_capacities = bins_remain_cap[valid_bins]
    fit_score = 1.0 / (remaining_capacities - item + 1e-6)
    waste_factor = np.exp(-remaining_capacities / np.mean(bins_remain_cap))
    priorities[valid_bins] = fit_score * waste_factor

    best_bin_index = np.argmin(bins_remain_cap[valid_bins] - item)
    priorities[valid_bins][best_bin_index] *= 2.0

    return priorities
