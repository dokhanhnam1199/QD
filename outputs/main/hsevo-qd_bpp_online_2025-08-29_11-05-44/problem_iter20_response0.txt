import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit score with waste-sensitive decay and age bonus."""
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return np.zeros_like(bins_remain_cap)

    fit_score = -np.abs(item - bins_remain_cap[valid_bins])
    waste = bins_remain_cap[valid_bins] - item
    age_bonus = np.arange(len(bins_remain_cap))[valid_bins] * 0.01
    total_waste = np.sum(bins_remain_cap[valid_bins] - item)

    k = 0.01
    priorities = np.zeros_like(bins_remain_cap)
    priorities[valid_bins] = (bins_remain_cap[valid_bins] * fit_score + age_bonus) * np.exp(-k * total_waste)
    return priorities
