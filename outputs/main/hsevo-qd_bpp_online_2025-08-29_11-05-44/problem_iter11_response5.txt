```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit score (remaining capacity) with exponentially decaying waste."""
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities

    waste = bins_remain_cap - item
    priorities[possible_bins] = (bins_remain_cap[possible_bins] - item) * np.exp(-waste[possible_bins]/10)
    return priorities
```
