import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit and exponentially decayed waste for robust bin selection."""
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)
    waste = bins_remain_cap[possible_bins] - item
    fit_score = -np.abs(bins_remain_cap[possible_bins] - item)
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = (bins_remain_cap[possible_bins] * fit_score) / (np.exp(waste) + 1e-9)
    return priorities
