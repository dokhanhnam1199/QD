```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit, waste (exponential decay), and age for bin selection."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities

    fit_scores = bins_remain_cap[valid_bins] - item
    waste = bins_remain_cap[valid_bins] - item
    age = np.arange(n_bins)[valid_bins]

    weights = {'fit': 0.6, 'waste': 0.3, 'age': 0.1}
    decay_rate = 0.005

    waste_scores = np.exp(-decay_rate * waste)
    age_scores = age

    priorities[valid_bins] = (weights['fit'] * fit_scores +
                              weights['waste'] * waste_scores +
                              weights['age'] * age_scores)
    return priorities
```
