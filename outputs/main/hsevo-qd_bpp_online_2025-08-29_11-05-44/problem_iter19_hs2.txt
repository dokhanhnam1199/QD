import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, total_waste: float = 0.0, bin_ages: np.ndarray = None, k: float = 0.04226602965352387, age_factor: float = 0.003667267723317399) -> np.ndarray:
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities
    waste = bins_remain_cap[possible_bins] - item
    fit_scores = -np.abs(bins_remain_cap[possible_bins] - item)
    capacity_weighted_preference = bins_remain_cap[possible_bins] * fit_scores
    if bin_ages is None:
        bin_ages = np.zeros(n_bins)
    age_boost = np.exp(-age_factor * bin_ages)
    preference = (capacity_weighted_preference) * np.exp(-k * total_waste) * age_boost
    priorities[possible_bins] = preference
    return priorities
