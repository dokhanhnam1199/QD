```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, fit_weight: float = 0.7, waste_weight: float = 0.2, age_weight: float = 0.1) -> np.ndarray:
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item

    if not np.any(possible_bins):
        return priorities

    fit_scores = -np.abs(bins_remain_cap[possible_bins] - item)
    waste = np.sum(bins_remain_cap[possible_bins])
    age_bonus = np.arange(len(bins_remain_cap))[possible_bins] * 0.001

    preferences = (fit_weight * fit_scores) + (waste_weight * waste) + (age_weight * age_bonus)
    priorities[possible_bins] = preferences

    return priorities
```
