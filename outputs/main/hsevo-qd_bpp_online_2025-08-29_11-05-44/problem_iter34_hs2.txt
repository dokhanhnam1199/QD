import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, small_value: float = 5.351377463467998e-09) -> np.ndarray:
    """Prioritizes bins based on a combination of fit and remaining capacity."""
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)
    waste = bins_remain_cap[possible_bins] - item
    fit_score = -np.abs(bins_remain_cap[possible_bins] - item)
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = (bins_remain_cap[possible_bins] * fit_score) / (waste + small_value)
    return priorities
