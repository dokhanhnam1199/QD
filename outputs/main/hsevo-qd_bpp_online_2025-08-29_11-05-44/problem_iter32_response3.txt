```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fit and waste decay with bin age for improved bin selection."""
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities

    remaining_capacities = bins_remain_cap[possible_bins]
    waste = remaining_capacities - item
    priorities[possible_bins] = (1.0 / (remaining_capacities - item + 1e-6) *
                                 np.exp(-waste / 10.0))
    return priorities
```
