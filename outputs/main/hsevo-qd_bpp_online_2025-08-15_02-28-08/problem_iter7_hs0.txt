import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.0004632857614330139) -> np.ndarray:
    priorities = np.zeros_like(bins_remain_cap)
    for i in range(len(bins_remain_cap)):
        if bins_remain_cap[i] >= item:
            priorities[i] = 1.0 / (bins_remain_cap[i] - item + epsilon)
        else:
            priorities[i] = 0.0
    return priorities
