def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    epsilon = 1e-9
    diff = bins_remain_cap - item
    priorities = np.where(diff >= 0, diff, np.inf)
    priorities = np.where(priorities == 0, epsilon, priorities)
    priorities = 1 / priorities
    priorities = np.where(diff < 0, 0, priorities)
    return priorities
