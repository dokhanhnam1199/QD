import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:

    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Rule 1: Infeasible bins get a very negative priority.
    infeasible_bins = bins_remain_cap < item
    priorities[infeasible_bins] = -1e9

    # Rule 2: Quantify "good fit" using relative wasted space and bin capacity ratio.
    available_bins = bins_remain_cap >= item
    if np.any(available_bins):
        wasted_space = bins_remain_cap[available_bins] - item
        relative_wasted_space = wasted_space / (item + 1e-9) # avoid division by zero if item is 0
        
        # Bin capacity ratio (remaining capacity / total capacity)
        bin_capacity_ratio = bins_remain_cap[available_bins] / np.max(bins_remain_cap) if np.max(bins_remain_cap) > 0 else 0 #handle edge case where max cap is zero

        # Good fit = low relative wasted space and large bin capacity ratio.  Invert and scale.
        priorities[available_bins] = 1.0 / (0.01 + relative_wasted_space) + bin_capacity_ratio * 0.5

        # Penalize bins which are empty or almost full, scaled by item size ratio
        occupancy = 1.0 - bin_capacity_ratio
        occupancy_penalty = np.abs(occupancy - 0.5) * (item / np.max(bins_remain_cap) if np.max(bins_remain_cap) > 0 else 0)

        priorities[available_bins] -= occupancy_penalty * 0.2

        # Reward bins approaching a target fullness (e.g., 75%), scaled by item size.
        target_fullness = 0.75 * np.max(bins_remain_cap) if np.max(bins_remain_cap) > 0 else 0.0
        target_range = 0.1 * np.max(bins_remain_cap) if np.max(bins_remain_cap) > 0 else 0.1 #Scale relative to maximum bin size.

        target_bonus = np.exp(-((bins_remain_cap[available_bins] - item - target_fullness)**2) / (2 * target_range**2))
        priorities[available_bins] += target_bonus * 0.5 * (item / np.max(bins_remain_cap) if np.max(bins_remain_cap) > 0 else 0)

    # Rule 3: Prioritize almost full bins to complete them, scale based on remaining cap
    almost_full = (bins_remain_cap >= item) & (bins_remain_cap - item < 0.1 * np.max(bins_remain_cap) if np.max(bins_remain_cap)>0 else bins_remain_cap >=item )
    priorities[almost_full] += 10 * (1 - bins_remain_cap[almost_full]/np.max(bins_remain_cap) if np.max(bins_remain_cap) > 0 else 1)

    # Rule 4: Add some noise to break ties randomly, scaled by item size.
    available_bins = bins_remain_cap >= item
    priorities[available_bins] += np.random.normal(0, 1e-6 * (item / np.max(bins_remain_cap) if np.max(bins_remain_cap) > 0 else 1), size=np.sum(available_bins))

    return priorities
