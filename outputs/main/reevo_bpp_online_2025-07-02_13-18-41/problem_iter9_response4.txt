import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Rule 1: Infeasible bins get a very negative priority.
    infeasible_bins = bins_remain_cap < item
    priorities[infeasible_bins] = -1e9

    # Rule 2: Prioritize bins that minimize relative wasted space.
    available_bins = bins_remain_cap >= item
    if np.any(available_bins):
        wasted_space = bins_remain_cap[available_bins] - item
        relative_wasted_space = wasted_space / (bins_remain_cap[available_bins] + 1e-9)  # Wasted space relative to *remaining* capacity

        priorities[available_bins] = 1.0 / (0.01 + relative_wasted_space) # Invert for priority

        # Rule 3: Bonus for bins nearing target fullness (75%) - relative to bin capacity
        bin_capacities = bins_remain_cap + item  # Approximates the initial bin capacity
        target_fullness = 0.75 * bin_capacities[available_bins]
        target_range = 0.1 * bin_capacities[available_bins]

        target_bonus = np.exp(-((bins_remain_cap[available_bins] - target_fullness)**2) / (2 * target_range**2))
        priorities[available_bins] += target_bonus * 0.5

    # Rule 4: Prioritize almost full bins to complete them.
    almost_full_threshold = 0.1 * np.max(bin_capacities) if np.max(bin_capacities) > 0 else 0.1 #Threshold relative to inital bin size
    almost_full = (bins_remain_cap >= item) & (bins_remain_cap - item < almost_full_threshold)
    priorities[almost_full] += 10

    # Rule 5: Small noise to break ties.
    priorities[available_bins] += np.random.normal(0, 1e-6, size=np.sum(available_bins))

    return priorities
