import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, near_fit_threshold: float = 0.9,
                 near_fit_bonus: float = 10.0, small_fragment_threshold: float = 0.1,
                 large_fragment_penalty: float = -5.0, complete_fill_bonus: float = 20.0,
                 capacity_ratio_weight: float = 2.0) -> np.ndarray:
    """Combines near-fit, fragmentation avoidance, and capacity ratio for bin selection."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    valid_bins = bins_remain_cap > 0
    if not np.any(valid_bins):
        return priorities

    possible_bins = bins_remain_cap >= item

    for i, remaining_cap in enumerate(bins_remain_cap):
        if remaining_cap < item:
            priorities[i] = -np.inf
            continue

        # Near Fit
        if item / remaining_cap >= near_fit_threshold:
            priorities[i] += near_fit_bonus

        # Complete Fill
        if item == remaining_cap:
            priorities[i] += complete_fill_bonus
        
        # Fragmentation
        new_remaining = remaining_cap - item
        if new_remaining > 0 and (new_remaining / bins_remain_cap[0]) < small_fragment_threshold:
            priorities[i] += large_fragment_penalty

        # Capacity Ratio
        priorities[i] += (item / remaining_cap) * capacity_ratio_weight

    # Give small bonus to all bins that can pack the item, to break ties if all bins are bad
    if not np.any(priorities[possible_bins] > -np.inf):
        priorities[possible_bins] += 0.001
    
    return priorities
