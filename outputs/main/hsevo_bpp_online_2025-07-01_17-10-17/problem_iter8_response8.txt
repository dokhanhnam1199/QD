import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins by waste, half-full target, and handles edge cases."""
    epsilon = 1e-9
    distances = np.abs(bins_remain_cap - item)
    priority = 1 / (distances + epsilon)

    target_fill = np.max(bins_remain_cap) / 2
    fill_distances = np.abs(bins_remain_cap - target_fill)
    priority = priority / (fill_distances + epsilon)

    priority[bins_remain_cap < item] = 0
    almost_full = bins_remain_cap > 0.9 * np.max(bins_remain_cap)
    priority[almost_full] *= 0.5

    return priority
