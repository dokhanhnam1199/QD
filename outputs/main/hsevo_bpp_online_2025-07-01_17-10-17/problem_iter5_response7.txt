import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on waste and target fill (around 75% full)."""

    epsilon = 1e-9
    distances = np.abs(bins_remain_cap - item)

    # Prioritize bins where item nearly fits.
    base_priority = 1 / (distances + epsilon)

    # Target fill level is 75% of bin capacity.
    target_fill = bins_remain_cap * 0.75
    fill_distance = np.abs((bins_remain_cap - item) - target_fill)
    fill_priority = 1 / (fill_distance + epsilon)

    priorities = base_priority + fill_priority

    # Invalidate bins that can't fit the item.
    priorities[bins_remain_cap < item] = 0

    return priorities
