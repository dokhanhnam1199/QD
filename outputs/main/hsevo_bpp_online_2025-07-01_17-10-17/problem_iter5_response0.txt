```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on waste, target fill, and efficient utilization."""
    epsilon = 1e-9
    distances = np.abs(bins_remain_cap - item)
    base_priority = 1 / (distances + epsilon) # Waste minimization

    fill_target_ratio = 0.75
    target_capacity = bins_remain_cap * fill_target_ratio
    target_priority = 1 / (np.abs((bins_remain_cap - item) - target_capacity) + epsilon) #Target Fill

    # Prioritize bins that provide more utilization
    utilization = 1 - (bins_remain_cap / np.max(bins_remain_cap))
    utilization_priority = utilization

    priorities = 0.5 * base_priority + 0.3 * target_priority + 0.2 * utilization_priority
    priorities[bins_remain_cap < item] = 0
    return priorities
```
