import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on wasted space and fill ratio."""
    epsilon = 1e-9
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    for i, capacity in enumerate(bins_remain_cap):
        if capacity >= item:
            wasted_space = capacity - item
            fill_ratio = item / (capacity + epsilon)
            priorities[i] = -wasted_space + 0.5 * fill_ratio #Wasted space + fill ratio
        else:
            priorities[i] = -np.inf
    return priorities
