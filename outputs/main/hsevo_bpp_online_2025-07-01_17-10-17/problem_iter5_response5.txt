```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritize bins based on waste minimization and target fill (around 75%)."""
    epsilon = 1e-9
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Waste minimization
    waste = bins_remain_cap - item
    waste_priority = 1 / (np.abs(waste) + epsilon)
    waste_priority[waste < 0] = 0

    # Target fill (around 75%)
    target_fill = bins_remain_cap * 0.75
    fill_diff = np.abs((bins_remain_cap - item) - target_fill)
    fill_priority = 1 / (fill_diff + epsilon)

    priorities = 0.6 * waste_priority + 0.4 * fill_priority
    priorities[bins_remain_cap < item] = 0
    
    return priorities
```
