import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritize bins based on waste, target fill (75%), and penalize almost full."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    possible_bins = bins_remain_cap >= item

    if not np.any(possible_bins):
        return priorities

    waste = bins_remain_cap[possible_bins] - item
    priorities[possible_bins] = 1.0 / (waste + 0.001) # Waste minimization

    target_fill = 0.75 * np.max(bins_remain_cap)
    fill_diff = np.abs((bins_remain_cap[possible_bins] - item) - target_fill)
    priorities[possible_bins] += np.exp(-0.5 * (fill_diff / np.max(bins_remain_cap))**2) # Target fill

    almost_full = bins_remain_cap > 0.9 * np.max(bins_remain_cap)
    priorities[almost_full] *= 0.9 # Penalize almost full

    return priorities
