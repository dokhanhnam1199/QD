```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Assign priorities based on waste and target (75% full) fill."""
    epsilon = 1e-9
    distances = np.abs(bins_remain_cap - item)
    base_priority = 1 / (distances + epsilon)

    target_fill = 0.75 * bins_remain_cap
    target_diff = np.abs((bins_remain_cap - item) - target_fill)
    target_priority = 1 / (target_diff + epsilon)

    priorities = base_priority + target_priority
    priorities[bins_remain_cap < item] = 0  # Invalidate bins

    return priorities
```
