```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on waste and fill level, avoiding overfilling."""

    epsilon = 1e-9
    distances = np.abs(bins_remain_cap - item)
    
    # High priority if capacity close to item size and bin fits.
    priority = 1 / (distances + epsilon)
    
    # Discourage almost-full bins.
    priority = priority * (bins_remain_cap > item * 1.1)

    # Encourage bins to be closer to half-full
    target_fill = np.max(bins_remain_cap)/2
    fill_distances = np.abs(bins_remain_cap - target_fill)
    priority = priority / (fill_distances + epsilon)
    
    priority[bins_remain_cap < item] = 0  # Zero out if bin can't fit

    return priority
```
