```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on waste and fill, favoring half-full bins."""

    epsilon = 1e-9
    distances = np.abs(bins_remain_cap - item)
    priority = 1 / (distances + epsilon) # Prioritize close fits

    target_fill = np.max(bins_remain_cap) / 2 # Set half-full target
    fill_distances = np.abs(bins_remain_cap - target_fill)
    priority = priority / (fill_distances + epsilon) # Encourage half-full

    priority[bins_remain_cap < item] = 0 # Invalidate bins that can't fit

    return priority
```
