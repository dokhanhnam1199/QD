```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritize bins: balance waste minimization and target fill (50%)."""
    epsilon = 1e-9
    waste = bins_remain_cap - item
    waste_priority = np.where(waste >= 0, 1 / (waste + epsilon), 0)

    target_fill = np.max(bins_remain_cap) / 2
    fill_distances = np.abs((bins_remain_cap - item) - target_fill)
    fill_priority = 1 / (fill_distances + epsilon)

    priorities = 0.6 * waste_priority + 0.4 * fill_priority
    priorities[bins_remain_cap < item] = 0
    return priorities
```
