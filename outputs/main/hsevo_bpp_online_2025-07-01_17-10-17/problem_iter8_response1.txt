```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on waste minimization and a target fill level (50%)."""

    epsilon = 1e-9
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Waste minimization
    waste = bins_remain_cap - item
    waste_priority = 1 / (np.abs(waste) + epsilon)
    waste_priority[waste < 0] = 0

    # Target fill (around 50%) - distance to half-full *after* placing the item
    half_full = bins_remain_cap / 2
    half_full_dist = np.abs(waste - half_full)
    half_full_priority = 1 / (half_full_dist + epsilon)


    priorities = 0.7 * waste_priority + 0.3 * half_full_priority
    priorities[bins_remain_cap < item] = 0

    return priorities
```
