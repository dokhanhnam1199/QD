```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Assign priorities to bins based on multiple factors."""

    epsilon = 1e-9
    
    # 1. Immediate Fit: Prioritize bins where the item fits.
    fit_priority = (bins_remain_cap >= item).astype(float)

    # 2. Waste Minimization: Minimize wasted space.
    waste = bins_remain_cap - item
    waste_priority = np.where(waste >= 0, 1 / (waste + epsilon), 0)

    # 3. Balance Fill: Favor bins nearing target fill level (e.g., 75%).
    target_fill_level = 0.75  # Adjustable target fill level
    target_capacity = bins_remain_cap * target_fill_level
    balance_priority = np.abs((bins_remain_cap - item) - target_capacity)
    balance_priority = np.where(bins_remain_cap >= item, 1 / (balance_priority + epsilon), 0)

    # 4. Combination of Priorities (Weighted): Combine the priorities
    priorities = (0.5 * fit_priority + 0.3 * waste_priority + 0.2 * balance_priority)
    
    return priorities
```
