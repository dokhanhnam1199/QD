```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and target fill, balancing weights."""
    epsilon = 1e-9
    distances = np.abs(bins_remain_cap - item)
    waste_priority = 1 / (distances + epsilon)

    target_fill_ratio = 0.75
    target_capacity = bins_remain_cap * target_fill_ratio
    target_priority = 1 / (np.abs((bins_remain_cap - item) - target_capacity) + epsilon)

    priorities = 0.6 * waste_priority + 0.4 * target_priority
    priorities[bins_remain_cap < item] = 0
    return priorities
```
