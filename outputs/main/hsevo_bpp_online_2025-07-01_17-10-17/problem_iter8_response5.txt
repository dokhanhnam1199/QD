```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on waste and target fill (75%)."""
    epsilon = 1e-9
    distances = np.abs(bins_remain_cap - item)
    priority = 1 / (distances + epsilon)

    target_fill = 0.75 * np.max(bins_remain_cap)
    fill_distances = np.abs(bins_remain_cap - target_fill)
    priority = 0.6 * priority + 0.4 / (fill_distances + epsilon)

    priority[bins_remain_cap < item] = 0
    return priority
```
