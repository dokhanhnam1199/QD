import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritize bins based on waste and target fill."""

    epsilon = 1e-9
    max_cap = np.max(bins_remain_cap)

    # Waste Minimization
    waste = bins_remain_cap - item
    waste_priority = np.where(waste >= 0, 1 / (waste + epsilon), 0)

    # Target Fill (e.g., aim for 75% full after packing)
    target_fill = 0.75 * max_cap
    fill_diff = np.abs((bins_remain_cap - item) - target_fill)
    fill_priority = np.exp(-fill_diff / max_cap)

    # Combine priorities (waste more important)
    priorities = 0.7 * waste_priority + 0.3 * fill_priority

    # Invalidate bins that can't fit item
    priorities[bins_remain_cap < item] = 0

    return priorities
