import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines relative waste with exponential weighting and normalization for robust bin prioritization."""
    bin_count = len(bins_remain_cap)
    priorities = np.zeros(bin_count)
    valid_bins = bins_remain_cap >= item
    if np.any(valid_bins):
        waste = bins_remain_cap[valid_bins] - item
        relative_waste = waste / (item + 1e-6)
        priorities[valid_bins] = np.exp(-5 * relative_waste)
        priorities /= np.sum(priorities)
    return priorities
