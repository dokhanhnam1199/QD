import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, small_value: float = 9.620177160509137e-05) -> np.ndarray:
    """Prioritizes bins based on inverse waste with exponential weighting."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    waste = bins_remain_cap[valid_bins] - item
    priorities[valid_bins] = np.exp(-waste / (item + small_value))
    return priorities
