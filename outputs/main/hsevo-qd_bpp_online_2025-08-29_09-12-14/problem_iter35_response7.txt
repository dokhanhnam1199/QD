import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on exponential of inverse waste, normalized by sum of scores."""
    feasible_bins = bins_remain_cap[bins_remain_cap >= item]
    if len(feasible_bins) == 0:
        return np.zeros(len(bins_remain_cap))
    waste = feasible_bins - item
    scores = np.exp(-waste / item)
    normalized_scores = scores / np.sum(scores)
    priorities = np.zeros(len(bins_remain_cap))
    indices = np.where(bins_remain_cap >= item)[0]
    for i, idx in enumerate(indices):
        priorities[idx] = normalized_scores[i]
    return priorities
