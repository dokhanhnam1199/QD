import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Calculates bin priorities based on waste-to-item-size ratio
    with exponential weighting and normalization.
    """
    bin_count = len(bins_remain_cap)
    priorities = np.zeros(bin_count)
    for i in range(bin_count):
        if bins_remain_cap[i] >= item:
            waste = bins_remain_cap[i] - item
            priorities[i] = np.exp(-5 * (item / (waste + 0.0001)))
        else:
            priorities[i] = 0.0
    
    if np.sum(priorities) > 0:
        priorities = priorities / np.sum(priorities)
    
    return priorities
