import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on waste-to-item ratio with exponential weighting."""
    bin_count = len(bins_remain_cap)
    priorities = np.zeros(bin_count)
    valid_bins = bins_remain_cap >= item
    if np.sum(valid_bins) == 0:
        return priorities
    remaining_capacities = bins_remain_cap[valid_bins]
    waste = remaining_capacities - item
    priorities[valid_bins] = np.exp(-1.0 / (waste + 1e-6))
    priorities /= np.sum(priorities)
    return priorities
