import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines relative waste with exponential weighting for robust bin selection."""
    bin_count = len(bins_remain_cap)
    priorities = np.zeros(bin_count)
    for i in range(bin_count):
        if bins_remain_cap[i] >= item:
            waste = bins_remain_cap[i] - item
            relative_waste = waste / item
            priorities[i] = np.exp(-5 * relative_waste)
    priorities = priorities / (np.sum(priorities) + 1e-8)
    return priorities
