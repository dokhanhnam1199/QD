import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    bin_count = len(bins_remain_cap)
    priorities = np.zeros(bin_count)
    for i in range(bin_count):
        if bins_remain_cap[i] >= item:
            waste = bins_remain_cap[i] - item
            weighted_waste = np.exp(-item / (waste + 1e-6))
            priorities[i] = weighted_waste
        else:
            priorities[i] = 0.0
    priorities = priorities / (np.sum(priorities) + 1e-9)
    return priorities
