import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines waste minimization and exponential weighting for bin prioritization.
    Prioritizes bins with lower waste after placing the item, using an exponential
    factor for nuanced selection and normalization for stability.
    """
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if np.any(valid_bins):
        waste = bins_remain_cap[valid_bins] - item
        priorities[valid_bins] = np.exp(-waste)
        priorities[valid_bins] /= np.sum(np.exp(-waste))
    return priorities
