def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority based on the percentage of the bin that would be filled by the item, favoring bins that would be filled to a high percentage without overflowing.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_percentage = float(item) / cap
            priority.append(fill_percentage)
        else:
            priority.append(-1)
    return priority
