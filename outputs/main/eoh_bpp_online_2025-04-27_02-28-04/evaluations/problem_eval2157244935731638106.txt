def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates priority scores for each bin based on the ratio of item size to remaining capacity if the item fits, otherwise assigns a negative priority.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority_score = item / cap  # Ratio of item size to remaining capacity
    else:
      priority_score = -1  # Assign a very low priority if the item doesn't fit

    priority.append(priority_score)
  return priority
