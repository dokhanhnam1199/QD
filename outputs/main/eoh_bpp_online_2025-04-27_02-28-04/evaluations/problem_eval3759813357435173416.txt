def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, favoring bins where the item fills a significant portion of the remaining space.}"""
  priorities = []
  for remain_cap in bins_remain_cap:
    if remain_cap >= item:
      priority = float(item) / remain_cap
    else:
      priority = -1 
    priorities.append(priority)
  return priority
