def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the percentage of the bin they fill, with bins that fill a higher percentage (without exceeding capacity) receiving higher priority.}"""
  priorities = []
  for remain_cap in bins_remain_cap:
    if remain_cap >= item:
      fill_percentage = float(item) / (remain_cap + 0.00001)  # Avoid division by zero
      priority = fill_percentage
    else:
      priority = -1
    priorities.append(priority)
  return priority
