def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the percentage of the remaining capacity that the item would fill, favoring fills closer to a target percentage.}"""
  priority = []
  target_fill_percentage = 0.75 # Target fill percentage (e.g., 75%)
  for cap in bins_remain_cap:
    if cap >= item:
      fill_percentage = item / cap
      priority_score = 1 / (abs(fill_percentage - target_fill_percentage) + 0.001)
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
