def priority_v2(item, bins_remain_cap):
    """{Assign a priority score to each bin based on the fill ratio after placing the item, favoring bins that become more full but penalizing those that would be overfilled, and using a simple heuristic to differentiate between acceptable bins.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = item / cap 
            priority.append(fill_ratio)
        else:
            priority.append(-1)
    return priority
