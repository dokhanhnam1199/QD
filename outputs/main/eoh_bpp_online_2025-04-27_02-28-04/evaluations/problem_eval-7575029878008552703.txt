def priority_v2(item, bins_remain_cap):
    """
    {Assigns a priority to each bin based on a combination of remaining capacity and the amount of wasted space if the item is placed in the bin, favoring bins that are a good fit and have higher remaining capacity.}
    """
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            priority = cap - waste*0.5
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
