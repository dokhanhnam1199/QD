def priority_v2(item, bins_remain_cap):
    """{Assign a priority to each bin based on the absolute difference between the item size and a fraction of the bin's remaining capacity, favoring bins where the item size closely matches a significant portion of the bin's capacity.}"""
    priorities = []
    for remain_cap in bins_remain_cap:
        if remain_cap >= item:
            target_capacity = remain_cap * 0.75  # Aim for filling around 75% of the bin
            priority = -abs(item - target_capacity)  # Minimize the difference
        else:
            priority = -float('inf')  # Very low priority if the item doesn't fit
        priorities.append(priority)
    return priority
