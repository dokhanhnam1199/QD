def priority_v3(item, bins_remain_cap):
    """{This algorithm calculates bin priorities based on a cost function that considers bin utilization and a bonus for bins nearing full capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            utilization = item / (100 - cap + item)
            bonus = 0
            if cap - item <= 10:  # Near full capacity bonus
                bonus = 100 - cap + item
            priority_score = utilization * 100 + bonus
            priority.append(priority_score)
        else:
            priority.append(-1)
    return priority
