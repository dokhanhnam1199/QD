def priority_v3(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on a weighted sum of remaining capacity and the exponential of utilization, heavily favoring bins that are almost full but still have space for the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      utilization = item / cap
      priority_score = (0.2 * (2**utilization)) + (0.8 * (cap / 100)) # Weighted score
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
