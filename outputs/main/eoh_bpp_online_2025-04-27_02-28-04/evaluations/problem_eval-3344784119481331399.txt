def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on the ratio of the item size to the remaining bin capacity, penalizing bins that would be nearly full or have very little space used after packing the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      ratio = item / cap
      priority.append(ratio * (1 - ratio)**2) # Penalizes near full bins.
    else:
      priority.append(-1)
  return priority
