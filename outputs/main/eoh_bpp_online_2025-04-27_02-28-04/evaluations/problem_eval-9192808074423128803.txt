def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on a weighted sum of the remaining capacity and the inverse of utilization, favoring bins that have sufficient space for the item and are not too empty.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      utilization = item / cap
      priority_score = (0.6 * (1-utilization)) + (0.4 * (cap / 100)) # Weighted score
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
