def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the square of the remaining capacity if the item fits, otherwise assigns a very low priority.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(cap * cap)  # Square the remaining capacity as priority
        else:
            priority.append(-1)  # Very low priority if item doesn't fit
    return priority
