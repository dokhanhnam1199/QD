def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins by considering the proportion of the item's size relative to the bin's remaining capacity and rewarding bins that closely match the item's size.}"""
    priorities = []
    for remain_cap in bins_remain_cap:
        if remain_cap >= item:
            capacity_ratio = float(item) / remain_cap
            priority = capacity_ratio 
        else:
            priority = -1
        priorities.append(priority)
    return priority
