def priority_v2(item, bins_remain_cap):
    """{Assign a priority score to each bin based on how close the item's size is to half of the remaining capacity of the bin, penalizing bins where the item is either too small or too large relative to half the capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            ideal_size = cap / 2.0
            difference = abs(item - ideal_size)
            priority.append(1.0 / (1 + difference)) # Smaller difference yields higher priority.
        else:
            priority.append(-1)  # Cannot fit
    return priority
