def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates a priority score for each bin based on a combination of remaining capacity and wasted space if the item is placed in the bin, prioritizing bins with sufficient space and minimizing waste.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      wasted_space = cap - item
      priority.append(cap * (1 / (wasted_space + 1))) 
    else:
      priority.append(-1)
  return priority
