def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins by considering the ratio of item size to remaining bin capacity, favoring bins where the item fills a significant portion of the remaining space while still fitting.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      ratio = item / cap
      priority.append(ratio)
    else:
      priority.append(-1)
  return priority
