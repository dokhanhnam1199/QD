def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority to each bin based on how close the item size is to the bin's remaining capacity, favoring bins where the item fills a significant portion of the available space, but penalizing bins that are too full.}"""
  priorities = []
  for remain_cap in bins_remain_cap:
    if remain_cap >= item:
      fill_ratio = float(item) / remain_cap
      priority = fill_ratio - abs(fill_ratio - 0.75) # Favor bins where item fills around 75% of space
    else:
      priority = -1
    priorities.append(priority)
  return priority
