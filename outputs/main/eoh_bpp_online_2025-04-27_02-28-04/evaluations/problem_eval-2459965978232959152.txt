def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on the ratio of the item size to the remaining bin capacity, favoring bins where the item fills a significant but not excessive portion of the remaining space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      ratio = item / cap
      priority.append(ratio * (1 - abs(ratio - 0.75))) #Favors ratio around 0.75 (item utilizes 75% of remaining cap).
    else:
      priority.append(-1)
  return priority
