def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns bins a score based on how close their remaining capacity is to the item size, penalizing bins significantly for insufficient capacity or large wasted space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      diff = abs(cap - item)
      priority_score = 1 / (1 + diff) 
    else:
      priority_score = -1000  # Very low priority if item doesn't fit

    priority.append(priority_score)
  return priority
