def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority to each bin based on the percentage of the bin that would be filled by the item, favoring bins that would be filled closest to a target percentage.}"""
  priorities = []
  bin_capacity = 1.0 
  target_fill_percentage = 0.75

  for remain_cap in bins_remain_cap:
    if remain_cap >= item:
      new_remain_cap = remain_cap - item
      fill_percentage = (bin_capacity - new_remain_cap) / bin_capacity
      priority = 1 - abs(fill_percentage - target_fill_percentage)
    else:
      priority = -1
    priorities.append(priority)
  return priority
