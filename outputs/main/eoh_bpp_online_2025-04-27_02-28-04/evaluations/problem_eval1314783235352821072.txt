def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin by rewarding bins whose remaining capacity is greater than or equal to the item size and penalizing those that are not, while favoring bins with smaller remaining capacity.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(1 / (1 + cap))  # Higher priority for bins with smaller remaining capacity.
    else:
      priority.append(-1)
  return priority
