def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority to each bin based on a combination of the wasted space if the item is placed in the bin and how full the bin would be, favoring bins that result in higher bin utilization.}"""
  priorities = []
  for remain_cap in bins_remain_cap:
    if remain_cap >= item:
      wasted_space = remain_cap - item
      bin_utilization = (1 - (float(wasted_space) / 100)) # Assuming bin capacity is 100. Making relative wasted space against bin size.
      priority = bin_utilization + (1.0 / (wasted_space + 0.00001)) # Avoid division by zero
    else:
      priority = -1  # Assign a very low priority if the item doesn't fit
    priorities.append(priority)
  return priority
