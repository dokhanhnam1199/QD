def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority to each bin based on the ratio of item size to remaining capacity if the item fits, otherwise assigns a very low priority.}"""
  priorities = []
  for remain_cap in bins_remain_cap:
    if remain_cap >= item:
      priority = float(item) / remain_cap
    else:
      priority = -1  # Assign a very low priority if the item doesn't fit
    priorities.append(priority)
  return priority
