import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm constructs a heuristic matrix by scaling the inverse of the distance matrix with a factor that penalizes longer distances, making shorter distances more attractive.}"""
    n = distance_matrix.shape[0]
    
    # Heuristic: inverse distance, scaled by a factor that emphasizes shorter distances
    heuristics_matrix = (1 / (distance_matrix + 1e-9)) * (1 / (np.mean(distance_matrix) + 1e-9))
    
    return heuristics_matrix
