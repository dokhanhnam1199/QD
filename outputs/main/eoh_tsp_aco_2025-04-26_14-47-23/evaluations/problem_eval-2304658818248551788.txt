import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm constructs a heuristic matrix by adaptive weighting of distance and pheromone trails, biasing towards shorter edges and iteratively refining pheromone levels based on ant colony system principles.}"""
    n = distance_matrix.shape[0]
    pheromone_matrix = np.ones_like(distance_matrix, dtype=float)
    alpha = 0.5  # Pheromone influence
    beta = 2.0  # Distance influence
    rho = 0.1  # Evaporation rate
    Q = 100  # Pheromone deposit constant
    num_ants = 15
    num_iterations = 60
    best_so_far_tour = None
    best_so_far_length = float('inf')

    for iteration in range(num_iterations):
        ant_tours = []
        ant_tour_lengths = []

        for ant in range(num_ants):
            current_node = np.random.randint(n)
            tour = [current_node]
            unvisited_nodes = set(range(n))
            unvisited_nodes.remove(current_node)

            while unvisited_nodes:
                probabilities = np.zeros(n)
                for neighbor in range(n):
                    if neighbor in unvisited_nodes:
                        probabilities[neighbor] = (pheromone_matrix[current_node, neighbor]**alpha) * ((1 / (distance_matrix[current_node, neighbor] + 1e-9))**beta)

                if np.sum(probabilities) == 0:
                    next_node = np.random.choice(list(unvisited_nodes))
                else:
                    probabilities /= np.sum(probabilities)
                    next_node = np.random.choice(range(n), p=probabilities)
                tour.append(next_node)
                unvisited_nodes.remove(next_node)
                current_node = next_node

            tour_length = 0
            for i in range(n - 1):
                tour_length += distance_matrix[tour[i], tour[i + 1]]
            tour_length += distance_matrix[tour[-1], tour[0]]

            ant_tours.append(tour)
            ant_tour_lengths.append(tour_length)

            if tour_length < best_so_far_length:
                best_so_far_length = tour_length
                best_so_far_tour = tour
        
        #Global Pheromone Update (Elitist Strategy): Only best tour deposits pheromone
        pheromone_matrix *= (1 - rho) 
        if best_so_far_tour:
            for i in range(n - 1):
                pheromone_matrix[best_so_far_tour[i], best_so_far_tour[i + 1]] += Q / best_so_far_length
                pheromone_matrix[best_so_far_tour[i + 1], best_so_far_tour[i]] += Q / best_so_far_length
            pheromone_matrix[best_so_far_tour[-1], best_so_far_tour[0]] += Q / best_so_far_length
            pheromone_matrix[best_so_far_tour[0], best_so_far_tour[-1]] += Q / best_so_far_length

    heuristics_matrix = pheromone_matrix / (distance_matrix + 1e-9)
    return heuristics_matrix
