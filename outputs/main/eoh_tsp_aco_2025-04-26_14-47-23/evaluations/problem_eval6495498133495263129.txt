import numpy as np

def heuristics_v2(distance_matrix):
    """
    {This algorithm iteratively samples random tours, and updates a heuristic matrix to favor edges that appear frequently in short tours, while penalizing edges that appear in long tours.}
    """
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    num_iterations = 1000

    for _ in range(num_iterations):
        # Generate a random tour
        tour = np.random.permutation(n)
        tour_length = 0
        for i in range(n - 1):
            tour_length += distance_matrix[tour[i], tour[i + 1]]
        tour_length += distance_matrix[tour[-1], tour[0]]

        # Update the heuristic matrix based on tour length
        for i in range(n - 1):
            heuristics_matrix[tour[i], tour[i + 1]] += 1 / tour_length
            heuristics_matrix[tour[i + 1], tour[i]] += 1 / tour_length
        heuristics_matrix[tour[-1], tour[0]] += 1 / tour_length
        heuristics_matrix[tour[0], tour[-1]] += 1 / tour_length
        

    return heuristics_matrix
