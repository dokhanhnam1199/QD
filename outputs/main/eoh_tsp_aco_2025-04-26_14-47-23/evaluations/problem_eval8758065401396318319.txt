import numpy as np

def heuristics_v2(distance_matrix):
    """
    {This algorithm iteratively samples random tours, and updates a heuristic matrix based on the frequency each edge appears in short tours relative to its distance.}
    """
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    num_samples = 1000

    for _ in range(num_samples):
        # Generate a random tour
        tour = np.random.permutation(n)
        tour_distance = 0
        for i in range(n - 1):
            tour_distance += distance_matrix[tour[i], tour[i + 1]]
        tour_distance += distance_matrix[tour[-1], tour[0]]

        # Update heuristics matrix
        for i in range(n - 1):
            heuristics_matrix[tour[i], tour[i + 1]] += 1 / (tour_distance * distance_matrix[tour[i], tour[i + 1]])
            heuristics_matrix[tour[i + 1], tour[i]] += 1 / (tour_distance * distance_matrix[tour[i + 1], tour[i]])
        heuristics_matrix[tour[-1], tour[0]] += 1 / (tour_distance * distance_matrix[tour[-1], tour[0]])
        heuristics_matrix[tour[0], tour[-1]] += 1 / (tour_distance * distance_matrix[tour[0], tour[-1]])

    return heuristics_matrix
