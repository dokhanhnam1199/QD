def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority to each bin based on the square root of its remaining capacity if the item fits, otherwise assigns a low priority, favoring bins that are neither too full nor too empty.}"""
  priority = []
  for cap in bins_remain_cap:
    if item <= cap:
      priority_score = cap**0.5
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
