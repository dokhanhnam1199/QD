def priority_v2(item, bins_remain_cap):
  """{Assign a priority score to each bin based on the wasted space if the item is placed into it, with lower wasted space resulting in higher priority, and return these scores as a list.}"""
  priorities = []
  for cap in bins_remain_cap:
    if item <= cap:
      wasted_space = cap - item
      priority = -wasted_space  # Lower waste is better
    else:
      priority = -float('inf')  # Item doesn't fit
    priorities.append(priority)
  return priority
