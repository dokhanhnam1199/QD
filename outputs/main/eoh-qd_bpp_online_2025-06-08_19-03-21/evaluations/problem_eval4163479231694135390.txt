def priority_v3(item, bins_remain_cap):
  """{This algorithm assigns priority based on the ratio of the item size to the remaining capacity of each bin after packing the item, favoring bins where the item fills a larger portion of the remaining space and penalizing bins that can't fit the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      remaining_cap = cap - item
      ratio = item / (remaining_cap + 0.0001)  # Avoid division by zero
      priority.append(ratio)
    else:
      priority.append(-1)
  return priority
