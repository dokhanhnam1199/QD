def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the "almost full" criteria, favoring bins that would be filled to a high percentage of their capacity after the item is placed.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_percentage = (1 - (cap - item) / 100) # 100 is bin capacity
      priority.append(fill_percentage)
    else:
      priority.append(-1)
  return priority
