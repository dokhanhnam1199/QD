def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on how close their remaining capacity is to the item size, while also considering the overall fullness of the bin.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append((item / cap) * (cap - item)) # Ratio of item to capacity, penalized by remaining waste.
        else:
            priority.append(-1)
    return priority
