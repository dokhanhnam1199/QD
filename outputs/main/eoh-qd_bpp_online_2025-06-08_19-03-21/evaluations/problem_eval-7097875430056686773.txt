def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates a priority score for each bin based on the percentage of remaining capacity that would be filled by the item, prioritizing bins that would be filled more completely.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_percentage = item / cap
            priorities.append(fill_percentage)
        else:
            priorities.append(-1e9)  # Very low priority if item doesn't fit
