def priority_v2(item, bins_remain_cap):
  """{Assign a priority score to each bin based on the waste ratio if the item is placed in that bin, penalizing bins with higher waste ratios and assigning a large negative value if the item doesn't fit.}"""
  priorities = []
  for cap in bins_remain_cap:
    if item <= cap:
      waste = cap - item
      waste_ratio = float(waste) / float(cap)
      priority = -waste_ratio
    else:
      priority = -1000  # Item doesn't fit, large penalty
    priorities.append(priority)
  return priority
