def priority_v2(item, bins_remain_cap):
  """{Assign a priority score to each bin based on the bin's remaining capacity relative to the item size, favoring bins where the item fits best without considering waste.}"""
  priority = []
  for cap in bins_remain_cap:
    if item <= cap:
      priority_score = item / cap  # Prioritize bins where the item fills a larger proportion of the remaining space
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
