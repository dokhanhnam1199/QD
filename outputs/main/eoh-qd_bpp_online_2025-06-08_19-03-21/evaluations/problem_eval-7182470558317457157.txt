def priority_v2(item, bins_remain_cap):
  """{Assign a priority score to each bin based on the inverse of the wasted space if the item were placed in that bin, penalizing bins that the item does not fit into.}"""
  priorities = []
  for cap in bins_remain_cap:
    if item <= cap:
      wasted_space = cap - item
      priority = 1.0 / (wasted_space + 1)  # Avoid division by zero
    else:
      priority = -1
    priorities.append(priority)
  return priority
