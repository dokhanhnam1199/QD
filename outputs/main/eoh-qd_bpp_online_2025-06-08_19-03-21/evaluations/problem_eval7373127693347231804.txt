def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, adding a bonus for bins that are already partially filled.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            ratio = item / cap
            fullness_bonus = (1 - (cap / 100))  # Linear bonus for fullness
            priority_score = ratio + fullness_bonus
        else:
            priority_score = -1
        priority.append(priority_score)
    return priority
