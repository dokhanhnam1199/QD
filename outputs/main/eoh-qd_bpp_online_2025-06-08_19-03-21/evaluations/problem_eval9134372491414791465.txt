def priority_v2(item, bins_remain_cap):
    """{This algorithm selects the bin with remaining capacity closest to the item size without exceeding it, prioritizing bins that result in the least remaining space.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority_score = 100 - cap + item # Priority for smallest wasted space
        else:
            priority_score = -1
        priority.append(priority_score)
    return priority
