def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the ratio of the item size to the square of the remaining capacity, encouraging efficient use of space while considering bin fullness.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority = item / (cap**2)
      priorities.append(priority)
    else:
      priorities.append(-float('inf'))
  return priority
