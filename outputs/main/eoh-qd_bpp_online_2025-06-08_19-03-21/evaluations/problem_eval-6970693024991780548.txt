def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates a priority score for each bin based on how close the remaining capacity is to the item size, penalizing bins that are either much larger or too small.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      diff = cap - item
      priority_score = 1 / (1 + diff)  # Inverse of the difference, closer is better
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
