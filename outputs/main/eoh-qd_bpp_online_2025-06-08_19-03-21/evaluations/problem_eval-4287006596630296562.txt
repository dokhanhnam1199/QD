def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates a priority score for each bin based on a combination of remaining capacity and how well the item fits, favoring bins with smaller remaining capacity and tighter fits.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fit_ratio = item / cap
            priority_score = 1 / cap + fit_ratio # Favor smaller cap and tighter fit
            priority.append(priority_score)
        else:
            priority.append(-1)
    return priority
