def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on the 'best fit' concept, favoring bins that, after packing the item, have the smallest remaining capacity without overfilling, and penalizes bins with a low fill ratio.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap < item:
      priorities.append(-1)  # Cannot fit
    else:
      remaining = cap - item
      fill_ratio = item / 100 if item > 0 else 0
      priority = -remaining + fill_ratio # Prioritize smaller remaining capacity and penalize low fill ratio 
      priorities.append(priority)
  return priority
