def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, penalizing bins with very small or very large remaining capacity relative to the item.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            ratio = item / cap
            if ratio > 1:
                ratio = 1
            priority_score = ratio - abs(ratio - 0.5) # peak at 0.5, decay to either direction.
        else:
            priority_score = -1
        priority.append(priority_score)
    return priority
