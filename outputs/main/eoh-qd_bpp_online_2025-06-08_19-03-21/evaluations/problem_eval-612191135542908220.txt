def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority to each bin based on the exponential decay of the remaining capacity after packing the item, incentivizing bins that can accommodate the item with minimal wasted space.}"""
  priorities = []
  for cap in bins_remain_cap:
    if item <= cap:
      remaining_cap = cap - item
      priority = 1 / (1 + remaining_cap**2)
    else:
      priority = -1
    priorities.append(priority)
  return priority
