def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on how close their remaining capacity is to half the bin's capacity, favoring bins that are nearly half-full and penalizing those that are nearly empty or full.}"""
  priority = []
  bin_capacity = 1  # Assume bin capacity is 1; adjust if needed
  half_capacity = bin_capacity / 2.0
  for cap in bins_remain_cap:
    if cap >= item:
      diff_from_half = abs(cap - half_capacity)
      priority.append(-diff_from_half)  # Lower difference is better, so negate
    else:
      priority.append(-float('inf'))  # Assign lowest priority if item doesn't fit
  return priority
