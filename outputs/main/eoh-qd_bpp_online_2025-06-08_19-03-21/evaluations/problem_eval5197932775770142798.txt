def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on the ratio of remaining capacity to the item size, penalizing bins that would become overly full after placing the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = float(item) / cap
      priority_score = cap * (1 - abs(fill_ratio - 0.5))  # Prioritize bins that become moderately full
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
