def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority based on the absolute difference between remaining capacity and item size, prioritizing bins where the item fits closest to the capacity without exceeding it.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(-abs(cap - item))
        else:
            priority.append(-float('inf'))
    return priority
