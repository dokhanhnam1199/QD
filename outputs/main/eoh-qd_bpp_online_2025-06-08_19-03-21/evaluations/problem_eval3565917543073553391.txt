def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority based on the ratio of item size to remaining bin capacity, favoring bins where the item fills a significant portion of the available space while still fitting.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(item / cap)  # Ratio of item size to remaining capacity
        else:
            priority.append(-1)  # Penalize bins that cannot fit the item
    return priority
