def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins by considering the ratio of item size to remaining capacity, penalizing bins that are significantly larger than the item.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            ratio = item / cap
            p = 100 * ratio  # Priority based on ratio, higher ratio = better fit
        else:
            p = -1  # Cannot fit, lowest priority
        priority.append(p)
    return priority
