def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on how much the bin will be filled if the item is placed, preferring bins that will be filled to between 75% and 95% capacity.}"""
    priorities = []
    bin_capacity = sum(bins_remain_cap) / len(bins_remain_cap)
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = (bin_capacity - cap + item) / bin_capacity
            if 0.75 <= fill_ratio <= 0.95:
                priority = fill_ratio
            else:
                priority = -abs(fill_ratio - 0.85)
            priorities.append(priority)
        else:
            priorities.append(-float('inf'))
    return priority
