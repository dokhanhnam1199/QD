import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Random Fit: assign random priority to bins that can accommodate the item."""
    mask = bins_remain_cap >= item
    random_scores = np.random.rand(bins_remain_cap.shape[0])
    priorities = np.where(mask, random_scores, -np.inf)
    return priorities
