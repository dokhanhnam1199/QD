def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Computes priority for each bin using First Fit strategy.

    Favors the earliest bin that can accommodate the item; bins that cannot fit receive -inf priority.

    Args:
        item: Size of the item to pack.
        bins_remain_cap: Remaining capacities of the bins.

    Returns:
        Array of priority scores for the bins; higher values indicate higher preference.
    """
    can_fit = bins_remain_cap >= item
    n = bins_remain_cap.size
    priorities = np.full(n, -np.inf)
    priorities[can_fit] = -np.arange(n)[can_fit]
    return priorities
