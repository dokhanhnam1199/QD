```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    new_remain = bins_remain_cap - item
    mask = new_remain >= 0
    scores = np.zeros_like(bins_remain_cap, dtype=float)
    if np.any(mask):
        alpha = 10.0 / (np.std(new_remain[mask]) + 1e-6)
        target = np.percentile(new_remain[mask], 25)
        scores[mask] = 1.0 / (1.0 + np.exp(alpha * (new_remain[mask] - target)))
    return scores
```
