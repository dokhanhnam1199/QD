import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.1) -> np.ndarray:
    """Epsilonâ€‘greedy priority for online bin packing."""
    feasible = bins_remain_cap >= item
    if not feasible.any():
        return np.full_like(bins_remain_cap, -np.inf, dtype=float)
    deterministic = np.zeros_like(bins_remain_cap, dtype=float)
    deterministic[feasible] = 1.0 / (bins_remain_cap[feasible] - item + 1e-9)
    random_part = np.zeros_like(bins_remain_cap, dtype=float)
    random_part[feasible] = np.random.rand(feasible.sum())
    priorities = (1.0 - epsilon) * deterministic + epsilon * random_part
    return priorities
