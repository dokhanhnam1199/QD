import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """First Fit priority based on earliest bin with sufficient capacity."""
    enough = bins_remain_cap >= item
    base = bins_remain_cap.shape[0)
    base = bins_remain_cap.shape[0] - np.arange(bins_remain_cap.shape[0])
    penalty = (bins_remain_cap - item) * 0.001
    priorities = np.where(enough, base - penalty, -np.inf)
    return priorities
