import numpy as np
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Assigns random priority to bins that can accommodate the item."""
    can_fit = bins_remain_cap >= item
    rand_vals = np.random.rand(bins_remain_cap.shape[0])
    priorities = np.where(can_fit, rand_vals, -np.inf)
    return priorities
