```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    max_cap = bins_remain_cap.max()
    fit_mask = bins_remain_cap >= item
    priorities = np.zeros_like(bins_remain_cap, dtype=np.float64)
    if np.any(fit_mask):
        alpha = 10.0
        mid = 0.5 * max_cap
        leftover = bins_remain_cap[fit_mask] - item
        priorities[fit_mask] = 1.0 / (1.0 + np.exp(alpha * (leftover - mid)))
    return priorities
```
