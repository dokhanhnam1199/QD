import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    feasible = bins_remain_cap >= item
    if not feasible.any():
        return np.full_like(bins_remain_cap, -np.inf, dtype=float)
    slack = bins_remain_cap[feasible] - item
    w_inv = 0.5
    w_lin = -0.2
    w_sq = -0.05
    epsilon = 0.1
    deterministic = w_inv / (slack + 1e-9) + w_lin * slack + w_sq * slack**2
    random_part = np.random.rand(feasible.sum())
    scores_feasible = (1 - epsilon) * deterministic + epsilon * random_part
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    priorities[feasible] = scores_feasible
    return priorities
