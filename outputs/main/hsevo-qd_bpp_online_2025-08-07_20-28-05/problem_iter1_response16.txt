import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    slack = bins_remain_cap - item
    usable = slack >= 0
    norm_slack = np.zeros_like(bins_remain_cap, dtype=float)
    norm_slack[usable] = slack[usable] / bins_remain_cap[usable]
    k = 12.0
    mu = 0.02
    scores = np.zeros_like(bins_remain_cap, dtype=float)
    scores[usable] = 1 / (1 + np.exp(k * (norm_slack[usable] - mu)))
    return scores
