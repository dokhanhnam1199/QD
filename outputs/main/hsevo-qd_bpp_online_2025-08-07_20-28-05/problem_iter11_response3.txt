import numpy as np

def priority_v2(item, bins_remain_cap, epsilon=0.15, offset=1.0, scale=1.0, beta=0.01):
    feasible = bins_remain_cap >= item
    scores = np.full(bins_remain_cap.shape, -np.inf, dtype=float)
    if not feasible.any():
        return scores
    leftover = bins_remain_cap[feasible] - item
    deterministic = scale / (leftover + offset) - beta * (leftover ** 2)
    random_part = np.random.rand(feasible.sum())
    scores[feasible] = (1 - epsilon) * deterministic + epsilon * random_part
    return scores
