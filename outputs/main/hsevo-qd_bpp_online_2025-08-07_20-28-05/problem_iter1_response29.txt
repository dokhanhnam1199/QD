def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    suitable = bins_remain_cap >= item
    fit_scores = np.where(suitable, -(bins_remain_cap - item), -1e9)
    beta = 10.0
    exp_vals = np.exp(beta * fit_scores)
    priorities = exp_vals / exp_vals.sum()
    return priorities
