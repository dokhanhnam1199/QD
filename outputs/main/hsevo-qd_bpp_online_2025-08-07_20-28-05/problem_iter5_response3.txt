import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray,
                epsilon: float = 0.2, beta: float = 0.02) -> np.ndarray:
    """Epsilon‑greedy tight‑fit with squared‑slack penalty for waste reduction."""
    feasible = bins_remain_cap >= item
    scores = np.full(bins_remain_cap.shape, -np.inf, dtype=float)
    if not np.any(feasible):
        return scores
    leftovers = bins_remain_cap[feasible] - item
    det = 1.0/(leftovers + 1.0) - beta * leftovers**2
    rand_part = np.random.rand(bins_remain_cap.size)
    scores[feasible] = (1 - epsilon) * det + epsilon * rand_part[feasible]
    return scores
