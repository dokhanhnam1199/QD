import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines inverse slack and negative slack with epsilonâ€‘greedy randomness."""
    epsilon = 0.2
    feasible = bins_remain_cap >= item
    scores = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    if np.any(feasible):
        slack = bins_remain_cap[feasible] - item
        inv = 1.0 / (slack + 1.0)
        neg = -slack
        scores[feasible] = 0.7 * inv + 0.3 * neg
    scores = (1 - epsilon) * scores + epsilon * np.random.rand(bins_remain_cap.size)
    return scores
