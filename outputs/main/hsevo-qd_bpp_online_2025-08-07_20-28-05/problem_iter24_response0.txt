import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.1) -> np.ndarray:
    caps = np.asarray(bins_remain_cap, dtype=float)
    n = caps.size
    scores = np.full(n, -np.inf, dtype=float)
    feasible = caps >= item
    if not np.any(feasible):
        return scores
    slack = caps[feasible] - item
    deterministic = 1.0 / (slack + 1.0)
    random_part = np.random.rand(feasible.sum())
    scores[feasible] = (1.0 - epsilon) * deterministic + epsilon * random_part
    return scores
