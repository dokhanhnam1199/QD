import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray,
                offset: float = 1e-6, beta: float = 0.01,
                scale: float = 1.0) -> np.ndarray:
    """Score bins by inverse slack minus quadratic slack penalty."""
    feasible = bins_remain_cap >= item
    scores = np.full(bins_remain_cap.shape, -np.inf, dtype=float)
    if not feasible.any():
        return scores
    slack = bins_remain_cap[feasible] - item
    scores[feasible] = scale / (slack + offset) - beta * slack**2
    return scores
