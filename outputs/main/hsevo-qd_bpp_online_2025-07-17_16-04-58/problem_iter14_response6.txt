import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on waste, fullness, and feasibility."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible = bins_remain_cap >= item
    if not np.any(feasible):
        return priorities

    wasted_space = bins_remain_cap - item
    wasted_space[~feasible] = np.inf
    priorities = -wasted_space #Minimizes wasted space

    fill_rate = (bins_remain_cap - item) / (bins_remain_cap + 1e-9)
    fill_rate[~feasible] = -np.inf
    priorities +=  fill_rate # Encourages fullness
    return priorities
