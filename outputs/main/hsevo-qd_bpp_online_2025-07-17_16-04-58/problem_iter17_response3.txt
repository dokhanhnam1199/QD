import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines wasted space minimization with a target capacity."""

    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    feasible_bins = bins_remain_cap >= item

    if not np.any(feasible_bins):
        return priorities

    priorities[feasible_bins] = 0.0

    wasted_space = bins_remain_cap[feasible_bins] - item

    # Target remaining capacity encourages filling bins to a desired level.
    target_capacity = 0.2
    capacity_deviation = np.abs(wasted_space - target_capacity)
    priorities[feasible_bins] = 1.0 / (capacity_deviation + 0.00001)

    # Almost-full boost encourages complete fills.
    almost_full_threshold = 0.05
    almost_full = wasted_space <= almost_full_threshold
    priorities[feasible_bins][almost_full] += 5.0

    # Fragmentation penalty discourages leaving too much space.
    large_capacity_threshold = 0.75
    large_capacity = bins_remain_cap[feasible_bins] > large_capacity_threshold
    priorities[feasible_bins][large_capacity] -= 1.0
    return priorities
