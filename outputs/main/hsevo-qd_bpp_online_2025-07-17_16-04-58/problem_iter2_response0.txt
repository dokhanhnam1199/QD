```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins that can fit the item, favors smaller waste, and avoids tiny remaining space."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    eligible_bins = bins_remain_cap >= item
    priorities[~eligible_bins] = -np.inf  # Infeasible bins get lowest priority

    waste = bins_remain_cap - item
    priorities[eligible_bins] = -np.abs(waste[eligible_bins]) - 100 * (waste[eligible_bins] < 0.1)
    return priorities
```
