```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    valid_bins = bins_remain_cap >= item
    
    if not np.any(valid_bins):
        return np.zeros_like(bins_remain_cap)
    
    eligible_bins_cap = bins_remain_cap[valid_bins]
    
    diffs = eligible_bins_cap - item
    
    exp_diffs = np.exp(diffs / np.max(eligible_bins_cap))
    
    priorities = np.zeros_like(bins_remain_cap)
    priorities[valid_bins] = exp_diffs / np.sum(exp_diffs)
    
    return priorities
```
