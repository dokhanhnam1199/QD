import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # 1. Feasibility: Disqualify bins that can't fit the item.
    infeasible_bins = bins_remain_cap < item
    priorities[infeasible_bins] = -np.inf  # Mark infeasible bins with lowest priority

    # 2. Encourage tight packing and penalize wasted space exponentially
    feasible_bins = ~infeasible_bins
    if np.any(feasible_bins):
        remaining_space = bins_remain_cap[feasible_bins] - item
        # Exponential decay of wasted space.  Smaller remaining space => higher priority.
        priorities[feasible_bins] = np.exp(-5 * remaining_space)

        # 3. Bin Utilization: Prefer bins that are already relatively full.
        utilization = (bins_remain_cap[feasible_bins] - remaining_space) / bins_remain_cap[feasible_bins] #current_fill/capacity
        priorities[feasible_bins] *= (1 + utilization) #give a boost based on utilization

        # 4. Small Item Preference: If the item is small relative to the bin size, slightly prefer bins with more remaining capacity
        # This can help distribute smaller items more evenly, reducing fragmentation.
        small_item_threshold = 0.2  #tuneable parameter: if item < 20% of bin, consider small
        large_capacity_boost = 0.1 #tuneable parameter: a small preference
        is_small_item = item < small_item_threshold * np.max(bins_remain_cap)  #adaptive threshold
        if is_small_item:
            priorities[feasible_bins] += large_capacity_boost * bins_remain_cap[feasible_bins] / np.max(bins_remain_cap)

    return priorities
