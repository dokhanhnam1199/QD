```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # 1. Feasibility: Disqualify bins that can't fit the item.
    infeasible_bins = bins_remain_cap < item
    priorities[infeasible_bins] = -np.inf  # Mark infeasible bins with lowest priority

    # 2. Encourage tight packing: Higher priority for bins where the item fills a larger proportion.
    feasible_bins = ~infeasible_bins
    if np.any(feasible_bins):
        fill_ratios = item / bins_remain_cap[feasible_bins]
        priorities[feasible_bins] = fill_ratios

        # Bonus 1: Minimize wasted space (exponential decay).  Scale the exponent.
        remaining_space = bins_remain_cap[feasible_bins] - item
        priorities[feasible_bins] += np.exp(-5 * remaining_space)

        # Bonus 2: Favor bins that are already relatively full (avoid creating many almost-empty bins).
        # This can be more effective than just fill_ratios because it accounts for the *absolute* amount filled.
        # Weigh this bonus lower than the wasted space bonus.
        current_fill = (bins_remain_cap[feasible_bins] - remaining_space) / np.max(bins_remain_cap) #Scale for stability
        priorities[feasible_bins] += 0.5 * current_fill

    return priorities
```
