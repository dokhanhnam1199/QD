import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    eps = 1e-9
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=np.float64)
    fit = bins_remain_cap >= item
    if not np.any(fit):
        return priorities
    remaining = bins_remain_cap[fit] - item
    weight1 = 1.0 / (remaining + eps)
    max_rem = bins_remain_cap.max() + eps
    norm_left = remaining / max_rem
    thresh = 0.2
    scale = 0.05
    weight2 = 1.0 / (1.0 + np.exp((norm_left - thresh) / scale))
    random_vals = np.random.rand(np.count_nonzero(fit))
    combined = weight1 * weight2 * (1 + 0.1 * random_vals)
    priorities[fit] = combined
    return priorities
