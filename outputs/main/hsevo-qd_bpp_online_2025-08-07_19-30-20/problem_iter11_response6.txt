import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Inverse remaining space weighting with tiny random tieâ€‘breaker for online bin packing."""
    priorities = np.full_like(bins_remain_cap, -np.inf, dtype=np.float64)
    fit = bins_remain_cap >= item
    if not np.any(fit):
        return priorities
    remaining = bins_remain_cap[fit] - item
    base = 1.0 / (remaining + 1.0)
    jitter = np.random.rand(np.count_nonzero(fit)) * 1e-9
    priorities[fit] = base + jitter
    return priorities
