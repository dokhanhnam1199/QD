```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    residual = bins_remain_cap - item
    mask = residual >= 0
    beta = 10.0
    scores = np.zeros_like(bins_remain_cap, dtype=float)
    scores[mask] = np.exp(-beta * residual[mask])
    return scores
```
