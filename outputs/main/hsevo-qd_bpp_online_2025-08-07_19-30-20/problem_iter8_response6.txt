import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritize bins with tight fit (inverse residual) plus capacity bias and jitter."""
    valid = bins_remain_cap >= item
    residual = bins_remain_cap - item
    inv_residual = np.where(valid, 1/(residual + 1e-9), 0.0)
    bias = 0.01 * bins_remain_cap
    jitter = np.random.rand(bins_remain_cap.size) * 1e-6 * inv_residual
    scores = np.where(valid, inv_residual + bias + jitter, -np.inf)
    return scores
