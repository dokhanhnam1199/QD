import numpy as np

# priority_v2: combine inverse-remaining weighting, linear bias, variance‑scaled jitter.
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritize tight fits using inverse remaining capacity, a linear bias, and variance‑scaled jitter."""
    eps = 1e-12
    priorities = np.full(bins_remain_cap.shape, -np.inf, dtype=np.float64)
    fit = bins_remain_cap >= item
    if not np.any(fit):
        return priorities
    remaining = bins_remain_cap[fit] - item
    weight = 1.0 / (remaining + eps)
    max_rem = bins_remain_cap.max() if bins_remain_cap.size else 1.0
    bias = 1.0 - remaining / (max_rem + eps)
    var_factor = np.clip(np.std(remaining) / (np.mean(remaining) + eps), 0.0, 1.0)
    jitter = var_factor * (np.random.rand(remaining.size) - 0.5)
    priorities[fit] = weight * bias * (1.0 + jitter)
    return priorities
