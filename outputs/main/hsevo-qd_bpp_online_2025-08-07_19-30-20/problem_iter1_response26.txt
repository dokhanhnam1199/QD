import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    caps = np.asarray(bins_remain_cap, dtype=float)
    leftover = caps - item
    valid = leftover >= 0
    scores = np.zeros_like(caps)
    if np.any(valid):
        left_norm = leftover[valid] / caps[valid]
        beta = 12.0
        scores[valid] = 1.0 / (1.0 + np.exp(beta * (left_norm - 0.5)))
    return scores
