```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin using epsilon-greedy.
    """
    epsilon = 0.1
    fit_mask = bins_remain_cap >= item
    deterministic = np.where(fit_mask, -(bins_remain_cap - item), -1e9)
    random_component = np.random.rand(bins_remain_cap.shape[0])
    priorities = (1 - epsilon) * deterministic + epsilon * random_component
    return priorities
```
