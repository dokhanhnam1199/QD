import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.1) -> np.ndarray:
    valid = bins_remain_cap >= item
    if np.random.rand() < epsilon:
        priorities = np.random.rand(*bins_remain_cap.shape)
        priorities[~valid] = -np.inf
    else:
        priorities = -(bins_remain_cap - item)
        priorities[~valid] = -np.inf
    return priorities
