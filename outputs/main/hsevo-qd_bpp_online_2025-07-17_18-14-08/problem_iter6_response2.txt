import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.
       This version incorporates adaptive weighting based on item size and bin capacities.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    valid_bins = bins_remain_cap > 0
    priorities = np.full(bins_remain_cap.shape, -np.inf)

    if np.any(valid_bins):
        remaining_capacities = bins_remain_cap[valid_bins]

        waste = np.maximum(0, remaining_capacities - item)
        waste_score = 1 / (waste + 1e-9)

        fit_score = np.exp(-(remaining_capacities - item)**2 / (2 * (item/3)**2))

        capacity_utilization = 1 - (remaining_capacities / np.max(bins_remain_cap))

        # Adaptive Weighting: Adjust weights based on item size relative to bin capacities
        total_capacity = np.sum(remaining_capacities)
        if total_capacity > 0:
          item_proportion = item / total_capacity
        else:
          item_proportion = 0  # Avoid division by zero if no bins are available.

        # If the item is small relative to the total remaining capacity, emphasize fit and capacity utilization.
        # If it's large, emphasize waste minimization.
        if item_proportion < 0.1:
            waste_weight = 0.2
            fit_weight = 0.5
            capacity_weight = 0.3
        elif item_proportion > 0.5:
            waste_weight = 0.7
            fit_weight = 0.2
            capacity_weight = 0.1
        else:
            waste_weight = 0.6
            fit_weight = 0.3
            capacity_weight = 0.1


        combined_score = waste_weight * waste_score + fit_weight * fit_score + capacity_weight * capacity_utilization

        priorities[valid_bins] = combined_score

    return priorities
