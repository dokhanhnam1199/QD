```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and a perfect fit bonus."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    
    # Waste minimization score
    waste = bins_remain_cap - item
    waste_score = np.where(waste >= 0, 1 / (waste + 1e-9), -np.inf)
    
    # Perfect fit bonus
    perfect_fit = np.isclose(waste, 0)
    waste_score[perfect_fit] = waste_score[perfect_fit] * 2  #Boost priority
    
    # Avoid bins nearly full but too small for the current item
    near_full = (bins_remain_cap < item) & (bins_remain_cap > 0)
    waste_score[near_full] = -np.inf
    
    priorities = waste_score
    
    return priorities
```
