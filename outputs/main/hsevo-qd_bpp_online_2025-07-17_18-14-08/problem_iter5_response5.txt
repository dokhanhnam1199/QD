```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines ratio and waste to prioritize bin selection."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    valid_bins = bins_remain_cap >= item
    
    if np.any(valid_bins):
        # Ratio of item size to remaining capacity.
        ratios = item / bins_remain_cap[valid_bins]
        
        # Waste calculation
        waste = bins_remain_cap[valid_bins] - item
        
        # Combine ratio and waste, normalized. Higher priority if small waste, small ratio
        priorities[valid_bins] = 1/(waste + 1e-9) - ratios
    
    return priorities
```
