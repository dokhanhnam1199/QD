import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Calculates priority for adding item to each bin."""
    valid_bins = bins_remain_cap > 0
    priorities = np.full(bins_remain_cap.shape, -np.inf)

    if np.any(valid_bins):
        remaining_capacities = bins_remain_cap[valid_bins]

        # Waste minimization
        waste = np.maximum(0, remaining_capacities - item)
        waste_score = 1 / (waste + 1e-9)

        # Gaussian fit score
        fit_score = np.exp(-(remaining_capacities - item)**2 / (2 * (item/3)**2))

        # Capacity utilization
        capacity_utilization = 1 - (remaining_capacities / np.max(bins_remain_cap))

        # Item size relative to remaining capacity
        item_capacity_ratio = item / remaining_capacities

        # Adaptive Weights
        item_size_ratio = item / np.max(bins_remain_cap)
        waste_weight = 0.5 if item_size_ratio < 0.3 else 0.3
        fit_weight = 0.4 if item_size_ratio > 0.5 else 0.6
        capacity_weight = 0.1

        # Combined Score
        combined_score = waste_weight * waste_score + fit_weight * fit_score + capacity_weight * capacity_utilization

        # Adjust priority based on the item/capacity ratio
        # Bins where item is close to bin capacity get a boost
        ratio_boost = np.where(item_capacity_ratio > 0.7, item_capacity_ratio, 0) # Apply boost when ratio > threshold

        priorities[valid_bins] = combined_score + ratio_boost

    return priorities
