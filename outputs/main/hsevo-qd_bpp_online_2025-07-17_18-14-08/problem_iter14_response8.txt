import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization, gaussian fit, and dynamic capacity utilization."""

    valid_bins = bins_remain_cap > 0
    priorities = np.full(bins_remain_cap.shape, -np.inf)

    if np.any(valid_bins):
        remaining_capacities = bins_remain_cap[valid_bins]
        waste = np.maximum(0, remaining_capacities - item)
        waste_score = 1 / (waste + 1e-9)
        fit_score = np.exp(-(remaining_capacities - item)**2 / (2 * (item/3)**2))
        capacity_utilization = 1 - (remaining_capacities / np.max(bins_remain_cap))

        # Dynamic Weighting based on item size relative to average bin capacity
        avg_capacity = np.mean(bins_remain_cap[bins_remain_cap > 0]) if np.any(bins_remain_cap > 0) else 1  #Avoid zero division
        item_ratio = item / avg_capacity
        waste_weight = 0.5 - np.clip(item_ratio - 0.2, 0, 0.3) # Decrease waste weight if item is large
        fit_weight = 0.4 + np.clip(item_ratio - 0.2, 0, 0.3)    # Increase fit weight if item is large
        capacity_weight = 0.1

        combined_score = waste_weight * waste_score + fit_weight * fit_score + capacity_weight * capacity_utilization
        priorities[valid_bins] = combined_score

    return priorities
