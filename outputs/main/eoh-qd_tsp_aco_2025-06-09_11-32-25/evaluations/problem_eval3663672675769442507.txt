import numpy as np

def heuristics_v2(distance_matrix):
    """
    {This algorithm samples multiple random tours, and for each edge, it counts how many times it appears in the top 10% shortest tours, normalizing by the number of these tours, and returning the normalized counts as a heuristic.}
    """
    n = distance_matrix.shape[0]
    num_samples = 1000
    tour_distances = []
    tours = []

    for _ in range(num_samples):
        # Generate a random tour
        tour = np.random.permutation(n)
        tour = np.append(tour, tour[0])  # Return to starting node

        # Calculate the distance of the tour
        distance = 0
        for i in range(n):
            distance += distance_matrix[tour[i], tour[i+1]]

        tour_distances.append(distance)
        tours.append(tour)

    # Find top 10% shortest tours
    num_top_tours = int(0.1 * num_samples)
    top_tour_indices = np.argsort(tour_distances)[:num_top_tours]
    top_tours = [tours[i] for i in top_tour_indices]

    edge_counts = np.zeros_like(distance_matrix, dtype=float)

    # Count edge occurrences in top tours
    for tour in top_tours:
        for i in range(n):
            edge_counts[tour[i], tour[i+1]] += 1
            edge_counts[tour[i+1], tour[i]] += 1  # since the matrix is symmetric

    # Normalize edge counts by the number of top tours
    heuristics_matrix = edge_counts / len(top_tours)

    return heuristics_matrix
