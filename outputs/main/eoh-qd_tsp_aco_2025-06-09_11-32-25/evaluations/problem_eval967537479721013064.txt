import numpy as np

def heuristics_v2(distance_matrix):
    """
    {This algorithm constructs a heuristic matrix by combining inverse distance, nearest neighbor information, and a penalty for long edges, then normalizes it.}
    """
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros((n, n))

    # Inverse distance heuristic
    inverse_distance = 1.0 / (distance_matrix + np.eye(n))  # Avoid division by zero

    # Nearest neighbor heuristic
    nearest_neighbors = np.argsort(distance_matrix, axis=1)
    nearest_neighbor_bonus = np.zeros((n, n))
    for i in range(n):
        for j in range(1, min(4, n)):  # Bonus for top 3 nearest neighbors
            neighbor = nearest_neighbors[i, j]
            nearest_neighbor_bonus[i, neighbor] = 1.0 #/ j  # Higher bonus for closer neighbors
            
    # Penalty for long edges
    mean_distance = np.mean(distance_matrix)
    long_edge_penalty = np.where(distance_matrix > 2 * mean_distance, -0.5, 0)  # Penalize edges significantly longer than average

    # Combine heuristics
    heuristics_matrix = inverse_distance + nearest_neighbor_bonus + long_edge_penalty

    # Normalize the heuristics matrix
    min_val = np.min(heuristics_matrix)
    max_val = np.max(heuristics_matrix)
    if max_val > min_val:
        heuristics_matrix = (heuristics_matrix - min_val) / (max_val - min_val)

    return heuristics_matrix
