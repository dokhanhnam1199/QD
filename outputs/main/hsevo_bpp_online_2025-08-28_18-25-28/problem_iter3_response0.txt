import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, temperature: float = 1.0) -> np.ndarray:
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)

    normalized_fit = (bins_remain_cap[possible_bins] - item) / bins_remain_cap[possible_bins]
    normalized_fit = np.maximum(normalized_fit, 0)

    diversity_reward = 1.0 / (np.sum(possible_bins) + 1e-6)
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = np.exp(normalized_fit / temperature) * diversity_reward
    priorities = priorities / np.sum(priorities)
    return priorities
