import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, temperature: float = 1.0) -> np.ndarray:
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)

    fit_scores = (bins_remain_cap[possible_bins] - item)
    fit_scores = np.exp(-fit_scores / temperature)
    
    diversity_scores = np.ones_like(fit_scores)
    
    combined_scores = fit_scores * diversity_scores
    
    normalized_scores = combined_scores / np.sum(combined_scores)
    
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = normalized_scores

    return priorities
