import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with softmax, dynamically adjusting temperature."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities

    remaining_caps = bins_remain_cap[valid_bins]
    fit_scores = item / remaining_caps
    exp_fit_scores = np.exp(-fit_scores)
    probabilities = exp_fit_scores / np.sum(exp_fit_scores)
    priorities[valid_bins] = probabilities
    
    temperature = 0.1 + (0.9 * np.random.rand())
    best_bin_index = np.argmax(priorities)
    priorities[best_bin_index] += temperature * 0.5

    return priorities
