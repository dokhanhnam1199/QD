import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, temperature: float = 1.0) -> np.ndarray:
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)

    bin_utilizations = 1.0 - (bins_remain_cap[possible_bins] / bins_remain_cap[possible_bins].max())
    
    bin_scores = np.exp(bin_utilizations / temperature)
    
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = bin_scores
    
    if np.sum(priorities) == 0:
        best_bin_index = np.argmin(bins_remain_cap[possible_bins] - item)
        priorities[possible_bins] = 0.0
        priorities[possible_bins[best_bin_index]] = 1.0
        return priorities
    
    priorities /= np.sum(priorities)

    return priorities
