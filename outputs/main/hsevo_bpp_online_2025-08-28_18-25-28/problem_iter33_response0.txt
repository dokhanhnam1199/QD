import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    bin_utilization = 1.0 - (bins_remain_cap / np.max(bins_remain_cap))
    temperature = 0.1 + 0.9 * np.mean(bin_utilization)
    temperature = np.clip(temperature, 0.1, 1.0)
    priorities = np.zeros_like(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return priorities
    unnormalized_priorities = np.exp((bins_remain_cap[possible_bins] - item) / temperature)
    priorities[possible_bins] = unnormalized_priorities / np.sum(unnormalized_priorities)
    return priorities
