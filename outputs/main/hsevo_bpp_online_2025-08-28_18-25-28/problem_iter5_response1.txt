import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with softmax exploration and a diversity reward."""
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)

    waste = bins_remain_cap[possible_bins] - item
    priorities = 1.0 / (waste + 1e-6)
    temperature = 1.0
    probabilities = np.exp(priorities / temperature)
    probabilities /= np.sum(probabilities)
    diversity_reward = 1.0 / (np.sum(possible_bins) + 1e-6)
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = probabilities * diversity_reward
    return priorities
