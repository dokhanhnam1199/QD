import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with a temperature-scaled softmax exploration."""
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)

    waste = bins_remain_cap[possible_bins] - item
    normalized_waste = 1.0 / (waste + 1e-6)
    temperature = 1.0 / (np.mean(bins_remain_cap[possible_bins]) + 1e-6)
    exp_values = np.exp(normalized_waste / temperature)
    softmax_values = exp_values / np.sum(exp_values)

    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = softmax_values
    best_bin_index = np.argmin(bins_remain_cap[possible_bins] - item)
    priorities[possible_bins][best_bin_index] *= 2.0

    return priorities
