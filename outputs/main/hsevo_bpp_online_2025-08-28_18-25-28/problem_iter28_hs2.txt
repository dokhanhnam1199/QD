import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, temperature_base: float = 10.512176944297995, temperature_clip_min: float = 0.5115543188787152, temperature_clip_max: float = 1.2482503031303407, priority_boost: float = 1.1491181434599018) -> np.ndarray:
    """Combines best-fit with temperature-scaled softmax, prioritizing bin utilization."""
    n_bins = len(bins_remain_cap)
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros(n_bins)
    remaining_capacities = bins_remain_cap[possible_bins]
    fit_scores = - (bins_remain_cap[possible_bins] - item)
    bin_utilizations = 1.0 - (bins_remain_cap / np.max(bins_remain_cap))
    temperature = 1.0 / (1.0 + np.mean(bin_utilizations))
    temperature = np.clip(temperature, temperature_clip_min, temperature_clip_max)
    probabilities = np.exp(fit_scores / temperature) / np.sum(np.exp(fit_scores / temperature))
    priorities = np.zeros(n_bins)
    priorities[possible_bins] = probabilities
    best_bin_index = np.argmax(priorities)
    priorities[best_bin_index] *= priority_boost
    return priorities
