import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines Newtonian potential with ratio-based approach."""
    # Newtonian component
    valid_bins = bins_remain_cap > 0
    safe_bins_remain_cap = np.where(valid_bins, bins_remain_cap, np.inf)
    potential = (safe_bins_remain_cap / np.abs(safe_bins_remain_cap - item))
    newtonian_priorities = np.where(safe_bins_remain_cap >= item, potential, -np.inf)
    newtonian_priorities = np.nan_to_num(newtonian_priorities, neginf=-np.inf)
    newtonian_priorities = np.where(~valid_bins, -np.inf, newtonian_priorities)

    # Ratio component
    ratios = item / bins_remain_cap
    ratio_priorities = -np.log(ratios)
    ratio_priorities = np.where(~valid_bins, -np.inf, ratio_priorities)

    # Hybrid: Weighted combination
    combined_priorities = 0.7 * newtonian_priorities + 0.3 * ratio_priorities

    return combined_priorities
