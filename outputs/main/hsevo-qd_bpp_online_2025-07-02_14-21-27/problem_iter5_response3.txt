import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines Newtonian-inspired potential with fill ratio, adaptive weight."""

    valid_bins = bins_remain_cap > 0
    safe_bins_remain_cap = np.where(valid_bins, bins_remain_cap, np.inf)
    
    potential = (safe_bins_remain_cap / np.abs(safe_bins_remain_cap - item))
    ratio = item / safe_bins_remain_cap
    
    # Adaptive weight based on item size vs. remaining capacity
    weight = np.clip(item / np.mean(safe_bins_remain_cap[safe_bins_remain_cap != np.inf]), 0, 1) #clamp to be safe

    priorities = (weight * potential) - ((1-weight) * ratio)
    priorities = np.where(safe_bins_remain_cap >= item, priorities, -np.inf)
    priorities = np.nan_to_num(priorities, neginf=-np.inf)
    priorities = np.where(~valid_bins, -np.inf, priorities)
    
    return priorities
