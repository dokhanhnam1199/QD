```python
import numpy as np

epsilon = 1e-6
ratio_min = 0.01
ratio_max = 100
weight_clip_min = 0.1
weight_clip_max = 0.9

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines ratio, Newtonian, and adaptive weighting for bin priority."""
    # Ratio component
    ratios = item / (bins_remain_cap + epsilon)  # Avoid division by zero
    ratios = np.clip(ratios, ratio_min, ratio_max)
    ratio_priorities = -np.log(ratios)

    # Newtonian component (with checks for valid bins)
    valid_bins = bins_remain_cap > 0
    safe_bins_remain_cap = np.where(valid_bins, bins_remain_cap, np.inf)
    potential = safe_bins_remain_cap / np.abs(safe_bins_remain_cap - item + epsilon)
    newton_priorities = np.where(safe_bins_remain_cap >= item, potential, -np.inf)
    newton_priorities = np.nan_to_num(newton_priorities, neginf=-np.inf)
    newton_priorities = np.where(~valid_bins, -np.inf, newton_priorities)

    # Adaptive Weighting (based on item size relative to average capacity)
    mean_cap = np.mean(bins_remain_cap) if bins_remain_cap.size > 0 else epsilon
    weight = np.clip(item / (mean_cap + epsilon), weight_clip_min, weight_clip_max)

    # Combined priorities, emphasizing exploration
    combined_priorities = (1 - weight) * newton_priorities + weight * ratio_priorities

    # Add small random number to promote exploration
    exploration = np.random.rand(len(bins_remain_cap)) * 0.01
    combined_priorities += exploration

    return combined_priorities
```
