import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines adaptive Newtonian potential and capacity ratio."""
    valid_bins = bins_remain_cap > 0
    safe_bins_remain_cap = np.where(valid_bins, bins_remain_cap, np.inf)

    # Newtonian component
    potential = safe_bins_remain_cap / np.abs(safe_bins_remain_cap - item)
    newtonian_priority = np.where(safe_bins_remain_cap >= item, potential, -np.inf)
    newtonian_priority = np.nan_to_num(newtonian_priority, neginf=-np.inf)

    # Capacity ratio component
    ratio_priority = np.log(np.clip(safe_bins_remain_cap / (item + 1e-9), 0.001, 1000))

    # Adaptive weighting based on item size relative to bin capacity
    weight = np.clip(item / (safe_bins_remain_cap + 1e-9), 0.0, 1.0)
    combined_priority = weight * newtonian_priority + (1 - weight) * ratio_priority

    # Proportion filled metric
    proportion_filled = 1 - (safe_bins_remain_cap / np.sum(safe_bins_remain_cap)) # Use sum to calculate the proportion instead of using mean/median

    # Combine the proportion-filled and combined priorities
    final_priority = combined_priority + 0.1 * proportion_filled

    priorities = np.where(~valid_bins, -np.inf, final_priority)
    return priorities
