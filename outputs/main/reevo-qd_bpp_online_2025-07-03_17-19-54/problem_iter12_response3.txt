import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    max_capacity = np.max(bins_remain_cap)

    # 1. Feasibility: Only consider bins that can fit the item
    feasible_bins = bins_remain_cap >= item
    priorities[~feasible_bins] = -np.inf  # Very low priority if it doesn't fit

    # 2. Best Fit Heuristic: Higher priority for bins with smallest remaining capacity after packing
    remaining_space_after_packing = bins_remain_cap - item
    priorities[feasible_bins] = -np.abs(remaining_space_after_packing[feasible_bins])

    # 3. Fill percentage heuristic: Promote bins that would be filled close to 100%
    fill_percentage = (max_capacity - remaining_space_after_packing) / max_capacity
    priorities[feasible_bins] += fill_percentage[feasible_bins]

    # 4. Penalize near-full bins to prevent fragmentation
    near_full_threshold = 0.95 * max_capacity
    near_full_bins = bins_remain_cap >= near_full_threshold
    priorities[near_full_bins] -= 0.1

    # 5. Bonus for empty bins (encourages starting new bins when appropriate, but reduced impact)
    empty_bins = bins_remain_cap == max_capacity
    priorities[empty_bins] += 0.02 # Reduced impact, as excessive empty bin creation is bad

    # 6. Prioritize bins with capacity close to item size, stronger effect.
    item_size_proximity = -np.abs(bins_remain_cap - item)
    priorities[feasible_bins] += 0.8 * (item_size_proximity[feasible_bins] / max_capacity) # Scale and add. Increased impact.

    # 7. Modified Best Fit: prioritize bins where the item fills over a certain percentage. Stronger effect.
    fill_threshold = 0.75 #item fills at least 75% of the bins current capacity
    high_fill_bins = (item / bins_remain_cap) >= fill_threshold
    priorities[high_fill_bins] += 0.3 #stronger bonus for filling bin by good amount

    # 8. Introduce a penalty for bins with very little space remaining after packing, to avoid tiny fragments.
    tiny_fragment_threshold = 0.05 * max_capacity
    tiny_fragment_bins = remaining_space_after_packing > 0  # Only consider bins where the item fits
    tiny_fragment_bins = tiny_fragment_bins & (remaining_space_after_packing <= tiny_fragment_threshold) # Check remaining cap.

    priorities[tiny_fragment_bins] -= 0.2

    return priorities
