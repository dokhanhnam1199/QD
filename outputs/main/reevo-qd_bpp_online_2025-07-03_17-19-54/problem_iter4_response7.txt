import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    max_capacity = np.max(bins_remain_cap)

    # 1. Feasibility: Only consider bins that can fit the item
    feasible_bins = bins_remain_cap >= item
    priorities[~feasible_bins] = -np.inf  # Very low priority if it doesn't fit

    if np.any(feasible_bins):
        # 2. Best Fit Heuristic: Higher priority for bins with smallest remaining capacity after packing
        remaining_space_after_packing = bins_remain_cap - item
        priorities[feasible_bins] += -np.abs(remaining_space_after_packing[feasible_bins])

        # 3. Fill percentage heuristic: Promote bins that would be filled close to 100%
        fill_percentage = (max_capacity - remaining_space_after_packing) / max_capacity
        priorities[feasible_bins] += fill_percentage[feasible_bins]

        # 4. Penalize near-full bins to prevent fragmentation (adjust threshold)
        near_full_threshold = 0.9 * max_capacity
        near_full_bins = (bins_remain_cap >= near_full_threshold) & feasible_bins
        priorities[near_full_bins] -= 0.05

        # 5. Bonus for bins that are empty
        empty_bins = bins_remain_cap == max_capacity
        priorities[empty_bins] += 0.025 #Slightly less priority than penalty to encourage filling existing bins first

        # 6. Prioritize bins with capacity close to item size to avoid large fragmentations
        close_fit_threshold = item * 1.1  # Allow 10% larger bin than item
        close_fit_bins = (bins_remain_cap >= item) & (bins_remain_cap <= close_fit_threshold)
        priorities[close_fit_bins] += 0.075 #Slightly higher bonus than empty bin

        # 7. Slightly penalize bins that would be left with very little space after packing
        very_little_space = 0.1 * max_capacity #Remaining space less than 10% of max capacity.
        bins_with_very_little_space = (remaining_space_after_packing > 0) & (remaining_space_after_packing <= very_little_space) & feasible_bins
        priorities[bins_with_very_little_space] -= 0.03 # Less penalty than near full penalty.

    return priorities
