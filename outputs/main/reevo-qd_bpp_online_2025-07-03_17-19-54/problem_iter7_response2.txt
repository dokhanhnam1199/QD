import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    max_capacity = np.max(bins_remain_cap)  # Maximum capacity of bins

    # 1. Feasibility: Only consider bins that can fit the item
    feasible_bins = bins_remain_cap >= item
    priorities[~feasible_bins] = -np.inf  # Very low priority if it doesn't fit

    # 2. Best Fit Heuristic: Higher priority for bins with smallest remaining capacity after packing
    remaining_space_after_packing = bins_remain_cap - item
    priorities[feasible_bins] = -np.abs(remaining_space_after_packing[feasible_bins])

    # 3. Fill percentage heuristic: Promote bins that would be filled close to 100%
    fill_percentage = (max_capacity - remaining_space_after_packing) / max_capacity
    priorities[feasible_bins] += fill_percentage[feasible_bins]

    # 4. Fragmentation control (less relevant for online, but still a small penalty)
    near_full_threshold = 0.95 * max_capacity
    near_full_bins = bins_remain_cap[feasible_bins] - item > 0  # Apply only to feasible bins
    near_full_bins_indices = np.where(feasible_bins)[0][near_full_bins]
    overfill_amount = (bins_remain_cap[near_full_bins_indices] - item) / max_capacity  # normalized overfill amounts
    priorities[near_full_bins_indices] -= 0.05 * overfill_amount  # Smaller penalty

    # 5. Bonus for bins that are empty
    empty_bins = bins_remain_cap == max_capacity
    priorities[empty_bins] += 0.1

    # 6. Item size proximity: Give a small boost if the item size is a significant portion of remaining capacity
    size_ratio = item / bins_remain_cap[feasible_bins]
    size_ratio = np.clip(size_ratio, 0, 1)  # Ensure ratio stays within 0 and 1
    size_proximity_boost = 0.05 * size_ratio # Small proportional bonus
    priorities[feasible_bins] += size_proximity_boost

    return priorities
