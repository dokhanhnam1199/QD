import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins, favoring those that fit the item well
    and penalizing those where the item doesn't fit.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fit_mask = bins_remain_cap >= item
    
    if np.any(fit_mask):
        waste = bins_remain_cap[fit_mask] - item
        priorities[fit_mask] = (bins_remain_cap[fit_mask]**2) / (waste + 0.00001)
    else:
        priorities = -np.abs(bins_remain_cap - item)

    return priorities
