```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fill ratio with waste penalty for online bin packing."""

    epsilon = 1e-9
    ratios = item / (bins_remain_cap + epsilon)
    fill_score = np.exp(-np.abs(ratios - 1) * 5)

    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-waste / (item + epsilon)), -1e9)

    priorities = 0.7 * fill_score + 0.3 * waste_penalty
    return priorities
```
