import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on capacity fit and waste, avoiding near-full bins."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fit_mask = bins_remain_cap >= item
    waste = bins_remain_cap[fit_mask] - item
    
    if waste.size > 0:
        priorities[fit_mask] = 1.0 / (waste + 0.00001)

    almost_full_mask = (bins_remain_cap > 0) & (bins_remain_cap - item < 0.1)
    priorities[almost_full_mask] = -100

    return priorities
