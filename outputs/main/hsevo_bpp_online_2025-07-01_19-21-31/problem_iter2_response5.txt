import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on near-fit, waste, and feasibility."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    infeasible = bins_remain_cap < item
    priorities[infeasible] = -np.inf

    waste = bins_remain_cap - item
    feasible = bins_remain_cap >= item

    # Prioritize bins with less wasted space
    priorities[feasible] = np.exp(-waste[feasible] / (item + epsilon))

    near_fit = (bins_remain_cap >= item) & (waste < 0.1)
    priorities[near_fit] += 2 # Boost near-fit
    return priorities
