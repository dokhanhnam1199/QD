```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin. 
    Combines best fit strategy with a bonus for smaller remaining capacity.
    """
    # Check if the item can fit in each bin and calculate priority
    valid_bins = bins_remain_cap >= item
    # Prioritize bins that fit the item tightly
    fit_score = -np.abs(bins_remain_cap - item)
    # Add a small bonus based on the remaining capacity to break ties
    bonus = bins_remain_cap * 1e-6
    # Combine fit score and bonus
    priority = np.where(valid_bins, fit_score + bonus, -np.inf)
    
    return priority
```
