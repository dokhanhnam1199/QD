def priority_v2(item: float, bins_remain_cap: np.ndarray, 
                epsilon: float = 1e-6, 
                bin_fullness_weight: float = 1.0) -> np.ndarray:
    """
    Combines bin fullness and capacity check for effective bin prioritization.
    Rewards fuller bins that can fit the item.
    """
    # Check if the bin has enough capacity for the item
    has_enough_capacity = bins_remain_cap >= item
    
    # Calculate bin fullness score, penalizing bins without enough capacity
    bin_fullness_score = np.where(has_enough_capacity, 
                                  1 - (bins_remain_cap - item) / (bins_remain_cap + epsilon), 
                                  -1)
    
    # Use bin fullness score as priority, weighted by bin_fullness_weight
    priority_scores = bin_fullness_weight * bin_fullness_score
    
    return priority_scores
