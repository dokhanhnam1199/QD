import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin. 
    Prioritizes bins that fit the item tightly, are more full, and have smaller remaining capacity.
    """
    # Check if the bin has enough capacity for the item
    has_enough_capacity = bins_remain_cap >= item
    
    # Calculate the priority score for bins with enough capacity
    # Higher priority for bins that are more full and fit the item tightly
    tight_fit_score = 1 / (bins_remain_cap - item + 1e-6)
    fullness_score = 1 / (bins_remain_cap + 1e-6)  # Prefer bins with smaller remaining capacity
    priority_scores = np.where(has_enough_capacity, tight_fit_score + fullness_score, -np.inf)
    
    return priority_scores
