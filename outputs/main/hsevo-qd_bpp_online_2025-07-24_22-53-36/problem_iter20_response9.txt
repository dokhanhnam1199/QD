import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin. 
    Rewards bins that are full, have enough capacity, and have remaining capacity close to the item size.
    """
    # Check if bin has enough capacity
    has_enough_capacity = bins_remain_cap >= item
    
    # Calculate bin fullness and remaining capacity scores
    bin_fullness_score = np.where(has_enough_capacity, 1 - ((bins_remain_cap - item) / (bins_remain_cap + 1e-6)), -1)
    remaining_cap_score = np.where(has_enough_capacity, 1 / (1 + np.abs(bins_remain_cap - item) / (item + 1e-6)), 0)
    
    # Combine scores with simple weighting
    priority_scores = bin_fullness_score + remaining_cap_score
    
    return priority_scores
