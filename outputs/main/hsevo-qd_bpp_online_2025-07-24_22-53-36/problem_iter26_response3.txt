import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines simplicity with effective bin prioritization using bin fullness and fit scores.
    """
    # Check if the bin has enough capacity for the item
    has_enough_capacity = bins_remain_cap >= item
    
    # Calculate remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Calculate bin fullness score, rewarding bins that are fuller
    bin_fullness_score = np.where(has_enough_capacity, 
                                  1 - (remaining_cap_after_addition / (bins_remain_cap + 1e-6)), 
                                  -1)
    
    # Calculate fit score, prioritizing bins that fit the item tightly
    fit_score = np.where(has_enough_capacity, 
                         1 / (1 + np.abs(remaining_cap_after_addition) / (item + 1e-6)), 
                         0)
    
    # Introduce a bias towards bins with smaller indices to break ties
    index_bias = -np.arange(len(bins_remain_cap)) / len(bins_remain_cap)
    
    # Combine scores with weights, focusing on simplicity and clarity
    weights = np.array([0.7, 0.3])  # Weights for bin_fullness_score and fit_score
    priority_scores = (weights[0] * bin_fullness_score + 
                       weights[1] * fit_score + 
                       0.0 * index_bias)  # Removed tunable weight for index_bias
    
    # Normalize priority scores using tanh for simplicity and robustness
    priority_scores = np.tanh(priority_scores)
    
    return priority_scores
