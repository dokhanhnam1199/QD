import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for adding an item to each bin, 
    combining bin fullness and fit scores with a simple, straightforward approach.
    """
    # Calculate remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Check if the bin has enough capacity
    has_enough_capacity = remaining_cap_after_addition >= 0
    
    # Calculate bin fullness score (rewarding full bins) and fit score (rewarding bins with remaining capacity close to the item size)
    bin_fullness_score = np.where(has_enough_capacity, 1 - (remaining_cap_after_addition / (bins_remain_cap + 1e-6)), -1)
    remaining_cap_score = np.where(has_enough_capacity, 1 / (1 + np.abs(remaining_cap_after_addition) / (item + 1e-6)), 0)
    
    # Combine scores with equal weights and introduce a bias towards bins with smaller indices to break ties
    priority_scores = 0.5 * bin_fullness_score + 0.5 * remaining_cap_score - np.arange(len(bins_remain_cap)) / len(bins_remain_cap) * 0.1
    
    # Normalize priority scores using tanh to avoid extreme values
    priority_scores = np.tanh(priority_scores)
    
    return priority_scores
