import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines bin fullness and remaining capacity scores effectively with tunable weights.
    Prioritizes bins that are more full and have enough capacity for the item.
    """
    # Calculate remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Check if the bin has enough capacity
    has_enough_capacity = remaining_cap_after_addition >= 0
    
    # Calculate priority scores for bins with enough capacity
    # Weighted sum of bin's fullness and remaining capacity after addition
    fullness_score = 1 - (bins_remain_cap / (bins_remain_cap + item))  # Score for bin fullness
    remaining_score = 1 - (remaining_cap_after_addition / (bins_remain_cap + 1e-6))  # Score for remaining capacity
    priority_scores_enough_cap = 0.6 * fullness_score + 0.4 * remaining_score  # Weighted sum
    
    # Assign low priority to bins without enough capacity
    priority_scores_not_enough_cap = -1 * np.ones_like(bins_remain_cap)
    
    # Combine priority scores
    priority_scores = np.where(has_enough_capacity, priority_scores_enough_cap, priority_scores_not_enough_cap)
    
    return priority_scores
