```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for adding an item to each bin.
    
    Prioritizes bins with the least remaining capacity after adding the item.
    """
    # Calculate the remaining capacity after adding the item to each bin
    remaining_cap_after_add = bins_remain_cap - item
    
    # Assign high priority to bins that can hold the item and have less remaining capacity after addition
    priorities = np.where(bins_remain_cap >= item, 
                          -remaining_cap_after_add,  # Less remaining capacity is better
                          -np.inf)  # Cannot hold the item
    
    return priorities
```
