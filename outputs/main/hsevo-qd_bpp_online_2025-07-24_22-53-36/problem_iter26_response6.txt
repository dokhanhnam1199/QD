```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for adding an item to each bin. 
    Prioritizes bins with enough capacity, high fullness, and remaining capacity close to the item size.
    """
    # Check if bin has enough capacity
    has_enough_capacity = bins_remain_cap >= item
    
    # Calculate scores rewarding bins that are full and have remaining capacity close to item size
    bin_fullness_score = np.where(has_enough_capacity, 1 - ((bins_remain_cap - item) / (bins_remain_cap + 1e-6)), -1)
    remaining_cap_score = np.where(has_enough_capacity, 1 / (1 + np.abs(bins_remain_cap - item) / (item + 1e-6)), 0)
    
    # Combine scores with fixed weights ( simplicity & avoiding tunable parameters )
    priority_scores = 0.5 * bin_fullness_score + 0.5 * remaining_cap_score
    
    return priority_scores
```
