import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    n = bins_remain_cap.shape[0]
    idx = np.arange(n)
    feasible = bins_remain_cap >= item
    order_prio = (n - idx).astype(bins_remain_cap.dtype)
    residual = (bins_remain_cap - item) * 1e-4
    base = np.where(feasible, order_prio, np.zeros_like(bins_remain_cap))
    return np.where(feasible, base + residual, base)
