import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    caps = bins_remain_cap.astype(float)
    space_left = caps - item
    valid = space_left >= 0
    priorities = np.full_like(space_left, -1e9, dtype=float)
    if np.any(valid):
        ratio = space_left[valid] / caps[valid]
        k = 12.0
        priorities[valid] = 1.0 / (1.0 + np.exp(-k * (1.0 - ratio)))
    return priorities
