```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Compute priorities for each bin using the Worst Fit strategy:
    - For bins that can accommodate the item, priority equals remaining capacity after placement.
    - For bins that cannot fit the item, priority is set to negative infinity to exclude them.
    """
    return np.where(bins_remain_cap >= item, bins_remain_cap - item, -np.inf)
```
