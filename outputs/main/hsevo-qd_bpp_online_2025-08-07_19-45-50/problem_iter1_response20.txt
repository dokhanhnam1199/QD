import numpy as np
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """First Fit priority: selects the earliest feasible bin and prefers tighter fit."""
    can_fit = bins_remain_cap >= item
    remaining = bins_remain_cap - item
    idx = np.arange(bins_remain_cap.shape[0])
    priority = np.where(can_fit, -remaining - idx * 1e-6, -np.inf)
    return priority
