import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    diff = bins_remain_cap - item
    feasible = diff >= 0
    if not np.any(feasible):
        return np.zeros_like(bins_remain_cap)
    target = np.mean(diff[feasible])
    k = 8.0
    exp_term = np.exp(np.clip(k * (diff - target), -200, 200))
    scores = 1 / (1 + exp_term)
    priorities = np.where(feasible, scores, 0.0)
    return priorities
