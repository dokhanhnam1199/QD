import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority based on slack, huge for exact fit, -inf for infeasible, deterministic jitter for tieâ€‘breaks."""
    priorities = np.full(bins_remain_cap.shape, -np.inf, dtype=float)
    exact_mask = bins_remain_cap == item
    valid_mask = bins_remain_cap >= item
    priorities[exact_mask] = 1e12
    valid_not_exact = valid_mask & ~exact_mask
    slack = bins_remain_cap[valid_not_exact] - item
    idx = np.arange(bins_remain_cap.size)
    jitter = 1e-9 * np.sin(item * (1 + idx))
    priorities[valid_not_exact] = -slack + jitter[valid_not_exact]
    return priorities
