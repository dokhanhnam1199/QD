import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Calculate inverse leftover capacity for each bin to prioritize tight fits."""
    eps = 1e-9
    leftover = bins_remain_cap - item
    fit = leftover >= 0
    priorities = np.where(fit, 1.0 / (leftover + eps), -np.inf)
    return priorities
