```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority for placing an item in each bin using Exact Fit First.
    Exact fit (zero remaining capacity) gets the highest priority.
    Bins where the item does not fit receive negative infinity priority.
    """
    feasible = bins_remain_cap >= item
    leftover = bins_remain_cap - item
    priorities = np.where(feasible, -leftover, -np.inf)
    return priorities
```
