import numpy as np
from typing import Optional

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.1, temperature: float = 0.1, random_state: Optional[int] = None) -> np.ndarray:
    """Combine waste score, temperature scaling, and epsilonâ€‘greedy random exploration."""
    bins = np.asarray(bins_remain_cap, dtype=float)
    feasible = bins >= item
    if not np.any(feasible):
        return np.full(bins.shape, -np.inf, dtype=float)
    waste = bins -0.0 - item
    deterministic = -waste / max(temperature, 1e-12)
    rng = np.random.default_rng(random_state)
    random_scores = rng.random(bins.shape)
    combined = np.where(feasible, (1 - epsilon) * deterministic + epsilon * random_scores, -np.inf)
    return combined
