import numpy as np
from typing import Optional

# Priority function combining deterministic waste and optional epsilonâ€‘greedy exploration
def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.0,
                random_state: Optional[int] = None) -> np.ndarray:
    """Priority = -waste for feasible bins; optional epsilonâ€‘greedy random scores."""
    feasible = bins_remain_cap >= item
    if epsilon > 0.0:
        rng = np.random.default_rng(random_state)
        if rng.random() < epsilon:
            rand_scores = rng.random(bins_remain_cap.shape[0])
            return np.where(feasible, rand_scores, -np.inf)
    waste = bins_remain_cap - item
    return np.where(feasible, -waste, -np.inf)
