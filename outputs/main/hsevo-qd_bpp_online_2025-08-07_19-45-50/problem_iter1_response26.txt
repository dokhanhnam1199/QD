```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    rem = bins_remain_cap - item
    valid = rem >= 0
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    k = 10.0
    priorities[valid] = 1.0 / (1.0 + np.exp(k * rem[valid] / item))
    return priorities
```
