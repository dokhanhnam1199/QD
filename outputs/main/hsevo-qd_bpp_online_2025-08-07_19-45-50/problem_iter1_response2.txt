import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Worst Fit priority: favor bins with the most remaining capacity that can accommodate the item."""
    feasible = bins_remain_cap >= item
    priorities = np.where(feasible, bins_remain_cap.astype(float), -np.inf)
    return priorities
