```python
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    suitable_bins = bins_remain_cap >= item
    if not np.any(suitable_bins):
        return np.zeros_like(bins_remain_cap)
    
    priorities = np.zeros_like(bins_remain_cap)
    
    worst_fit_candidates = bins_remain_cap[suitable_bins]
    
    
    min_remaining_capacity = np.min(worst_fit_candidates)
    
    
    priorities[suitable_bins] = (worst_fit_candidates - min_remaining_capacity)
    
    
    priorities[bins_remain_cap < item] = -np.inf
    
    
    return priorities
```
