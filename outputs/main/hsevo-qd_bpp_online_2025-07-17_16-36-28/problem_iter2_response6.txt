```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines snugness, fragmentation avoidance, and utilization."""

    wasted_space = bins_remain_cap - item
    priorities = np.where(wasted_space < 0, -np.inf, 0.0)

    # Snugness: smaller waste is better
    snugness = 1 / (wasted_space + 1e-9)
    snugness = np.nan_to_num(snugness, nan=0.0, posinf=0.0, neginf=0.0)

    # Fragmentation risk: avoid almost full
    sigmoid_input = 5 * (bins_remain_cap - item) / bins_remain_cap
    fragmentation_risk = 1 / (1 + np.exp(-sigmoid_input))

    # Utilization: Prefer higher utilization
    utilization = item / bins_remain_cap
    utilization = np.clip(utilization, 0, 1)

    priorities = np.where(wasted_space >= 0, snugness * (1 - fragmentation_risk) + 0.5 * utilization, priorities)

    #add small exploration noise.
    priorities += np.random.normal(0, 0.001, size=priorities.shape)
    return priorities
```
