```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fit, wasted space, and spreading items."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    valid_bins = bins_remain_cap - item >= 0
    if not np.any(valid_bins):
        return priorities

    priorities[valid_bins] = 1.0  # Base priority for fitting

    wasted_space = bins_remain_cap[valid_bins] - item
    priorities[valid_bins] += 1.0 / (1e-6 + wasted_space)  # Minimize wasted space

    priorities[valid_bins] += bins_remain_cap[valid_bins] / (np.sum(bins_remain_cap[valid_bins]) + 1e-6)  # Spread items

    priorities[valid_bins] += np.random.rand(np.sum(valid_bins)) * 0.01 # add a small random component

    return priorities
```
