import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fit, wasted space, and fill level."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    valid_bins = bins_remain_cap >= item

    if not np.any(valid_bins):
        return priorities

    remaining_cap_after_add = bins_remain_cap - item
    wasted_space = remaining_cap_after_add[valid_bins]

    priorities[valid_bins] = 1.0 / (1e-6 + wasted_space)  # Minimize wasted space
    priorities[valid_bins] += np.exp(-np.abs(remaining_cap_after_add[valid_bins] - np.mean(bins_remain_cap) / 2))  # Target half-full
    priorities[valid_bins] += np.random.rand(np.sum(valid_bins)) * 0.01 # Add small random
    return priorities
