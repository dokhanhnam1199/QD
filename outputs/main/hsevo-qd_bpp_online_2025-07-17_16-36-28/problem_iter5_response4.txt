import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fit ratio, energy landscape, and wasted space."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fit_mask = bins_remain_cap >= item

    # Fit ratio component
    fit_ratio = item / bins_remain_cap[fit_mask]
    priorities[fit_mask] = (1 - np.abs(fit_ratio - 0.7)) - 0.01 * (bins_remain_cap[fit_mask] - item)
    # Energy landscape component
    priorities[fit_mask] /= (bins_remain_cap[fit_mask] + 1e-9)
    priorities[~fit_mask] = -np.inf
    return priorities
