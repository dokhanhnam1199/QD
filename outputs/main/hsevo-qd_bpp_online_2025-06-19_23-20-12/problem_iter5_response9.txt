import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fill ratio, waste, and handles infeasibility."""

    potential_bins = bins_remain_cap >= item
    if not np.any(potential_bins):
        return np.zeros_like(bins_remain_cap)

    waste = bins_remain_cap - item
    waste[waste < 0] = np.inf
    waste_normalized = waste / np.max(bins_remain_cap)

    fill_ratio = (bins_remain_cap - waste) / bins_remain_cap
    fill_ratio = np.clip(fill_ratio, 0, 1)  # Ensure fill_ratio is within [0, 1]
    
    # Incorporate penalty for near-full bins to prevent creating tiny wastes
    near_full_penalty = np.where(bins_remain_cap < 1.1 * item, 0.2, 0) # or any reasonable constant instead of 1.1

    # Weighted combination of fill ratio and waste normalization.
    weight_fill = 0.6
    weight_waste = 0.4

    priorities = np.where(potential_bins, weight_fill * fill_ratio - weight_waste * waste_normalized - near_full_penalty, -np.inf)

    return priorities
