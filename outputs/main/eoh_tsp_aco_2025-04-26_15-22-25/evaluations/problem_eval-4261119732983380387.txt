import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm uses a combination of nearest neighbor and simulated annealing principles to generate a promising edge matrix for the TSP problem.}"""

    num_nodes = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)

    for i in range(num_nodes):
        for j in range(num_nodes):
            if i == j:
                heuristics_matrix[i, j] = 0
            else:
                heuristics_matrix[i, j] = 1 / distance_matrix[i, j] if distance_matrix[i, j] > 0 else 0

    return heuristics_matrix
