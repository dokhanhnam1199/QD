```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for adding an item to each bin.
    Prioritizes bins where the item fits well and adds noise to break ties.
    """
    # Calculate the priority score for each bin where the item fits
    scores = np.where(bins_remain_cap >= item, 
                       1 - np.abs(bins_remain_cap - item) / (bins_remain_cap + item), 
                       0)
    # Add a small random noise to break ties
    noise = np.random.uniform(0, 1e-6, size=len(bins_remain_cap))
    return scores + noise
```
