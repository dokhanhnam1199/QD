import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm combines greedy nearest neighbor path construction with pheromone updates based on path quality and a scaling factor based on edge distance to build a heuristic matrix favoring edges in good solutions.}"""
    n = distance_matrix.shape[0]
    pheromone_matrix = np.ones_like(distance_matrix)
    alpha = 0.7  # Pheromone influence
    beta = 1.5  # Distance influence
    rho = 0.3  # Evaporation rate
    Q = 150  # Pheromone deposit constant
    num_ants = n
    num_iterations = 100
    distance_scale = 0.1

    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)

    for iteration in range(num_iterations):
        all_paths = []
        all_path_lengths = []

        for ant in range(num_ants):
            start_node = ant % n
            current_node = start_node
            unvisited_nodes = set(range(n))
            unvisited_nodes.remove(start_node)
            path = [start_node]
            path_length = 0

            while unvisited_nodes:
                # Nearest Neighbor biased by Pheromones
                probabilities = []
                candidates = list(unvisited_nodes)
                for neighbor in candidates:
                    pheromone = pheromone_matrix[current_node, neighbor]
                    distance = distance_matrix[current_node, neighbor]
                    probabilities.append((pheromone**alpha) / (distance**beta))
                probabilities = np.array(probabilities)
                probabilities /= np.sum(probabilities)
                next_node = np.random.choice(candidates, p=probabilities)
                
                path.append(next_node)
                path_length += distance_matrix[current_node, next_node]
                unvisited_nodes.remove(next_node)
                current_node = next_node

            path.append(start_node)
            path_length += distance_matrix[current_node, start_node]
            all_paths.append(path)
            all_path_lengths.append(path_length)

        # Update pheromone trails based on path quality
        pheromone_matrix *= (1 - rho)  # Evaporation

        # Deposit pheromone on edges of all paths, weighted by path length
        for path, path_length in zip(all_paths, all_path_lengths):
            for i in range(n):
                node1 = path[i]
                node2 = path[i+1]
                pheromone_matrix[node1, node2] += (Q / path_length) * (1 + distance_scale * (1 - (distance_matrix[node1, node2] / np.max(distance_matrix))))
                pheromone_matrix[node2, node1] += (Q / path_length) * (1 + distance_scale * (1 - (distance_matrix[node2, node1] / np.max(distance_matrix))))

    # Aggregate pheromone information into the heuristics matrix
    heuristics_matrix = pheromone_matrix
    return heuristics_matrix
