import numpy as np

def heuristics_v2(distance_matrix):
  """{This algorithm uses a combination of inverse distance and shortest-edge inclusion frequency from sampled greedy tours, with an added component that prefers edges connecting nodes that are relatively far apart in the greedy tours, to guide edge selection.}"""
  n = distance_matrix.shape[0]
  num_samples = 1000
  edge_counts = np.zeros_like(distance_matrix)
  tour_distances = np.zeros_like(distance_matrix)

  for _ in range(num_samples):
    start_node = np.random.randint(n)
    tour = [start_node]
    unvisited = list(range(n))
    unvisited.remove(start_node)

    while unvisited:
      current_node = tour[-1]
      nearest_neighbor = min(unvisited, key=lambda x: distance_matrix[current_node, x])
      tour.append(nearest_neighbor)
      unvisited.remove(nearest_neighbor)
    
    for i in range(n):
      u = tour[i]
      v = tour[(i + 1) % n]
      edge_counts[u, v] += 1
      edge_counts[v, u] += 1

      # Calculate distance between nodes in the tour
      u_index = tour.index(u)
      v_index = tour.index(v)
      tour_dist = abs(u_index - v_index)
      tour_distances[u, v] += tour_dist
      tour_distances[v, u] += tour_dist

  edge_frequencies = edge_counts / num_samples
  inverse_distances = 1.0 / (distance_matrix + 1e-9)  # Avoid division by zero
  normalized_tour_distances = tour_distances / num_samples
  heuristics_matrix = edge_frequencies * inverse_distances * (1 + normalized_tour_distances/n)
  return heuristics_matrix
