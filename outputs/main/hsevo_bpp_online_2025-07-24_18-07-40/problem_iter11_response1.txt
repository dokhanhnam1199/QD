```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Priority score is the negative remaining capacity for bins that can fit the item, 
    with a bonus for nearly full bins. Encourages best-fit strategy.
    """
    # Calculate priority scores with a best-fit strategy and bonus for nearly full bins
    can_fit = bins_remain_cap >= item
    priority = np.where(can_fit, -(bins_remain_cap - item) + 0.1 * (bins_remain_cap - item < 0.1), -np.inf)
    
    return priority
```
