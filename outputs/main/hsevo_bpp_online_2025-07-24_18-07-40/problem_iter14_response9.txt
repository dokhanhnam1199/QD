import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines best-fit strategy with a bonus for nearly full bins and normalized remaining capacity.
    """
    # Best-fit strategy
    best_fit = -np.abs(bins_remain_cap - item)
    
    # Bonus for bins that are nearly full after adding the item
    remaining_cap = np.where(bins_remain_cap >= item, bins_remain_cap - item, np.inf)
    bonus = -remaining_cap / np.max(remaining_cap + 1e-6)
    
    # Combine best-fit with bonus and normalize
    priority = best_fit + 0.5 * bonus
    valid_bins = bins_remain_cap >= item
    if np.any(valid_bins):
        max_priority = np.max(priority[valid_bins])
        min_priority = np.min(priority[valid_bins])
        priority = np.where(valid_bins, (priority - min_priority) / (max_priority - min_priority + 1e-6), -np.inf)
    else:
        priority = -np.inf * np.ones_like(priority)
    
    return priority
