```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Priority score for each bin based on best-fit strategy and bonus for nearly full bins.
    """
    # Calculate priority score: favor bins with remaining capacity closest to item size and higher utilization
    max_capacity = 1.0  # Uniform initial capacity for all bins
    # Best-fit strategy
    best_fit = -np.abs(bins_remain_cap - item)
    # Bonus for nearly full bins
    utilization_bonus = (max_capacity - bins_remain_cap) / max_capacity
    
    priority = best_fit + 0.5 * utilization_bonus
    
    # Set priority to negative infinity if bin cannot hold the item
    priority[bins_remain_cap < item] = -np.inf
    
    return priority
```
