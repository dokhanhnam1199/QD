```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority score for each bin by combining Best Fit strategy and nuanced scoring.
    Prioritizes bins with just enough capacity and uses exponential normalization.
    """
    # Check if item can fit into the bin
    valid_bins = (bins_remain_cap >= item).astype(float)
    
    # Calculate priority using Best Fit strategy with exponential normalization
    priority = np.where(valid_bins > 0, np.exp(-(bins_remain_cap - item) / bins_remain_cap), -1)
    
    return priority
```
