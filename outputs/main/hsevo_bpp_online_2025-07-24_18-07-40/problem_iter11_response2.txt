import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Priority score calculation: best-fit strategy with bonus for nearly full bins.
    """
    # Calculate priority based on remaining capacity and a bonus for nearly full bins
    can_hold_item = bins_remain_cap >= item
    priority = np.where(can_hold_item, -bins_remain_cap + (bins_remain_cap - item) / bins_remain_cap, -np.inf)
    return priority
