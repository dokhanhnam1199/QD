```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit heuristic.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin. Higher scores indicate higher priority.
    """
    # Calculate space left if item is placed in each bin
    space_left = bins_remain_cap - item
    # Valid bins are those with non-negative space left
    # Priority is negative space left (so bins with less space left have higher priority)
    return np.where(
        bins_remain_cap >= item,
        -space_left,  # Equivalent to item - bins_remain_cap
        -np.inf
    )
```
