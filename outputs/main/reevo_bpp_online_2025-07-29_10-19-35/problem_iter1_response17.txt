```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores that favor bins with remaining capacity just above the item size."""
    # Mask bins that can fit the item
    can_fit = bins_remain_cap >= item
    # Calculate priority as negative of leftover space (Best Fit), encouraging minimal waste
    priority = np.where(can_fit, -(bins_remain_cap - item), -np.inf)
    return priority
```
