import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit heuristic.
    
    The priority for each bin is the negative of the remaining capacity after placing the item,
    which prioritizes bins that fit the item with minimal leftover space. Bins that cannot fit
    the item are assigned a priority of negative infinity.
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
    
    Returns:
        Array of priority scores for each bin.
    """
    priorities = np.full_like(bins_remain_cap, fill_value=-np.inf, dtype=np.float64)
    valid_mask = bins_remain_cap >= item
    if np.any(valid_mask):
        priorities[valid_mask] = -(bins_remain_cap[valid_mask] - item)
    return priorities
