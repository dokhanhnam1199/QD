import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function using a combination of best fit and exact fit prioritization."""
    valid_mask = bins_remain_cap >= item
    diff = bins_remain_cap - item
    
    # Base score: negative diff to prioritize smaller leftover space (best fit)
    base_score = np.where(valid_mask, -diff, -np.inf)
    
    # Bonus for exact fits: add a small epsilon to differentiate exact fits from near fits
    exact_fit_bonus = np.where(np.isclose(bins_remain_cap, item), 1e-6, 0.0)
    
    # Combine base score and exact fit bonus
    priority = base_score + exact_fit_bonus
    
    return priority
