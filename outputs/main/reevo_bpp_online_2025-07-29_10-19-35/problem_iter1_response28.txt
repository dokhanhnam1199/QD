```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function implementing Best Fit strategy for online Bin Packing Problem.
    
    Assigns higher priority to bins that can fit the item with minimal leftover space.
    Bins that cannot fit the item are assigned a very low priority (-inf).
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
    
    Returns:
        Array of priority scores for each bin.
    """
    # Create a mask for bins that can fit the item
    feasible_mask = bins_remain_cap >= item
    
    # Prioritize bins by minimizing leftover space: 
    # priority = -(leftover space) = item - remain_cap (for feasible bins)
    # Non-feasible bins get -inf to ensure they are not selected
    priorities = np.where(feasible_mask, item - bins_remain_cap, -np.inf)
    
    return priorities
```
