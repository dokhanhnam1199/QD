import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function implementing Best Fit strategy for online Bin Packing.
    
    Bins that can fit the item are prioritized by how tightly they fit it (smaller slack),
    with the highest priority given to bins that would be completely filled.
    """
    slack = bins_remain_cap - item
    valid = slack >= 0
    return np.where(valid, -slack, -np.inf)
