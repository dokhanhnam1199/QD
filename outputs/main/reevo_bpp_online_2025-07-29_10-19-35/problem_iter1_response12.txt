import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority using Best Fit strategy: higher priority for bins with least leftover space."""
    leftover = bins_remain_cap - item
    can_fit = leftover >= 0
    return np.where(can_fit, -leftover, -np.inf)
