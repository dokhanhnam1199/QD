import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit heuristic.

    The priority is highest for bins that can fit the item with the least
    remaining capacity after placement. Bins that cannot fit the item are
    assigned a very low priority.

    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.

    Returns:
        Array of priority scores for each bin.
    """
    can_fit = bins_remain_cap >= item
    # Prioritize bins with minimal leftover space (Best Fit)
    # item - remain_cap is equivalent to -leftover, so higher is better
    priorities = np.where(
        can_fit,
        item - bins_remain_cap,
        -np.inf
    )
    return priorities
