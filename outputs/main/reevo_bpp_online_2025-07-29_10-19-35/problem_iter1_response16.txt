import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function using Best Fit heuristic with a tie-breaker for larger remaining capacities."""
    space_left = bins_remain_cap - item
    valid = space_left >= 0
    # Best Fit priority: maximize -(space_left)
    # Tie-breaker: small preference for bins with larger remaining capacity
    scores = np.where(valid, -space_left + 0.01 * bins_remain_cap, -np.inf)
    return scores
