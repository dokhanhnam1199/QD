```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on a combination of residual space and bin utilization.

    Bins that can fit the item are prioritized by:
    1. Minimizing the remaining capacity after packing (residual space).
    2. Preferring bins that are already more filled (lower current remaining capacity).
    This is achieved by scoring valid bins as `-2 * remaining_cap + item`, which balances both factors.
    """
    valid = bins_remain_cap >= item
    scores = np.where(valid, -2 * bins_remain_cap + item, -np.inf)
    return scores
```
