import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    feasible = bins_remain_cap >= item
    # Threshold for large items (assuming bin capacity normalized to 1.0)
    threshold = 0.5  
    if item > threshold:
        # Prioritize bins with least remaining capacity (Best Fit for large items)
        return np.where(feasible, -bins_remain_cap, -np.inf)
    else:
        # Prioritize bins with most remaining capacity (Worst Fit for small items)
        return np.where(feasible, bins_remain_cap, -np.inf)
