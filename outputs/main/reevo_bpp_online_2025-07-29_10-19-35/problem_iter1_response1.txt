import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function based on Best Fit heuristic: prioritize bins that leave the least remaining space after packing the item.
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
    
    Returns:
        Array of priority scores for each bin.
    """
    feasible = bins_remain_cap >= item
    priority = np.full_like(bins_remain_cap, -np.inf, dtype=np.float64)
    # Assign higher priority (less negative) to bins with smaller residual space (item - remaining)
    priority[feasible] = item - bins_remain_cap[feasible]
    return priority
