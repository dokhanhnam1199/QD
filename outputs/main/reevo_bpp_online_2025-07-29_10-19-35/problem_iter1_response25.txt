import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority based on Best Fit heuristic: prioritize bins that can fit the item and have the smallest remaining capacity."""
    can_fit = bins_remain_cap >= item
    # Assign priority as negative remaining capacity for bins that can fit, -infinity otherwise
    priorities = np.where(can_fit, -bins_remain_cap, -np.inf)
    return priorities
