```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function based on minimizing leftover space with a tie-breaker for larger bins.
    
    Assigns priority scores to bins by calculating the negative of the leftover space after placing
    the item (favoring tighter fits). In case of ties, bins with slightly larger remaining capacity
    are prioritized to encourage consolidation.
    """
    slack = bins_remain_cap - item
    valid_mask = slack >= 0
    # Base priority: -slack (best fit), with a small tie-breaking term
    priority = np.where(valid_mask, -slack + 1e-7 * bins_remain_cap, -np.inf)
    return priority
```
