```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Bins that can fit the item get a higher priority
    can_fit = remaining_cap_after_addition >= 0
    
    # Prioritize bins that have just enough capacity for the item
    priorities = np.where(can_fit, -np.abs(remaining_cap_after_addition), -np.inf)
    
    # To avoid division by zero, replace zero remaining capacity with a small value
    bins_remain_cap = np.maximum(bins_remain_cap, 1e-9)
    
    # Also consider the ratio of item size to bin remaining capacity
    ratios = item / bins_remain_cap
    log_ratios = np.log(ratios)
    ratio_priorities = -log_ratios
    
    # Combine the two priorities
    priorities = np.where(can_fit, priorities + ratio_priorities, -np.inf)
    
    return priorities
```
