import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines exact fit priority with a scaled Best Fit strategy using normalized slack.
    Prioritizes exact fits, then bins with minimal normalized slack, ensuring clear hierarchy.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Exact Fit: Highest priority (score 1.0)
    exact_fit_mask = np.isclose(bins_remain_cap, item)
    priorities[exact_fit_mask] = 1.0

    # Best Fit for non-exact fits: Prioritize bins with minimal normalized slack.
    # Consider bins that can fit the item and are not exact fits.
    can_fit_mask = (bins_remain_cap >= item) & ~exact_fit_mask
    fit_indices = np.where(can_fit_mask)[0]

    if len(fit_indices) > 0:
        remaining_after_fit = bins_remain_cap[fit_indices] - item
        current_capacities = bins_remain_cap[fit_indices]

        # Calculate normalized slack: (remaining_capacity_after_fit) / (current_bin_capacity)
        # Lower normalized slack is better. Add epsilon for numerical stability.
        normalized_slack = remaining_after_fit / (current_capacities + 1e-9)

        # Assign priorities: scores are higher for smaller normalized slack.
        # Map (1.0 - normalized_slack) to a range clearly below 1.0, e.g., [0.5, 0.99].
        # This ensures exact fits (1.0) are always preferred.
        best_fit_scores = 1.0 - normalized_slack
        # Scale scores to be in the range [0.5, 0.99]
        scaled_best_fit_priorities = 0.5 + best_fit_scores * 0.49 

        priorities[fit_indices] = scaled_best_fit_priorities

    return priorities
