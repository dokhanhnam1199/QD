import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Prioritizes exact fits, then bins with minimal normalized slack.
    Uses a stable inverse relationship for non-exact fits.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    suitable_bins_mask = bins_remain_cap >= item
    
    if not np.any(suitable_bins_mask):
        return priorities

    suitable_bins_remain_cap = bins_remain_cap[suitable_bins_mask]
    
    # Calculate slack for suitable bins
    slack = suitable_bins_remain_cap - item

    # Prioritize exact fits with a score of 1.0
    exact_fit_mask_local = slack == 0
    priorities[suitable_bins_mask][exact_fit_mask_local] = 1.0

    # For non-exact fits, use inverse of slack + 1 to give higher priority to smaller slack
    # This is similar to priority_v0 but ensures exact fits are handled separately and more favorably.
    non_exact_fit_mask_local = slack > 0
    if np.any(non_exact_fit_mask_local):
        # Adding 1.0 to the slack ensures that bins with zero slack (already handled)
        # would get a score of 1.0 if they were in this calculation.
        # Adding a small epsilon prevents division by zero for floating point inaccuracies.
        priorities[suitable_bins_mask][non_exact_fit_mask_local] = 1.0 / (slack[non_exact_fit_mask_local] + 1.0 + 1e-9)

    # Normalize the priorities of non-exact fits so the highest non-exact fit priority is capped below 1.0
    # This ensures exact fits remain distinctly prioritized.
    if np.any(non_exact_fit_mask_local):
        max_non_exact_priority = np.max(priorities[suitable_bins_mask][non_exact_fit_mask_local])
        if max_non_exact_priority > 0:
            priorities[suitable_bins_mask][non_exact_fit_mask_local] /= max_non_exact_priority
            # Scale down non-exact fits to be clearly less than exact fits (if any exact fits exist)
            if np.any(exact_fit_mask_local):
                priorities[suitable_bins_mask][non_exact_fit_mask_local] *= 0.9 
            # Ensure exact fits are still 1.0
            priorities[suitable_bins_mask][exact_fit_mask_local] = 1.0


    return priorities
