import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    priorities = np.zeros_like(bins_remain_cap)
    for i, bin_cap in enumerate(bins_remain_cap):
        if bin_cap >= item:
            priorities[i] = 1.0 / (bin_cap - item + 1e-9)
    return priorities
