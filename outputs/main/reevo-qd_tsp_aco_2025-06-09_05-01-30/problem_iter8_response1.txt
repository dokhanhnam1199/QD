import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristic function for TSP using a combination of factors, sparsification,
    and normalization based on global context.

    Args:
        distance_matrix (np.ndarray): A 2D numpy array representing the distance matrix.

    Returns:
        np.ndarray: A 2D numpy array of the same shape as distance_matrix,
                    representing the heuristic values (prior indicators) for each edge.
    """

    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros_like(distance_matrix, dtype=float)

    # Calculate nearest neighbors for each node
    nearest_neighbors = np.argsort(distance_matrix, axis=1)

    # Calculate average distance from each node to all other nodes
    avg_distances = np.mean(distance_matrix, axis=1)

    for i in range(n):
        for j in range(n):
            if i != j:
                distance = distance_matrix[i, j]

                # Heuristic factor 1: Inverse distance (node proximity)
                heuristic_factor_1 = 1 / (distance + 1e-9)

                # Heuristic factor 2: Inverse rank of nearest neighbors (neighbor rank)
                nn_rank = np.where(nearest_neighbors[i] == j)[0][0] + 1
                heuristic_factor_2 = 1 / nn_rank

                # Heuristic factor 3: Relative distance compared to average (global distance context)
                heuristic_factor_3 = (avg_distances[i] + avg_distances[j]) / (2 * (distance + 1e-9))

                # Combine the factors
                heuristic_matrix[i, j] = heuristic_factor_1 * heuristic_factor_2 * heuristic_factor_3

    # Sparsify the matrix: Keep only k nearest neighbors for each node
    k = int(np.sqrt(n))  # Keep sqrt(n) nearest edges to each node
    for i in range(n):
        row = heuristic_matrix[i, :]
        indices = np.argsort(row)[::-1]  # Indices of the highest values in descending order
        keep_indices = indices[:k]

        mask = np.zeros(n, dtype=bool)
        mask[keep_indices] = True
        heuristic_matrix[i, ~mask] = 0

    # Normalize the heuristic values based on the maximum heuristic value in the entire matrix
    max_heuristic = np.max(heuristic_matrix)
    if max_heuristic > 0:
        heuristic_matrix = heuristic_matrix / max_heuristic
    else:
        # If all heuristic values are zero, assign a small uniform value for exploration
        heuristic_matrix = np.ones_like(heuristic_matrix) * 1e-9

    return heuristic_matrix
