import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristic function for TSP using a combination of normalized inverse rank and distance,
    with a sharpened nearest neighbor decay. Sparsifies the matrix for efficient sampling.

    Args:
        distance_matrix (np.ndarray): A 2D numpy array representing the distance matrix.

    Returns:
        np.ndarray: A 2D numpy array of the same shape as distance_matrix,
                    representing the heuristic values (prior indicators) for each edge.
    """

    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros_like(distance_matrix, dtype=float)

    # Calculate nearest neighbors for each node
    nearest_neighbors = np.argsort(distance_matrix, axis=1)

    for i in range(n):
        for j in range(n):
            if i != j:
                distance = distance_matrix[i, j]

                # Heuristic factor 1: Inverse distance, normalized. Avoid division by zero
                heuristic_factor_1 = 1 / (distance + 1e-9)

                # Heuristic factor 2: Sharpened nearest neighbor decay.
                nn_rank = np.where(nearest_neighbors[i] == j)[0][0] + 1  # Rank from 1 to N-1
                heuristic_factor_2 = np.exp(-nn_rank)  # Sharper decay

                # Combine factors (product)
                heuristic_matrix[i, j] = heuristic_factor_1 * heuristic_factor_2

    # Sparsify the matrix: keep only top K edges for each node
    k = min(int(np.sqrt(n)), n-1)  # Keep sqrt(n) nearest edges to each node, or up to n-1
    for i in range(n):
        row = heuristic_matrix[i, :]
        indices = np.argsort(row)[::-1]  # Indices of the highest values in descending order
        keep_indices = indices[:k]

        mask = np.zeros(n, dtype=bool)
        mask[keep_indices] = True
        heuristic_matrix[i, ~mask] = 0  # Set unpromising edges to zero

    # Normalize the heuristic values
    max_heuristic = np.max(heuristic_matrix)
    if max_heuristic > 0:
        heuristic_matrix = heuristic_matrix / max_heuristic
    else:
        heuristic_matrix = np.ones_like(heuristic_matrix) * 1e-9 # Small uniform values if all zero

    return heuristic_matrix
