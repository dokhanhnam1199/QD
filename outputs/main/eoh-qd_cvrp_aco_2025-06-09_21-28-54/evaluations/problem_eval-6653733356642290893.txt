import numpy as np

def heuristics_v2(distance_matrix, coordinates, demands, capacity):
    """
    {This algorithm samples a set of random routes, evaluates them based on a combination of distance and capacity constraints, and constructs a heuristic matrix that favors shorter edges and penalizes edges used in routes exceeding capacity.}
    """
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix)
    num_samples = 1000
    distance_weight = 0.5
    capacity_penalty = 100

    for _ in range(num_samples):
        # Generate a random permutation of customers (excluding depot)
        customer_indices = np.random.permutation(np.arange(1, n))
        
        # Create routes based on capacity constraints
        routes = []
        current_route = [0]
        current_capacity = 0
        
        for customer in customer_indices:
            if current_capacity + demands[customer] <= capacity:
                current_route.append(customer)
                current_capacity += demands[customer]
            else:
                current_route.append(0)
                routes.append(current_route)
                current_route = [0, customer]
                current_capacity = demands[customer]

        current_route.append(0)
        routes.append(current_route)
        
        # Update the heuristic matrix with edges present in the route
        route_capacity_exceeded = False
        for route in routes:
            route_capacity = 0
            for node in route[1:-1]:
                route_capacity += demands[node]
            if route_capacity > capacity:
                route_capacity_exceeded = True
                break
        
        for route in routes:
            for i in range(len(route) - 1):
                node1 = route[i]
                node2 = route[i+1]
                edge_distance = distance_matrix[node1, node2]
                
                if route_capacity_exceeded:
                    heuristics_matrix[node1, node2] -= capacity_penalty
                    heuristics_matrix[node2, node1] -= capacity_penalty
                else:
                    heuristics_matrix[node1, node2] += (1 - distance_weight) + distance_weight / (1 + edge_distance)
                    heuristics_matrix[node2, node1] += (1 - distance_weight) + distance_weight / (1 + edge_distance)
                
    return heuristics_matrix
