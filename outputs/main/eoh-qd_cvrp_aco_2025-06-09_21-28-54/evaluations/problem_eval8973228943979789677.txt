import numpy as np

def heuristics_v2(distance_matrix, coordinates, demands, capacity):
    """{This algorithm samples random feasible routes, evaluates their quality based on distance and capacity constraints, and uses a reinforcement learning approach to update a heuristic matrix, favoring edges appearing in better solutions and penalizing those in poor solutions.}"""

    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros((n, n))
    num_samples = 1000  # Number of random samples to generate
    learning_rate = 0.1

    for _ in range(num_samples):
        # Generate a random permutation of customers (excluding the depot)
        customer_indices = np.arange(1, n)
        np.random.shuffle(customer_indices)

        routes = []
        current_route = [0]  # Start at the depot
        current_demand = 0

        for customer in customer_indices:
            if current_demand + demands[customer] <= capacity:
                current_route.append(customer)
                current_demand += demands[customer]
            else:
                current_route.append(0)  # Return to depot
                routes.append(current_route)
                current_route = [0, customer]  # Start a new route
                current_demand = demands[customer]

        current_route.append(0)  # Return to depot
        routes.append(current_route)

        # Calculate the total distance of the routes
        total_distance = 0
        for route in routes:
            for i in range(len(route) - 1):
                total_distance += distance_matrix[route[i], route[i + 1]]

        # Normalize the total distance to a range between 0 and 1
        normalized_distance = 1 / (1 + total_distance)
        
        # Update the heuristics matrix based on edge usage and total distance
        # Reinforcement Learning: Reward good edges, penalize bad edges
        for route in routes:
            for i in range(len(route) - 1):
                node1 = route[i]
                node2 = route[i + 1]
                
                # Update based on normalized distance (higher is better)
                heuristics_matrix[node1, node2] += learning_rate * normalized_distance
                heuristics_matrix[node2, node1] += learning_rate * normalized_distance # Ensure symmetry
                
                # Optional: Penalize edges in longer routes by subtracting
                # heuristics_matrix[node1, node2] -= learning_rate * (1-normalized_distance)
                # heuristics_matrix[node2, node1] -= learning_rate * (1-normalized_distance)


    return heuristics_matrix
