import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, small_space_threshold: float = 0.1, small_space_penalty: float = 10, almost_full_threshold: float = 0.9, almost_full_reward: float = 10) -> np.ndarray:
    """Calculate bin priorities considering feasibility, best fit, small space avoidance, and almost full utilization."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # 1. Feasibility: Eliminate infeasible bins.
    infeasible_mask = item > bins_remain_cap
    priorities[infeasible_mask] = -np.inf

    # 2. Best Fit: Prioritize bins with minimal remaining space.
    feasible_mask = ~infeasible_mask
    remaining_space = bins_remain_cap[feasible_mask] - item
    priorities[feasible_mask] = -remaining_space

    # 3. Small Space Avoidance: Penalize bins creating tiny spaces.
    very_small_space_mask = (remaining_space > 0) & (remaining_space < small_space_threshold)
    priorities[feasible_mask][very_small_space_mask] -= small_space_penalty

    # 4. Almost Full Reward: Reward bins approaching full capacity.
    almost_full_mask = (bins_remain_cap[feasible_mask] - item) / bins_remain_cap[feasible_mask] >= almost_full_threshold
    priorities[feasible_mask][almost_full_mask] += almost_full_reward

    return priorities
