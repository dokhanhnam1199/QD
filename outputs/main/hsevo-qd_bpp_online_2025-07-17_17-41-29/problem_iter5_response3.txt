```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Assigns priority to bins based on feasibility, best fit, and a fill ratio."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Infeasible bins get -inf priority
    infeasible_mask = item > bins_remain_cap
    priorities[infeasible_mask] = -np.inf

    # Calculate remaining space and prioritize smaller remaining space (Best Fit)
    feasible_mask = ~infeasible_mask
    remaining_space = bins_remain_cap[feasible_mask] - item
    priorities[feasible_mask] = -remaining_space

    #Prioritize fuller bins (ratio of item size to bin capacity)
    fill_ratio = item / bins_remain_cap[feasible_mask]
    priorities[feasible_mask] += fill_ratio

    return priorities
```
