def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    priorities = np.zeros_like(bins_remain_cap)
    fullest_bins_mask = bins_remain_cap >= (np.max(bins_remain_cap) - item)
    priorities[fullest_bins_mask] = 1
    priorities[bins_remain_cap >= item] += 0.5
    return priorities
