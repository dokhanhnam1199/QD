import numpy as np

def heuristics_v2(distance_matrix: np.ndarray, coordinates: np.ndarray, demands: np.ndarray, capacity: int) -> np.ndarray:
    """
    A heuristics function for Solving Capacitated Vehicle Routing Problem (CVRP) via stochastic solution sampling.
    Improved version with refined heuristics, sparsification, and learned components.

    Args:
        distance_matrix (np.ndarray): Distance matrix (shape: n by n).
        coordinates (np.ndarray): Euclidean coordinates of nodes (shape: n by 2).
        demands (np.ndarray): Customer demands (shape: n).
        capacity (int): Vehicle capacity.

    Returns:
        np.ndarray: Prior indicators of how promising it is to include each edge in a solution (same shape as distance_matrix).
    """
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((n, n))

    # Calculate savings for each pair of nodes (Clarke-Wright Savings Heuristic)
    savings_matrix = np.zeros((n, n))
    for i in range(1, n):
        for j in range(i + 1, n):
            savings_matrix[i, j] = distance_matrix[0, i] + distance_matrix[0, j] - distance_matrix[i, j]
            savings_matrix[j, i] = savings_matrix[i, j]

    # Demand Pairing Preference: Incentivize pairings whose sum is close to capacity, but under.
    demand_pairing_preference = np.zeros((n, n))
    for i in range(1, n):
      for j in range(1,n):
        if i != j:
          combined_demand = demands[i] + demands[j]
          if combined_demand <= capacity:
            demand_pairing_preference[i, j] = np.exp(-abs(capacity - combined_demand) / capacity)
          else:
            demand_pairing_preference[i, j] = 0.0

    # Combine distance, savings, and demand considerations
    for i in range(n):
        for j in range(n):
            if i == j:
                heuristic_matrix[i, j] = 0  # No self-loops
            else:
                # Distance factor (shorter distances are preferred)
                distance_factor = np.exp(-distance_matrix[i, j] / np.mean(distance_matrix))

                # Savings factor (higher savings are preferred)
                savings_factor = savings_matrix[i, j] / (np.max(savings_matrix) + 1e-6) # Normalizing

                # Demand factor (prioritize connections that help balance vehicle load).
                demand_penalty = 0.0
                if i != 0 and j != 0:
                    combined_demand = demands[i] + demands[j]
                    if combined_demand > capacity:
                        demand_penalty = 1.0  # Heavy penalty if exceeds
                    else:
                        demand_penalty = (combined_demand / capacity)**2 # Moderate penalty

                demand_factor = 1 - demand_penalty # Modified Demand factor

                # Combine the factors
                heuristic_matrix[i, j] = distance_factor * savings_factor * demand_factor  # Multiplicative Combination
                heuristic_matrix[i, j] *= demand_pairing_preference[i,j]
                # Adding proximity to depot
                heuristic_matrix[i, j] *= (1 / (distance_matrix[0,i] + distance_matrix[0,j] + 1e-6))**0.5 # Prefer nodes closer to depot

    # Adaptive sparsification: adjust k based on problem size
    k = int(np.sqrt(n))  # Adjust k as needed; square root scaling
    # Ensure k is at least 1 and less than n
    k = max(1, min(k, n - 1))

    for i in range(n):
        # Get the indices of the top k promising edges for node i
        top_k_indices = np.argsort(heuristic_matrix[i, :])[::-1][:k]

        # Set the heuristic value to 0 for all other edges
        for j in range(n):
            if j not in top_k_indices:
                heuristic_matrix[i, j] = 0

    # Learned Component (more sophisticated: boost edges based on node centrality and demand)
    node_centrality = np.sum(1 / (distance_matrix + 1e-6), axis=1)  # Sum of inverse distances to each node

    demand_boost = np.zeros((n, n))
    for i in range(1, n):
        for j in range(1, n):
            demand_boost[i, j] = (demands[i] * demands[j]) / (capacity**2) * (node_centrality[i] + node_centrality[j])  # Centrality-weighted demand

    heuristic_matrix = heuristic_matrix + 0.05 * demand_boost  # Add a small proportion of the demand boost

    # Final touch: Emphasize edges connected to the depot, but less aggressively
    for i in range(1, n):
        heuristic_matrix[0, i] *= 1.1
        heuristic_matrix[i, 0] *= 1.1

    # Introduce a small amount of noise to encourage exploration
    noise = np.random.normal(0, 0.01 * np.mean(heuristic_matrix), size=(n, n))
    heuristic_matrix = heuristic_matrix + noise

    return heuristic_matrix
