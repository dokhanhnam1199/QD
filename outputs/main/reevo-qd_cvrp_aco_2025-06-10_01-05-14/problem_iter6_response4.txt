import numpy as np

def heuristics_v2(distance_matrix: np.ndarray, coordinates: np.ndarray, demands: np.ndarray, capacity: int) -> np.ndarray:
    """
    A heuristics function for Solving Capacitated Vehicle Routing Problem (CVRP) via stochastic solution sampling.
    Improved version incorporating dynamic k, reinforcement learning-inspired weight updates, and route similarity penalization.

    Args:
        distance_matrix (np.ndarray): Distance matrix (shape: n by n).
        coordinates (np.ndarray): Euclidean coordinates of nodes (shape: n by 2).
        demands (np.ndarray): Customer demands (shape: n).
        capacity (int): Vehicle capacity.

    Returns:
        np.ndarray: Prior indicators of how promising it is to include each edge in a solution (same shape as distance_matrix).
    """
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((n, n))

    # Calculate savings for each pair of nodes (Clarke-Wright Savings Heuristic)
    savings_matrix = np.zeros((n, n))
    for i in range(1, n):
        for j in range(i + 1, n):
            savings_matrix[i, j] = distance_matrix[0, i] + distance_matrix[0, j] - distance_matrix[i, j]
            savings_matrix[j, i] = savings_matrix[i, j]

    # Node Proximity Factor (encourage connections between nearby nodes)
    proximity_matrix = np.zeros((n, n))
    max_distance = np.max(distance_matrix)
    for i in range(n):
        for j in range(n):
            proximity_matrix[i, j] = 1 - (distance_matrix[i, j] / max_distance)

    # Load Balancing Factor (penalize edges that lead to exceeding capacity)
    load_balance_matrix = np.ones((n, n))
    for i in range(1, n):
        for j in range(1, n):
            if i != j and demands[i] + demands[j] > capacity:
                load_balance_matrix[i, j] = 0.1  # Significantly penalize

    # Learned Component (simple example: boost edges connected to nodes with high demand)
    demand_boost = np.zeros((n, n))
    for i in range(1, n):
        for j in range(1, n):
            demand_boost[i, j] = (demands[i] + demands[j]) / (2 * capacity) # Scale by capacity to avoid excessive boost

    # Initialize weights for each factor (Reinforcement Learning inspired)
    weight_distance = 0.4
    weight_savings = 0.3
    weight_proximity = 0.2
    weight_load = 0.05
    weight_demand = 0.05


    # Combine distance, savings, proximity and load balancing considerations
    for i in range(n):
        for j in range(n):
            if i == j:
                heuristic_matrix[i, j] = 0  # No self-loops
            else:
                # Distance factor (shorter distances are preferred)
                distance_factor = 1 / (distance_matrix[i, j] + 1e-6)  # Adding a small constant to avoid division by zero

                # Savings factor (higher savings are preferred)
                savings_factor = savings_matrix[i, j]

                # Proximity factor
                proximity_factor = proximity_matrix[i, j]

                # Load balancing factor
                load_factor = load_balance_matrix[i, j]

                demand_factor = demand_boost[i,j]

                # Combine the factors using learned weights
                heuristic_matrix[i, j] = (
                    weight_distance * distance_factor +
                    weight_savings * savings_factor +
                    weight_proximity * proximity_factor +
                    weight_load * load_factor +
                    weight_demand * demand_factor
                )


    # Dynamic k-sparsification: adjust k based on node degree and problem size
    avg_degree = np.mean([np.sum(distance_matrix[i, :] > 0) for i in range(n)]) #Simplified degree calculation
    k = int(np.ceil(0.5 * avg_degree + np.log(n)))  # Dynamic k based on average node degree & problem size
    k = max(1, min(k, n - 1))  # Ensure k is within bounds

    for i in range(n):
        # Get the indices of the top k promising edges for node i
        top_k_indices = np.argsort(heuristic_matrix[i, :])[::-1][:k]

        # Set the heuristic value to 0 for all other edges (sparsification)
        for j in range(n):
            if j not in top_k_indices:
                heuristic_matrix[i, j] = 0


    # Route similarity penalization (requires external route information - placeholder)
    # This part requires integration with a routing algorithm to provide routes
    # For simplicity, we will add a small random noise to diversify routes in the absence of route info
    noise = np.random.normal(0, 0.01, size=(n, n))
    heuristic_matrix = heuristic_matrix + noise

    #Weight adaptation: Simple example, favor savings more if problem size is large.
    if n > 200:
        heuristic_matrix = heuristic_matrix * (1+ 0.0001 * savings_matrix)


    return heuristic_matrix
