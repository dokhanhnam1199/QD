import numpy as np

def heuristics_v2(distance_matrix: np.ndarray, coordinates: np.ndarray, demands: np.ndarray, capacity: int) -> np.ndarray:
    """
    A heuristics function for Solving Capacitated Vehicle Routing Problem (CVRP) via stochastic solution sampling.

    Args:
        distance_matrix (np.ndarray): Distance matrix (shape: n by n).
        coordinates (np.ndarray): Euclidean coordinates of nodes (shape: n by 2).
        demands (np.ndarray): Customer demands (shape: n).
        capacity (int): Vehicle capacity.

    Returns:
        np.ndarray: Prior indicators of how promising it is to include each edge in a solution (same shape as distance_matrix).
    """
    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((n, n))

    # Calculate savings for each pair of nodes (Clarke-Wright Savings Heuristic)
    savings_matrix = np.zeros((n, n))
    for i in range(1, n):
        for j in range(i + 1, n):
            savings_matrix[i, j] = distance_matrix[0, i] + distance_matrix[0, j] - distance_matrix[i, j]
            savings_matrix[j, i] = savings_matrix[i, j]

    # Combine distance, savings, and demand considerations
    for i in range(n):
        for j in range(n):
            if i == j:
                heuristic_matrix[i, j] = 0  # No self-loops
            else:
                # Distance factor (shorter distances are preferred) - normalized
                distance_factor = 1 / (distance_matrix[i, j] + 1e-6)
                distance_factor = distance_factor / np.max(1 / (distance_matrix + 1e-6))


                # Savings factor (higher savings are preferred) - normalized
                savings_factor = savings_matrix[i, j]
                savings_factor = savings_factor / np.max(savings_matrix)

                # Demand factor (prioritize connections that help balance vehicle load).
                demand_factor = 1.0
                if (i != 0 and j != 0):
                    if (demands[i] + demands[j] > capacity):
                        demand_factor = 0.0
                    else:
                        # soft penalty based on how close we are to capacity
                        demand_factor = max(0.1, 1 - (demands[i] + demands[j]) / capacity)

                # Depot proximity boost
                depot_proximity = (1 / (distance_matrix[i, 0] + 1e-6) + 1 / (distance_matrix[j, 0] + 1e-6)) if (i !=0 and j!=0) else 0
                depot_proximity = depot_proximity / np.max([1 / (distance_matrix[k, 0] + 1e-6) for k in range(1,n)]) if n > 1 else 0

                # Combine the factors
                heuristic_matrix[i, j] = distance_factor * savings_factor * demand_factor * (1 + 0.2 * depot_proximity)

    # Adaptive sparsification: adjust k based on problem size
    k = int(2 * np.log(n))  # Adjust k as needed; logarithmic scaling seems appropriate
    k = max(1, min(k, n - 1))

    for i in range(n):
        top_k_indices = np.argsort(heuristic_matrix[i, :])[::-1][:k]

        for j in range(n):
            if j not in top_k_indices:
                heuristic_matrix[i, j] = 0

    # Learned Component: boost edges connected to nodes with high demand, considering depot - Removed total demand calculation, simplified
    demand_boost = np.zeros((n, n))

    for i in range(1, n):
        for j in range(1, n):
            # Prioritize connecting high demand nodes to the depot
            depot_proximity = (1 / (distance_matrix[i, 0] + 1e-6) + 1 / (distance_matrix[j, 0] + 1e-6)) if (i !=j) else 0

            # Boost edges connecting high-demand nodes to each other or to the depot
            demand_component = (demands[i] + demands[j]) / (2 * capacity)  # Scale demand relative to capacity
            demand_boost[i, j] = demand_component * (1 + 0.1 * depot_proximity)

    # Scale the demand boost to avoid dominating other heuristics
    demand_boost_scale = 0.10
    heuristic_matrix = heuristic_matrix + demand_boost_scale * demand_boost

    # Introducing Capacity-Aware Edge Penalization - more aggressive
    for i in range(1, n):
        for j in range(1, n):
            if i != j:
                combined_demand = demands[i] + demands[j]
                if combined_demand > capacity:
                    penalty_factor = (combined_demand - capacity) / capacity
                    heuristic_matrix[i, j] *= max(0.05, 1 - 0.5 * penalty_factor)  # Reduce the heuristic value more aggressively
                    heuristic_matrix[j, i] *= max(0.05, 1 - 0.5 * penalty_factor)

    return heuristic_matrix
