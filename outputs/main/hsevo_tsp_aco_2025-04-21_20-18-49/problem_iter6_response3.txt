import numpy as np

def heuristics_v2(distance_matrix: np.ndarray,
                  attractiveness_exponent: float = 2.0,
                  sparsification_factor: float = 4.733362140013519,
                  centrality_exponent: float = 0.5,
                  temperature_multiplier: float = 1.0) -> np.ndarray:
    """Enhanced heuristics using inverse distance, centrality, attractiveness, and adaptive temperature."""
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    temperature = temperature_multiplier * np.median(distance_matrix)
    inverse_distance = 1.0 / (distance_matrix + np.eye(n))
    
    # Node attractiveness based on inverse distance
    node_attractiveness = np.sum(inverse_distance, axis=0)
    node_attractiveness = node_attractiveness / np.mean(node_attractiveness)  # Normalized attractiveness

    # Node centrality (degree centrality based on inverse distance)
    node_centrality = np.sum(inverse_distance, axis=1)
    node_centrality = node_centrality / np.mean(node_centrality)  # Normalized centrality
    
    # Edge-specific factors (combining inverse distance, attractiveness, and centrality)
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristics[i, j] = (inverse_distance[i, j]**attractiveness_exponent) * \
                                   (node_attractiveness[i] * node_attractiveness[j]) * \
                                   (node_centrality[i]**centrality_exponent * node_centrality[j]**centrality_exponent) * \
                                   np.exp(-distance_matrix[i, j] / temperature)  # Temperature-controlled exploration

    # Adaptive sparsification based on mean heuristic value
    threshold = np.mean(heuristics) / sparsification_factor
    heuristics[heuristics < threshold] = 0.0  # Sparsify by setting low-promise edges to zero

    return heuristics
