import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    TSP heuristic: Combines inverse distance, node attractiveness,
    dynamic temperature scaling, and adaptive sparsification.
    """
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix, dtype=float)
    temperature = np.median(distance_matrix)

    inverse_distance = 1.0 / (distance_matrix + np.eye(n))
    node_attractiveness = np.sum(inverse_distance, axis=0)
    node_attractiveness = 1.0 / (node_attractiveness / np.mean(node_attractiveness))

    for i in range(n):
        for j in range(n):
            if i != j:
                heuristics[i, j] = (inverse_distance[i, j]**2) * \
                                   (node_attractiveness[i] * node_attractiveness[j]) * \
                                    np.exp(-distance_matrix[i, j] / temperature)
    # Normalize heuristic values
    heuristics = heuristics / np.max(heuristics)
    # Adaptive threshold for sparsification
    threshold = np.mean(heuristics) / 5  # dynamic based on the mean
    heuristics[heuristics < threshold] = 0

    return heuristics
