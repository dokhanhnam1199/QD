import numpy as np

def heuristics_v2(distance_matrix: np.ndarray,
                  attractiveness_exponent: float = 2.0,
                  centrality_exponent: float = 1.0,
                  sparsification_factor: float = 4.733362140013519,
                  temperature_coefficient: float = 1.0) -> np.ndarray:
    """Enhanced heuristics using inverse distance, centrality, attractiveness, adaptive temperature, and dynamic range scaling."""
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix, dtype=float)

    # Adaptive temperature based on distance distribution
    temperature = np.median(distance_matrix) * temperature_coefficient

    # Inverse distance calculation with self-loop avoidance
    inverse_distance = 1.0 / (distance_matrix + np.eye(n))

    # Node attractiveness based on inverse distance sum
    node_attractiveness = np.sum(inverse_distance, axis=0)
    node_attractiveness = node_attractiveness / np.mean(node_attractiveness)

    # Node centrality based on inverse distance sum (normalized)
    node_centrality = np.sum(inverse_distance, axis=1)
    node_centrality = node_centrality / np.mean(node_centrality)

    # Edge attractiveness calculation, incorporating all factors
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristics[i, j] = (inverse_distance[i, j]**attractiveness_exponent) * \
                                   (node_attractiveness[i] * node_attractiveness[j]) * \
                                   (node_centrality[i]**centrality_exponent + node_centrality[j]**centrality_exponent) * \
                                   np.exp(-distance_matrix[i, j] / temperature)

    # Dynamic range scaling and sparsification
    max_heuristic = np.max(heuristics)
    if max_heuristic > 0: # avoid division by zero
        heuristics = heuristics / max_heuristic  # Scale to [0, 1]
    threshold = np.mean(heuristics) / sparsification_factor
    heuristics[heuristics < threshold] = 0

    return heuristics
