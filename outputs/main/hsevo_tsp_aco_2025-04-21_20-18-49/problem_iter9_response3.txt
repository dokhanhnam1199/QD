import numpy as np

def heuristics_v2(distance_matrix: np.ndarray,
                  attractiveness_exponent: float = 2.0,
                  sparsification_factor: float = 4.733362140013519,
                  centrality_exponent: float = 0.5,
                  temperature_factor: float = 1.0) -> np.ndarray:
    """Enhanced heuristics using inverse distance, centrality, attractiveness, and adaptive temperature."""
    n = distance_matrix.shape[0]
    heuristics = np.zeros_like(distance_matrix, dtype=float)

    # Adaptive temperature based on the range of distances
    temperature = np.median(distance_matrix) * temperature_factor

    # Inverse distance with a small offset to avoid division by zero
    inverse_distance = 1.0 / (distance_matrix + np.eye(n) * 1e-6)

    # Node attractiveness based on total inverse distance
    node_attractiveness = np.sum(inverse_distance, axis=0)
    node_attractiveness = node_attractiveness / np.mean(node_attractiveness)

    # Node centrality based on eigenvector centrality
    # Normalize distance matrix for stability
    normalized_distance = distance_matrix / np.max(distance_matrix)
    eigenvalues, eigenvectors = np.linalg.eig(normalized_distance)
    node_centrality = np.abs(eigenvectors[:, np.argmax(eigenvalues)])
    node_centrality = node_centrality / np.mean(node_centrality)


    # Combine factors
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristics[i, j] = (inverse_distance[i, j]**attractiveness_exponent) * \
                                   (node_attractiveness[i] * node_attractiveness[j]) * \
                                   (node_centrality[i]**centrality_exponent * node_centrality[j]**centrality_exponent) * \
                                   np.exp(-distance_matrix[i, j] / temperature)

    # Adaptive sparsification based on the distribution of heuristic values
    threshold = np.mean(heuristics) / sparsification_factor
    heuristics[heuristics < threshold] = 0

    return heuristics
