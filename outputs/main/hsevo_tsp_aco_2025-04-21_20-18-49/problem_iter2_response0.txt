```python
import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Combines inverse distance, gravitational attraction, and randomness.
    """
    n = distance_matrix.shape[0]
    epsilon = 1e-9

    inverse_distance = 1 / (distance_matrix + epsilon)
    gravitational_attraction = 1 / ((distance_matrix**2) + epsilon)
    randomness = np.random.rand(n, n) * (1.0 / n)
    randomness = (randomness + randomness.T) / 2

    heuristic_matrix = (
        0.5 * inverse_distance +
        0.3 * gravitational_attraction +
        0.2 * randomness
    )

    np.fill_diagonal(heuristic_matrix, 0)
    return heuristic_matrix
```
