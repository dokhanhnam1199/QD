import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristics for TSP using sparsification, degree centrality,
    and controlled randomness, with refined parameter tuning and normalization.
    """
    n = distance_matrix.shape[0]

    # Initialize heuristic matrix with inverse distance
    heuristic_matrix = 1 / (distance_matrix + 1e-9)

    # Node degree centrality heuristic
    degree_centrality = np.sum(1 / (distance_matrix + 1e-9), axis=1)
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristic_matrix[i, j] *= (degree_centrality[i] + degree_centrality[j]) / 2

    # Adaptive sparsification with dynamic threshold
    for i in range(n):
        # Exclude the node itself when finding nearest neighbors
        distances = distance_matrix[i, :]
        nearest_neighbors_indices = np.argsort(distances)[1:min(6, n)]  # Top 5 nearest neighbors, excluding self, handle small n
        avg_neighbor_distance = np.mean(distances[nearest_neighbors_indices]) if len(nearest_neighbors_indices) > 0 else np.mean(distances)
        dynamic_threshold = 1.3 * avg_neighbor_distance  # Tuned multiplier - decreased to encourage exploration

        for j in range(n):
            if i != j and distance_matrix[i, j] > dynamic_threshold:
                heuristic_matrix[i, j] *= 0.1  # Increased penalty - focus more on near neighbors, and more severely

    # Incorporate randomness
    randomness_factor = 0.01  # Further reduced randomness
    random_matrix = np.random.rand(n, n) * randomness_factor
    heuristic_matrix += random_matrix

    # Enhanced normalization and scaling
    heuristic_matrix = heuristic_matrix / (distance_matrix + 1e-9)**0.3  # Adjusted exponent - decreased

    # Row-wise normalization to prioritize local choices after sparsification
    for i in range(n):
        row_sum = np.sum(heuristic_matrix[i, :])
        if row_sum > 0:
            heuristic_matrix[i, :] /= row_sum

    max_val = np.max(heuristic_matrix)
    min_val = np.min(heuristic_matrix)

    if max_val > min_val:
        heuristic_matrix = (heuristic_matrix - min_val) / (max_val - min_val)  # Proper normalization
    else:
        heuristic_matrix = np.ones_like(heuristic_matrix) * 0.5

    return heuristic_matrix
