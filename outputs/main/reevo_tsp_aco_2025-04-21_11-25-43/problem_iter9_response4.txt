import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristics for TSP based on distance, degree centrality, and adaptive sparsification.
    """
    n = distance_matrix.shape[0]

    # Distance-squared scaling
    heuristic_matrix = 1 / (distance_matrix**2 + 1e-9)

    # Degree centrality heuristic (geometric mean)
    degree_centrality = np.sum(1 / (distance_matrix + 1e-9), axis=1)
    geo_mean_centrality = np.zeros((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                geo_mean_centrality[i, j] = np.sqrt(degree_centrality[i] * degree_centrality[j])
    heuristic_matrix *= geo_mean_centrality

    # Adaptive sparsification based on percentile threshold
    threshold_percentile = 75  # Adjust this percentile as needed
    threshold = np.percentile(distance_matrix[distance_matrix > 0], threshold_percentile)  # Exclude zero distances

    sparse_mask = distance_matrix > threshold
    heuristic_matrix[sparse_mask] = 0

    # Local normalization
    for i in range(n):
        row = heuristic_matrix[i, :]
        max_row = np.max(row)
        min_row = np.min(row)
        if max_row > min_row:
            heuristic_matrix[i, :] = (row - min_row) / (max_row - min_row)
        else:
            heuristic_matrix[i, :] = np.ones_like(row) * 0.5
            

    # Control Randomness - adding a small amount of noise
    noise_level = 0.01
    heuristic_matrix += np.random.normal(0, noise_level, size=(n, n))
    heuristic_matrix = np.clip(heuristic_matrix, 0, 1)

    return heuristic_matrix
