import numpy as np

def heuristics_v2(distance_matrix: np.ndarray) -> np.ndarray:
    """
    Heuristics for TSP based on distance, degree centrality, geometric mean centrality,
    adaptive sparsification with percentile threshold, and targeted normalization.
    """
    n = distance_matrix.shape[0]

    # Initialize the heuristic matrix with the inverse of the distance
    heuristic_matrix = 1 / (distance_matrix + 1e-9)

    # Degree centrality heuristic
    degree_centrality = np.sum(1 / (distance_matrix + 1e-9), axis=1)
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristic_matrix[i, j] *= (degree_centrality[i] + degree_centrality[j]) / 2

    # Geometric mean centrality heuristic
    geometric_mean_centrality = np.exp(np.mean(np.log(distance_matrix + 1e-9), axis=1))
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristic_matrix[i, j] *= np.sqrt(geometric_mean_centrality[i] * geometric_mean_centrality[j])

    # Adaptive sparsification: dynamic percentile threshold
    edge_values = distance_matrix[np.triu_indices_from(distance_matrix, k=1)]
    sparsification_threshold = np.percentile(edge_values, 70)  # Adjust percentile as needed, slightly lower than v1

    for i in range(n):
        for j in range(n):
            if distance_matrix[i, j] > sparsification_threshold:
                heuristic_matrix[i, j] = 0

    # Targeted normalization: Normalize by both row and column means, and geometric mean
    row_means = np.mean(distance_matrix, axis=1, keepdims=True)
    col_means = np.mean(distance_matrix, axis=0, keepdims=True)

    for i in range(n):
        for j in range(n):
            if i != j:
                heuristic_matrix[i, j] /= (row_means[i, 0] + col_means[0, j] + geometric_mean_centrality[i] + geometric_mean_centrality[j] + 1e-9) # Inclusion of geometric mean

    # Controlled randomness for exploration
    randomness_factor = 0.005  # Adjust randomness factor as needed, lower than v1
    heuristic_matrix += randomness_factor * np.random.rand(n, n)

    # Final adjustment based on the inverse square of the distance
    heuristic_matrix = heuristic_matrix / (distance_matrix**1.8 + 1e-9) # reduced exponent

    # Normalize the heuristic matrix to a range of 0 to 1
    max_val = np.max(heuristic_matrix)
    min_val = np.min(heuristic_matrix)

    if max_val > min_val:
        heuristic_matrix = (heuristic_matrix - min_val) / (max_val - min_val)
    else:
        heuristic_matrix = np.ones_like(heuristic_matrix) * 0.5

    return heuristic_matrix
