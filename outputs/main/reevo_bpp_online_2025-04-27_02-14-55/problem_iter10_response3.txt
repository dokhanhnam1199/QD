import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, bin_contents: list[list[float]]) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        bin_contents: A list of lists, where each inner list represents the items
                      currently in a bin.  bin_contents[i] holds the items in bin i.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    num_bins = len(bins_remain_cap)

    # Highest priority: Exact fit
    exact_fit = (bins_remain_cap == item)
    priorities[exact_fit] = 1000.0

    # High priority: Near perfect fit, scaled by item size
    near_perfect_fit_threshold = 0.01  # Define a threshold for "near perfect"
    near_perfect_fit = (bins_remain_cap >= item) & (bins_remain_cap <= item * (1 + near_perfect_fit_threshold))
    priorities[near_perfect_fit] = 900.0 - (bins_remain_cap[near_perfect_fit] - item) / item * 200  # Scale based on waste

    # Feasibility check and HUGE penalty for infeasible bins
    infeasible = bins_remain_cap < item
    priorities[infeasible] = -1e10  # Even stronger penalty

    # Dynamic Fill Ratio Reward: Gaussian around optimal fill, scaled heavily
    can_accommodate = (bins_remain_cap >= item)
    if np.any(can_accommodate):
        fill_ratio = item / bins_remain_cap[can_accommodate]
        gaussian_std = 0.04  # Adjust std for the gaussian curve - narrowed
        gaussian_reward = np.exp(-0.5 * ((fill_ratio - 1.0) / gaussian_std) ** 2)
        priorities[can_accommodate] += gaussian_reward * 800.0  # Scale Gaussian reward aggressively

    # Fragmentation Penalty: Penalize creating small remaining spaces
    small_space_threshold = 0.15  # Define "small space" relative to bin size
    fragmented_bins = (bins_remain_cap >= item) & ((bins_remain_cap - item) <= small_space_threshold * np.max(bins_remain_cap))
    if np.any(fragmented_bins):
        waste = bins_remain_cap[fragmented_bins] - item
        penalty = (waste / (small_space_threshold * np.max(bins_remain_cap))) * 400  # Scale penalty
        priorities[fragmented_bins] -= penalty

    # Encourage using existing bins, more heavily
    non_empty = bins_remain_cap < np.max(bins_remain_cap)
    priorities[non_empty] += 75.0

    # Similarity Bonus: Reward bins with similar-sized items.  Use actual bin contents.
    for i in range(num_bins):
        if can_accommodate[i] and bin_contents[i]:  # Check if bin can accommodate and isn't empty
            avg_item_size = np.mean(bin_contents[i])
            similarity_threshold = 0.2  # Reduce threshold slightly
            if abs(avg_item_size - item) <= similarity_threshold * item:
                priorities[i] += 120.0  # Increased bonus

    # Discourage bins with way more capacity than the item size, more aggressive penalty
    large_capacity = (bins_remain_cap >= item) & (bins_remain_cap > 2 * item)  # Increased factor
    priorities[large_capacity] -= 80  # Increase the penalty

    # Moderate penalty for bins that are nearly full after placement
    almost_full_threshold = 0.1 * np.max(bins_remain_cap)
    almost_full = can_accommodate & ((bins_remain_cap - item) <= almost_full_threshold)
    priorities[almost_full] -= 50
    
    return priorities
