def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on how full a bin would become after placing the item, favoring bins that become as full as possible without exceeding capacity.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(cap - item)  # Prioritize smaller remaining capacity
    else:
      priority.append(float('inf'))
  return priority
