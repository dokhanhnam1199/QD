def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins by considering both the absolute remaining capacity and the relative waste if the item is placed in the bin, penalizing bins with small remaining capacity and high waste.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            priority = cap - (waste * waste)  # Penalize waste more severely
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
