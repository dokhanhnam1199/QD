def priority_v2(item, bins_remain_cap):
  """{This algorithm selects the bin whose remaining capacity is closest to half the item size, penalizing bins that are too full or too empty relative to this target.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      target_capacity = item / 2.0
      diff = abs(cap - target_capacity)
      priority = 1.0 / (1 + diff)
    else:
      priority = -1
    priorities.append(priority)
  return priority
