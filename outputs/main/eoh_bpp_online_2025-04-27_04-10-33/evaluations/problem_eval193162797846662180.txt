def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on how close the bin's remaining capacity is to the item's size, normalized by the total capacity, with a penalty for bins too small and a preference for bins slightly larger than the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      diff = cap - item
      priority.append(1.0 / (diff + 1))
    else:
      priority.append(-1)
  return priority
