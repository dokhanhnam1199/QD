def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins that can accommodate the item and then selects the bin with the least remaining capacity after placing the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(-1 * (cap - item))  # Prioritize bins with smaller remaining capacity after placement
    else:
      priority.append(-100000)  # extremely low priority if item does not fit
  return priority
