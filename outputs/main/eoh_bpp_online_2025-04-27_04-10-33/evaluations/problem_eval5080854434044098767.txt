def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, favoring bins where the item fills the bin as much as possible without exceeding capacity.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(item / cap)
    else:
      priority.append(-float('inf'))
  return priority
