def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns a priority score to each bin based on the remaining capacity after placing the item, favoring bins where the item fits best and avoids excessive fragmentation.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(cap - item)
    else:
      priority.append(-1)
  return priority
