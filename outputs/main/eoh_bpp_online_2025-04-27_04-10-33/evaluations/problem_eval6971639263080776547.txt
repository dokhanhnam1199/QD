def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority of each bin for an item based on the wasted space if the item is put into the bin, favoring bins with less wasted space and penalizing bins that are too full.}"""
    priorities = []
    bin_capacity = 1  # Assuming bin capacity is 1
    for cap in bins_remain_cap:
        if cap >= item:
            wasted_space = cap - item
            priority = 1 / (wasted_space + 0.0001) if wasted_space <= 0.5 else -1
        else:
            priority = -1
        priorities.append(priority)
    return priority
