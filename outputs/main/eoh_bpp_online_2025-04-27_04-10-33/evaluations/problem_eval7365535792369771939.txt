def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the wasted space if the item were placed in the bin, favoring bins with minimal waste.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      wasted_space = cap - item
      priority.append(-wasted_space) # Lower wasted space equals higher priority
    else:
      priority.append(-float('inf'))
  return priority
