def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins by considering both the percentage of the bin filled by the item and the absolute remaining space left after placing the item.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_percentage = item / cap
            remaining_space = cap - item
            # Combine fill percentage and remaining space into a single score
            priority.append(fill_percentage * (1 / (remaining_space + 1)))  # Adding 1 to prevent division by zero
        else:
            priority.append(-1)
    return priority
