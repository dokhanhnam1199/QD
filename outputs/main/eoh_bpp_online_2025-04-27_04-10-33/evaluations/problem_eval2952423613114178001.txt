def priority_v2(item, bins_remain_cap):
    """{Assign a priority score to each bin based on a combination of remaining capacity and how well the item fits, prioritizing bins with enough space and a tighter fit.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append((cap - item) + (1/(cap+1)))  # Reward tighter fit and overall capacity. small value added for the second term avoid /0
        else:
            priority.append(-1)  # Penalize bins with insufficient capacity
    return priority
