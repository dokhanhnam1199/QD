def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on the wasted space if the item is placed in the bin, prioritizing bins with the least waste relative to the item size.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      priority = item / (waste + 0.0001) # Avoid division by zero and prioritize less waste relative to item size
    else:
      priority = -1
    priorities.append(priority)
  return priority
