def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates priority scores for bins based on the waste generated if the item is placed in the bin, penalizing both small and large waste amounts.}"""
    priorities = []
    bin_capacity = 1 # Assuming bin capacity is 1
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            priority = 1 / (0.1 + abs(waste - (bin_capacity / 4))) #Penalize both small and large waste amounts
        else:
            priority = -1
        priorities.append(priority)
    return priority
