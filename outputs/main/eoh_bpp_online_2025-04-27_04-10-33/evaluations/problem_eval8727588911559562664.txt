def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, favoring bins where the item fills a significant portion of the remaining space without causing excessive waste.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(item / (cap + 0.001))  # Ratio of item size to remaining capacity, avoids division by zero
        else:
            priority.append(-1)
    return priority
