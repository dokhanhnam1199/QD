def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins by rewarding bins that can accommodate the item and favoring bins with more remaining capacity, encouraging efficient space utilization.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      remaining = cap - item
      priority_score = cap # reward bins with more remaining capacity
      if remaining < 0.1:
        priority_score -= 5  # Penalize almost full
      priority.append(priority_score)
    else:
      priority.append(-1000)
  return priority
