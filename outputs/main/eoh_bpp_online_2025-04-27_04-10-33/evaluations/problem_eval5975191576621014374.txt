def priority_v3(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on how much space the item will leave behind, favoring bins that will be filled to a level above a certain threshold after the item is placed.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      remaining_after_fit = cap - item
      fill_ratio_after = item / (cap)
      if fill_ratio_after >= 0.7:
            priority.append(1 - remaining_after_fit/cap)
      else:
          priority.append(-remaining_after_fit)
    else:
      priority.append(-100000)
  return priority
