def priority_v3(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the percentage of bin utilization after packing the item, favoring bins that result in higher utilization without exceeding capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            utilization = item / cap
            priority.append(utilization)
        else:
            priority.append(-1)
    return priority
