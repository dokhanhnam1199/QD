def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the squared ratio of remaining capacity to item size, favoring bins where the remaining capacity is close to the item size, and penalizing bins where the item doesn't fit.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append((float(item) / cap)**2)
        else:
            priority.append(-1)
    return priority
