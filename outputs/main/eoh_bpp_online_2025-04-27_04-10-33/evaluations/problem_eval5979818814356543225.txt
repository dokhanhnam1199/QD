def priority_v3(item, bins_remain_cap):
    """{This algorithm calculates the priority of each bin based on how close the remaining capacity is to the item size, while also considering a penalty for bins that are already too full.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            priority.append(item / (waste + 0.1) + cap/100)
        else:
            priority.append(-1)  # Item doesn't fit

    return priority
