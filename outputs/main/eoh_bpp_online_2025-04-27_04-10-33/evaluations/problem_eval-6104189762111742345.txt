def priority_v3(item, bins_remain_cap):
  """{This algorithm prioritizes bins by balancing remaining capacity, item size fit ratio, and a bonus for bins that are utilized well.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fit_ratio = item / cap # Ratio of item size to remaining capacity
      utilization_bonus = fit_ratio * 100 # Bonus for better utilization

      priority.append(utilization_bonus + (100 - cap)) # higher utilization bonus + smaller remaining capacity
    else:
      priority.append(-100000)  # extremely low priority if item does not fit
      
  return priority
