def priority_v3(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on how fully the item would fill them, aiming for high utilization while still ensuring the item fits.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            # Prioritize bins that would be filled to a high percentage.
            priority_score = item / cap
            priority.append(priority_score)
        else:
            priority.append(-1)  # Item doesn't fit
    return priority
