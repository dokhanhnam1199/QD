def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the square of the remaining capacity after placing the item, penalizing smaller capacities and infeasible bins.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append(1.0 / ( (cap - item)**2 + 0.001) ) 
    else:
      priority.append(-100000)
  return priority
