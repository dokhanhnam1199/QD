def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on the ratio of item size to remaining capacity, favoring bins where the item fills a significant portion of the available space without overfilling.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority_score = item / cap
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
