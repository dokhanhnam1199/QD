def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority of each bin based on the ratio of item size to remaining capacity, adjusted by a penalty for bins that are significantly larger than the item.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            priority.append(item / cap - (waste / (10 * cap))) # Ratio of item/cap, penalize waste
        else:
            priority.append(-1)  # Item doesn't fit

    return priority
