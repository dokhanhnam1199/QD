def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates a bin's priority based on how close the item's size is to the golden ratio proportion of the bin's remaining capacity, favoring bins where the item fills a golden-ratio-related portion.}"""
    priority = []
    golden_ratio = 1.618
    for cap in bins_remain_cap:
        if cap >= item:
            ratio = item / cap
            diff = abs(ratio - (1 / golden_ratio))
            priority.append(1.0 / (diff + 0.001))
        else:
            priority.append(-1)
    return priority
