def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on how close the remaining capacity is to the item size, penalizing bins significantly larger than the item.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      difference = cap - item
      priority = 1 / (1 + difference + (difference**2)) # Penalize large differences
    else:
      priority = -1
    priorities.append(priority)
  return priority
