def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates priority scores for bins based on a combination of remaining capacity and a bonus for bins that can perfectly fit the item, then returns the priority scores.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      if cap == item:
        priority.append(cap * 2)  # perfect fit bonus
      else:
        priority.append(cap)
    else:
      priority.append(-1)  # Invalid bin
  return priority
