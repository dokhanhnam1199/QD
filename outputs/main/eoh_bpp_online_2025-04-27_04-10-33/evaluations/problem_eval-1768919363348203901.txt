def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority based on how well the item fills the bin relative to the bin's total capacity, favoring bins that are filled more completely by the item.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = float(item) / cap
            priority.append(fill_ratio)
        else:
            priority.append(-1)
    return priority
