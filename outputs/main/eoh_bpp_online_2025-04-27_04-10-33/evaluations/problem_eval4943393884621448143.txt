def priority_v3(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on how well the item fills the bin, penalizing both excessive waste and insufficient space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = item / cap
      priority.append(fill_ratio) # prioritize bins that are filled well
    else:
      priority.append(-100000)  # extremely low priority if item does not fit
  return priority
