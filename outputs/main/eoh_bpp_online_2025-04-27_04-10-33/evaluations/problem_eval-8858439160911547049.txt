def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins by considering the wasted space if the item is placed in each bin, normalized by the initial capacity, and penalizes bins that cannot accommodate the item.}"""
    priorities = []
    bin_capacity = 1
    for cap in bins_remain_cap:
        if cap >= item:
            wasted_space = cap - item
            priority = 1 - (wasted_space / bin_capacity)
        else:
            priority = -1000
        priorities.append(priority)
    return priority
