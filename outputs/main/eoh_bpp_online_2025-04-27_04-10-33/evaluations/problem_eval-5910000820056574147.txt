def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on how close the item size is to a fraction of the bin's remaining capacity, aiming for balanced utilization.}"""
    priorities = []
    target_fraction = 0.75 # Try to make the item equal this fraction of the remaining capacity
    for cap in bins_remain_cap:
        if cap >= item:
            #How similar is the item size to a fraction of the remaining capacity?
            priority = -abs(item - target_fraction * cap)
        else:
            priority = -float('inf')
        priorities.append(priority)
    return priority
