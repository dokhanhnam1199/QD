def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority based on the absolute difference between the item size and the remaining capacity of each bin, favoring bins with the smallest difference if the item fits, and assigning a very low priority if the item doesn't fit.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(-abs(cap - item))
        else:
            priority.append(-float('inf'))
    return priority
