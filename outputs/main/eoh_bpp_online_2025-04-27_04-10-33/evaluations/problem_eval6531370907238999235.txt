def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on a combination of remaining capacity and the waste it would create, favoring bins that are a good fit for the item and avoid excessive waste.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      priority = cap + (1/(waste + 0.0001)) #add small const to avoid zero division
    else:
      priority = -1
    priorities.append(priority)
  return priority
