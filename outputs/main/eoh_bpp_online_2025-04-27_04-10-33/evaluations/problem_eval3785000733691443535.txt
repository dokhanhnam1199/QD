def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins by assessing the degree to which the item fills the bin, preferring bins that are neither too full nor too empty, but close to half full after placement.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = (item + 0.001) / (cap + 0.001)  # Ratio of item size to bin capacity after placing the item. Adding small value to prevent dividing by zero.
      ideal_ratio = 0.5
      priority.append(1.0 - abs(fill_ratio - ideal_ratio))  # Prioritize bins where fill_ratio is close to ideal_ratio.
    else:
      priority.append(-1)  # Item doesn't fit
  return priority
