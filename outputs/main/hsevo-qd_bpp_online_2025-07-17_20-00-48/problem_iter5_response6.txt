import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines utilization, fragmentation, and potential well for priority."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    small_number = 1e-9
    # Bin Utilization: Favor well-utilized bins.
    utilization = item / (bins_remain_cap + small_number)
    priorities += utilization

    # Fragmentation: Discourage small remaining spaces.
    frag_threshold_ratio = 5 #Adjust to item size
    frag_threshold = item / frag_threshold_ratio
    frag_penalty = np.exp(-np.abs(bins_remain_cap - item - frag_threshold) / (frag_threshold + small_number))
    fit_mask = bins_remain_cap >= item
    priorities[fit_mask] -= 0.3 * frag_penalty[fit_mask]

    # Potential Well: Favor bins slightly larger than the item.
    diff = bins_remain_cap - item
    potential_well = np.exp(-(diff**2) / (2 * (item/3)**2))
    priorities += potential_well

    # Infeasible bins get -inf priority
    priorities[bins_remain_cap < item] = -np.inf

    return priorities
