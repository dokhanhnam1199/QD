import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines utilization, fragmentation, and potential well concepts."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # 1. Utilization: Prioritize fuller bins after item placement.
    utilization = (item / (np.max(bins_remain_cap) + 1e-9))
    fit_mask = bins_remain_cap >= item
    if np.any(fit_mask):
        priorities[fit_mask] += (1 - (bins_remain_cap[fit_mask] - item) / (np.max(bins_remain_cap) + 1e-9)) * utilization

    # 2. Fragmentation: Discourage small remaining spaces.
    frag_threshold = item / 4.0
    if np.any(fit_mask):
      frag_penalty = np.exp(-np.abs(bins_remain_cap[fit_mask] - item - frag_threshold) / (frag_threshold + 1e-9))
      priorities[fit_mask] -= 0.2 * frag_penalty

    # 3. Potential Well: Favor bins slightly larger than item.
    diff = bins_remain_cap - item
    potential_well = np.exp(-(diff**2) / (2 * (item/3)**2))
    priorities += potential_well

    # Infeasible bin handling
    priorities[bins_remain_cap < item] = -np.inf

    return priorities
