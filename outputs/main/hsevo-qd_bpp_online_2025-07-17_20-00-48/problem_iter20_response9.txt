```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines utilization, fragmentation, and best fit while adapting to bin fill level."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    max_cap = np.max(bins_remain_cap)
    fit_mask = bins_remain_cap >= item

    # Hard constraint: Item must fit.
    priorities[~fit_mask] = -np.inf

    if not np.any(fit_mask):
        return priorities

    # 1. Utilization, adjusted for existing fill.
    utilization = item / max_cap
    remaining_capacity_after_fit = bins_remain_cap - item
    current_fill = 1 - bins_remain_cap / max_cap # bins fill level
    utilization_score = (1 - remaining_capacity_after_fit / max_cap) * utilization * (1 + current_fill)
    priorities[fit_mask] += utilization_score[fit_mask]

    # 2. Fragmentation penalty, adaptive.
    tiny_fragment_threshold = 0.05 * max_cap
    small_fragment_threshold = 0.2 * max_cap
    tiny_fragment_penalty = -2.0 - 2.0 * utilization * (1 + current_fill) # more penalized when the bin is fuller.
    small_fragment_penalty = -0.5 - 0.5 * utilization * (1 + current_fill)
    tiny_fragment_mask = (remaining_capacity_after_fit > 0) & (remaining_capacity_after_fit <= tiny_fragment_threshold)
    small_fragment_mask = (remaining_capacity_after_fit > tiny_fragment_threshold) & (remaining_capacity_after_fit <= small_fragment_threshold)
    priorities[tiny_fragment_mask] += tiny_fragment_penalty
    priorities[small_fragment_mask] += small_fragment_penalty
    
    # 3. Best Fit, with bonus that increases with smaller differences.
    diffs = bins_remain_cap - item
    min_diff = np.min(diffs[diffs >= 0]) if np.any(diffs >= 0) else np.inf
    best_fit_bonus = np.where(diffs == min_diff, 1.0 + 1.0/(1 + utilization * (1 + current_fill)) , 0) # adapt the bonus based on utilisation
    priorities += best_fit_bonus

    return priorities
```
