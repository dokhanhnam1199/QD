```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Hybrid priority: capacity, waste, best-fit bonus."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fit_mask = bins_remain_cap >= item

    if not np.any(fit_mask):
        return priorities - np.inf

    waste = bins_remain_cap[fit_mask] - item
    priorities[fit_mask] = bins_remain_cap[fit_mask] / (1 + waste)

    # Best-fit bonus.
    best_fit_idx = np.argmin(waste)
    priorities[fit_mask][best_fit_idx] += 1.0  # Add bonus.

    return priorities
```
