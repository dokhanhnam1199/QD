```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines ratio-based priority with a potential well for better bin selection."""
    ratios = item / (bins_remain_cap + 1e-9)
    priorities = -np.log(ratios) # Ratio-based priority

    diff = bins_remain_cap - item
    potential_well = np.exp(-(diff**2) / (2 * (item/3)**2))
    priorities += 0.5 * potential_well # Add potential well, reduced influence

    priorities[bins_remain_cap < item] = -np.inf
    return priorities
```
