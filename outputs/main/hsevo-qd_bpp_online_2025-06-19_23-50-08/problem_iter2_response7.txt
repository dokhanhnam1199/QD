```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins considering fill rate, waste, and infeasibility."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fit = item <= bins_remain_cap
    waste = bins_remain_cap - item
    
    priorities[~fit] = -np.inf  # Infeasible bins get lowest priority
    
    # Combination of fill rate and waste penalty for feasible bins
    fill_rate = item / bins_remain_cap[fit]
    waste_penalty = np.exp(-waste[waste >= 0] * 5)  # Reduced waste penalty
    priorities[fit] = fill_rate + waste_penalty # Combines fill rate & waste
    
    return priorities
```
