import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fill rate and wasted space, with infeasibility penalty."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fit = item <= bins_remain_cap
    waste = bins_remain_cap - item
    priorities[~fit] = -np.inf
    
    #Combines fill rate with waste penalty and a baseline log ratio.
    priorities[fit] = (item / bins_remain_cap[fit]) + np.exp(-waste[waste >= 0] * 10) - np.log(item / bins_remain_cap[fit])
    return priorities
