```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins by fill rate, waste, and infeasibility."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fit = item <= bins_remain_cap
    waste = bins_remain_cap - item
    
    priorities[~fit] = -np.inf  # Infeasible bins get lowest priority

    # Combine fill ratio and waste penalty for feasible bins.  Adjust the waste penalty
    # to be less aggressive than v0, but still present.
    
    priorities[fit] = (item / bins_remain_cap[fit]) + np.exp(-waste[waste >= 0] * 5)

    return priorities
```
