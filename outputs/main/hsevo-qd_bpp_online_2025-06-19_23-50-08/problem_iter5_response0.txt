import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fill rate, waste, and remaining capacity compared to average item size."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fit = item <= bins_remain_cap
    waste = bins_remain_cap - item
    priorities[~fit] = -np.inf
    priorities[fit] = (item / bins_remain_cap[fit]) + np.exp(-waste[waste >= 0] * 5)
    avg_item_size = np.mean(item)
    remaining_after_fit = bins_remain_cap[fit] - item
    priorities[fit] += np.exp(-np.abs(remaining_after_fit - avg_item_size) * 2)
    return priorities
