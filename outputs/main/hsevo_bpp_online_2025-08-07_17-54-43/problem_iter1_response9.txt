import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, temperature: float = 0.5) -> np.ndarray:
    """Softmax-based fit priority for online bin packing.

    Args:
        item: Size of the incoming item.
        bins_remain_cap: Remaining capacities of each bin.
        temperature: Softmax temperature controlling exploration vs exploitation.

    Returns:
        Priority scores for each bin, summing to 1 if any bin can accommodate the item.
    """
    can_fit = bins_remain_cap >= item
    leftover = bins_remain_cap - item
    safe_temp = max(temperature, 1e-8)
    scores = np.exp(-leftover / safe_temp) * can_fit
    total = scores.sum()
    priorities = scores / total if total > 0 else scores
    return priorities
