import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority for each bin using a Best Fit strategy."""
    overflow = bins_remain_cap - item
    mask = overflow >= 0
    priority = np.full(bins_remain_cap.shape, -np.inf, dtype=float)
    idx = np.arange(bins_remain_cap.shape[0])
    priority[mask] = -overflow[mask] - 1e-9 * idx[mask]
    return priority
