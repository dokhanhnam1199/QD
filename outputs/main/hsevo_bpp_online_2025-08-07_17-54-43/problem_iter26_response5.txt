```python
import numpy as np

# Simple deterministic best‑fit priority for online bin packing.
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Deterministic best‑fit: prioritize smallest remaining capacity; tie‑break by index."""
    n = bins_remain_cap.size
    scores = np.full(n, -np.inf, dtype=float)
    feasible = bins_remain_cap >= item
    if not np.any(feasible):
        return scores
    slack = bins_remain_cap[feasible] - item
    base = -slack
    idx = np.flatnonzero(feasible)
    base += idx * 1e-9  # tiny index bias for deterministic tie‑breaking
    scores[feasible] = base
    return scores
```
