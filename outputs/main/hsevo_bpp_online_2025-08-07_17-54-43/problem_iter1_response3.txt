import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Almost Full Fit: prioritize bins that leave the smallest slack after adding the item."""
    fits = bins_remain_cap >= item
    slack = bins_remain_cap - item
    priorities = np.where(fits, -slack, -np.inf)
    return priorities
