import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Best-fit with near-full boost, static ε‑greedy exploration, and deterministic tie‑breaking."""
    epsilon = 0.05
    scores = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    feasible = bins_remain_cap >= item
    if not np.any(feasible):
        return scores
    slack = bins_remain_cap[feasible] - item
    base = -slack
    near_full_thresh = max(0.02, 0.05 * item)
    boost = 0.5
    base[slack <= near_full_thresh] += boost
    idx = np.where(feasible)[0]
    deterministic = base + 1e-9 * idx
    if np.random.rand() < epsilon:
        scores[feasible] = np.random.rand(slack.size)
    else:
        scores[feasible] = deterministic
    return scores
