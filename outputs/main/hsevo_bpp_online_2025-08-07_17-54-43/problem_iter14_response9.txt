```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Static epsilon‑greedy best‑fit using occupancy fraction and softmax."""
    epsilon = 0.05
    capacity = 1.0
    n = bins_remain_cap.shape[0]
    priorities = np.full(n, -np.inf, dtype=float)
    feasible = bins_remain_cap >= item
    if not np.any(feasible):
        return priorities
    if np.random.rand() < epsilon:
        rand_scores = np.full(n, -np.inf, dtype=float)
        rand_scores[feasible] = np.random.rand(np.count_nonzero(feasible))
        return rand_scores
    occupancy = item + (capacity - bins_remain_cap[feasible])
    max_occ = np.max(occupancy)
    exp_scores = np.exp(occupancy - max_occ)
    softmax = exp_scores / np.sum(exp_scores)
    priorities[feasible] = softmax
    return priorities
```
