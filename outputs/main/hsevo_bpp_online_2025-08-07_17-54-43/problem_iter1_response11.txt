import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    residual = bins_remain_cap - item
    feasible = residual >= 0
    priority = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    priority[feasible] = -residual[feasible]
    return priority
