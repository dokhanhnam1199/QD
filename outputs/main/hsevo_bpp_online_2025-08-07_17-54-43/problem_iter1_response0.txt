import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """First Fit priority: higher score for earlier feasible bins."""
    idx = np.arange(bins_remain_cap.shape[0])
    feasible = bins_remain_cap >= item
    priorities = np.where(feasible, -idx, -np.inf)
    return priorities
