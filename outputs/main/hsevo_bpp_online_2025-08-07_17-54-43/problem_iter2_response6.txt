import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.1) -> np.ndarray:
    """Best‑fit priority with ε‑greedy exploration."""
    feasible = bins_remain_cap >= item
    waste = bins_remain_cap - item
    base_priority = np.where(feasible, 1.0/(waste + 1e-12), -np.inf)
    random_priority = np.random.rand(bins_remain_cap.size)
    return np.where(feasible, (1 - epsilon) * base_priority + epsilon * random_priority, -np.inf)
