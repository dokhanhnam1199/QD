import numpy as np

# Best-fit scoring with ε‑greedy exploration and near‑full/exact‑fit boosts
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Best-fit scoring with ε‑greedy exploration and near‑full/exact‑fit boosts, prioritizing bins with minimal slack while exploring occasionally.
    """
    if not hasattr(priority_v2, "epsilon"):
        priority_v2.epsilon = 0.1
        priority_v2.near_full_thresh = 0.05
        priority_v2.near_full_boost = 0.5
        priority_v2.exact_fit_boost = 1.0
    eps = priority_v2.epsilon
    feasible = bins_remain_cap >= item
    n = bins_remain_cap.size
    scores = np.full(n, -np.inf, dtype=float)
    if not np.any(feasible):
        return scores
    if np.random.rand() < eps:
        scores[feasible] = np.random.rand(feasible.sum())
    else:
        slack = bins_remain_cap[feasible] - item
        base = -slack
        near_full = slack <= priority_v2.near_full_thresh
        base[near_full] += priority_v2.near_full_boost
        exact_fit = slack == 0
        base[exact_fit] += priority_v2.exact_fit_boost
        scores[feasible] = base
    return scores
