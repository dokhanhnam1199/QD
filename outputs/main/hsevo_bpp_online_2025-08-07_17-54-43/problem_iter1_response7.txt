import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Random Fit priority: assign equal random priority to all bins that can accommodate the item."""
    feasible = bins_remain_cap >= item
    priorities = np.where(feasible, np.random.random(random_state=None, size=bins_remain_cap.shape), -np.inf)
    return priorities
