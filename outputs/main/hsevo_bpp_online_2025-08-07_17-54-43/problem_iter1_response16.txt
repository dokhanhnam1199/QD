```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    free = bins_remain_cap - item
    fit = free >= 0
    priorities = np.zeros_like(bins_remain_cap)
    if fit.any():
        f = free[fit]
        max_f = f.max()
        norm = (max_f - f) / (max_f + 1e-9)
        priorities[fit] = 1.0 / (1.0 + np.exp(-12.0 * norm))
        priorities[(fit) & (free == 0)] = 1.0
    return priorities
```
