import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    epsilon = 0.1
    n = bins_remain_cap.size
    priorities = np.full(n, -np.inf, dtype=float)
    valid = bins_remain_cap >= item
    if np.any(valid):
        priorities[valid] = item / bins_remain_cap[valid]
    if np.random.rand() < epsilon:
        random_scores = np.random.rand(n)
        random_scores[~valid] = -np.inf
        priorities = random_scores
    return priorities
