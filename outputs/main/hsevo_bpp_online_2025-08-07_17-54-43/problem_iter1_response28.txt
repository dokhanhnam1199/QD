import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.1) -> np.ndarray:
    """Returns priority with which we want to add item to each bin using epsilon-greedy."""
    n = bins_remain_cap.shape[0]
    priorities = np.full(n, -np.inf, dtype=float)
    feasible = bins_remain_cap >= item
    if not np.any(feasible):
        return priorities
    waste = bins_remain_cap[feasible] - item
    bestfit = -waste
    rand_vals = np.random.rand(bestfit.size)
    explore = np.random.rand(bestfit.size) < epsilon
    combined = np.where(explore, rand_vals, bestfit)
    priorities[feasible] = combined
    return priorities
