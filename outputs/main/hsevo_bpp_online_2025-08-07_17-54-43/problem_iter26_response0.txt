import numpy as np

# Adaptive epsilon‑greedy best‑fit with near‑full boost and ratio penalty
def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Score bins using negative slack, a small ratio penalty, and a near‑full boost. Occasionally explore random bins with decaying epsilon."""
    if not hasattr(priority_v2, "_epsilon"):
        priority_v2._epsilon = 0.2
        priority_v2._epsilon_min = 0.01
        priority_v2._epsilon_decay = 0.995
        priority_v2._step = 0
    priority_v2._step += 1
    priority_v2._epsilon = max(
        priority_v2._epsilon_min,
        priority_v2._epsilon * priority_v2._epsilon_decay,
    )
    eps = priority_v2._epsilon
    scores = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    feasible = bins_remain_cap >= item
    if not np.any(feasible):
        return scores
    slack, base = bins_remain_cap[feasible] - item, None
    slack = bins_remain_cap[feasible] - item
    base = -slack
    ratio = item / bins_remain_cap[feasible]
    ratio_weight = 0.02
    near_full_thresh = max(0.02, 0.05 * item)
    boost = 0.5
    bonus = np.where(slack <= near_full_thresh, boost, 0.0)
    deterministic = base + ratio_weight * ratio + bonus
    if np.random.rand() < eps:
        scores[feasible] = np.random.rand(feasible.sum())
    else:
        jitter = np.random.rand(feasible.sum()) * 1e-6
        scores[feasible] = deterministic + jitter
    return scores
