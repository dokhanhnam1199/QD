```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.05) -> np.ndarray:
    slack = bins_remain_cap.astype(float) - item
    eps = 1e-9
    feasible = slack >= 0
    base = np.where(feasible, 1.0 / (slack + eps), -np.inf)
    rand = np.random.rand(bins_remain_cap.shape[0])
    scores = np.where(feasible, (1 - epsilon) * base + epsilon * rand, -np.inf)
    return scores
```
