import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Random Fit priority: assign random scores to bins that can accommodate the item."""
    feasible = bins_remain_cap >= item
    priorities = np.full(bins_remain_cap.shape, -np.inf, dtype=float)
    priorities[feasible] = np.random.rand(feasible.sum())
    return priorities
