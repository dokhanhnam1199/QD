import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Simple best‑fit with ε‑greedy, ratio tie‑breaker and capped near‑full boost."""
    feasible = bins_remain_cap >= item
    scores = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    if not np.any(feasible):
        return scores
    slack = bins_remain_cap[feasible] - item
    base = -slack
    ratio = item / bins_remain_cap[feasible]
    ratio_weight = 0.02
    near_full_threshold = 0.05
    near_full_boost = np.where(slack <= near_full_threshold * bins_remain_cap[feasible], 0.1, 0.0)
    boost = np.minimum(near_full_boost, 0.2)
    deterministic = base + ratio_weight * ratio + boost
    epsilon = 0.05
    if np.random.rand() < epsilon:
        scores[feasible] = np.random.rand(feasible.sum())
    else:
        jitter = np.random.uniform(0, 1e-6, size=deterministic.shape)
        scores[feasible] = deterministic + jitter
    return scores
