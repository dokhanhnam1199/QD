import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin using Random Fit.

    In Random Fit, we consider all bins that can accommodate the item and assign
    them equal priority. Then, a random bin from this subset is chosen.
    This function assigns a priority of 1 to all bins that can fit the item,
    and 0 to those that cannot.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of remaining capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    # Assign priority 1 to bins that can accommodate the item
    can_fit_mask = bins_remain_cap >= item
    priorities[can_fit_mask] = 1.0
    return priorities
