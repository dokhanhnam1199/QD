import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin using Almost Full Fit.

    Almost Full Fit prioritizes bins that are close to being full but can still
    accommodate the item. It favors bins with remaining capacity slightly larger
    than the item size.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap)

    # Calculate the "fit" for each bin. A smaller value means a tighter fit.
    # We only consider bins that can actually fit the item.
    can_fit_mask = bins_remain_cap >= item
    
    # If no bin can fit the item, return all zeros (or handle as an error/new bin creation)
    if not np.any(can_fit_mask):
        return priorities

    # For bins that can fit the item, calculate a priority score.
    # The strategy is to give a high priority to bins where the remaining capacity
    # is just enough or slightly more than the item.
    # We want to maximize `bins_remain_cap - item`, but penalize cases where
    # the remaining capacity is much larger than the item.
    
    # A simple way to achieve "almost full fit" is to prioritize bins
    # where (bins_remain_cap - item) is minimized.
    # However, we need to differentiate between bins that are almost full
    # and bins that are very full (and thus not a good fit).

    # Let's try to prioritize bins where `bins_remain_cap - item` is small.
    # The ideal scenario is when `bins_remain_cap - item` is 0 (perfect fit).
    # We can use a function that peaks at 0 and decreases as the difference
    # increases. A negative quadratic might work, but we need to be careful
    # about the range.

    # Consider the "slack" or "wasted space" if the item is placed in the bin: `slack = bins_remain_cap - item`
    # We want to minimize this slack.

    # A common approach in "best fit" style heuristics is to give higher priority
    # to bins with the smallest remaining capacity that can still fit the item.
    # So, the priority could be inversely related to `bins_remain_cap` among those that fit.
    # However, "Almost Full Fit" suggests we want bins that are *already* relatively full.

    # Let's consider the bins that *can* fit the item.
    # For these bins, we want to find the one that will be "almost full" *after*
    # the item is placed. This means the remaining capacity *after* placement
    # should be small, but non-negative.

    # Let's define priority as a value that increases as the remaining capacity
    # after placing the item (`bins_remain_cap - item`) decreases,
    # but only for bins that can accommodate the item.

    # To make it "Almost Full Fit", we might want to slightly penalize bins
    # that are *too* empty, and strongly prefer bins that are just right.

    # Let's try: priority = 1 / (1 + (bins_remain_cap - item))
    # This gives higher priority to smaller (bins_remain_cap - item) values.
    # It's capped at 1 when bins_remain_cap == item.
    # It approaches 0 as (bins_remain_cap - item) gets large.

    # We only apply this to bins that can fit the item.
    
    # Calculate potential remaining capacity for fitting bins
    potential_remaining_caps = bins_remain_cap[can_fit_mask] - item

    # Calculate priorities for fitting bins: Higher priority for smaller remaining capacity
    # Using `1 / (1 + slack)` encourages smaller positive slacks.
    # If slack is 0 (perfect fit), priority is 1.
    # If slack is positive, priority is between 0 and 1.
    # Adding a small epsilon to the denominator to avoid division by zero if `item` exactly fills a bin and `bins_remain_cap` is 0 (though unlikely with positive item sizes).
    priorities[can_fit_mask] = 1.0 / (1.0 + potential_remaining_caps + 1e-9)

    # To further emphasize "almost full", we can also consider how full the bin
    # is *before* placing the item. A bin that is already very full (low `bins_remain_cap`)
    # might be preferred.
    
    # Let's modify the priority to give a boost to bins that are already more full.
    # We can multiply the previous priority by a factor that increases with `bins_remain_cap`.
    # However, this could conflict if a very full bin cannot fit the item.

    # A better approach for "Almost Full Fit" might be to prioritize bins
    # whose remaining capacity is closest to `item`, but still greater than or equal to it.
    # So, we want to minimize `bins_remain_cap - item`.

    # Let's re-evaluate the scoring:
    # We are selecting from bins where `bins_remain_cap >= item`.
    # Among these, we want the one that, after adding the item, leaves the least amount of space.
    # This means we want to minimize `bins_remain_cap - item`.

    # If we want to assign a *higher* priority to bins that result in *less* remaining space,
    # we can invert the value of `bins_remain_cap - item`.
    # A common way to do this is `1 / (bins_remain_cap - item + epsilon)`.
    # However, this can lead to very large values if `bins_remain_cap - item` is close to 0.

    # Let's use a Gaussian-like function centered at `item` for the bin capacities.
    # Or more simply, prioritize bins where `bins_remain_cap` is small but sufficient.
    
    # Alternative interpretation of "Almost Full Fit":
    # Prioritize bins that are already quite full (high occupancy).
    # Among these, pick the one that can fit the item.

    # Let's try prioritizing bins by their original remaining capacity, but only
    # for those that can fit the item. The bins with the smallest `bins_remain_cap`
    # that are still >= `item` are the best candidates. This is Best Fit.

    # For "Almost Full Fit", maybe it's about finding a bin that, *after* the item is added,
    # becomes "almost full" relative to its original capacity or some threshold.

    # Let's consider the "urgency" of filling a bin. A bin that is currently at capacity `C`
    # and can fit an item `i` leaving `C-i` capacity is prioritized if `C-i` is small.

    # Consider the inverse of the remaining capacity *after* adding the item.
    # Priority = 1 / (bins_remain_cap[i] - item + epsilon)
    # This favors bins that leave less space.
    
    # We need to ensure that if a bin is too small, it gets zero or very low priority.
    
    # Let's try a score that peaks when `bins_remain_cap` is just above `item`.
    # Consider `f(x) = max(0, C - x)` where `x` is the remaining capacity, `C` is the bin capacity, `item` is the item size.
    # This doesn't directly use `bins_remain_cap`.

    # Let's focus on `bins_remain_cap`. We want to prioritize bins where `bins_remain_cap`
    # is relatively small, but still greater than or equal to `item`.
    
    # A common heuristic for "Almost Full Fit" is to give priority proportional
    # to the remaining capacity of the bin, but only for bins that can fit the item.
    # This seems counter-intuitive to "almost full".

    # Let's use a score that is high when `bins_remain_cap` is slightly larger than `item`.
    # And rapidly drops off as `bins_remain_cap` increases beyond `item`.

    # Consider the function: `priority = exp(-alpha * (bins_remain_cap - item))` for `bins_remain_cap >= item`.
    # `alpha` controls how quickly priority drops. Larger `alpha` means stronger preference for tighter fits.
    # If `bins_remain_cap - item` is 0, priority is `exp(0) = 1`.
    # If `bins_remain_cap - item` is small and positive, priority is slightly less than 1.
    # If `bins_remain_cap - item` is large, priority approaches 0.

    # Let's set alpha = 1 for simplicity.
    alpha = 1.0
    
    # Apply the scoring only to bins that can fit the item.
    # For bins that cannot fit, their priority remains 0.
    priorities[can_fit_mask] = np.exp(-alpha * (bins_remain_cap[can_fit_mask] - item))

    # This prioritizes bins that are closer to fitting the item.
    # The 'almost full' aspect comes from the fact that if a bin's remaining capacity is
    # much larger than `item`, the priority will be significantly lower than for a bin
    # where `bins_remain_cap` is just `item` or slightly more.

    # To make it more "almost full" specifically, we might want to ensure
    # that bins that are *already* very full (i.e., `bins_remain_cap` is small,
    # but still fits `item`) get a bonus.
    # This current formula `exp(-(rem_cap - item))` inherently gives higher scores
    # to smaller `rem_cap` values (given they are >= item).

    # Let's consider an example: item = 0.3
    # Bin capacities: [0.5, 0.4, 1.0, 0.35, 0.2]
    # can_fit_mask: [True, True, True, True, False]
    # bins_remain_cap[can_fit_mask]: [0.5, 0.4, 1.0, 0.35]
    # rem_cap - item: [0.2, 0.1, 0.7, 0.05]
    # priorities[can_fit_mask]: [exp(-0.2), exp(-0.1), exp(-0.7), exp(-0.05)]
    # Priorities: [0.818, 0.904, 0.496, 0.951]

    # The highest priority goes to the bin with remaining capacity 0.35, which is closest to 0.3.
    # This seems to align well with "Almost Full Fit" â€“ it finds a bin that will be
    # "almost full" after the item is packed, by choosing the bin that leaves the least
    # amount of space.

    return priorities
