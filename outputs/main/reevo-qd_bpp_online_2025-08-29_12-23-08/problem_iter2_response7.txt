```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.1) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        epsilon: Probability of exploration.  Item-specific rate.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    n_bins = len(bins_remain_cap)
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Calculate the "greedy" priority (best fit)
    for i in range(n_bins):
        if bins_remain_cap[i] >= item:
            priorities[i] = 1.0 / (bins_remain_cap[i] - item + 1e-6)  # Smaller waste is better
        else:
            priorities[i] = 0.0

    # Exploration: Add a small random value to each bin with probability epsilon
    if np.random.rand() < epsilon * (1 + item):  # Item-specific exploration rate
        priorities += np.random.rand(n_bins) * 0.1  # Add a small random bonus

    return priorities
```
