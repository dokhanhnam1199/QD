import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_capacities = bins_remain_cap - item
    
    # Assign a very low priority to bins that cannot fit the item
    priorities = np.where(remaining_capacities >= 0, 
                          # Combine multiple factors: the remaining capacity and the absolute difference between the remaining capacity and the item size
                          remaining_capacities / (1 + np.abs(remaining_capacities - item)) + 
                          # Encourage using bins that are more full (lower remaining capacity) first
                          1 / (1 + bins_remain_cap), 
                          -1e6)
    
    # Introduce a controlled noise to valid bins for exploration-exploitation balance
    valid_bins = priorities != -1e6
    noise = np.random.uniform(-0.01, 0.01, size=len(bins_remain_cap))
    priorities = np.where(valid_bins, priorities + noise, priorities)
    
    return priorities
