import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 1e-8, seed: int = 0) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        epsilon: Small value to control the level of randomness.
        seed: Random seed for reproducibility.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Set random seed for reproducibility
    np.random.seed(seed)
    
    # Calculate the remaining capacity after adding the item
    remaining_capacities = bins_remain_cap - item
    
    # Assign a very low priority to bins that cannot fit the item
    priorities = np.where(remaining_capacities >= 0, -np.abs(remaining_capacities), -1e6)
    
    # Normalize priorities for bins that can fit the item
    valid_bins = remaining_capacities >= 0
    if np.any(valid_bins):
        valid_priorities = priorities[valid_bins]
        min_priority, max_priority = np.min(valid_priorities), np.max(valid_priorities)
        if max_priority != min_priority:
            normalized_priorities = (valid_priorities - min_priority) / (max_priority - min_priority)
            # Add adaptive noise to the normalized priorities
            noise = np.random.uniform(-epsilon, epsilon, size=len(valid_priorities))
            scaling_factor = 1 + np.abs(max_priority - min_priority)
            normalized_priorities += noise * scaling_factor
            priorities[valid_bins] = normalized_priorities
        else:
            priorities[valid_bins] = 1.0
    
    return priorities
