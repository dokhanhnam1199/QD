import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 1e-6) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        epsilon: Small value to control the level of randomness and prevent division by zero.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_capacities = bins_remain_cap - item
    
    # Assign a very low priority to bins that cannot fit the item
    priorities = np.where(remaining_capacities >= 0, -remaining_capacities, -1e6)
    
    # Normalize the priorities for valid bins and add controlled noise
    valid_bins = remaining_capacities >= 0
    if np.any(valid_bins):
        min_priority, max_priority = np.min(priorities[valid_bins]), np.max(priorities[valid_bins])
        if max_priority != min_priority:
            priorities[valid_bins] = (priorities[valid_bins] - min_priority) / (max_priority - min_priority + epsilon)
        else:
            priorities[valid_bins] = 1.0  # All valid bins have the same priority, set to 1.0
        priorities[valid_bins] += np.random.uniform(-epsilon, epsilon, np.sum(valid_bins))
        priorities[valid_bins] = np.clip(priorities[valid_bins], 0, 1)
    
    return priorities
