import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, bin_capacity: float) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of remaining capacities for each bin.
        bin_capacity: The total capacity of each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_capacities = bins_remain_cap - item
    
    # Assign a very low priority to bins that cannot fit the item
    valid_bins = remaining_capacities >= 0
    
    # Calculate the priority based on how well the item fills the bin
    priorities = np.where(valid_bins, 
                          (bins_remain_cap - remaining_capacities) / bin_capacity + 
                          (1 - (remaining_capacities / bins_remain_cap)), 
                          -1e6)
    
    # Add a small controlled noise to valid priorities for exploration-exploitation balance
    priorities = np.where(valid_bins, priorities + np.random.normal(0, 0.01, size=len(priorities)), priorities)
    
    return priorities
