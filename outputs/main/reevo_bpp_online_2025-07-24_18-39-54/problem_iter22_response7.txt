import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, capacity: float) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of remaining capacities for each bin.
        capacity: Maximum capacity of a bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Normalize the remaining capacities to ensure numerical stability
    normalized_bins_remain_cap = bins_remain_cap / capacity
    
    # Assign a very low priority to bins that cannot fit the item
    priorities = np.where(bins_remain_cap >= item, 
                          -np.abs(normalized_bins_remain_cap - item / capacity), -1e6)
    
    # Introduce a controlled noise to balance exploration-exploitation
    noise = np.random.uniform(-0.01, 0.01, size=len(bins_remain_cap))
    priorities = np.where(bins_remain_cap >= item, priorities + noise, priorities)
    
    return priorities
