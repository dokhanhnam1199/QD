```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 1e-9, noise_scale: float = 0.0) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        epsilon: Small value to prevent division by zero.
        noise_scale: Scale of the random noise.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_capacities = bins_remain_cap - item
    
    # Assign a very low priority to bins that cannot fit the item
    priorities = np.where(remaining_capacities >= 0, remaining_capacities, -1e6)
    
    # Add controlled noise for exploration only to bins that can fit the item
    valid_bins = priorities != -1e6
    if noise_scale > 0:
        priorities = np.where(valid_bins, priorities + noise_scale * np.random.rand(*priorities.shape), priorities)
    
    return priorities
```
