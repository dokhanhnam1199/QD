import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, bin_capacities: np.ndarray) -> np.ndarray:
    """
    Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of remaining capacities for each bin.
        bin_capacities: The capacities of the bins.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """

    # Feature 1: Whether the bin can fit the item
    can_fit = (bins_remain_cap >= item).astype(float)
    
    # Feature 2: Normalized remaining capacity after adding the item
    remaining_capacities = np.maximum(bins_remain_cap - item, 0) / bin_capacities
    
    # Feature 3: Normalized fullness of the bin before adding the item
    bin_fullness = 1 - (bins_remain_cap / bin_capacities)
    
    # Feature 4: Normalized absolute difference between the remaining capacity and the item size
    diff = 1 - np.abs(bins_remain_cap - item) / (np.maximum(bins_remain_cap, item) + 1e-6)
    
    # Combine features into a priority score with weights
    priorities = can_fit * (0.25 * remaining_capacities + 0.25 * bin_fullness + 0.5 * diff)
    
    # Add controlled noise for exploration
    noise = np.random.normal(0, 0.005, size=len(bins_remain_cap))
    priorities += noise
    
    return priorities
