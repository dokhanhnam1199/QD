```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 1e-6, noise_scale: float = 0.01) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        epsilon: Small value to prevent division by zero.
        noise_scale: Scale of random noise to add for exploration.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_capacities = bins_remain_cap - item
    
    # Assign a very low priority to bins that cannot fit the item
    valid_bins = remaining_capacities >= 0
    
    # Normalize priorities for valid bins
    priorities = np.where(valid_bins, -np.abs(bins_remain_cap - item) / (item + epsilon), -1e6)
    
    # Add a small noise to priorities for exploration
    priorities = np.where(valid_bins, priorities + noise_scale * np.random.rand(*priorities.shape), priorities)
    
    return priorities
```
