import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, noise_scale: float = 0.01) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        noise_scale: Scale of the noise to be added for exploration.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Directly use the remaining capacity after adding the item as priority for valid bins
    priorities = np.where(bins_remain_cap >= item, bins_remain_cap - item, -1e6)
    
    # Add noise to valid bins for exploration-exploitation balance
    valid_bins = priorities != -1e6
    noise = np.random.uniform(-noise_scale, noise_scale, size=len(bins_remain_cap))
    priorities = np.where(valid_bins, priorities + noise, priorities)
    
    return priorities
