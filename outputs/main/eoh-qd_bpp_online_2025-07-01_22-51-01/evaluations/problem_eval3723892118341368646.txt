def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority based on a combination of remaining capacity, waste, and a penalty for bins that are too full.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            if waste == 0:
                p = 100  # Perfect fit
            else:
                p = 50 + (1 / (waste + 0.01)) # Prioritize bins with smaller waste
        else:
            p = -1000  # Heavily penalize bins that can't fit the item
        priority.append(p)
    return priority
