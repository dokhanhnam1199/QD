def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on a combination of remaining capacity and how well the item fits, preferring bins that have enough space for the item and are close to being filled after packing the item.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority.append( (1 / (cap - item + 0.00001)) + (cap / sum(bins_remain_cap)) )
    else:
      priority.append(-1)
  return priority
