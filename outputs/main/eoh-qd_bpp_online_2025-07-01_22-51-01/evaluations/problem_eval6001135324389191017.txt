def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on the remaining capacity of each bin after placing the item, favoring bins that leave a smaller gap but avoiding bins where the item doesn't fit or the gap is too small.}"""
  priority = []
  for cap in bins_remain_cap:
    if item > cap:
      priority.append(-1)
    else:
      gap = cap - item
      if gap < 0.1:
        priority.append(-1)
      else:
        priority.append(1/(gap + 0.0001))
  return priority
