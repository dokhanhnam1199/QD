import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm iteratively samples random tours, improves them using 2-opt, and updates a matrix indicating the frequency with which each edge appears in the improved tours.}"""
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    num_iterations = 1000

    def calculate_tour_length(tour):
        length = 0
        for i in range(n - 1):
            length += distance_matrix[tour[i], tour[i + 1]]
        length += distance_matrix[tour[-1], tour[0]]
        return length

    def two_opt_swap(tour, i, k):
        new_tour = tour[:i] + tour[i:k+1][::-1] + tour[k+1:]
        return new_tour
    
    def two_opt(tour):
        best_tour = tour
        best_length = calculate_tour_length(tour)
        improved = True
        while improved:
            improved = False
            for i in range(1, n - 1):
                for k in range(i + 1, n):
                    new_tour = two_opt_swap(best_tour, i, k)
                    new_length = calculate_tour_length(new_tour)
                    if new_length < best_length:
                        best_tour = new_tour
                        best_length = new_length
                        improved = True
                        break
                if improved:
                    break
        return best_tour
        
    for _ in range(num_iterations):
        # Generate a random tour
        tour = np.random.permutation(n)
        
        # Improve the tour using 2-opt
        improved_tour = two_opt(tour)

        # Update the heuristics matrix
        for i in range(n - 1):
            heuristics_matrix[improved_tour[i], improved_tour[i+1]] += 1
            heuristics_matrix[improved_tour[i+1], improved_tour[i]] += 1
        heuristics_matrix[improved_tour[-1], improved_tour[0]] += 1
        heuristics_matrix[improved_tour[0], improved_tour[-1]] += 1

    return heuristics_matrix
