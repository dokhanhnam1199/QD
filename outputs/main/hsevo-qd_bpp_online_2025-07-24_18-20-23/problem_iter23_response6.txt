import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines relative fit and proportion of capacity used with nuanced normalization.
    Prioritizes bins with remaining capacity closest to item size and balances fullness.
    """
    epsilon = 1e-6  # Small value for numerical stability
    valid_bins = bins_remain_cap >= item
    diff = np.abs(bins_remain_cap - item)
    priority = np.where(valid_bins, -diff + 1 / (bins_remain_cap + epsilon), -np.inf)
    
    # Introduce a Gaussian-like term to further differentiate among viable bins
    gaussian_term = np.exp(-((bins_remain_cap - item) / item) ** 2)
    priority = np.where(valid_bins, priority + 0.2 * gaussian_term, priority)
    
    # Normalize priority scores for valid bins
    valid_bins = priority != -np.inf
    if np.any(valid_bins):
        max_priority = np.max(priority[valid_bins])
        priority[valid_bins] = np.exp(-(max_priority - priority[valid_bins]))
    
    return priority
