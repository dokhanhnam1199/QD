import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines relative fit, proportion of capacity used, and nuanced differentiation for online bin packing.
    Prioritizes bins where the item fits well and differentiates among viable options effectively.
    """
    # Calculate priority score based on relative fit and proportion of capacity used
    epsilon = 1e-6  # Small value for numerical stability
    valid_bins = bins_remain_cap >= item
    relative_fit = 1 - np.abs(bins_remain_cap - item) / (item + epsilon)
    proportion_used = item / (bins_remain_cap + epsilon)
    priority = np.where(valid_bins, relative_fit * proportion_used, -1.0)
    
    # Normalize priority scores for valid bins and further differentiation
    max_priority = np.max(priority)
    if max_priority > -1.0:  
        priority = np.where(priority > -1.0, priority / max_priority, priority)
        # Introduce nuanced differentiation among valid bins similar to priority_v1
        diff = np.abs(bins_remain_cap - item)
        weight_for_diff = 1.0  # Adjusted weight for differentiation
        priority = np.where(valid_bins, priority - weight_for_diff * diff / (item + epsilon), priority)
        # Normalize again after differentiation
        max_priority = np.max(priority)
        priority = np.where(priority > -1.0, priority / max_priority, priority)
    
    return priority
