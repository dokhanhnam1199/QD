import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin using a combined heuristic of relative fit and proportion used with nuanced normalization.
    """
    # Calculate priority score combining relative fit and proportion used
    priorities = np.where(bins_remain_cap >= item,
                          (1 - np.abs(bins_remain_cap - item) / item) * (item / (bins_remain_cap - item + item)),
                          -1.0)
    
    # Normalize priorities
    max_priority = np.max(priorities)
    if max_priority > -1.0:  
        priorities = np.where(priorities > -1.0, priorities / max_priority, priorities)

    # Slightly reward bins with lower remaining capacity and penalize those with very low utilization
    priorities = np.where((bins_remain_cap >= item) & (bins_remain_cap <= item * 2),
                          priorities * 1.1,  # Reward bins with relatively low remaining capacity
                          priorities)
    priorities = np.where((bins_remain_cap >= item) & ((item / (bins_remain_cap - item + item)) < 0.5),
                          priorities * 0.8,  # Penalize bins that will be less than half full
                          priorities)

    return priorities
