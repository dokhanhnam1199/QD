def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority score for each bin based on remaining capacity and item size.
    Prioritizes bins that can fit the item with minimal remaining capacity.
    """
    # Calculate the remaining capacity after adding the item
    remaining_cap = bins_remain_cap - item
    
    # Bins that can fit the item have higher priority based on remaining capacity
    valid_bins = remaining_cap >= 0
    
    # Assign high priority to valid bins with minimal remaining capacity
    priority = np.where(valid_bins, -remaining_cap, -np.inf)
    
    return priority
