```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for adding an item to each bin.
    Combines elements of 'First-Fit Decreasing' and 'Best-Fit' heuristics.
    """
    # Prioritize bins where the item fits and leaves minimal remaining capacity
    priorities = np.where(bins_remain_cap >= item, 
                          1 / (bins_remain_cap - item + 1e-6),  # Best-Fit element: prioritize bins with least remaining capacity
                          -1.0)  # Penalize bins that can't fit the item
    
    # Adjust priorities to ensure bins that can't fit the item have lower priority
    priorities = np.where(bins_remain_cap < item, priorities - np.max(priorities), priorities)
    
    return priorities
```
