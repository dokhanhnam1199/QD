def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines relative fit, proportion of capacity used, and nuanced normalization for online bin packing.
    Prioritizes bins based on remaining capacity while ensuring numerical stability.
    """
    epsilon = 1e-6  # Small value for numerical stability
    valid_bins = bins_remain_cap >= item
    
    # Calculate relative fit and proportion used
    relative_fit = 1 - np.abs(bins_remain_cap - item) / (item + epsilon)
    proportion_used = item / (bins_remain_cap + epsilon)
    
    # Combine relative fit and proportion used with fullness priority
    fullness_priority = 1 - bins_remain_cap / (bins_remain_cap + item + epsilon)
    priority = np.where(valid_bins, relative_fit * proportion_used + 0.4 * fullness_priority, -1.0)
    
    # Normalize priority scores for valid bins
    max_priority = np.max(priority)
    if max_priority > -1.0:  
        priority = np.where(priority > -1.0, priority / max_priority, priority)
    
    # Differentiate among bins that can fit the item
    priority = np.where((bins_remain_cap >= item) & (bins_remain_cap > item * 2),
                        priority * 0.9,  # Slightly reduce priority for bins with high remaining capacity
                        priority)
    
    return priority
