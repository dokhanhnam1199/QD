```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines 'Best-Fit' and nuanced prioritization strategies with numerical stability.
    """
    # Prioritize bins where the item fits, using Best-Fit element with epsilon for stability
    priorities = np.where(bins_remain_cap >= item, 
                          np.exp(-(bins_remain_cap - item)) + 1 / (bins_remain_cap - item + 1e-6), 
                          -1e10)  # Penalize bins that can't fit the item
    
    return priorities
```
