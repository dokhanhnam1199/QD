import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.0005808774091321647) -> np.ndarray:
    """
    Returns priority score for each bin based on remaining capacity and item size.
    Prioritizes bins that can fit the item and have remaining capacity closest to the item size.
    """
    # Check if bins can fit the item
    valid_bins = bins_remain_cap >= item
    
    # Calculate the absolute difference between remaining capacity and item size
    diff = np.abs(bins_remain_cap - item)
    
    # Assign high priority to valid bins with remaining capacity closest to the item size
    priority = np.where(valid_bins, -diff + 1 / (bins_remain_cap + epsilon), -np.inf)
    
    return priority
