```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines 'Best-Fit' and 'First-Fit Decreasing' heuristics with numerical stability.
    Prioritizes bins where the item fits with minimal remaining capacity.
    """
    # Calculate priority based on remaining capacity and item size
    priorities = np.where(bins_remain_cap >= item, 
                          np.exp(-np.abs(bins_remain_cap - item)) / (bins_remain_cap - item + 1e-6),  # Combine Best-Fit and numerical stability
                          -np.inf)  # Penalize bins that can't fit the item
    
    return priorities
```
