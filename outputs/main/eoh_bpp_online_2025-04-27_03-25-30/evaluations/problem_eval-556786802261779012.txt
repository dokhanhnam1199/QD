def priority_v3(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on a ratio of remaining capacity to item size, favoring bins with a closer fit.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority_score = item / cap # Ratio of item size to remaining capacity
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
