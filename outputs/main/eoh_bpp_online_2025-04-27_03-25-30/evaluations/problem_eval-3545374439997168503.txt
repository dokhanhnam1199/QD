def priority_v3(item, bins_remain_cap):
    """{This algorithm assigns a priority to each bin based on how full the bin would be if the item were placed in it, preferring bins that would be filled to a high percentage but not over capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_percentage = item / (cap + item)
            priority.append(fill_percentage)
        else:
            priority.append(-1)
    return priority
