def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates a priority score for each bin based on the ratio of the item size to the remaining capacity, penalized by the absolute difference between them.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority_score = (item / cap) - abs(item - cap)
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
