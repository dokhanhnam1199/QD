def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority based on how full a bin would be after adding the item, favoring bins that become nearly full but not overfull.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = item / cap
            priority_score = fill_ratio
        else:
            priority_score = -1
        priority.append(priority_score)
    return priority
