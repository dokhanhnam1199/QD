def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, favoring bins where the item fills a substantial portion of the remaining space.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = float(item) / cap
            priorities.append(priority)
        else:
            priorities.append(-1)
    return priority
