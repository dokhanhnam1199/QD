def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on the ratio of item size to remaining capacity, penalizing bins that are either too full or have significantly more capacity than the item size.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      ratio = float(item) / cap
      priority_score = ratio - abs(ratio - 0.75) # Prefer ratios around 0.75, penalize deviations
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
