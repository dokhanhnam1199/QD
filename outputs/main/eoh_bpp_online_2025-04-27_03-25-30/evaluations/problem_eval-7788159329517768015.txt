def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns a priority to each bin based on the 'fit' status of the item and bin, penalizing bins that cannot fit, otherwise rewarding bins that can fit with scores dependent on the ratio of item size to remaining capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(item / cap)
        else:
            priority.append(-1)
    return priority
