def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority based on how close the item's size is to a fraction (e.g., half, third, quarter) of the remaining capacity of each bin.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      diff_from_half = abs(item - cap / 2.0)
      diff_from_third = abs(item - cap / 3.0)
      diff_from_quarter = abs(item - cap / 4.0)
      
      priority_score = 1.0 / (min(diff_from_half, diff_from_third, diff_from_quarter) + 1e-9) # Smaller difference, higher priority
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
