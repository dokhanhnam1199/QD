def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on how much the item fills the bin, favoring bins that are filled more completely and avoiding bins that would be overfilled after the item is placed, then selecting the best possible bin and providing a score for each bin.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      fill_ratio = item / cap
      priority.append(fill_ratio)
    else:
      priority.append(-1)
  return priority
