def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns a priority to each bin based on how well the item fits relative to a target fill percentage of the bin.}"""
    priority = []
    bin_capacity = 1  # Assume bin capacity is 1, adjust as needed
    target_fill = 0.75  # Aim to fill bins to 75% capacity

    for cap in bins_remain_cap:
        if cap >= item:
            ideal_remaining = bin_capacity * (1 - target_fill)
            priority.append(-abs(cap - ideal_remaining)) # penalize deviations from ideal remaining capacity.
        else:
            priority.append(-1000) # Assign a very low priority to bins that can't fit the item.

    return priority
