def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on remaining capacity, penalizing bins with too much or too little remaining space, favoring bins that offer a close fit.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            capacity_ratio = cap / 100  # Remaining capacity as a ratio of bin size
            priority_score = 1 - abs(capacity_ratio - 0.5) #Penalize too small or too big
            priority.append(priority_score)
        else:
            priority.append(-1)
    return priority
