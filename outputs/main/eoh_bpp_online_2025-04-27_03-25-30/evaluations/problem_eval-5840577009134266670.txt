def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on how close the item size is to the remaining capacity of the bin, preferring bins that are nearly full after the item is placed.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority_score = 1.0 / (abs(cap - item) + 1e-9)
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
