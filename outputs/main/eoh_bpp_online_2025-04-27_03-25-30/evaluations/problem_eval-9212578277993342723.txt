def priority_v3(item, bins_remain_cap):
    """{This algorithm assigns a priority to each bin based on how full the bin will be after placing the item, preferring bins that will be filled to a higher percentage of their capacity.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_percentage = item / (cap + 1e-9)  # Avoid division by zero
            priority.append(fill_percentage)  # Prioritize higher fill percentage
        else:
            priority.append(-1)
    return priority
