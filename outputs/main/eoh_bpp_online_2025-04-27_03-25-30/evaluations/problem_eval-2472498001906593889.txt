def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority of each bin based on the square of the remaining capacity after placing the item, penalizing both very empty and nearly full bins.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            remaining_cap_after = cap - item
            priority_score = - (remaining_cap_after ** 2) # Penalize high remaining capacity by squaring it
            priority.append(priority_score)
        else:
            priority.append(-1)
    return priority
