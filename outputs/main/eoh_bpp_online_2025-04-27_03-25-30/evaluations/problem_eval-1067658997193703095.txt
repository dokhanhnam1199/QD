def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, combined with a bonus for bins that are already mostly full.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            capacity_ratio = item / cap
            fullness_ratio = (1 - (cap / 100)) # Assuming bin capacity is 100
            priority_score = capacity_ratio + fullness_ratio
            priority.append(priority_score)
        else:
            priority.append(-1)
    return priority
