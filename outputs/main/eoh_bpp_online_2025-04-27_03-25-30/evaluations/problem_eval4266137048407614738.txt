def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the absolute difference between the item size and remaining capacity, preferring bins where the remaining capacity is closest to the item size, but only if the bin can fit the item.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(100 - abs(cap - item))
        else:
            priority.append(-1)
    return priority
