def priority_v2(item, bins_remain_cap):
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            fill_ratio = item / 100
            nearly_full_penalty = 0
            if cap - item <= 10:
                nearly_full_penalty = (10 - (cap - item)) * 0.05  # Penalty increases as bin gets closer to full
            priority_score = (1 / (waste + 0.0001)) * fill_ratio - nearly_full_penalty
        else:
            priority_score = -1
        priority.append(priority_score)
    return priority
