def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins by considering the ratio of item size to remaining capacity, favoring bins where the item fills a significant portion of the remaining space without exceeding it.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = item / cap
            priority.append(fill_ratio)
        else:
            priority.append(-1)
    return priority
