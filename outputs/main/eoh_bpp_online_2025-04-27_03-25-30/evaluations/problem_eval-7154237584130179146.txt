def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins that have enough capacity for the item and then selects the bin with the smallest index.}"""
    priority = []
    for i, cap in enumerate(bins_remain_cap):
        if cap >= item:
            priority_score = 1000 - i  # Prioritize lower index if fits
        else:
            priority_score = -1
        priority.append(priority_score)
    return priority
