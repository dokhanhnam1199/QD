def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on a heuristic function considering remaining capacity and item size ratio, favoring bins with remaining capacity close to the item size.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority_score = cap / item
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
