def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on how closely the item size matches the bin's remaining capacity, preferring near-perfect fits to minimize wasted space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority_score = 1.0 / (abs(cap - item) + 0.0001)  # Higher score for closer fit
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
