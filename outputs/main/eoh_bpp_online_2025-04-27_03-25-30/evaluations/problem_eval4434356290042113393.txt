def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the ratio of item size to remaining capacity, favoring bins where the item fills a significant portion of the remaining space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      priority_score = item / cap
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
