def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin based on a combination of remaining capacity and wasted space, prioritizing bins that can accommodate the item without excessive waste.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      wasted_space = cap - item
      priority_score = cap - wasted_space**2 #Prefer not to have too much wasted space
    else:
      priority_score = -1 
    priority.append(priority_score)
  return priority
