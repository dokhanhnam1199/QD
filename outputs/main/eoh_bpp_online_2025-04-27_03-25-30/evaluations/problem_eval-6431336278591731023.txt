def priority_v2(item, bins_remain_cap):
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            relative_waste = item / (cap + 1e-9) # Avoid division by zero
            penalty = (cap > 90) # Penalize nearly empty bins
            priority_score = (1 - relative_waste) - (penalty * 0.5) #Adjustments could be necessary in the constansts, according to the distribution of item sizes.
            priority.append(priority_score)
        else:
            priority.append(-1)
    return priority
