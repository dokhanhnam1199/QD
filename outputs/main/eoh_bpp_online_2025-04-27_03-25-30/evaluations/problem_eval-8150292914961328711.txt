def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns a priority to each bin based on how well the item fills the bin, favoring bins that are filled to a higher percentage of their capacity after the item is placed.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(item / (cap + 1e-9))  # Avoid division by zero
        else:
            priority.append(-1)

    return priority
