def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns a priority to each bin based on how full it would be after packing the item, favoring bins that become mostly full but not overfull.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            fill_ratio = item / cap
            if fill_ratio > 1:
                priority.append(-1)  # Overfull penalty
            else:
                priority.append(1 - abs(fill_ratio - 0.9)) # Closer to 90% fill, higher priority
        else:
            priority.append(-1)
    return priority
