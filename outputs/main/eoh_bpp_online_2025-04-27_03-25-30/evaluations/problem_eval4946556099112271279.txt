def priority_v3(item, bins_remain_cap):
  """{This algorithm prioritizes bins primarily based on fullness, with a secondary consideration for minimizing wasted space and a reduced penalty for creating small remaining spaces.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      waste = cap - item
      fullness = 1 - (cap / 100)
      penalty = 0
      if waste < 10:
        penalty = (10 - waste) / 20
      priority_score = fullness * 2 + (1 / (waste + 1)) - penalty
      priority.append(priority_score)
    else:
      priority.append(-1)
  return priority
