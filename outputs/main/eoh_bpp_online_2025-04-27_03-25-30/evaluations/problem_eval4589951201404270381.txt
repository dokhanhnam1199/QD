def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on the absolute difference between the item size and half of the bin's remaining capacity, favoring bins where the item fills approximately half the remaining space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      diff = abs(item - cap / 2)
      priority_score = 1 / (diff + 0.0001)
    else:
      priority_score = -1
    priority.append(priority_score)
  return priority
