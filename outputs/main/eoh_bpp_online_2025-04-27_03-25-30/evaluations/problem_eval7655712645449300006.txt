def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on how closely filling them with the item results in the bin being nearly full, while still ensuring the item fits.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      remaining_cap = cap - item
      # Prioritize bins that, after packing, have a remaining capacity close to zero.
      priority = 1.0 / (remaining_cap + 0.0001)
      priorities.append(priority)
    else:
      priorities.append(-1)  # Assign low priority to bins that cannot fit the item
  return priority
