def priority_v3(item, bins_remain_cap):
    """{This algorithm assigns a priority to each bin based on the wasted space if the item is placed in it, favoring bins with waste close to a target waste value.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            target_waste = 10
            penalty_factor = 0.5
            priority.append(1 / (abs(waste - target_waste) * penalty_factor + 1))
        else:
            priority.append(-1)
    return priority
