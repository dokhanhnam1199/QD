def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns a priority to each bin based on how well the item fills the bin, rewarding bins that are nearly full after packing the item and penalizing those with significant or negligible remaining space.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            waste = cap - item
            if waste > 15:  # Penalize excessive waste
                priority.append(0.2)
            elif waste < 3: # Heavily penalize very little waste
                priority.append(0.05)
            else:
                priority.append(1 / (waste + 0.5))
        else:
            priority.append(-1)
    return priority
