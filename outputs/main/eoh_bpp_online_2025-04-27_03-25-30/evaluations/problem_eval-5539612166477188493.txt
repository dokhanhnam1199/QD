def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the ratio of remaining capacity to the item size, penalized by the potential waste if the item is placed.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            wasted_space = cap - item
            priority_score = (cap / item) - (wasted_space / (item + 1e-9))
            priority.append(priority_score)
        else:
            priority.append(-1)  # Assign a low priority if the item doesn't fit
    return priority
