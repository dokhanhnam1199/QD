def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority of each bin based on a combination of remaining capacity, wasted space, and bin fullness.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            wasted_space = cap - item
            fullness = 1 - (cap / sum(bins_remain_cap)) if sum(bins_remain_cap) > 0 else 1
            priority_score = (1 / (wasted_space + 0.0001)) + fullness
        else:
            priority_score = -1
        priority.append(priority_score)
    return priority
