import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray,
                almost_full_threshold: float = 0.9,
                almost_full_boost: float = 0.5,
                balance_threshold_low: float = 0.3,
                balance_threshold_high: float = 0.6,
                small_item_threshold_fraction: float = 0.1,
                small_item_boost: float = 0.3,
                random_factor: float = 0.01,
                not_fit_penalty: float = -1e9) -> np.ndarray:
    """Combines FFD, almost full, balanced bin, and small remainder preferences."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    bin_max_cap = np.max(bins_remain_cap)

    # Heuristic 1: First Fit Decreasing (FFD)
    for i, cap in enumerate(bins_remain_cap):
        if cap >= item:
            wasted_space = cap - item
            priorities[i] = 1.0 / (1.0 + wasted_space)
        else:
            priorities[i] = not_fit_penalty

    # Heuristic 2: Almost Full Bin
    is_almost_full = bins_remain_cap / bin_max_cap < (1 - almost_full_threshold)
    for i, is_full in enumerate(is_almost_full):
        if is_full and bins_remain_cap[i] >= item:
            priorities[i] += almost_full_boost

    # Heuristic 3: Balanced Bin
    bin_fill_levels = (bin_max_cap - bins_remain_cap) / bin_max_cap
    is_balanced = (bin_fill_levels >= balance_threshold_low) & (bin_fill_levels <= balance_threshold_high)
    for i, is_bal in enumerate(is_balanced):
        if is_bal and bins_remain_cap[i] >= item:
            priorities[i] += 0.2  # Balanced bin boost

    # Heuristic 4: Small Item & Small Remainder
    small_item_threshold = small_item_threshold_fraction * bin_max_cap
    if item < small_item_threshold:
        for i, cap in enumerate(bins_remain_cap):
            remainder = cap - item
            if 0 < remainder < small_item_threshold:
                priorities[i] += small_item_boost

    # Heuristic 5: Random Factor
    priorities += np.random.rand(len(bins_remain_cap)) * random_factor
    return priorities
