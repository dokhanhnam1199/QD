import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines FFD, almost full bin filling, small remainder preference, and bin diversity."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    not_fit_penalty = -1e9
    almost_full_threshold = 0.9
    almost_full_boost = 0.5
    small_item_threshold = 0.1 * np.max(bins_remain_cap)
    random_factor = 0.005

    # FFD-based priority (primary factor)
    for i, cap in enumerate(bins_remain_cap):
        if cap >= item:
            wasted_space = cap - item
            priorities[i] = 1.0 / (1.0 + wasted_space)
        else:
            priorities[i] = not_fit_penalty

    # Almost full bin boost
    is_almost_full = bins_remain_cap / np.max(bins_remain_cap) < (1 - almost_full_threshold)
    for i, is_full in enumerate(is_almost_full):
        if is_full and bins_remain_cap[i] >= item:
            priorities[i] += almost_full_boost

    # Small item remainder preference
    if item < small_item_threshold:
        for i, cap in enumerate(bins_remain_cap):
            remainder = cap - item
            if 0 < remainder < small_item_threshold:
                priorities[i] += 0.3

    # Bin Diversity: Give preference to bins with higher current fill level.
    fill_levels = 1 - (bins_remain_cap / np.max(bins_remain_cap))
    for i in range(len(bins_remain_cap)):
        if bins_remain_cap[i] >= item:
            priorities[i] += 0.1 * fill_levels[i]


    # Introduce a bit of randomness to break ties and explore the search space
    priorities += np.random.rand(len(bins_remain_cap)) * random_factor
    return priorities
