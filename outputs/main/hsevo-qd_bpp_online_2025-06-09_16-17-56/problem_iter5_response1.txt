import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines wasted space, almost full, balance, small item heuristics."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Heuristic 1: Wasted Space
    for i, cap in enumerate(bins_remain_cap):
        if cap >= item:
            wasted_space = cap - item
            priorities[i] = 1.0 / (1.0 + wasted_space)
        else:
            priorities[i] = -1e9

    # Heuristic 2: Almost Full
    almost_full_threshold = 0.9
    is_almost_full = bins_remain_cap / np.max(bins_remain_cap) < (1 - almost_full_threshold)
    for i, is_full in enumerate(is_almost_full):
        if is_full and bins_remain_cap[i] >= item:
            priorities[i] += 0.5

    # Heuristic 3: Balance
    balance_threshold_low = 0.2
    balance_threshold_high = 0.8
    bin_fill_levels = (np.max(bins_remain_cap) - bins_remain_cap) / np.max(bins_remain_cap)
    is_balanced = (bin_fill_levels >= balance_threshold_low) & (bin_fill_levels <= balance_threshold_high)
    for i, is_bal in enumerate(is_balanced):
        if is_bal and bins_remain_cap[i] >= item:
            priorities[i] += 0.25

    # Heuristic 4: Small Items
    small_item_threshold = 0.1 * np.max(bins_remain_cap)
    if item < small_item_threshold:
        for i, cap in enumerate(bins_remain_cap):
            remainder = cap - item
            if 0 < remainder < small_item_threshold:
                priorities[i] += 0.3

    # Heuristic 5: Randomization
    priorities += np.random.rand(len(bins_remain_cap)) * 0.01
    return priorities
