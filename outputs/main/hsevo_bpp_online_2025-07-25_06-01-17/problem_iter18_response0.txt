import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, alpha: float = 0.4, beta: float = 0.3, gamma: float = 0.3) -> np.ndarray:
    """
    An evolved priority function for the online Bin Packing Problem, incorporating adaptive weighting, 
    multiple objectives, and nuanced problem understanding.

    Parameters:
    item (float): The size of the current item to be packed.
    bins_remain_cap (np.ndarray): The remaining capacities of the available bins.
    alpha (float): Weight for the current item's fit priority (default: 0.4).
    beta (float): Weight for the bin utilization priority (default: 0.3).
    gamma (float): Weight for the bin capacity utilization priority (default: 0.3).

    Returns:
    np.ndarray: Priority scores for each bin, where higher scores indicate a better choice for the current item.
    """

    # Calculate the difference between the remaining capacity of each bin and the item size
    capacity_diff = np.abs(bins_remain_cap - item)
    
    # Invert the differences to obtain a priority score for the current item's fit
    fit_priority = 1 / (1 + capacity_diff)
    
    # If the item size exceeds the remaining capacity, set the priority to 0
    fit_priority[bins_remain_cap < item] = 0
    
    # Calculate the bin utilization, considering both the current item and future potential
    avg_historic_size = np.mean(bins_remain_cap)  # Using mean remaining capacity as a proxy for historic item sizes
    utilization_priority = (bins_remain_cap + avg_historic_size) / (bins_remain_cap + item + avg_historic_size)
    
    # Calculate the bin capacity utilization priority, emphasizing the efficient use of bin space
    bin_capacity_utilization = bins_remain_cap / (bins_remain_cap + item)
    capacity_utilization_priority = np.where(bin_capacity_utilization > 0.5, 1 - bin_capacity_utilization, bin_capacity_utilization)
    
    # Calculate the overall priority, balancing the current item's fit, bin utilization, and capacity utilization
    priority = alpha * fit_priority + beta * utilization_priority + gamma * capacity_utilization_priority
    
    # Introduce an adaptive weighting mechanism, adjusting the priority based on problem complexities
    complexity_factor = 1 - (np.sum(bins_remain_cap) / (len(bins_remain_cap) * np.max(bins_remain_cap)))
    priority *= (1 + complexity_factor)
    
    # Ensure the priority scores are not NaN (e.g., due to division by zero)
    priority = np.nan_to_num(priority)
    
    # Assign high priority to bins with enough capacity and a good fit
    priority = np.where(bins_remain_cap >= item, priority, -1)
    
    return priority
