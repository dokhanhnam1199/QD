import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins by balancing waste, capacity, and best-fit with a parameter."""
    eligible_bins = bins_remain_cap >= item
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    if np.any(eligible_bins):
        waste = bins_remain_cap[eligible_bins] - item
        priorities[eligible_bins] = -waste

        capacity_factor = bins_remain_cap[eligible_bins] / np.max(bins_remain_cap)
        priorities[eligible_bins] += capacity_factor

        best_fit_bonus = np.exp(-np.abs(waste) / (0.2 * item + 1e-6))
        priorities[eligible_bins] += best_fit_bonus

        # Tunable parameter to adjust the importance of capacity
        capacity_weight = 0.5
        priorities[eligible_bins] += capacity_weight * (bins_remain_cap[eligible_bins] / np.max(bins_remain_cap))


    else:
        priorities[:] = -1e9

    return priorities
