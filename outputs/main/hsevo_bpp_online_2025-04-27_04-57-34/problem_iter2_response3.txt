import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Assigns bin priorities considering fit, fill ratio, and fragmentation."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fits = bins_remain_cap >= item

    if np.any(fits):
        remaining_space = bins_remain_cap[fits] - item
        priorities[fits] = (item / bins_remain_cap[fits]) + (1 / (1 + remaining_space))
        nearly_full = remaining_space < 0.1
        priorities[fits][nearly_full] *= 0.9
    else:
        priorities = -bins_remain_cap
        priorities = priorities - np.min(priorities) + 1e-9

    return priorities
