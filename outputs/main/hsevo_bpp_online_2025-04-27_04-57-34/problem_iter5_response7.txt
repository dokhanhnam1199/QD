import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fill ratio and avoids excessive fragmentation."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fits = bins_remain_cap >= item

    if np.any(fits):
        priorities[fits] = item / bins_remain_cap[fits]  # Fill ratio
        remaining_space = bins_remain_cap[fits] - item
        nearly_full = remaining_space < 0.1
        priorities[fits][nearly_full] *= 0.9 # avoid nearly_full
    else:
        priorities = -bins_remain_cap # prioritize bins to open
        priorities = priorities - np.min(priorities) + 1e-9

    return priorities
