import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Calculate bin priorities balancing waste, capacity utilization, and a fill bonus."""

    eligible_bins = bins_remain_cap >= item
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    if np.any(eligible_bins):
        waste = bins_remain_cap[eligible_bins] - item
        priorities[eligible_bins] = -waste # Prioritize smaller waste

        # Capacity utilization
        capacity_ratio = item / bins_remain_cap[eligible_bins]
        priorities[eligible_bins] += np.clip(capacity_ratio, 0, 1)

        # Fill bonus: bonus when item almost fills the bin
        best_fit_threshold = 0.05 * item
        best_fit_bonus = np.where(waste <= best_fit_threshold, 0.5, 0.0)
        priorities[eligible_bins] += best_fit_bonus

        priorities[eligible_bins] += 0.001  #Small boost for tie-breaking

    else:
        priorities[:] = -1e9 # Big penalty for bins that can't fit item

    return priorities
