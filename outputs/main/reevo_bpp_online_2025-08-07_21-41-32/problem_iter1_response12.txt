import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Priority function implementing the Worst‑Fit strategy for online bin packing.

    Parameters
    ----------
    item : float
        Size of the item to be placed.
    bins_remain_cap : np.ndarray
        Array of current remaining capacities for each bin.

    Returns
    -------
    np.ndarray
        Array of priority scores; bins that can hold the item receive a
        priority equal to the remaining capacity after the placement.
        Non‑fitting bins are assigned a priority of -inf to exclude them.
    """
    # Prepare array of priorities; start with -inf so that non‑fitting bins
    # are never chosen unless all bins fail.
    priorities = np.full(bins_remain_cap.shape, -np.inf, dtype=float)

    # Identify bins that have enough capacity to hold the item
    can_fit = bins_remain_cap >= item

    # Compute priority as the remaining capacity after the item would be added
    priorities[can_fit] = bins_remain_cap[can_fit] - item

    return priorities
