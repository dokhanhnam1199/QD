import numpy as np

def heuristics_v2(distance_matrix, coordinates, demands, capacity):
    """{This algorithm constructs routes greedily based on a weighted sum of distance and demand, favoring shorter distances and smaller demands.}"""
    n = len(demands)
    heuristics_matrix = np.zeros_like(distance_matrix)
    
    for i in range(n):
        for j in range(n):
            if i != j:
                heuristics_matrix[i][j] = 1 / (distance_matrix[i][j] * (demands[j] if j !=0 else 1)) 

    return heuristics_matrix
