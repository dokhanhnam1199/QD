import numpy as np
import random

def heuristics_v2(distance_matrix, coordinates, demands, capacity):
    """
    {This algorithm estimates edge inclusion probabilities by generating a large number of random feasible routes and counting how often each edge appears in these routes.}
    """
    n = len(demands)
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    num_samples = 1000

    for _ in range(num_samples):
        # Generate a random permutation of customers
        customer_indices = list(range(1, n))
        random.shuffle(customer_indices)

        # Construct routes based on the permutation and capacity constraints
        routes = []
        current_route = [0]  # Start at the depot
        current_load = 0

        for customer in customer_indices:
            if current_load + demands[customer] <= capacity:
                current_route.append(customer)
                current_load += demands[customer]
            else:
                current_route.append(0)  # Return to depot
                routes.append(current_route)
                current_route = [0, customer]  # Start a new route
                current_load = demands[customer]

        current_route.append(0) # Return to depot from last customer
        routes.append(current_route)

        # Update the heuristics matrix based on edges used in the routes
        for route in routes:
            for i in range(len(route) - 1):
                node1 = route[i]
                node2 = route[i+1]
                heuristics_matrix[node1, node2] += 1
                heuristics_matrix[node2, node1] += 1  # Assuming symmetric distance matrix

    # Normalize the heuristics matrix by the number of samples
    heuristics_matrix /= num_samples

    return heuristics_matrix
