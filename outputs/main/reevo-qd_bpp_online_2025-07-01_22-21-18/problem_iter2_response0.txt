import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    
    # Prioritize bins that can fit the item snugly (First-Fit Decreasing-like behavior)
    fit_mask = bins_remain_cap >= item
    
    if np.any(fit_mask):
        slack = bins_remain_cap[fit_mask] - item
        priorities[fit_mask] = 1 / (slack + 1e-6)  # Prioritize smaller slack

        # Reward near-full bins after placing the item
        near_full_capacity = 0.95  # Target percentage for near-full (e.g., 95%)
        near_full_boost = 10.0      # Boost for almost full bins
        post_fill_levels = bins_remain_cap[fit_mask] - item #capacity after hypothetically placing item
        near_full_threshold = (1 - near_full_capacity) * 1.0
        near_full_mask = post_fill_levels <= near_full_threshold
        
        
        priorities[fit_mask][near_full_mask] += near_full_boost
    


    # Penalize bins that can't fit the item heavily. Allow overflow if no other options are available, but discourage it strongly.
    overflow_mask = bins_remain_cap < item
    if np.any(overflow_mask):
          priorities[overflow_mask] = - (item - bins_remain_cap[overflow_mask]) / (np.max(bins_remain_cap) + 1e-6) #penalize the amount of overfill relative to biggest bin

    
    return priorities
