import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm iteratively samples random tours, evaluates their lengths, and updates a heuristic matrix to favor edges appearing in shorter tours.}"""
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    num_iterations = 1000
    
    for _ in range(num_iterations):
        # Generate a random tour
        tour = np.random.permutation(n)
        
        # Calculate the tour length
        tour_length = 0
        for i in range(n - 1):
            tour_length += distance_matrix[tour[i], tour[i+1]]
        tour_length += distance_matrix[tour[n-1], tour[0]]
        
        # Update the heuristics matrix
        for i in range(n - 1):
            heuristics_matrix[tour[i], tour[i+1]] += (1 / tour_length)
            heuristics_matrix[tour[i+1], tour[i]] += (1 / tour_length) # Symmetric
        heuristics_matrix[tour[n-1], tour[0]] += (1 / tour_length)
        heuristics_matrix[tour[0], tour[n-1]] += (1 / tour_length)

    return heuristics_matrix
