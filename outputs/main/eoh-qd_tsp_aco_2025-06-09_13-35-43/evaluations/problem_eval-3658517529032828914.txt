import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm generates a large number of random tours, evaluates their lengths, and then assigns higher probabilities to edges that appear frequently in shorter tours, using an exponential score function based on tour length.}"""
    n = distance_matrix.shape[0]
    num_samples = 1000
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    tour_lengths = []
    tours = []

    for _ in range(num_samples):
        tour = np.random.permutation(n)
        tour_length = 0
        for i in range(n - 1):
            tour_length += distance_matrix[tour[i], tour[i+1]]
        tour_length += distance_matrix[tour[-1], tour[0]]
        tour_lengths.append(tour_length)
        tours.append(tour)

    # Use an exponential score function
    avg_length = np.mean(tour_lengths)
    scores = np.exp([-length / avg_length for length in tour_lengths])
    total_score = np.sum(scores)
    weights = [score / total_score for score in scores]

    for i, tour in enumerate(tours):
        weight = weights[i]
        for j in range(n - 1):
            u = tour[j]
            v = tour[j+1]
            heuristics_matrix[u, v] += weight
            heuristics_matrix[v, u] += weight  # Ensure symmetry
        heuristics_matrix[tour[-1], tour[0]] += weight
        heuristics_matrix[tour[0], tour[-1]] += weight # Ensure symmetry

    return heuristics_matrix
