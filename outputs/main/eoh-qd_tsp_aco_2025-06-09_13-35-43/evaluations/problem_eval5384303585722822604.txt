import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm constructs a heuristics matrix by iteratively sampling random tours, and giving higher weight to edges present in shorter tours within a limited search space.}"""
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros((n, n))
    num_samples = 1000
    
    for _ in range(num_samples):
        # Generate a random tour
        tour = np.random.permutation(n)
        tour = np.append(tour, tour[0]) # return to starting node

        # Calculate the length of the tour
        tour_length = 0
        for i in range(n):
            tour_length += distance_matrix[tour[i], tour[i+1]]
        
        # Use the inverse of the tour length as weight
        weight = 1 / (tour_length + 1e-6)  # Avoid division by zero

        # Increment the heuristics matrix for each edge in the current sampled tour, weighted by tour length
        for i in range(n):
            heuristics_matrix[tour[i], tour[i+1]] += weight
            heuristics_matrix[tour[i+1], tour[i]] += weight

    return heuristics_matrix
