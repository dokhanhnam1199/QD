import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm generates a large number of random tours, evaluates their lengths, and then assigns weights based on the exponential of the negative tour length, returning a matrix representing the likelihood of each edge's inclusion in an optimal tour.}"""
    n = distance_matrix.shape[0]
    num_samples = 1000
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
    tour_lengths = []
    tours = []

    for _ in range(num_samples):
        tour = np.random.permutation(n)
        tour_length = 0
        for i in range(n - 1):
            tour_length += distance_matrix[tour[i], tour[i+1]]
        tour_length += distance_matrix[tour[-1], tour[0]]
        tour_lengths.append(tour_length)
        tours.append(tour)

    # Use exponential of negative tour length as weight
    tour_lengths = np.array(tour_lengths)
    weights = np.exp(-tour_lengths / tour_lengths.std())  # Normalize by standard deviation
    weights /= weights.sum()

    for i, tour in enumerate(tours):
        weight = weights[i]
        for j in range(n - 1):
            u = tour[j]
            v = tour[j+1]
            heuristics_matrix[u, v] += weight
            heuristics_matrix[v, u] += weight  # Ensure symmetry
        heuristics_matrix[tour[-1], tour[0]] += weight
        heuristics_matrix[tour[0], tour[-1]] += weight # Ensure symmetry

    return heuristics_matrix
