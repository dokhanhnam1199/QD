import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm constructs a heuristics matrix by iteratively sampling random tours, and incrementing the corresponding entry in the matrix inversely proportional to the tour length for each edge present in a sampled tour.}"""
    n = distance_matrix.shape[0]
    heuristics_matrix = np.zeros((n, n))
    num_samples = 1000
    
    for _ in range(num_samples):
        # Generate a random tour
        tour = np.random.permutation(n)
        tour = np.append(tour, tour[0]) # return to starting node

        # Calculate the length of the tour
        tour_length = 0
        for i in range(n):
            tour_length += distance_matrix[tour[i], tour[i+1]]
        
        # Increment the heuristics matrix for each edge in the current sampled tour, weighted by inverse tour length
        weight = 1.0 / tour_length if tour_length > 0 else 0.0 # Avoid division by zero.
        for i in range(n):
            heuristics_matrix[tour[i], tour[i+1]] += weight
            heuristics_matrix[tour[i+1], tour[i]] += weight

    return heuristics_matrix
