import numpy as np

def heuristics_v2(distance_matrix):
    """{This algorithm calculates a heuristic matrix by subtracting each distance from the maximum distance in the matrix and then normalizing the result.}"""
    max_distance = np.max(distance_matrix)
    heuristics_matrix = (max_distance - distance_matrix) / (max_distance + 1e-9)
    return heuristics_matrix
