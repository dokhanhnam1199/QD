import numpy as np

def heuristics_v2(distance_matrix):
    """
    {This algorithm creates a matrix where each element represents the inverse of the square of the corresponding distance in the distance matrix, adjusted by a scaling factor and exponentiation to further enhance contrast between short and long distances.}
    """
    heuristics_matrix = 1 / ((distance_matrix**2) + 1e-9)
    return heuristics_matrix
