import numpy as np

def heuristics_v2(distance_matrix):
  """{This algorithm samples multiple solutions using a greedy nearest neighbor approach with probabilistic acceptance of edges based on edge distance and a temperature parameter that decreases over iterations, favoring shorter edges as the temperature cools.}"""
  n = distance_matrix.shape[0]
  num_samples = 100
  heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)
  temperature = 1.0
  cooling_rate = 0.95

  for _ in range(num_samples):
    start_node = np.random.randint(n)
    current_node = start_node
    unvisited_nodes = set(range(n))
    unvisited_nodes.remove(current_node)
    path = [current_node]

    while unvisited_nodes:
      best_neighbor = None
      min_distance = float('inf')
      for neighbor in unvisited_nodes:
        distance = distance_matrix[current_node, neighbor]
        if distance < min_distance:
          min_distance = distance
          best_neighbor = neighbor

      # Probabilistic acceptance based on temperature
      if np.random.rand() < np.exp(-(min_distance / temperature)):
        nearest_neighbor = best_neighbor
      else:
        # Choose a random neighbor if not accepting the best one
        nearest_neighbor = np.random.choice(list(unvisited_nodes))
        min_distance = distance_matrix[current_node, nearest_neighbor]

      path.append(nearest_neighbor)
      unvisited_nodes.remove(nearest_neighbor)
      current_node = nearest_neighbor

    path.append(start_node)

    for i in range(n):
      node1 = path[i]
      node2 = path[i+1]
      heuristics_matrix[node1, node2] += 1
      heuristics_matrix[node2, node1] += 1

    temperature *= cooling_rate

  mean_distance = np.mean(distance_matrix)

  for i in range(n):
    for j in range(n):
      if i != j:
        heuristics_matrix[i, j] = (heuristics_matrix[i, j] / num_samples) / (distance_matrix[i, j] / mean_distance)
      else:
        heuristics_matrix[i, j] = 0

  return heuristics_matrix
