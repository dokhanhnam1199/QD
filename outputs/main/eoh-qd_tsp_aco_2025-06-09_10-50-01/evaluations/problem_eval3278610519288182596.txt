import numpy as np

def heuristics_v2(distance_matrix):
    """
    {This algorithm samples multiple random tours, then aggregates the edge frequencies across all sampled tours to create a heuristic matrix, indicating the promise of each edge.}
    """
    n = distance_matrix.shape[0]
    num_samples = 100
    heuristics_matrix = np.zeros_like(distance_matrix, dtype=float)

    for _ in range(num_samples):
        # Generate a random tour
        tour = np.random.permutation(n)
        tour = np.append(tour, tour[0])  # Return to starting node

        # Update heuristics matrix based on edges in this tour
        for i in range(n):
            u = tour[i]
            v = tour[i + 1]
            heuristics_matrix[u, v] += 1
            heuristics_matrix[v, u] += 1  # Ensure symmetry

    # Normalize the heuristic matrix by the number of samples
    heuristics_matrix /= num_samples

    return heuristics_matrix
