def priority_v2(item, bins_remain_cap):
    """{This algorithm prioritizes bins based on the remaining capacity, favoring bins with sufficient space for the item and penalizing those that would lead to significant wasted space.}"""
    priorities = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority = 1 / (cap - item + 0.0001) # Avoid division by zero
        else:
            priority = -float('inf')  # Assign a very low priority if the bin is too small
        priorities.append(priority)
    return priority
