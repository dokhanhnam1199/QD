def priority_v2(item, bins_remain_cap):
    """{This algorithm assigns priority to bins based on the remaining capacity after placing the item, favoring bins with smaller remaining capacity that can still accommodate the item and penalizing bins that would be filled exactly.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            remaining_cap = cap - item
            if remaining_cap == 0:
                priority.append(-1)  # Penalize perfect fit
            else:
                priority.append(1 / remaining_cap)  # Higher priority to bins with smaller remaining capacity
        else:
            priority.append(-float('inf'))  # Assign lowest priority if item doesn't fit
    return priority
