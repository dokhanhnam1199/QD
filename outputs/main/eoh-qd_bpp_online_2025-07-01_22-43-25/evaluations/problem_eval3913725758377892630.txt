def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates a priority score for each bin based on the remaining capacity and the item size, favoring bins where the item fits best without wasting too much space.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append(item / cap) 
        else:
            priority.append(-1)
    return priority
