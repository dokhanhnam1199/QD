def priority_v2(item, bins_remain_cap):
  """{This algorithm calculates the priority of each bin for an item based on a weighted sum of the remaining capacity and the wasted space if the item is placed in the bin, favoring bins with sufficient capacity and minimal wasted space.}"""
  priority = []
  for cap in bins_remain_cap:
    if cap >= item:
      wasted_space = cap - item
      priority.append(cap*0.6 - wasted_space*0.4)
    else:
      priority.append(-1000) 
  return priority
