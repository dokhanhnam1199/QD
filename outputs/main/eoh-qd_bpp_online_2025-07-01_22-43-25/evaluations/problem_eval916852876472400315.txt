def priority_v2(item, bins_remain_cap):
  """{This algorithm prioritizes bins based on a combination of remaining capacity and wasted space if the item is placed in the bin.}"""
  priorities = []
  for cap in bins_remain_cap:
    if cap >= item:
      wasted_space = cap - item
      priority = cap + 1.0 / (wasted_space + 0.1)  # Add a small constant to avoid division by zero
    else:
      priority = -1  # Assign a negative priority if the item doesn't fit
    priorities.append(priority)
  return priority
