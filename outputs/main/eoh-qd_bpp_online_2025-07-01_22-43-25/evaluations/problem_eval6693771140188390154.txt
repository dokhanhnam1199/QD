def priority_v2(item, bins_remain_cap):
    """{This algorithm calculates the priority of each bin by considering the remaining capacity of the bin and how well the item fits, favoring bins with enough space and a tighter fit.}"""
    priority = []
    for cap in bins_remain_cap:
        if cap >= item:
            priority.append((cap - item) / (cap + item))
        else:
            priority.append(-1)
    return priority
