def priority_v2(item, bins_remain_cap):
  """{This algorithm assigns priority to each bin based on a combination of remaining capacity, a penalty if the item doesn't fit, and a bonus if the bin is empty.}"""
  priorities = []
  for cap in bins_remain_cap:
    if item <= cap:
      priority = cap - item  # Remaining capacity after packing (lower is better)
      if cap == bins_remain_cap[0]:
          priority += 0.1 # give the first bin in the array slight preference
      
    else:
      priority = -1000 # very low priority if it doesn't fit
    priorities.append(priority)
  return priority
