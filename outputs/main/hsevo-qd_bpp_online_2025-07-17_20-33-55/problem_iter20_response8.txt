import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fill ratio, waste, feasibility, and dynamic bin selection."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item
    
    # Fill ratio score
    fill_ratios = item / bins_remain_cap
    
    # Residual capacity after insertion
    residual_capacity = bins_remain_cap - item
    
    # Relative residual capacity
    relative_residual = residual_capacity / item
    
    # Target relative residual and combine fill ratio
    target_relative_residual = 0.75
    priorities[feasible_bins] = fill_ratios[feasible_bins] / (1 + np.abs(relative_residual[feasible_bins] - target_relative_residual))

    #Almost Full Bins - Encourages near-full packing
    almost_full_threshold = 0.95
    almost_full_bins = (bins_remain_cap - item) / bins_remain_cap < (1-almost_full_threshold)
    priorities[feasible_bins & almost_full_bins] += 1

    # Low Waste Bins - Discourages excessive waste
    low_waste_threshold = 0.1
    low_waste_bins = (bins_remain_cap - item) / np.max(bins_remain_cap) < low_waste_threshold
    priorities[feasible_bins & low_waste_bins] -= 0.5

    # Infeasible penalty
    priorities[~feasible_bins] = -np.inf
    
    # Add small noise for differentiation
    priorities += np.random.rand(len(bins_remain_cap)) * 1e-9
    
    return priorities
