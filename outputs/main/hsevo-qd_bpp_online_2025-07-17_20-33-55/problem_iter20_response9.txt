import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fill ratio, adaptive waste bonus, and fullness bonus."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    # 1. Infeasible penalty
    priorities[~feasible_bins] = -np.inf

    # 2. Fill ratio maximization
    fill_ratios = item / bins_remain_cap
    priorities[feasible_bins] = fill_ratios[feasible_bins]

    # 3. Low waste bonus with adaptive scaling
    waste = bins_remain_cap - item
    normalized_waste = waste / bins_remain_cap
    low_waste_threshold = 0.1
    low_waste_bins = normalized_waste < low_waste_threshold
    bonus_scale = 0.2 * (1 - normalized_waste[low_waste_bins])  #Scale bonus inversely to waste
    priorities[feasible_bins & low_waste_bins] += bonus_scale

    # 4. Fullness Bonus: Prioritize bins that are closer to being full.
    priorities[feasible_bins] += 0.1 * (1 - normalized_waste[feasible_bins])

    # 5. Small random number for tie-breaking
    priorities += np.random.rand(len(bins_remain_cap)) * 1e-9

    return priorities
