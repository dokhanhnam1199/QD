import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fill ratio, waste minimization, and feasibility."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    # Infeasible penalty
    priorities[~feasible_bins] = -np.inf

    # Fill ratio
    fill_ratios = item / bins_remain_cap
    priorities[feasible_bins] = fill_ratios[feasible_bins]

    # Low waste bonus
    waste_ratio = (bins_remain_cap - item) / np.max(bins_remain_cap)
    low_waste_threshold = 0.2
    low_waste_bins = waste_ratio < low_waste_threshold
    priorities[feasible_bins & low_waste_bins] += 0.2

    # Add a small random number for tie-breaking
    priorities += np.random.rand(len(bins_remain_cap)) * 1e-9

    return priorities
