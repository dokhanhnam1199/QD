```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    can_fit = bins_remain_cap >= item
    priorities[~can_fit] = -np.inf

    if np.any(can_fit):
        diff = bins_remain_cap - item
        std_dev = item / 3
        priorities[can_fit] = np.exp(-(diff[can_fit]**2) / (2 * std_dev**2))

        # Tie-breaking bonus for emptier bins, scaling with item size
        max_cap = np.max(bins_remain_cap)
        if max_cap > 0:  # Avoid division by zero
            priorities[can_fit] += 0.1 * (bins_remain_cap[can_fit] / max_cap) * item
        else:
            priorities[can_fit] += 0 # if max_cap is zero just don't add anything
    return priorities
```
