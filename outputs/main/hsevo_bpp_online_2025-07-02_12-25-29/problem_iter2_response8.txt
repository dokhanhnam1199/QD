import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins by combining fit proximity and capacity."""

    difference = bins_remain_cap - item
    difference = np.where(difference >= 0, difference, np.inf)

    priorities = np.where(difference != np.inf, np.exp(-5 * difference) * (bins_remain_cap**0.5), -np.inf)

    return priorities
