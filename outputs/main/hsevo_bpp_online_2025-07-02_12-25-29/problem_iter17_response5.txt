import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on close fit, target fill, and early exit."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item

    # Early exit: If there's a bin that fits the item perfectly, prioritize it highly
    perfect_fit = np.isclose(bins_remain_cap, item)
    if np.any(perfect_fit):
        priorities[perfect_fit] = 10.0  # High priority for perfect fit
        return priorities

    if not np.any(feasible_bins):
        if len(priorities) > 0:
            priorities[:] = -1e9
            priorities[np.argmin(bins_remain_cap)] = 0
        return priorities

    close_fit = np.abs(bins_remain_cap - item)
    close_fit_priority = np.exp(-7.0 * close_fit)
    priorities += close_fit_priority * feasible_bins

    remaining_after_fit = bins_remain_cap - item
    fill_level = 1.0 - (remaining_after_fit / 1.0) #scale remaining capacity to 0-1
    target_fill = 0.75
    target_bonus = np.exp(-10.0 * np.abs(fill_level - target_fill))

    priorities += 0.5 * target_bonus * feasible_bins

    # Moderate penalty for leaving too much space
    bin_capacity = 1.0
    space_left_penalty = np.where(remaining_after_fit > 0.5 * bin_capacity, -0.5, 0.0) * feasible_bins
    priorities += space_left_penalty
    
    return priorities
