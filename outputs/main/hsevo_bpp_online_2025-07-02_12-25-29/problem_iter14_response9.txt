import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines target fill, fit, and fragmentation with no-fit handling."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    bin_capacity = 1.0

    # Early exit if item doesn't fit in any bin
    if np.all(bins_remain_cap < item):
        # Desperation strategy: put in smallest bin.
        min_cap_idx = np.argmin(bins_remain_cap)
        priorities[min_cap_idx] = 1.0
        return priorities

    item_fits = bins_remain_cap >= item
    remaining_after_fit = bins_remain_cap - item
    fill_level_after_fit = (bin_capacity - remaining_after_fit) / bin_capacity

    # Target fill bonus with exponential decay
    target_fill = 0.8
    fill_level_priority = np.exp(-5 * np.abs(fill_level_after_fit - target_fill)) * item_fits
    priorities += fill_level_priority

    # Fragmentation penalty
    fragmentation_threshold = 0.15
    fragmentation_penalty = np.where(item_fits & (remaining_after_fit/bin_capacity < fragmentation_threshold), -0.3, 0.0)
    priorities += fragmentation_penalty

    # Larger bin bonus (if item fits)
    priorities[item_fits] += 0.1

    #Waste calculation
    waste = np.where(item_fits, remaining_after_fit, np.inf)
    waste_priority = np.where(item_fits, np.exp(-2 * waste), 0)
    priorities += waste_priority

    return priorities
