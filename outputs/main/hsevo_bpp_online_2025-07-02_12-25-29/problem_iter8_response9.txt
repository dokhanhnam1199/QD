import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fit, fullness, and a target fullness, robustly."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item
    priorities[~feasible_bins] = -np.inf  # Infeasible bins get -inf

    if not np.any(feasible_bins):
        return priorities

    # Close fit priority
    close_fit = np.abs(bins_remain_cap[feasible_bins] - item)
    priorities[feasible_bins] = np.exp(-close_fit)

    # Fullness reward
    remaining_after_fit = bins_remain_cap[feasible_bins] - item
    nearly_full = np.exp(-np.abs(remaining_after_fit))
    priorities[feasible_bins] += nearly_full

    # Bonus for bins close to a target fullness after placement
    target_remaining = 0.1
    almost_full_bonus = np.exp(-np.abs(remaining_after_fit - target_remaining))
    priorities[feasible_bins] += almost_full_bonus * 0.5

    return priorities
