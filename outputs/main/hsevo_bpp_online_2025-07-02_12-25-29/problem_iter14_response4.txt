import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins: close fit, target fill, and no fit handling."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    item_fits = bins_remain_cap >= item

    if not np.any(item_fits):
        priorities[:] = -1e9
        priorities[np.argmin(bins_remain_cap)] = 0
        return priorities

    remaining_after_fit = bins_remain_cap - item
    # Close fit (nearly full)
    nearly_full = np.exp(-15 * np.abs(remaining_after_fit)) * item_fits
    priorities += 2.0 * nearly_full

    #Target fill.  Bin capacity is assumed 1.56 here
    fill_level = 1.0 - (remaining_after_fit / 1.56)
    target_fill = 0.8
    target_bonus = np.exp(-10.0 * np.abs(fill_level - target_fill))
    priorities += 0.5 * target_bonus * item_fits

    return priorities
