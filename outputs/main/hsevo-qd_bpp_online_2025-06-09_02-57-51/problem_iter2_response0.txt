import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines Gaussian priority with fill-up bonus and desperation handling."""

    priorities = np.zeros_like(bins_remain_cap)
    gaussian_width = np.mean(bins_remain_cap) / 5
    gaussian_priority = np.exp(-((bins_remain_cap - item)**2) / (2 * gaussian_width**2))

    remaining_after_add = bins_remain_cap - item
    valid_bins = remaining_after_add >= 0

    if np.any(valid_bins):
      fill_ratios = (bins_remain_cap[valid_bins] - remaining_after_add[valid_bins]) / bins_remain_cap[valid_bins]
      gaussian_priority[valid_bins] += 10*np.exp(-10*(remaining_after_add[valid_bins]/item)**2)
    else:
      min_cap = np.min(bins_remain_cap)
      gaussian_priority = (bins_remain_cap == min_cap).astype(float)

    randomness_strength = 0.001
    priorities = gaussian_priority + np.random.rand(len(bins_remain_cap)) * randomness_strength
    return priorities
