import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:

    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    
    can_fit = bins_remain_cap >= item
    
    if np.any(can_fit):
        remaining_space_after_fit = bins_remain_cap[can_fit] - item
        
        # Prioritize tight fits (smaller remaining space)
        priorities[can_fit] = 1.0 / (remaining_space_after_fit + 1e-9)  # Inverse of remaining space, avoid division by zero

        # Boost priority for bins that are almost full, but only if item fits
        almost_full = (bins_remain_cap[can_fit] < 1.5 * item) & can_fit[can_fit] #Slightly relaxed almost full condition
        priorities[can_fit][almost_full] *= 5  # Give stronger priority boost if almost full
        
        # Slightly encourage use of emptier bins, but not as much as tight fits
        normalized_remaining_capacity = bins_remain_cap[can_fit] / np.max(bins_remain_cap)
        priorities[can_fit] += 0.05 * normalized_remaining_capacity  # Add small bonus, decreased weight

        # Add some randomization
        priorities[can_fit] += np.random.rand(np.sum(can_fit)) * 0.1
    else:
        # if none of the bins fit, strongly penalize all bins
        priorities = -1e9 * np.ones_like(bins_remain_cap)
        diff = item - bins_remain_cap
        priorities = -diff

    return priorities
