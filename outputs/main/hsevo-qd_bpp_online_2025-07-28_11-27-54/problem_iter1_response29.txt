```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores based on item/bin size ratio for eligible bins.
    
    Bins that can fit the item are scored by the ratio of item size to bin's
    remaining capacity (higher ratio = better fit). Ineligible bins receive
    negative infinity to ensure they are not selected.
    """
    return np.where(
        bins_remain_cap >= item,
        item / bins_remain_cap,
        -np.inf
    )
```
