import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function implementing best-fit heuristic for online Bin Packing Problem.
    
    Bins that can fit the item are prioritized by the tightness of the fit (smallest leftover space).
    Bins that cannot fit the item receive negative infinity priority.
    
    Args:
        item: Size of item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
    
    Returns:
        Array of priority scores for each bin.
    """
    leftover = bins_remain_cap - item
    can_fit = leftover >= 0
    priority = np.where(can_fit, -leftover, -np.inf)  # Best-fit: minimize leftover
    return priority
