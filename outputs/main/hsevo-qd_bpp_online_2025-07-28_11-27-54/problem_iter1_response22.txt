import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit heuristic.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    valid_mask = bins_remain_cap >= item
    # Prioritize bins that can fit the item with minimal leftover space
    # This is equivalent to -(remaining_capacity - item) for valid bins
    priorities = np.where(valid_mask, -(bins_remain_cap - item), -np.inf)
    return priorities
