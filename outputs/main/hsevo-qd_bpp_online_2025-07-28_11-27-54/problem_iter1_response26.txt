```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on a Best Fit heuristic with exponential weighting.
    
    Bins that can fit the item are prioritized by how tightly they would pack it (exponentially 
    decaying reward for smaller leftover space). Invalid bins receive negative infinite priority.
    """
    remaining_after = bins_remain_cap - item
    valid = remaining_after >= 0
    # Exponentially reward smaller leftover space: priority = exp(-remaining_after)
    priority = np.where(valid, np.exp(-remaining_after), -np.inf)
    return priority
```
