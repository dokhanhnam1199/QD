import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, cannot_fit_score: float = -42480.31610371126) -> np.ndarray:
    """Returns priority scores for bins using a Best Fit heuristic.
    
    Bins that can fit the item are scored by how little space remains after placement.
    Bins that cannot fit the item receive a very low priority score.
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
        cannot_fit_score: The score assigned to bins that cannot fit the item.
    
    Returns:
        Array of priority scores for each bin.
    """
    can_fit = bins_remain_cap >= item
    fit_scores = item - bins_remain_cap
    return np.where(can_fit, fit_scores, cannot_fit_score)
