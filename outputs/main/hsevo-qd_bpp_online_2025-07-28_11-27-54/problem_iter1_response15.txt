import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function combining best-fit efficiency with a preference for more filled bins.
    
    Scores bins by rewarding minimal leftover space (best-fit) and penalizing large remaining
    capacity (encouraging consolidation into denser bins). Invalid bins (capacity < item) 
    receive -inf priority.
    """
    valid = bins_remain_cap >= item
    leftover = bins_remain_cap - item
    # Primary term: -leftover (prioritize minimal leftover)
    # Secondary term: -0.1*bins_remain_cap (favor bins already more filled)
    combined = -leftover - 0.1 * bins_remain_cap
    combined[~valid] = -np.inf
    return combined
