import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores based on minimizing squared space left after placement.
    
    Bins that can fit the item are prioritized by how closely they fit (smallest
    remaining space after placement). This is achieved by using negative squared
    space left as the priority score, which penalizes larger leftover space.
    """
    can_fit = bins_remain_cap >= item
    space_left = bins_remain_cap - item
    # Use negative squared space left for priorities of fitting bins, -inf otherwise
    return np.where(can_fit, -space_left**2, -np.inf)
