import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Adaptive scoring combining residual minimization and normalized fragmentation penalty.
    
    Uses feasible bin statistics and item-relative remaining space to balance space
    utilization and fragment avoidance via smooth exponential scoring.
    """
    feasible = bins_remain_cap >= item
    if not feasible.any():
        return np.full_like(bins_remain_cap, -np.inf, dtype=np.float64)
    
    r = bins_remain_cap - item
    feasible_r = r[feasible]
    
    mu = feasible_r.mean()
    std = feasible_r.std()
    eps = 1e-6
    tau = mu + std + eps  # Dynamic scale from central tendency and spread
    
    # Normalized remaining capacity relative to item size
    relative_r = r / (item + eps)
    
    # Smooth penalty term using normalized relative capacity
    penalty_term = np.exp(-relative_r / tau)
    
    # Adaptive weight balancing objectives
    beta = item / tau
    
    # Multi-objective score blending residual and fragmentation terms
    residual_objective = -r
    fragmentation_objective = -beta * penalty_term
    score = residual_objective + fragmentation_objective
    
    return np.where(feasible, score, -np.inf).astype(np.float64)
