import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function that favors bins leading to the tightest fit.
    
    Bins that can fit the item are prioritized by how small the remaining
    capacity will be after placing the item. This implements a Best Fit strategy.
    """
    remaining_after = bins_remain_cap - item
    mask = remaining_after >= 0
    # Use negative remaining_after to prioritize smaller leftover space
    priority = np.where(mask, -remaining_after, -np.inf)
    return priority
