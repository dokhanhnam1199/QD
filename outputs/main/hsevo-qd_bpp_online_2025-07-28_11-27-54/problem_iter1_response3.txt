import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function that prioritizes bins with remaining capacity just above the item size,
    while encouraging efficient packing through a ratio-based scoring system."""
    can_fit = bins_remain_cap >= item
    priority = np.full_like(bins_remain_cap, -np.inf)
    # Calculate priority as the ratio of item size to bin's remaining capacity.
    # This rewards bins where the item fills a larger proportion of available space.
    priority[can_fit] = item / bins_remain_cap[can_fit]
    return priority
