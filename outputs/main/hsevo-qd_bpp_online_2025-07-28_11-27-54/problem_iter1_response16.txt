import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority function implementing Best Fit heuristic for online Bin Packing.
    
    Bins that can fit the item are prioritized by how closely they match the item size,
    i.e., bins with remaining capacity just enough to accommodate the item receive the
    highest priority. This minimizes leftover space and aims to reduce the total number of bins used.
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
    
    Returns:
        Array of priority scores for each bin.
    """
    # Mask bins that can fit the item
    mask = bins_remain_cap >= item
    # Calculate priority: highest score for bins that can fit with minimal leftover space
    return np.where(mask, -(bins_remain_cap - item), -np.inf)
