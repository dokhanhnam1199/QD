```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority scores for bins based on Best Fit strategy.
    
    Higher priority is given to bins that minimize leftover space after placing the item.
    Bins that cannot fit the item are assigned -infinity priority.
    """
    leftover = bins_remain_cap - item
    valid = bins_remain_cap >= item
    priority = np.where(valid, -leftover, -np.inf)
    return priority
```
