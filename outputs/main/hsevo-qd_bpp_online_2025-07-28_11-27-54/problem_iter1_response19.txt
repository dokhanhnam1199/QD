```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority based on item size classification: best fit for large items, worst fit for small."""
    LARGE_ITEM_THRESHOLD = 0.5
    is_large = item > LARGE_ITEM_THRESHOLD
    can_fit = bins_remain_cap >= item
    
    if is_large:
        # Best fit: prioritize bins with smallest remaining capacity that can fit the item
        priority = np.where(can_fit, -(bins_remain_cap - item), -np.inf)
    else:
        # Worst fit: prioritize bins with largest remaining capacity that can fit the item
        priority = np.where(can_fit, bins_remain_cap, -np.inf)
    
    return priority
```
