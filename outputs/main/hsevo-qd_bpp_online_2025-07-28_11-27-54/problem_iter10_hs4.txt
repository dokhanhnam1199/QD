import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, non_fit_score= -927712.5583672648) -> np.ndarray:
    """Returns priority scores for bins using a Best Fit heuristic.
    
    Bins that can fit the item are scored by how little remaining space they have after placement.
    Bins that cannot fit the item receive a configurable low priority score.
    
    Args:
        item: Size of the item to be packed.
        bins_remain_cap: Array of remaining capacities for each bin.
        non_fit_score: Score for bins that cannot fit the item. Defaults to negative infinity.
    
    Returns:
        Array of priority scores for each bin.
    """
    can_fit = bins_remain_cap >= item
    fit_scores = item - bins_remain_cap
    return np.where(can_fit, fit_scores, non_fit_score)
