import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:

    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    bin_capacity = np.max(bins_remain_cap)

    # 1. Tight Fit Priority, Scaled by Item Size
    fit_mask = bins_remain_cap >= item
    if np.any(fit_mask):
        slack = bins_remain_cap[fit_mask] - item
        priorities[fit_mask] = 1 / (slack + 1e-6)
        priorities[fit_mask] *= (1 + 0.75 * item / bin_capacity)
    else:
        priorities[:] = -np.inf

    # 2. Refined Fill Level Penalties
    remaining_capacity = bins_remain_cap - item
    too_full_penalty = np.exp( -5 * (remaining_capacity / bin_capacity)) * (remaining_capacity >= 0)
    too_empty_penalty = np.exp(-5 * (bins_remain_cap / bin_capacity)) * (remaining_capacity >= 0) * (bins_remain_cap > bin_capacity * 0.35) 
    
    priorities -= (too_full_penalty + too_empty_penalty)

    # 3. Dynamic Occupancy-Based Reward
    occupied_ratio = (bin_capacity - bins_remain_cap) / bin_capacity
    moderate_occupancy = np.logical_and(occupied_ratio > 0.15, occupied_ratio < 0.85)
    priorities[moderate_occupancy] += 0.15 * (1 - np.abs(occupied_ratio[moderate_occupancy] - 0.5) / 0.5)

    # 4. Encourage Filling Partially Full Bins (Dynamically Scaled)
    filled_threshold = 0.2 * bin_capacity
    already_filled = (bin_capacity - bins_remain_cap) > filled_threshold
    priorities[already_filled] += 0.05 * (item / bin_capacity)

    return priorities
