import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Random Fit priority for online Bin Packing.

    Returns a random priority for each bin that can accommodate the item.
    Bins that cannot fit the item receive -inf priority to ensure they are never chosen.

    Args:
        item: Size of the incoming item.
        bins_remain_cap: Array of remaining capacities for each open bin.

    Returns:
        ndarray of priority scores, one per bin.
    """
    # Feasibility mask: bins that can fit the item
    feasible = bins_remain_cap >= item

    # Sample a uniform random number for each bin
    rand_scores = np.random.rand(bins_remain_cap.shape[0])

    # Infeasible bins get -inf priority (worst possible)
    priorities = np.where(feasible, rand_scores, -np.inf)

    return priorities
