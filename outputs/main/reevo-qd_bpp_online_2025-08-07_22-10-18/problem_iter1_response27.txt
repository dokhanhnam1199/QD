import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Random Fit priority function for online bin packing.

    For each bin that has enough remaining capacity for the incoming item, a
    random priority value in the range [0, 1) is assigned. Bins that cannot
    accommodate the item receive a priority of -inf, ensuring they are never
    selected. This effectively implements Random Fit: the algorithm randomly
    selects one of the eligible bins.

    Args:
        item: Size of the item to pack.
        bins_remain_cap: 1‑D array of remaining capacities for each bin.

    Returns:
        A 1‑D array of priority scores with the same shape as bins_remain_cap.
    """
    # Eligible bins are those whose remaining capacity is at least the item size
    eligible = bins_remain_cap >= item

    # Assign random scores to eligible bins; others get -inf so they will be ignored
    random_scores = np.where(
        eligible,
        np.random.rand(bins_remain_cap.size),  # random in [0, 1)
        -np.inf
    )

    return random_scores
