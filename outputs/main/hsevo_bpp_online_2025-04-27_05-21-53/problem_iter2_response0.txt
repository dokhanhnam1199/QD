import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines capacity utilization and waste minimization."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    for i, cap in enumerate(bins_remain_cap):
        if cap >= item:
            new_cap = cap - item
            fill_ratio = 1 - (new_cap / cap)
            waste = new_cap 
            priorities[i] = fill_ratio**2  + np.exp(-waste) 
        else:
            priorities[i] = -1e9

    return priorities
