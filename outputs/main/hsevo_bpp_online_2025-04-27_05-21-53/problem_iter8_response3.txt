import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins using fullness, exploration, and adaptive bonuses."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    remaining_capacity = bins_remain_cap - item

    # Infeasible bins get the lowest priority
    priorities[remaining_capacity < 0] = -np.inf

    feasible_bins = remaining_capacity >= 0

    if np.any(feasible_bins):
        # 1. Fullness score (Fill Ratio)
        utilization = item / bins_remain_cap[feasible_bins]
        priorities[feasible_bins] += utilization

        # 2. Exploration Bonus: Adaptive based on remaining capacity.
        exploration_factor = (bins_remain_cap[feasible_bins] / np.max(bins_remain_cap[feasible_bins]))
        priorities[feasible_bins] += 0.3 * exploration_factor

        # 3. Almost Full Bonus:  Adaptive threshold based on item size.
        almost_full_threshold = 0.1 * item  # Smaller items need tighter fits
        almost_full = feasible_bins & (remaining_capacity >= 0) & (remaining_capacity <= almost_full_threshold)
        priorities[almost_full] += 1.2

        # 4. Fragmentation Penalty: Discourage small remainders, scaled by item size
        small_remainder_threshold = 0.2 * item
        small_remainder = feasible_bins & (remaining_capacity > 0) & (remaining_capacity <= small_remainder_threshold)
        priorities[small_remainder] -= 0.8

        # 5. Perfect Fit Bonus: Large bonus for perfect fits
        perfect_fit = np.abs(remaining_capacity) < 1e-6
        if np.any(perfect_fit & feasible_bins):
            priorities[feasible_bins][perfect_fit[feasible_bins]] += 10

    return priorities
