import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines Best Fit with a soft penalty for waste, inspired by Softmax-Fit.
    Prioritizes bins that leave minimal remaining capacity, with a boost for perfect fits.
    """
    valid_bins_mask = bins_remain_cap >= item
    output_priorities = np.zeros_like(bins_remain_cap, dtype=float)

    if not np.any(valid_bins_mask):
        return output_priorities

    valid_bins_cap = bins_remain_cap[valid_bins_mask]

    # Calculate a score: inverse of remaining capacity after fit, favoring tighter fits.
    # Add a small epsilon to prevent division by zero.
    fit_scores = 1.0 / (valid_bins_cap - item + 1e-9)

    # Introduce a bonus for perfect fits to strongly prioritize them.
    perfect_fit_mask = (valid_bins_cap - item) < 1e-9
    fit_scores[perfect_fit_mask] *= 5.0  # Significant bonus for perfect fits

    # Apply softmax to normalize scores into probabilities/priorities.
    # This provides a smoother distribution than pure inverse,
    # while still emphasizing bins with higher fit_scores.
    exp_scores = np.exp(fit_scores)
    priorities = exp_scores / np.sum(exp_scores)

    output_priorities[valid_bins_mask] = priorities

    return output_priorities
