import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization with a worst-fit penalty, handling empty bins."""
    possible_bins = bins_remain_cap >= item
    if np.sum(possible_bins) == 0:
        return np.zeros_like(bins_remain_cap)
    waste = bins_remain_cap[possible_bins] - item
    sigmoid_scale = 5.0
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = 1.0 / (1 + np.exp(-waste * sigmoid_scale))
    best_bin_index = np.argmin(waste)
    worst_bin_index = np.argmax(waste)
    priorities[possible_bins][best_bin_index] += 1.0
    priorities[possible_bins][worst_bin_index] -= 0.5
    return priorities
