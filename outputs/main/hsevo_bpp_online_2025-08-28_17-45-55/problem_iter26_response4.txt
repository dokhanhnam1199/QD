import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with a mild worst-fit penalty and exploration."""
    n_bins = len(bins_remain_cap)
    priorities = np.zeros(n_bins)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities

    waste = bins_remain_cap[valid_bins] - item
    priorities[valid_bins] = 1.0 / (waste + 1e-9)

    worst_bin_index = np.argmax(waste)
    priorities[np.where(valid_bins)[0][worst_bin_index]] += 0.1

    exploration_rate = 0.1
    if np.random.rand() < exploration_rate:
        random_bin_index = np.random.choice(n_bins)
        priorities[random_bin_index] = np.max(priorities) + 1

    return priorities
