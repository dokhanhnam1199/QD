import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fit and minimizes waste using a sigmoid function."""
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return np.zeros_like(bins_remain_cap)

    fit_levels = item / bins_remain_cap[valid_bins]
    priorities = np.zeros_like(bins_remain_cap)
    priorities[valid_bins] = 1 / (1 + np.exp(-5 * (1 - fit_levels)))
    return priorities
