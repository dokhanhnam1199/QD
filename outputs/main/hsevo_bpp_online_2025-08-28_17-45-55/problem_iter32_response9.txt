import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with sigmoid scaling and a dynamic exploration boost."""
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return np.zeros_like(bins_remain_cap)

    normalized_waste = (bins_remain_cap[valid_bins] - item) / bins_remain_cap[valid_bins]
    waste_std = np.std(normalized_waste)
    sigmoid_scale = 5.0
    sigmoid_input = normalized_waste * sigmoid_scale + waste_std
    sigmoid_scaled_waste = 1 / (1 + np.exp(-sigmoid_input))
    priorities = np.zeros_like(bins_remain_cap)
    priorities[valid_bins] = sigmoid_scaled_waste
    best_bin_index = np.argmin(bins_remain_cap[valid_bins] - item)
    priorities[valid_bins][best_bin_index] += 1.0
    return priorities
