import numpy as np

sigmoid_scale = 1.0
best_bin_boost = 1.0

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return np.zeros_like(bins_remain_cap)

    waste = bins_remain_cap[valid_bins] - item
    std_dev_waste = np.std(waste) if len(waste) > 1 else 0.0
    
    priorities = np.zeros_like(bins_remain_cap)
    
    if std_dev_waste > 0:
        sigmoid_scaled_waste = 1 / (1 + np.exp(-waste * sigmoid_scale / std_dev_waste))
    else:
        sigmoid_scaled_waste = np.ones_like(waste)
    
    priorities[valid_bins] = sigmoid_scaled_waste
    
    best_bin_index = np.argmin(waste)
    priorities[np.where(valid_bins)[0][best_bin_index]] += best_bin_boost
    
    return priorities
