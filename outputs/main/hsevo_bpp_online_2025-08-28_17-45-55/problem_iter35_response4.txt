import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with modulated sigmoid scaling based on waste standard deviation and a best bin boost."""
    sigmoid_scale = 1.0
    best_bin_boost = 1.0
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if np.any(valid_bins):
        waste = bins_remain_cap[valid_bins] - item
        waste_std = np.std(waste)
        sigmoid_mod = 1.0 + waste_std
        sigmoid_scaled_waste = 1 / (1 + np.exp(-waste * sigmoid_scale * sigmoid_mod))
        priorities[valid_bins] = sigmoid_scaled_waste
        best_bin_index = np.argmin(waste)
        priorities[np.where(valid_bins)[0][best_bin_index]] += best_bin_boost
    return priorities
