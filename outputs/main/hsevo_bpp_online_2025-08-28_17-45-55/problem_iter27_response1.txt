import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Enhanced priority function for online bin packing."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if np.any(valid_bins):
        waste = bins_remain_cap[valid_bins] - item
        best_bin_index = np.argmin(waste)
        sigmoid_scaled_waste = 1 / (1 + np.exp(-waste * 2))
        priorities[valid_bins] = sigmoid_scaled_waste
        
        penalty = 0.1
        for i in range(len(valid_bins)):
            if i != best_bin_index:
                priorities[valid_bins][i] -= penalty * waste[i]

        priorities[np.where(valid_bins)[0][best_bin_index]] += 1.0
    return priorities
