import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, bin_capacity: float) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        bin_capacity: Capacity of each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Mask for bins that can accommodate the item
    valid_bins_mask = remaining_cap_after_addition >= 0
    
    # Assign high priority to valid bins and low priority to invalid bins
    priorities = np.where(valid_bins_mask, remaining_cap_after_addition, -np.inf)
    
    # Normalize and apply non-linear transformation to priorities for valid bins
    valid_priorities = priorities[valid_bins_mask]
    if len(valid_priorities) > 0:
        min_priority = np.min(valid_priorities)
        max_priority = np.max(valid_priorities)
        if max_priority != min_priority:
            normalized_priorities = (valid_priorities - min_priority) / (max_priority - min_priority)
            # Apply a non-linear transformation to amplify differences
            transformed_priorities = np.power(normalized_priorities, 3)
            # Weigh the priorities by the fraction of bin capacity used
            fraction_used = (bin_capacity - valid_priorities - item) / bin_capacity
            weighted_priorities = transformed_priorities * fraction_used
            priorities = np.where(valid_bins_mask, weighted_priorities, priorities)
    
    return priorities
