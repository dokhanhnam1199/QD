import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, bin_capacity: float) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of remaining capacities for each bin.
        bin_capacity: The capacity of each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_after_addition = bins_remain_cap - item
    
    # Bins that can fit the item get a higher priority
    can_fit = remaining_after_addition >= 0
    
    # Prioritize bins that have just enough capacity for the item
    priorities = np.where(can_fit, -np.abs(remaining_after_addition), -np.inf)
    
    # Calculate the ratio of item size to bin remaining capacity
    epsilon = 1e-6
    ratios = item / (bins_remain_cap + epsilon)
    
    # Calculate the fill rate of the bin before adding the item
    fill_rates = 1 - bins_remain_cap / bin_capacity
    
    # Combine the three factors: the ability to fit the item, the ratio of item size to bin remaining capacity, and the fill rate
    priorities = np.where(can_fit, priorities + np.log(1 + ratios) + np.log(1 + fill_rates), priorities)
    
    return priorities
