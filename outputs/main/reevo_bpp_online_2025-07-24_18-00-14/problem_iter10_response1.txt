import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, bin_capacity: float) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        bin_capacity: The capacity of each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Calculate the remaining capacity after adding the item
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Identify bins that can accommodate the item
    valid_bins_mask = remaining_cap_after_addition >= 0
    
    # Assign a base priority score to each bin based on its remaining capacity ratio and item size ratio
    base_priority = np.where(valid_bins_mask, 
                              (1 - remaining_cap_after_addition / bin_capacity) * (item / bin_capacity), 
                              -np.inf)
    
    # Apply a non-linear transformation to the base priority for valid bins to amplify differences
    priorities = np.where(valid_bins_mask, np.exp(base_priority), base_priority)
    
    # Normalize the priorities for valid bins to ensure they are within a comparable range
    valid_priorities = priorities[valid_bins_mask]
    if len(valid_priorities) > 0:
        min_priority = np.min(valid_priorities)
        max_priority = np.max(valid_priorities)
        if max_priority != min_priority:
            priorities = np.where(valid_bins_mask, (priorities - min_priority) / (max_priority - min_priority + 1e-8), priorities)
    
    return priorities
