import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines space-time curvature and log ratios for priority.
    Prioritizes near-full bins capable of holding the item, 
    balancing capacity utilization and item fit."""
    bins_remain_cap = np.where(bins_remain_cap <= 0, 1e-9, bins_remain_cap)

    gravitational_effect = item / bins_remain_cap**2
    fill_factor = (1 - (bins_remain_cap - item) / bins_remain_cap)
    fill_factor = np.where(bins_remain_cap < item, -np.inf, fill_factor)

    log_ratios = np.log(item / bins_remain_cap)
    
    priorities = gravitational_effect + fill_factor - 0.5 * log_ratios
    return priorities
