```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines gravitational effect and log ratio for bin priority."""

    # Avoid division by zero
    bins_remain_cap = np.where(bins_remain_cap <= 0, 1e-9, bins_remain_cap)

    # Gravitational effect, prioritizing bins with smaller remaining capacity.
    gravitational_effect = item / bins_remain_cap**2

    # Log ratio of item size to remaining capacity.
    ratios = item / bins_remain_cap
    log_ratios = np.log(ratios)

    # Combine both effects with a weighting.
    priorities = 0.7 * gravitational_effect - 0.3 * log_ratios

    return priorities
```
