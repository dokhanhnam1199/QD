import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fill ratio, remaining capacity, and waste minimization."""
    min_capacity = 1e-9
    bins_remain_cap = np.where(bins_remain_cap <= 0, min_capacity, bins_remain_cap)

    fill_ratio = item / bins_remain_cap
    fill_ratio = np.where(bins_remain_cap < item, -np.inf, fill_ratio)

    remaining_capacity_utilization = bins_remain_cap / np.sum(bins_remain_cap)

    waste = np.abs(bins_remain_cap - item)

    priorities = fill_ratio + remaining_capacity_utilization - waste
    return priorities
