import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines fill ratio, remaining capacity, and feasibility."""
    min_capacity = 1e-9
    bins_remain_cap = np.where(bins_remain_cap <= 0, min_capacity, bins_remain_cap)

    # Fill ratio, penalize infeasible bins
    fill_ratio = item / bins_remain_cap
    fill_ratio = np.where(bins_remain_cap < item, -np.inf, fill_ratio)

    # Remaining capacity utilization
    capacity_utilization = 1 - (bins_remain_cap - item) / bins_remain_cap
    capacity_utilization = np.where(bins_remain_cap < item, -np.inf, capacity_utilization)

    priorities = fill_ratio + capacity_utilization
    return priorities
