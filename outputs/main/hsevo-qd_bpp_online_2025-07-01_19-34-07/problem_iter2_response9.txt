```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines gravitational effect and log ratio for priority."""
    # Gravitational effect: item size relative to remaining capacity squared.
    bins_remain_cap = np.where(bins_remain_cap <= 0, 1e-9, bins_remain_cap)
    gravitational_effect = item / bins_remain_cap**2

    # Log ratio: Logarithm of the ratio of item size to remaining capacity.
    ratios = item / bins_remain_cap
    log_ratios = np.log(ratios)

    # Combine effects
    priorities = gravitational_effect - log_ratios
    return priorities
```
