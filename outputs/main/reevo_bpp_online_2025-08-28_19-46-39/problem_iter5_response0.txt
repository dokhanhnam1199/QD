import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, epsilon: float = 0.1) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.
    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.
        epsilon: Exploration rate.
    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap)
    eligible_bins = bins_remain_cap >= item

    if np.any(eligible_bins):
        # Prioritize tightest fit (smallest remaining capacity after placing item)
        fit_scores = bins_remain_cap[eligible_bins] - item
        best_fit_bin_index = np.argmin(fit_scores)
        best_fit_bin_index = np.where(eligible_bins)[0][best_fit_bin_index]
        priorities[best_fit_bin_index] = 1.0

        # Epsilon-greedy exploration: assign random priority to other eligible bins
        if np.random.rand() < epsilon:
            other_eligible_bins = eligible_bins.copy()
            other_eligible_bins[best_fit_bin_index] = False  # Exclude best fit
            if np.any(other_eligible_bins):
                random_bin_indices = np.where(other_eligible_bins)[0]
                random_index = np.random.choice(random_bin_indices)
                priorities[random_index] = np.random.rand()

    return priorities
