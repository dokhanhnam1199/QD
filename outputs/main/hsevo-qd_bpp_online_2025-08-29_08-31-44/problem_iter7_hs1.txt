import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray,
                epsilon: float = 8.99618466852198e-07,
                sigmoid_weight: float = 1.2543832576085587,
                sigmoid_threshold: float = 0.6698816139068566) -> np.ndarray:
    """Combines best-fit with reciprocal waste for improved bin packing."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    waste = bins_remain_cap[valid_bins] - item
    priorities[valid_bins] = 1.0 / (waste + epsilon)
    fit_scores = (bins_remain_cap[valid_bins] - item) / bins_remain_cap[valid_bins]
    sigmoid_scores = 1 / (1 + np.exp(-sigmoid_weight * (fit_scores - sigmoid_threshold)))
    priorities[valid_bins] = priorities[valid_bins] * sigmoid_scores
    return priorities
