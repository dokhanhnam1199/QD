import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with reciprocal waste, boosting the best-fit bin."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    waste = bins_remain_cap[valid_bins] - item
    priorities[valid_bins] = 1.0 / (waste + 1e-9)
    best_fit_index = np.argmin(bins_remain_cap[valid_bins] - item)
    priorities[valid_bins] += 0.5 * (1.0 / (bins_remain_cap[valid_bins] + 1e-9))
    priorities[np.where(valid_bins)[0][best_fit_index]] += 2.0
    return priorities
