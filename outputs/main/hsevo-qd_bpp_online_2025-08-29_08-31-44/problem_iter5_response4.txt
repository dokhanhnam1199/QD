import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with a sigmoid function based on bin fill ratio."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities

    distances = bins_remain_cap[valid_bins] - item
    fit_scores = (bins_remain_cap[valid_bins] - item) / item
    bin_utilizations = bins_remain_cap[valid_bins] / bins_remain_cap[valid_bins]
    sigmoid_scores = 1 / (1 + np.exp(-5 * (bin_utilizations - 0.5)))
    combined_scores = 0.7 * sigmoid_scores + 0.3 * fit_scores
    priorities[valid_bins] = combined_scores
    return priorities
