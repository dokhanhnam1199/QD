import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, small_value: float = 0.0007839633992636663) -> np.ndarray:
    """
    Combines best-fit (reciprocal of waste) with a validity check.
    Prioritizes bins that fit the item and minimize remaining space.
    """
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if np.sum(valid_bins) == 0:
        return priorities
    priorities[valid_bins] = 1.0 / (bins_remain_cap[valid_bins] - item + small_value)
    return priorities
