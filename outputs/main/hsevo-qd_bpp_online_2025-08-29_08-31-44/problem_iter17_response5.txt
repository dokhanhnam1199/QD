import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines reciprocal waste with a sigmoid-scaled fit score for robust bin selection."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    waste = bins_remain_cap[valid_bins] - item
    fit = item / bins_remain_cap[valid_bins]
    sigmoid_score = 1.0 / (1.0 + np.exp(-5 * (fit - 0.5)))
    priorities[valid_bins] = 1.0 / (waste + 1e-9) + sigmoid_score * 0.5
    return priorities
