import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    priorities = np.zeros_like(bins_remain_cap)
    feasible_bins = bins_remain_cap >= item
    if not np.any(feasible_bins):
        return priorities

    fill_ratios = bins_remain_cap[feasible_bins] / (bins_remain_cap[feasible_bins] + item)
    
    priorities[feasible_bins] = np.exp(-((bins_remain_cap[feasible_bins] - item) / item)**2) * fill_ratios
    
    best_bin_index = np.argmax(priorities[feasible_bins])
    best_bin_index_abs = np.where(feasible_bins)[0][best_bin_index]
    
    priorities[best_bin_index_abs] += 1.0

    return priorities
