```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit reciprocal waste with a smoothed bin utilization score."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item

    if np.sum(valid_bins) == 0:
        return priorities

    waste = bins_remain_cap[valid_bins] - item + 1e-6
    fit_score = 1.0 / waste
    fill_ratio = bins_remain_cap[valid_bins] / np.sum(bins_remain_cap)
    sigmoid_fill = 1.0 / (1.0 + np.exp(-fill_ratio))
    
    priorities[valid_bins] = 0.6 * fit_score + 0.4 * sigmoid_fill
    
    return priorities
```
