import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with sigmoid-based fill ratio for adaptive prioritization."""
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)
    remaining_capacities = bins_remain_cap[possible_bins]
    fit_scores = 1.0 / (1.0 + np.exp(-5 * (remaining_capacities - item)))
    bin_utilizations = bins_remain_cap[possible_bins] / bins_remain_cap.max()
    sigmoid_scores = 1 / (1 + np.exp(-5 * (bin_utilizations - 0.5)))
    combined_scores = fit_scores * sigmoid_scores
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = combined_scores
    return priorities
