import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with a sigmoid-scaled waste to prioritize bins."""
    priorities = np.zeros_like(bins_remain_cap)
    suitable_bins = bins_remain_cap >= item
    if np.any(suitable_bins):
        waste = bins_remain_cap[suitable_bins] - item
        priorities[suitable_bins] = 1.0 / (waste + 1e-6)
        fill_ratios = (bins_remain_cap[suitable_bins] - item) / bins_remain_cap[suitable_bins]
        sigmoid_scores = 1.0 / (1.0 + np.exp(-5 * (fill_ratios - 0.5)))
        best_fit_index = np.argmin(waste)
        priorities[suitable_bins] = np.where(np.arange(len(suitable_bins)) == best_fit_index, priorities[suitable_bins] + sigmoid_scores, priorities[suitable_bins])
    return priorities
