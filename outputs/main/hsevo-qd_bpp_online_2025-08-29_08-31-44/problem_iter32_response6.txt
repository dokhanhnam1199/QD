import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit (reciprocal of waste) with a sigmoid-scaled utilization score."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if np.sum(valid_bins) == 0:
        return priorities

    priorities[valid_bins] = 1.0 / (bins_remain_cap[valid_bins] - item + 1e-6)
    fill_ratios = (bins_remain_cap - item) / bins_remain_cap
    sigmoid_scores = 1.0 / (1.0 + np.exp(-5 * (fill_ratios - 0.5)))
    priorities[valid_bins] += 0.5 * sigmoid_scores[valid_bins]
    return priorities
