import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines sigmoid fit score with a reciprocal waste component."""
    possible_bins = bins_remain_cap >= item
    if not np.any(possible_bins):
        return np.zeros_like(bins_remain_cap)
    
    fit_scores = (bins_remain_cap[possible_bins] - item) / bins_remain_cap[possible_bins]
    sigmoid_scores = 1 / (1 + np.exp(-5 * (fit_scores - 0.5)))
    
    waste_scores = 1 / (item + 1e-6)
    
    priorities = np.zeros_like(bins_remain_cap)
    priorities[possible_bins] = 0.7 * sigmoid_scores + 0.3 * waste_scores
    
    return priorities
