import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes best-fit bin and boosts it with a sigmoid function."""
    priorities = np.zeros_like(bins_remain_cap)
    feasible_bins = bins_remain_cap >= item
    if not np.any(feasible_bins):
        return priorities

    sorted_bins = np.argsort(bins_remain_cap[feasible_bins])
    best_fit_index = sorted_bins[0]
    
    priorities[feasible_bins] = 1.0 / (bins_remain_cap[feasible_bins] - item + 1e-6)
    
    priorities[np.where(feasible_bins)[0][best_fit_index]] += 0.5
    
    return priorities
