import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with a sigmoid-based fill ratio prioritization and reciprocal waste."""
    priorities = np.zeros_like(bins_remain_cap)
    feasible_bins = bins_remain_cap >= item
    if not np.any(feasible_bins):
        return priorities

    fill_ratios = (bins_remain_cap[feasible_bins] - item) / bins_remain_cap[feasible_bins]
    sigmoid_scores = 1.0 / (1 + np.exp(-5 * (0.5 - fill_ratios)))
    waste = bins_remain_cap[feasible_bins] - item
    reciprocal_waste = 1.0 / (waste + 1e-9)

    priorities[feasible_bins] = 0.7 * sigmoid_scores + 0.3 * reciprocal_waste
    best_fit_index = np.argmin(bins_remain_cap[feasible_bins] - item)
    priorities[np.where(feasible_bins)[0][best_fit_index]] += 1.0

    return priorities
