import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit with reciprocal waste for improved bin packing."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    waste = bins_remain_cap[valid_bins] - item
    priorities[valid_bins] = 1.0 / (waste + 1e-9)
    fit_scores = (bins_remain_cap[valid_bins] - item) / bins_remain_cap[valid_bins]
    sigmoid_scores = 1 / (1 + np.exp(-5 * (fit_scores - 0.5)))
    priorities[valid_bins] = priorities[valid_bins] * sigmoid_scores
    return priorities
