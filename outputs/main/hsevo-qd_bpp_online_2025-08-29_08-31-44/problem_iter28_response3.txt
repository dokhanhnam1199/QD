import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, weight_feasible: float = 6.505319006384634, weight_best_fit: float = 7.964486371100595) -> np.ndarray:
    priorities = np.zeros_like(bins_remain_cap)
    feasible_bins = bins_remain_cap >= item
    if np.any(feasible_bins):
        best_fit_index = np.argmin(bins_remain_cap[feasible_bins] - item)
        priorities[feasible_bins] = weight_feasible
        priorities[np.where(bins_remain_cap >= item)[0][best_fit_index]] = weight_best_fit
    return priorities
