import numpy as np
import random

def heuristics_v2(distance_matrix, coordinates, demands, capacity):
    """
    {This algorithm generates a set of random routes satisfying capacity constraints, evaluates the routes using a score function that prioritizes routes with fewer vehicles and lower total distance, and uses the frequency of edges appearing in high-scoring routes to create a heuristic matrix.}
    """
    n = len(demands)
    num_samples = 1000
    heuristics_matrix = np.zeros_like(distance_matrix)
    route_scores = []
    routes = []

    for _ in range(num_samples):
        # Generate a random route
        unvisited = list(range(1, n))
        current_route = [0]
        current_capacity = capacity
        route = []
        while unvisited:
            next_node = random.choice(unvisited)
            if demands[next_node] <= current_capacity:
                current_route.append(next_node)
                current_capacity -= demands[next_node]
                unvisited.remove(next_node)
            else:
                current_route.append(0)
                route.append(current_route)
                current_route = [0, next_node]
                current_capacity = capacity - demands[next_node]
                unvisited.remove(next_node)
        current_route.append(0)
        route.append(current_route)

        # Calculate the cost of the route
        cost = 0
        for r in route:
            for i in range(len(r) - 1):
                cost += distance_matrix[r[i]][r[i+1]]
        
        num_vehicles = len(route)
        
        # Define a score function that considers both cost and number of vehicles
        # A higher score indicates a better route
        score = 1 / (cost + 1e-6) + 1 / (num_vehicles + 1e-6) # adding small number to avoid division by zero

        route_scores.append(score)
        routes.append(route)

    # Normalize route scores
    normalized_scores = np.array(route_scores)
    
    # Calculate the median score
    median_score = np.median(normalized_scores)
    
    # Identify good routes (those with score >= median)
    good_route_indices = np.where(normalized_scores >= median_score)[0]
    
    # Update heuristics matrix based on good routes
    for index in good_route_indices:
        route = routes[index]
        for r in route:
             for i in range(len(r) - 1):
                heuristics_matrix[r[i]][r[i+1]] += 1
                heuristics_matrix[r[i+1]][r[i]] += 1  # Symmetry

    return heuristics_matrix
