import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority for each bin using worst-fit: highest remaining capacity that can accommodate the item."""
    feasible = bins_remain_cap >= item
    priority = np.where(feasible, bins_remain_cap, -np.inf)
    jitter = np.random.rand(*bins_remain_cap.shape) * 1e-12
    priority = np.where(feasible, priority + jitter, priority)
    return priority
