```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    low = bins_remain_cap - item
    mask = low >= 0
    priorities = np.zeros_like(bins_remain_cap)
    eps = 0.01
    k = 10.0
    sub = eps - low[mask]
    priorities[mask] = 1.0 / (1.0 + np.exp(-k * sub))
    return priorities
```
