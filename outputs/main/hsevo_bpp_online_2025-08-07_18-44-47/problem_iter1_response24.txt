import numpy as np

def priority_v2(item: float, priority_strategy="ExactFitFirst", bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin using Exact Fit First.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    feasible = bins_remain_cap >= item
    leftovers = bins_remain_cap - item
    priorities = np.where(feasible, -leftovers, -np.inf)
    exact_fit = bins_remain_cap == item
    if np.any(exact_fit):
        priorities[exact_fit] = np.inf
    return priorities
