import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Score bins: inverse residual capacity with tiny random tie‑breaker; infeasible → -inf."""
    priors = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    feasible = bins_remain_cap >= item
    if not np.any(feasible):
        return priors
    residual = bins_remain_cap[feasible] - item
    epsilon = 1e-9
    base = 1.0 / (residual + epsilon)
    rng = np.random.default_rng()
    noise = rng.random(base.shape) * 1e-6
    priors[feasible] = base + noise
    return priors
