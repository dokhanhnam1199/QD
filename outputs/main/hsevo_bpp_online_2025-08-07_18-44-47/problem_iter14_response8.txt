import numpy as np

_item_counter = 0

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority based on leftover capacity with Gumbel temperature decay, item-size jitter, and deterministic index tieâ€‘breaker."""
    global _item_counter
    _item_counter += 1
    bins = np.asarray(bins_remain_cap, dtype=float)
    n = bins.size
    priority = np.full(n, -np.inf, dtype=float)
    if n == 0:
        return priority
    remaining = bins - item
    feasible = remaining >= 0
    if not np.any(feasible):
        return priority
    eps = 1e-12
    base_temp = 0.5
    decay = 0.001
    temp = max(0.05, base_temp * np.exp(-decay * _item_counter))
    rng = np.random.default_rng()
    gumbel = -np.log(-np.log(rng.random(feasible.sum())))
    idx = np.arange(n)
    jitter = rng.random(feasible.sum()) * 0.01 * item
    priority[feasible] = -remaining[feasible] / temp + gumbel + jitter + idx[feasible] * 1e-12 - idx[feasible] * 1e-6
    return priority
