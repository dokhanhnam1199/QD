import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Compute bin priorities using inverse leftover, index bias, deterministic goldenâ€‘ratio jitter and adaptive random jitter."""
    caps = np.asarray(bins_remain_cap, dtype=float)
    n = caps.size
    if n == 0:
        return np.empty(0, dtype=float)
    eps = 1e-9
    remaining = caps - item
    feasible = remaining >= 0
    priority = np.full(n, -np.inf, dtype=float)
    priority[feasible] = 1.0 / (remaining[feasible] + eps)
    idx = np.arange(n, dtype=float)
    priority[feasible] -= idx[feasible] * 1e-5
    phi = 0.6180339887498949
    jitter_det = np.mod(phi * (idx + 1) * item, 1.0) * 0.01 * item
    priority[feasible] += jitter_det[feasible]
    max_cap = caps.max()
    rng = np.random.default_rng()
    jitter_rand = rng.random(n) * 0.01 * item * (remaining + eps) / (max_cap + eps)
    priority[feasible] += jitter_rand[feasible]
    return priority
