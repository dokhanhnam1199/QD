import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    temperature = 0.05
    residual = bins_remain_cap - item
    mask = residual >= 0
    scores = np.zeros_like(bins_remain_cap, dtype=np.float64)
    scores[mask] = np.exp(-residual[mask] / temperature)
    total = scores.sum()
    if total > 0:
        scores = scores / total
    return scores
