import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    epsilon = 1e-9
    leftover = bins_remain_cap - item
    fit = bins_remain_cap >= item
    base = np.where(fit, 1.0/(leftover+epsilon), -np.inf)
    noise = np.random.rand(len(bins_remain_cap)) * 0.1 * item
    priorities = base + noise
    return priorities
