import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """First-Fit priority: earlier bins with sufficient space get higher score."""
    can_fit = bins_remain_cap >= item
    idx = np.arange(bins_remain_cap.shape[0])
    priorities = np.where(can_fit, bins_remain_cap.shape[0] - idx, 0.0)
    return priorities
