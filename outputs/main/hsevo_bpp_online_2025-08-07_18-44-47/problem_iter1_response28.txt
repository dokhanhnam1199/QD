import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Epsilonâ€‘greedy priority for online bin packing."""
    epsilon = 0.12
    base = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    feasible = bins_remain_cap >= item
    base[feasible] = 1.0 / (bins_remain_cap[feasible] - item + 1e-12)
    rand = np.random.rand(bins_remain_cap.shape[0])
    priorities = (1 - epsilon) * base + epsilon * rand
    return priorities
