import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    priors = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    feasible = bins_remain_cap >= item
    if not np.any(feasible):
        return priors
    residual = bins_remain_cap[feasible] - item
    epsilon = 1e-9
    base = 1.0 / (residual + epsilon)
    tie = (np.nonzero(feasible)[0].astype(float) + 1) * 1e-12
    priors[feasible] = base + tie
    return priors
