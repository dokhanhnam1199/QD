import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Inverse leftover capacity with deterministic index tie‑breaker; infeasible bins get -inf."""
    if bins_remain_cap.size == 0:
        return np.empty_like(bins_remain_cap, dtype=float)
    fits = bins_remain_cap >= item
    leftover = bins_remain_cap - item
    eps = 1e-12
    inv = 1.0 / (leftover + eps)
    # deterministic tie‑breaker: tiny increasing offset per bin
    tie = 1e-12 * item * np.arange(bins_remain_cap.shape[0])
    priorities = np.where(fits, inv + tie, -np.inf)
    return priorities
