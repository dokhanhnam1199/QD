```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Inverse leftover score with tiny random jitter for tie-breaking."""
    eps = 1e-9
    residual = bins_remain_cap - item
    feasible = residual >= 0
    score = np.where(feasible, 1.0 / (residual + eps), -np.inf)
    jitter = np.random.random(bins_remain_cap.shape) * 1e-6
    return score + jitter
```
