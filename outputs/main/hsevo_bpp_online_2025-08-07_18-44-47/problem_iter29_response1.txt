import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Best‑fit score with tiny index bias, golden‑ratio jitter, and variance‑scaled random jitter."""
    # Convert input to NumPy array and compute remaining capacity after placing the item.
    bins = np.asarray(bins_remain_cap, dtype=float)
    n = bins.size
    if n == 0:
        return np.empty(0, dtype=float)
    remaining = bins - item
    feasible = remaining >= 0
    scores = np.full(n, -np.inf, dtype=float)
    if not feasible.any():
        return scores
    eps = 1e-12
    idx = np.arange(n, dtype=float)
    # Inverse leftover encourages tight packing (best‑fit).
    inv_leftover = 1.0 / (remaining[feasible] + eps)
    # Tiny index bias for deterministic tie‑breaking (prefer lower index).
    bias = -idx[feasible] * 1e-9
    # Deterministic golden‑ratio jitter adds slight perturbation.
    phi = (1 + np.sqrt(5)) / 2
    phi_jitter = ((phi * (idx[feasible] + 1) * item) % 1.0) * 1e-6 * item
    # Random jitter scaled by the variance of remaining capacities.
    std = np.std(remaining[feasible])
    mean = np.mean(remaining[feasible]) + eps
    coeff = std / mean
    rng = np.random.default_rng()
    rand_jitter = rng.random(feasible.sum()) * item * 0.005 * (1.0 + coeff)
    scores[feasible] = inv_leftover + bias + phi_jitter + rand_jitter
    return scores
