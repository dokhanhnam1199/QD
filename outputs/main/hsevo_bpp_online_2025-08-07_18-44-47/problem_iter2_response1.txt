import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    EPSILON = 0.1
    SMALL_EPS = 1e-6
    RANDOM_NOISE = 0.01
    feasible = bins_remain_cap >= item
    residual = bins_remain_cap - item
    deterministic = np.where(feasible, 1.0 / (residual + SMALL_EPS), -np.inf)
    deterministic += np.where(feasible, np.random.rand(*bins_remain_cap.shape) * RANDOM_NOISE, 0)
    random_scores = np.where(feasible, np.random.rand(*bins_remain_cap.shape), -np.inf)
    return random_scores if np.random.rand() < EPSILON else deterministic
