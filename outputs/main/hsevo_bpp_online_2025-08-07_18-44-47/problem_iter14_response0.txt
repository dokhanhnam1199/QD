import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines inverse leftover, index bias, random jitter, golden‑ratio and sinusoidal tie‑breakers for bin priority.
    """
    bins_remain_cap = np.asarray(bins_remain_cap, dtype=float)
    n = bins_remain_cap.size
    if n == 0:
        return np.empty(0, dtype=float)
    eps = 1e-9
    remaining = bins_remain_cap - item
    feasible = remaining >= 0
    priority = np.full(n, -np.inf, dtype=float)
    idx = np.arange(n, dtype=float)
    primary = 1.0 / (remaining + eps)
    bias = -idx * 1e-6
    rng = np.random.default_rng()
    random_jitter = rng.random(n) * 0.01 * item
    phi = 0.6180339887498949
    golden_jitter = ((phi * (idx + 1) * item) % 1.0) * 1e-6 * item
    sin_jitter = (np.sin((item + idx) * 7.0) + 1) * 0.5 * 1e-9 * item
    priority[feasible] = primary[feasible] + bias[feasible] + random_jitter[feasible] + golden_jitter[feasible] + sin_jitter[feasible]
    return priority
