import numpy as np

_rng = np.random.default_rng()

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combine inverse leftover, load‑balance, variance‑scaled jitter, tiny bias."""
    n = bins_remain_cap.size
    if n == 0:
        return np.empty(0, dtype=float)
    eps = 1e-12
    remaining = bins_remain_cap - item
    feasible = remaining >= 0
    priority = np.full(n, -np.inf, dtype=float)
    if not feasible.any():
        return priority
    # Primary best‑fit score: higher for smaller leftover space
    base = 1.0 / (remaining[feasible] + eps)
    # Secondary term: prefers bins that become fuller after placement
    load_balance = item / (item + remaining[feasible] + eps)
    # Jitter scaled by variance of remaining capacities among feasible bins
    var_rem = np.var(remaining[feasible])
    jitter_scale = item * 0.005 * (1.0 + np.sqrt(var_rem))
    jitter = (_rng.random(remaining.shape) - 0.5) * jitter_scale
    # Tiny deterministic tie‑breaker (lower index slightly favored)
    bias = -np.arange(n)[feasible] * 1e-6
    priority[feasible] = base + load_balance + jitter + bias
    return priority
