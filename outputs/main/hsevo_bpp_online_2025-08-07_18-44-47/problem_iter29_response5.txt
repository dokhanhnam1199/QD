import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Score bins with inverse leftover, tightness, varianceâ€‘scaled jitter, and tiny bias."""
    bins = np.asarray(bins_remain_cap, dtype=float)
    n = bins.size
    if n == 0:
        return np.empty(0, dtype=float)
    remaining = bins - item
    feasible = remaining >= 0
    scores = np.full(n, -np.inf, dtype=float)
    if not feasible.any():
        return scores
    eps = 1e-12
    rem = remaining[feasible]
    inv_leftover = 1.0 / (rem + eps)
    tightness = item / (item + rem + eps)
    idx = np.arange(n, dtype=float)[feasible]
    phi = (1 + np.sqrt(5.0)) / 2.0
    phi_jitter = ((phi * (idx + 1.0) * item) % 1.0) * 1e-6 * item
    bias = -idx * 1e-6
    mean = np.mean(rem)
    std = np.std(rem)
    coeff = std / (mean + eps)
    jitter_weight = 0.01 * (1.0 + coeff)
    rand_jitter = np.random.random(rem.shape) * jitter_weight * item
    scores[feasible] = inv_leftover + tightness + phi_jitter + rand_jitter + bias
    return scores
