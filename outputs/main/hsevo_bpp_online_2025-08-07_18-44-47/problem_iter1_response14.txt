```python
import numpy as np

def priority_v2(item, bins_remain_cap):
    pri = np.full_like(bins_remain_cap, -np.inf, dtype=float)
    exact_mask = bins_remain_cap == item
    pri[exact_mask] = 1e6
    fit_mask = bins_remain_cap > item
    pri[fit_mask] = 1 - (bins_remain_cap[fit_mask] - item) / bins_remain_cap[fit_mask]
    return pri
```
