import numpy as np

_rng = np.random.default_rng(42)

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Score bins by inverse leftover, tiny index bias, and deterministic jitter."""
    n = bins_remain_cap.size
    if n == 0:
        return np.empty(0, dtype=float)
    eps = 1e-9
    remaining = bins_remain_cap - item
    feasible = remaining >= 0
    priority = np.full(n, -np.inf, dtype=float)
    inv = 1.0 / (remaining[feasible] + eps)
    bias = -np.arange(n)[feasible] * 1e-5
    jitter = _rng.random(n)[feasible] * 0.01 * item
    priority[feasible] = inv + bias + jitter
    return priority
