import numpy as np

# deterministic RNG for reproducible jitter
_rng = np.random.default_rng(42)


def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Score feasible bins: inverse leftover, deterministic index bias, and jitter."""
    n = bins_remain_cap.size
    if n == 0:
        return np.empty(0, dtype=float)
    eps = 1e-9
    remaining = bins_remain_cap - item
    feasible = remaining >= 0
    priority = np.full(n, -np.inf, dtype=float)
    # base score: favor bins with little leftover space
    priority[feasible] = 1.0 / (remaining[feasible] + eps)
    # tieâ€‘break: slight negative bias for lower indices, plus a tiny positive offset
    idx = np.arange(n)
    priority[feasible] += -idx[feasible] * 1e-5 + idx[feasible] * 1e-12
    # reproducible jitter proportional to item size
    jitter = _rng.random(n) * 0.01 * item
    priority[feasible] += jitter[feasible]
    return priority
