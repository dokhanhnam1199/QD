import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines Best Fit's tightness and a logarithmic fullness bonus multiplicatively.
    This prioritizes bins that are already quite full and can snugly fit the item.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    fittable_bins_mask = bins_remain_cap >= item

    if not np.any(fittable_bins_mask):
        return priorities

    fittable_bins_remain_cap = bins_remain_cap[fittable_bins_mask]

    # Best Fit Score: Higher for bins with minimal remaining capacity after fitting.
    # Adding epsilon for numerical stability.
    best_fit_scores = 1.0 / (fittable_bins_remain_cap - item + 1e-6)

    # Fullness Bonus: A bonus reflecting how full the bin is *before* fitting the item.
    # Uses a logarithmic scale to give diminishing returns for very full bins,
    # but still rewards moderately full bins. Add 1 to avoid log(0).
    # We use the *inverse* of remaining capacity to represent fullness.
    fullness_bonus = np.log1p(1.0 / (fittable_bins_remain_cap + 1e-6))

    # Combined Score: Multiplicatively combine Best Fit and Fullness Bonus.
    # This prioritizes bins that are both a good fit (low residual space)
    # and are already quite full.
    combined_scores = best_fit_scores * (1.0 + fullness_bonus * 0.5) # Adjust weight for fullness bonus

    # Normalize priorities to a [0, 1] range.
    max_score = np.max(combined_scores)
    if max_score > 1e-9:
        priorities[fittable_bins_mask] = combined_scores / max_score
    else:
        # If all scores are near zero, assign a small uniform priority to fittable bins.
        priorities[fittable_bins_mask] = 0.1

    return priorities
