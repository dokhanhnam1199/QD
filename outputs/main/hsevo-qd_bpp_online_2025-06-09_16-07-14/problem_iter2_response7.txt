import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins using a Gaussian distribution and logarithmic scaling."""

    # Gaussian component: prioritize bins with remaining capacity close to item size
    mu = item
    sigma = item / 3.0
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))

    # Logarithmic component: encourages filling bins, avoids almost-full bins
    ratios = item / bins_remain_cap
    log_ratios = np.log(ratios + 1e-6)  # Avoid log(0)
    logarithmic = -log_ratios

    # Combine both components
    priorities = gaussian + 0.2 * logarithmic # Weighing to control balance.

    # Never pick impossible bins.
    priorities[bins_remain_cap < item] = -np.inf

    return priorities
