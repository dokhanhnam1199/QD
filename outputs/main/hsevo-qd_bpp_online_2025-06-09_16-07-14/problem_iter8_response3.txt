import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines Gaussian fit, capacity ratio, and almost-full bin bonus."""

    mu = item
    sigma = item / 3.0
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))

    ratios = item / (bins_remain_cap + 1e-9)
    priorities = gaussian + 0.1 * (1 - ratios)

    # Almost full bin bonus
    almost_full_threshold = 0.1 # If bin is filled over 90%
    almost_full_bonus = 0.5
    almost_full_mask = bins_remain_cap <= (1 - almost_full_threshold) * item
    priorities[almost_full_mask] += almost_full_bonus

    priorities[bins_remain_cap < item] = -np.inf

    return priorities
