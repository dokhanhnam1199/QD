import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins using a Gaussian fit, capacity ratio, and adaptive perturbation."""

    mu = item
    sigma = item / 3.0
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))

    # Adaptive perturbation based on remaining capacity
    max_cap = np.max(bins_remain_cap)
    if max_cap > 0:
        perturbation = 0.1 * (bins_remain_cap / max_cap)
    else:
        perturbation = 0 # Avoid division by zero
    
    priorities = gaussian + perturbation

    # Explicitly avoid bins that can't fit the item
    priorities[bins_remain_cap < item] = -np.inf

    return priorities
