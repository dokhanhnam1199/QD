import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines Gaussian fit with near-full bin bonus and capacity ratio."""

    mu = item
    sigma = item / 3  # Reduced sigma for sharper peak
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))

    capacity_ratio = item / bins_remain_cap
    capacity_priority = -np.log(capacity_ratio)

    almost_full_bonus = np.where(bins_remain_cap <= 1.1 * item, 0.5, 0)
    
    priorities = gaussian + 0.5 * capacity_priority + almost_full_bonus
    priorities[bins_remain_cap < item] = -np.inf
    return priorities
