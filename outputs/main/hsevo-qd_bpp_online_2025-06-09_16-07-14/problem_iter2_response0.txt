import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines Gaussian and logarithmic approaches with feasibility check."""
    # Gaussian component: Prioritizes bins close to item size.
    mu = item
    sigma = item / 3.0
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))

    # Logarithmic component: Discourages using almost full bins.
    ratios = item / bins_remain_cap
    log_ratios = np.log(ratios)
    logarithmic = -log_ratios

    # Combined priority
    priorities = gaussian + 0.2 * logarithmic #Weighing

    # Feasibility check
    priorities[bins_remain_cap < item] = -np.inf # Never pick impossible bins

    return priorities
