```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines Gaussian fit with ratio, adaptively. Gaussian focuses on ideal fit, ratio encourages fuller bins.
    Avoid impossible bins.
    """
    mu = item
    sigma = item / 3.0
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))

    ratios = item / bins_remain_cap
    log_ratios = np.log(ratios)
    ratio_priority = -log_ratios * 0.1 # Scale it so it doesn't dominate Gaussian

    priorities = gaussian + ratio_priority

    priorities[bins_remain_cap < item] = -np.inf # Never pick impossible bins.

    return priorities
```
