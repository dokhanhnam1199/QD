import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins using a Gaussian distribution and log ratio, avoiding overfill."""

    mu = item
    sigma = item / 3.0
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))

    ratios = item / bins_remain_cap
    log_ratios = np.log(ratios)
    log_priority = -log_ratios * 0.1 # Scale down to prevent overpowering the gaussian

    priorities = gaussian + log_priority

    priorities[bins_remain_cap < item] = -np.inf

    return priorities
