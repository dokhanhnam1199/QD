import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Calculates bin priorities based on Gaussian fit and remaining capacity, avoids impossible bins."""
    mu = item
    sigma = item / 3.0
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))
    perturbation = 0.1 * (bins_remain_cap / np.max(bins_remain_cap))
    priorities = gaussian + perturbation
    priorities[bins_remain_cap < item] = -np.inf
    return priorities
