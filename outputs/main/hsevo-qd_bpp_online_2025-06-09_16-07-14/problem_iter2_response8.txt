```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines Gaussian and log-ratio heuristics with feasibility check."""

    # Gaussian component
    mu = item
    sigma = item / 3.0
    gaussian = np.exp(-((bins_remain_cap - mu) ** 2) / (2 * sigma**2))

    # Log-ratio component (dampened)
    ratios = item / bins_remain_cap
    log_ratios = np.log(np.maximum(ratios, 1e-6)) # Avoid log(0)
    log_priority = -0.1 * log_ratios # Reduce impact
    
    #Combined
    priorities = gaussian + log_priority
    
    #Feasibility
    priorities[bins_remain_cap < item] = -np.inf
    
    return priorities
```
