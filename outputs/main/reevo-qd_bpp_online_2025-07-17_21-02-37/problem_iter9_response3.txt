```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:

    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Identify bins that can fit the item
    can_fit = bins_remain_cap >= item

    if np.any(can_fit):
        # Base priority for bins that can fit
        priorities[can_fit] = 1.0

        # Calculate remaining capacity after placing the item
        remaining_after_placement = bins_remain_cap[can_fit] - item

        # Prioritize tighter fits, stronger reward, adjusted scaling based on remaining capacity distribution.
        mean_remaining = np.mean(bins_remain_cap)
        std_remaining = np.std(bins_remain_cap)

        priority_scale = 5.0
        if std_remaining > 0.1 * mean_remaining:  # Adapt reward based on variance
           priority_scale = 7.5

        priorities[can_fit] += priority_scale / (remaining_after_placement + 0.01 * mean_remaining)


        # Bonus for bins that become nearly full after placing the item (reduces fragmentation), stronger reward. Window adjusted.
        nearly_full = (remaining_after_placement >= 0) & (remaining_after_placement <= 0.05 * np.max(bins_remain_cap))
        fragment_bonus = 10.0
        if std_remaining < 0.05 * mean_remaining:
            fragment_bonus = 12.5
        priorities[can_fit][nearly_full] += fragment_bonus


        # Discourage using empty bins unless the item is very large relative to bin size. Stronger penalty. Adaptive threshold.
        empty_bin_indices = np.where(bins_remain_cap == np.max(bins_remain_cap))[0]  # Assuming initial capacity is max.
        empty_and_fits = np.intersect1d(np.where(can_fit)[0], empty_bin_indices)

        if item <= 0.3 * np.max(bins_remain_cap): #Reduced Threshold
            priorities[empty_and_fits] -= 8.0  # Stronger discouragement
        elif len(empty_and_fits) > 0:
            priorities[empty_and_fits] += 1.0 # encourage if item is large and fits in an empty bin

        # Add small random value to break ties, only when needed.
        feasible_bins = np.where(can_fit)[0]
        if len(feasible_bins) > 1:
            priorities[feasible_bins] += np.random.rand(len(feasible_bins)) * 0.005 # Smaller randomness


    else:
        # No bin can fit the item. Assign very low priority.
        priorities[:] = -10.0

    #Penalize bins with larger remaining capacity, but less strongly than rewards. Adjusted Penalty
    capacity_penalty = 0.0005
    if std_remaining > 0.2 * mean_remaining:
        capacity_penalty = 0.00075
    priorities -= capacity_penalty * bins_remain_cap


    return priorities
```
