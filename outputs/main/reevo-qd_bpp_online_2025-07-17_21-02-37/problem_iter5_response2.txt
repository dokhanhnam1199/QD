```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Identify bins that can fit the item
    can_fit = bins_remain_cap >= item

    if np.any(can_fit):
        # Base priority for bins that can fit
        priorities[can_fit] = 1.0

        # Calculate remaining capacity after placing the item
        remaining_after_placement = bins_remain_cap[can_fit] - item

        # Strongly reward near-full bins
        nearly_full = (remaining_after_placement >= 0) & (remaining_after_placement <= 0.1 * np.max(bins_remain_cap))
        priorities[can_fit][nearly_full] += 5.0  # Higher reward

        # Prioritize tighter fits, more aggressively.
        priorities[can_fit] += 3.0 / (remaining_after_placement + 0.001)

        # Penalize emptier bins, scaling with item size
        empty_bin_indices = np.where(bins_remain_cap == np.max(bins_remain_cap))[0]
        empty_and_fits = np.intersect1d(np.where(can_fit)[0], empty_bin_indices)

        if item <= 0.3 * np.max(bins_remain_cap):
            priorities[empty_and_fits] -= 2.0 * (1 - (item / (0.3 * np.max(bins_remain_cap))))  # Stronger penalty, scaled.
        elif len(empty_and_fits) > 0:
             priorities[empty_and_fits] += 0.5 # Reward using empty bins for large objects

        # Small random value for tie-breaking only
        priorities[can_fit] += np.random.rand(np.sum(can_fit)) * 0.01

    else:
        # No bin can fit the item. Assign very low priority.
        priorities[:] = -10.0

    return priorities
```
