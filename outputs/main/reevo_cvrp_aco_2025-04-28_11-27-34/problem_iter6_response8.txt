import numpy as np

def heuristics_v2(distance_matrix: np.ndarray, coordinates: np.ndarray, demands: np.ndarray, capacity: int) -> np.ndarray:
    """
    Heuristic function for Capacitated Vehicle Routing Problem (CVRP) using distance, angle, demand, and stochastic factors.

    Args:
        distance_matrix (np.ndarray): Distance matrix between nodes (n x n).
        coordinates (np.ndarray): Euclidean coordinates of nodes (n x 2).
        demands (np.ndarray): Demand of each customer (n).
        capacity (int): Vehicle capacity.

    Returns:
        np.ndarray: Prior indicators of how promising it is to include each edge in a solution (n x n).
    """

    n = distance_matrix.shape[0]
    heuristic_matrix = np.zeros((n, n))

    # 1. Distance factor
    distance_factor = 1 / (distance_matrix + 1e-6)

    # 2. Angle factor (Direct calculation)
    angle_factor = np.ones((n, n))
    for i in range(n):
        for j in range(n):
            if i != j:
                # Angle between depot -> i and i -> j
                v1 = coordinates[i] - coordinates[0]  # Vector from depot to i
                v2 = coordinates[j] - coordinates[i]  # Vector from i to j

                norm1 = np.linalg.norm(v1)
                norm2 = np.linalg.norm(v2)

                if norm1 > 0 and norm2 > 0:
                    v1 = v1 / norm1
                    v2 = v2 / norm2
                    dot_product = np.dot(v1, v2)
                    angle = np.arccos(np.clip(dot_product, -1.0, 1.0))
                    angle_factor[i, j] = 1 - (angle / np.pi) # Smaller angle is better, scale to [0, 1]

    # 3. Demand factor
    demand_factor = np.ones((n, n))
    for i in range(n):
        for j in range(n):
            if i != j and i != 0 and j != 0:
                demand_factor[i, j] = np.exp(-(demands[i] + demands[j]) / (2 * capacity))

    # Adaptive Weighting (Adjusted)
    problem_size_weight = min(1.0, n / 150.0)  # Scale problem size
    distance_weight = 0.5 + 0.3 * problem_size_weight
    angle_weight = 0.3 - 0.1 * problem_size_weight
    demand_weight = 0.2 - 0.2 * problem_size_weight


    # Controlled Randomness (Scaled to Distance)
    random_matrix = np.random.normal(0, 0.05 * distance_matrix, size=(n, n))
    random_matrix = np.clip(random_matrix, -0.1, 0.1)  # Clipping

    # Combine factors multiplicatively
    heuristic_matrix = (distance_factor ** distance_weight) * \
                       (angle_factor ** angle_weight) * \
                       (demand_factor ** demand_weight) + random_matrix

    # Adaptive Sparsification (Quantile and Demand-Aware)
    demand_threshold = np.quantile(demands[1:], 0.75) # Remove depot demand
    distance_threshold = np.quantile(distance_matrix[distance_matrix > 0], 0.9 - 0.1 * (demand_threshold / np.max(demands[1:]))) #Adjust distance threshold

    heuristic_matrix[distance_matrix > distance_threshold] = 0
    np.fill_diagonal(heuristic_matrix, 0)

    # Normalize heuristic values
    max_heuristic = np.max(heuristic_matrix)
    if max_heuristic > 0:
        heuristic_matrix = heuristic_matrix / max_heuristic

    return heuristic_matrix
