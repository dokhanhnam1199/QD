```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization with a fill ratio consideration."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste minimization (higher is better, penalize bins that can't fit)
    waste = bins_remain_cap - item
    waste_priority = np.where(waste >= 0, np.exp(-waste), -np.inf)
    priorities += waste_priority

    # Fill ratio (higher is better, up to a point)
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_priority = np.where(fill_ratio <= 1, fill_ratio, -np.inf)
    priorities += fill_priority

    return priorities
```
