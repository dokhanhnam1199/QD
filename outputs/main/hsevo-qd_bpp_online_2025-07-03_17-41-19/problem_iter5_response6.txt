import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization, fill level, and stability for bin selection."""

    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-waste), -np.inf)

    fill_ratio = item / (bins_remain_cap + 1e-9)
    fill_reward = np.exp(fill_ratio) if np.all(fill_ratio <= 1) else -np.inf

    stability = -np.abs(bins_remain_cap - item)

    return waste_penalty + fill_reward + stability
