```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and fill level with bin utilization."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste: Prioritize bins where the item fits well.
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-2*waste), -np.inf)
    priorities += waste_penalty

    # Fill Level: Encourage filling bins as much as possible.
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.where(fill_ratio <= 1, fill_ratio, -np.inf)
    priorities += fill_reward

    # Bin Utilization: Encourage using bins that are already partially filled.
    bin_utilization = 1 - bins_remain_cap
    priorities += 0.5 * bin_utilization # scale bin utilization

    return priorities
```
