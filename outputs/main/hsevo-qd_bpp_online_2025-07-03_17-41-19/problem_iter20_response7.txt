```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins balancing waste, fill level, & stability."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    feasible_bins = bins_remain_cap >= item
    if not np.any(feasible_bins):
        return priorities

    waste = bins_remain_cap - item
    waste[waste < 0] = np.inf  # Mark infeasible bins
    waste_priority = -waste
    priorities += waste_priority * 0.5

    fill_ratio = item / bins_remain_cap
    fill_priority = fill_ratio
    priorities += fill_priority * 0.3

    stability = -np.abs(bins_remain_cap - item)
    priorities += stability * 0.2

    priorities[~feasible_bins] = -np.inf
    return priorities
```
