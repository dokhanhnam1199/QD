import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and fill level for priority."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste: Exponential penalty for waste, inf for overfill
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-waste), -np.inf)
    priorities += waste_penalty

    # Fill Level: Encourage filling, penalize overfill
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.where(fill_ratio <= 1, fill_ratio, -np.inf)
    priorities += fill_reward

    # Normalize priorities to ensure stability and avoid large numbers.
    priorities = (priorities - np.min(priorities)) / (np.max(priorities) - np.min(priorities) + epsilon)

    return priorities
