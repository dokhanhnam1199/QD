```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and fill level with a stability factor."""

    epsilon = 1e-6
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-waste), -np.inf)

    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.exp(fill_ratio)

    stability = -np.abs(bins_remain_cap - item) * 0.1

    priorities = waste_penalty + fill_reward + stability
    return priorities
```
