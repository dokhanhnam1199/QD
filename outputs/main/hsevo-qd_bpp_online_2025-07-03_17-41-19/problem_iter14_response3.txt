import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization, fill ratio, and balance for bin selection."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9
    alpha = 0.6 # Weight of waste vs fill

    # Waste minimization
    waste = bins_remain_cap - item
    waste_priority = np.where(waste >= 0, np.exp(-waste / (item + epsilon)), -np.inf)
    priorities += alpha * waste_priority

    # Fill ratio
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_priority = np.where(fill_ratio <= 1, np.exp(fill_ratio), -np.inf)
    priorities += (1 - alpha) * fill_priority
    
    # Balance bin utilization
    bin_utilization = 1 - (bins_remain_cap / (np.max(bins_remain_cap) + epsilon))
    balance_priority = np.exp(-np.abs(bin_utilization - 0.5))
    priorities += 0.1 * balance_priority

    return priorities
