import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and fill ratio optimization for bin selection."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste minimization: Exponential penalty for waste, inf for overfill.
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-waste), -np.inf)
    priorities += waste_penalty

    # Fill ratio optimization: Encourages filling, penalizes overfill.
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.where(fill_ratio <= 1, np.exp(fill_ratio), -np.inf)
    priorities += fill_reward

    return priorities
