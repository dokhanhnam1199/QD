```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization, fill level, and stability."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-6
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-waste), -np.inf)
    priorities += waste_penalty

    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.where(fill_ratio <= 1, fill_ratio, -np.inf)
    priorities += fill_reward

    stability = -np.abs(bins_remain_cap - item) * 0.1
    priorities += stability

    return priorities
```
