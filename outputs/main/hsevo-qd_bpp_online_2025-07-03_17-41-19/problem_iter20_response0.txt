import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and fill level, with explicit handling of overfill."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste minimization with a harsher penalty for bins too small.
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-5 * waste / item), -np.inf)
    priorities += waste_penalty

    # Fill level reward, scaled by remaining capacity.
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = (1 / (1 + np.exp(5 - 10 * fill_ratio))) * (1 - bins_remain_cap / np.max(bins_remain_cap))
    priorities += fill_reward

    return priorities
