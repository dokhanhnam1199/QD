```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and fill level, with a focus on simplicity."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste minimization: Favor bins with minimal waste
    waste = bins_remain_cap - item
    waste_priority = np.where(waste >= 0, -waste, -np.inf)  # linear penalty for waste

    # Fill level: Encourage filling bins close to full
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_priority = fill_ratio

    priorities = waste_priority + fill_priority # Combine waste and fill

    return priorities
```
