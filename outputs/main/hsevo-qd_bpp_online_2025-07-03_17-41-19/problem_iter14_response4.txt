```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, waste_exp_scale: float = 1.0, fill_exp_scale: float = 1.0, stability_scale: float = 1.0, epsilon: float = 1e-6) -> np.ndarray:
    """Combines waste minimization, fill level, and stability to prioritize bins."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    # Waste: Exponential penalty for waste.
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-waste * waste_exp_scale), -np.inf)
    priorities += waste_penalty

    # Fill Level: Exponential reward for fill ratio.
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.where(fill_ratio <= 1, np.exp(fill_ratio * fill_exp_scale), -np.inf)
    priorities += fill_reward

    # Stability (Closeness to Item Size):
    stability = -np.abs(bins_remain_cap - item) * stability_scale
    priorities += stability

    return priorities
```
