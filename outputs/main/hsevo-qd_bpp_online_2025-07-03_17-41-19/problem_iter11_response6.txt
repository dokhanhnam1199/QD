import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and fill ratio for bin selection."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste: Prioritize bins that fit the item well.
    waste = bins_remain_cap - item
    waste_priority = np.where(waste >= 0, np.exp(-waste), -np.inf) # smaller waste -> better priority
    priorities += waste_priority

    # Fill ratio: item_size/bin_remain_cap. The larger the better, but should be less than 1.
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_priority = np.where(fill_ratio <= 1, fill_ratio, -np.inf)
    priorities += fill_priority

    return priorities
