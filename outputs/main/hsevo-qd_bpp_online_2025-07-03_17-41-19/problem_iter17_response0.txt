import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization and fill level considerations for bin selection."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste: Exponential penalty for wasted space, hard penalty for overfill.
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-10 * waste / item), -np.inf)
    priorities += waste_penalty

    # Fill Level: Sigmoid reward for filling bins, avoid overfilling.
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = 1 / (1 + np.exp(5 - 10 * fill_ratio))
    priorities += fill_reward

    return priorities
