```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Return bin priorities, combining waste, fill level, and stability."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste and Fill Level
    waste = bins_remain_cap - item
    fit_mask = waste >= 0
    waste_penalty = np.where(fit_mask, np.exp(-waste), -np.inf)

    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.where(fill_ratio <= 1, np.exp(fill_ratio), -np.inf)

    priorities += waste_penalty + fill_reward

    # Stability (Best-Fit)
    stability = -np.abs(bins_remain_cap - item)
    priorities += 0.5 * stability

    return priorities
```
