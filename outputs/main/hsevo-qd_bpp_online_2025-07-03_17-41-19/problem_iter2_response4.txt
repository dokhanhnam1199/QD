```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization with fill ratio to prioritize bins."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste: Prioritize bins with minimal waste, but only if it fits
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-waste), -np.inf)
    priorities += waste_penalty

    # Fill Level: Encourages filling bins
    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.where(fill_ratio <= 1, fill_ratio, -np.inf) # Linear fill reward, no exponential.
    priorities += fill_reward

    return priorities
```
