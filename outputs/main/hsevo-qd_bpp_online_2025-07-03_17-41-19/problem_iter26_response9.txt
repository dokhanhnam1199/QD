import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste, fill level, and capacity for bin selection."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    waste = bins_remain_cap - item
    fit_mask = waste >= 0
    priorities = np.where(fit_mask, -waste**2, -np.inf)

    fill_ratio = item / (bins_remain_cap + epsilon)
    fill_reward = np.where(fit_mask & (fill_ratio <= 1), np.exp(1 - np.abs(1 - fill_ratio)), -np.inf)
    priorities = np.where(fit_mask, priorities + fill_reward, priorities)

    capacity_factor = bins_remain_cap / (np.max(bins_remain_cap) + epsilon)
    priorities = np.where(fit_mask, priorities + 0.1 * np.exp(capacity_factor), priorities)

    return priorities
