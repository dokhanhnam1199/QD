import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization, fill ratio, and item density."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste minimization
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, -waste, -np.inf)
    priorities += waste_penalty

    # Fill ratio
    fill_ratio = item / (bins_remain_cap + epsilon)
    priorities += fill_ratio

    # Item density approximation
    item_density = item / (np.mean(bins_remain_cap) + epsilon)
    priorities += item_density

    return priorities
