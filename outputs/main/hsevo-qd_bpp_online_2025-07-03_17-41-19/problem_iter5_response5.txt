import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization with bin utilization for online bin packing."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    epsilon = 1e-9

    # Waste: Prioritize bins with minimal waste.
    waste = bins_remain_cap - item
    waste_penalty = np.where(waste >= 0, np.exp(-5 * waste / (item + epsilon)), -np.inf)
    priorities += waste_penalty

    # Bin Utilization: Reward bins that are already partially filled.
    utilization = 1 - bins_remain_cap
    utilization_reward = np.tanh(utilization)
    priorities += 0.5 * utilization_reward

    return priorities
