```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines feasibility, waste minimization, and a fragmentation penalty."""
    small_value = 0.000664
    fragmentation_penalty = 0.1
    priorities = np.where(bins_remain_cap >= item, 1.0 / (bins_remain_cap - item + small_value), 0.0)
    priorities = priorities - fragmentation_penalty * (bins_remain_cap < item)
    return priorities
```
