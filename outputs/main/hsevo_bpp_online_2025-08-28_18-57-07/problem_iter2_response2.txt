import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines inverse remaining capacity with a check for feasibility."""
    priorities = np.zeros_like(bins_remain_cap)
    feasible_bins = bins_remain_cap >= item
    if np.sum(feasible_bins) > 0:
        remaining_capacities = bins_remain_cap[feasible_bins]
        waste = remaining_capacities - item
        priorities[feasible_bins] = 1.0 / (waste + 1e-6)
    return priorities
