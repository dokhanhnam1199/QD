import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit inverse waste with a relative fragmentation penalty for improved bin packing."""
    feasible_bins = bins_remain_cap >= item
    priorities = np.zeros_like(bins_remain_cap)
    if np.any(feasible_bins):
        waste = bins_remain_cap[feasible_bins] - item
        fragmentation_penalty = 0.05 * (1 - waste / bins_remain_cap[feasible_bins])
        priorities[feasible_bins] = 1.0 / (waste + 1e-6) - fragmentation_penalty
    return priorities
