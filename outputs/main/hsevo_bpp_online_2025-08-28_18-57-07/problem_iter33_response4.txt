```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    small_value = 0.000664
    fragmentation_penalty = 0.05
    priorities = np.where(bins_remain_cap >= item,
                          1.0 / (bins_remain_cap - item + small_value) -
                          fragmentation_penalty * (1 - (bins_remain_cap - item) / bins_remain_cap),
                          0.0)
    return priorities
```
