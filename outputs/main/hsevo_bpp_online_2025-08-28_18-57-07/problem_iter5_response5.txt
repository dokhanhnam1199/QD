```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines feasibility check with inverse remaining capacity for stable bin selection."""
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    priorities[valid_bins] = 1.0 / (bins_remain_cap[valid_bins] - item + 1e-6)
    return priorities
```
