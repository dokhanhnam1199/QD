```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines feasibility with inverse remaining capacity and a waste factor."""
    small_value = 1e-9
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    
    waste = bins_remain_cap[valid_bins] - item
    priorities[valid_bins] = (1.0 / (waste + small_value)) * (1.0 - (waste / (bins_remain_cap[valid_bins] + small_value)))
    return priorities
```
