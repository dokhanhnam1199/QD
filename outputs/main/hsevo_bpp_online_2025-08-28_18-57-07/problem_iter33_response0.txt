import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    small_value = 1e-6
    priorities = np.zeros_like(bins_remain_cap)
    feasible_bins = bins_remain_cap >= item
    if np.any(feasible_bins):
        remaining_capacities = bins_remain_cap[feasible_bins]
        waste = remaining_capacities - item
        fragmentation_penalty = 0.2 * (1 - waste / bins_remain_cap[feasible_bins])
        priorities[feasible_bins] = 1.0 / (waste + small_value) - fragmentation_penalty
    return priorities
