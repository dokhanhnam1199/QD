import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes fit and minimizes waste, with a relative fragmentation penalty."""
    small_value = 1e-6
    priorities = np.where(bins_remain_cap >= item,
                         (bins_remain_cap - item) / (bins_remain_cap + small_value) -
                         0.05 * (1 - (bins_remain_cap - item) / bins_remain_cap),
                         0.0)
    return priorities
