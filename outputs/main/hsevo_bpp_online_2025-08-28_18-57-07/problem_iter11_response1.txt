```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines feasibility with inverse waste, prioritizing bins with minimal waste."""
    small_value = 0.000664
    priorities = np.zeros_like(bins_remain_cap)
    suitable_bins = bins_remain_cap >= item
    if not np.any(suitable_bins):
        return priorities
    priorities[suitable_bins] = 1.0 / (bins_remain_cap[suitable_bins] - item + small_value)
    return priorities
```
