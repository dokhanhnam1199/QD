import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines best-fit inverse waste with a fragmentation penalty for bin selection."""
    feasible_bins = bins_remain_cap >= item
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    if np.any(feasible_bins):
        waste = bins_remain_cap[feasible_bins] - item
        priorities[feasible_bins] = 1.0 / (waste + 0.0001)
        fragmentation_penalty = 0.05 * (1 - (waste) / bins_remain_cap[feasible_bins])
        priorities[feasible_bins] -= fragmentation_penalty
    return priorities
