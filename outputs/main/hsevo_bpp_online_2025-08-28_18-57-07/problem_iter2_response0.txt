```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines inverse remaining capacity with a check for valid bins.
    Prioritizes bins with less remaining capacity after placing the item.
    """
    priorities = np.zeros_like(bins_remain_cap)
    valid_bins = bins_remain_cap >= item
    if not np.any(valid_bins):
        return priorities
    priorities[valid_bins] = 1.0 / (bins_remain_cap[valid_bins] - item + 1e-9)
    return priorities
```
