```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines inverse remaining capacity with a capacity check for stability."""
    capacities = np.copy(bins_remain_cap)
    capacities[capacities < item] = 0
    valid_bins = capacities > 0
    priorities = np.zeros_like(bins_remain_cap)
    if np.any(valid_bins):
        priorities[valid_bins] = 1.0 / (capacities[valid_bins] - item + 1e-6)
    return priorities
```
