```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines feasibility check with inverse remaining capacity and a small value to avoid division by zero."""
    small_value = 1e-9
    priorities = np.where(bins_remain_cap >= item, 1.0 / (bins_remain_cap - item + small_value), 0.0)
    return priorities
```
