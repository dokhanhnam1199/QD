import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, small_value: float = 0.000848929912194034, fragmentation_penalty: float = 0.02103602761461921) -> np.ndarray:
    """Combines feasibility and waste minimization with a fragmentation penalty."""
    priorities = np.where(bins_remain_cap >= item, 1.0 / (bins_remain_cap - item + small_value) - fragmentation_penalty * (1 - (bins_remain_cap - item) / bins_remain_cap), 0.0)
    return priorities
