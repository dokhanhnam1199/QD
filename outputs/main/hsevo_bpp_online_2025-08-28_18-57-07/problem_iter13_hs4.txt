import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray, small_value: float = 0.7517483940495092) -> np.ndarray:
    """Combines best-fit (waste minimization) with feasibility check."""
    priorities = np.where(bins_remain_cap >= item, 1.0 / (bins_remain_cap - item + small_value), 0.0)
    return priorities
