import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins considering fit and capacity balance."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    for i, remaining_capacity in enumerate(bins_remain_cap):
        if item <= remaining_capacity:
            remaining_space = remaining_capacity - item
            fit_score = 1.0 / (remaining_space + 0.01)  # Smaller remaining space is better

            # Penalize bins that are either too full or not full enough
            capacity_utilization = item / remaining_capacity
            utilization_penalty = np.abs(capacity_utilization - 0.75)  # Target 75% utilization

            priorities[i] = fit_score - utilization_penalty

        else:
            priorities[i] = -np.inf

    return priorities
