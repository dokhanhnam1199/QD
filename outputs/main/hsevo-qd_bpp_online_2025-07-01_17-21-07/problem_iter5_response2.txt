import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins based on fit and fragmentation, enhanced."""
    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    for i, remaining_capacity in enumerate(bins_remain_cap):
        if item <= remaining_capacity:
            remaining_space = remaining_capacity - item
            fit_score = 1.0 / (remaining_space + 0.01)

            # Penalize disproportionate remaining space relative to item size.
            size_ratio_penalty = np.abs((remaining_space) / (item + 0.01))

            priorities[i] = fit_score - size_ratio_penalty
        else:
            priorities[i] = -np.inf

    if np.any(np.isfinite(priorities)):
        priorities = (priorities - np.min(priorities[np.isfinite(priorities)])) / \
                     (np.max(priorities[np.isfinite(priorities)]) - np.min(priorities[np.isfinite(priorities)]))
    else:
        priorities = np.zeros_like(bins_remain_cap, dtype=float)

    return priorities
