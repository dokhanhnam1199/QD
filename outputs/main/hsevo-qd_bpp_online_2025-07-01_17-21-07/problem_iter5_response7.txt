import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Return priority score for each bin.
    Combines best-fit with a capacity-aware ratio.
    """
    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    for i, remaining_capacity in enumerate(bins_remain_cap):
        if item <= remaining_capacity:
            remaining_space = remaining_capacity - item
            fit_score = 1.0 / (remaining_space + 0.01)
            capacity_ratio = item / remaining_capacity
            priorities[i] = fit_score * capacity_ratio
        else:
            priorities[i] = -np.inf
    return priorities
