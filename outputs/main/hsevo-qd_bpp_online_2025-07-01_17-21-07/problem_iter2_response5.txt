```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Prioritizes bins considering fit and balance."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    for i, remaining_capacity in enumerate(bins_remain_cap):
        if item <= remaining_capacity:
            remaining_space = remaining_capacity - item
            fit_score = 1.0 / (remaining_space + 0.01)
            size_ratio = item / remaining_capacity
            balance_penalty = np.abs(size_ratio - 0.66)
            priorities[i] = fit_score - balance_penalty
        else:
            priorities[i] = -np.inf

    return priorities
```
