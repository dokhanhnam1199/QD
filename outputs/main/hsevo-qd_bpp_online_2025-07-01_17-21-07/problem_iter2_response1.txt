import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Calculate priority based on fit and balance."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)

    for i, remaining_capacity in enumerate(bins_remain_cap):
        if item <= remaining_capacity:
            remaining_space = remaining_capacity - item
            fit_score = 1.0 / (remaining_space + 0.01)

            # Penalize bins where the item is too small relative to the capacity, or remaining too large
            size_ratio_penalty = np.abs((item / remaining_capacity) - 0.66) # targetting fill rate (1 - 0.66 = 0.33)
            waste_penalty = remaining_space / (np.max(bins_remain_cap)+0.01) # avoid dividing by zero.

            priorities[i] = fit_score - size_ratio_penalty - waste_penalty
        else:
            priorities[i] = -np.inf

    return priorities
