import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Combines waste minimization, bin utilization, item size, randomness, & capacity-based velocity."""

    priorities = np.zeros_like(bins_remain_cap, dtype=float)
    original_capacity = np.max(bins_remain_cap)

    # Infeasible bin handling
    infeasible = bins_remain_cap < item
    priorities[infeasible] = -np.inf

    # Waste Minimization with exponential scaling
    waste = bins_remain_cap - item
    waste[infeasible] = np.inf  # Ensure waste is infinite for infeasible bins
    priorities += np.exp(-waste)

    # Bin utilization balancing
    bin_fraction = bins_remain_cap / original_capacity
    priorities += -(bin_fraction - 0.5)**2

    # Item size awareness
    priorities += item / original_capacity

    # Capacity-based Velocity (similar to v1 but scaled)
    velocity = np.exp(-np.abs(bins_remain_cap - item) / original_capacity)
    priorities += velocity * (1 - bin_fraction)

    # Random Perturbation
    priorities += np.random.normal(0, 0.01, size=bins_remain_cap.shape)

    return priorities
