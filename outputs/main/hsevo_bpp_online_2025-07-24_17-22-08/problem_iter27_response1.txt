import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Enhances non-linear prioritization with a dynamic penalty term that adapts to the remaining capacity and item size.
    """
    # Calculate valid bins and remaining capacity after adding the item
    valid_bins = (bins_remain_cap >= item).astype(float)
    remain_cap_after_add = bins_remain_cap - item
    
    # Dynamic non-linear prioritization balancing tight fit and future capacity
    tight_fit_score = np.exp(-remain_cap_after_add / (item + 1e-6))  # Avoid division by zero
    future_capacity_score = 1 - (remain_cap_after_add / bins_remain_cap) ** 2  # Penalize bins that are too full or too empty relative to their original capacity
    
    # Combine scores with a weighted average for better adaptability
    priority = np.where(valid_bins == 1, 
                         0.7 * tight_fit_score + 0.3 * future_capacity_score,  # Balance between tight fit and future capacity
                         -np.inf)
    
    # Normalize priority scores for enhanced adaptability across different item sizes
    max_priority = np.max(priority)
    if max_priority != -np.inf:
        priority = priority / max_priority
    
    return priority
