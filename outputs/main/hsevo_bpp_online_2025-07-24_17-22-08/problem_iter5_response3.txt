import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority score for each bin. 
    Prioritizes bins with remaining capacity closest to the item size using a non-linear function.
    """
    # Identify bins that are too small for the item and give them a very low priority
    priority = np.where(bins_remain_cap >= item, np.exp(-(bins_remain_cap - item)), -np.inf)
    
    # Return the priority scores
    return priority
