import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin based on a balance between tight fitting and future capacity.
    
    Combines non-linear functions to prioritize bins with remaining capacity just enough for the item.
    """
    # Identify bins that are too small and give them a very low priority
    priority = np.where(bins_remain_cap >= item, np.exp(-(bins_remain_cap - item)), -np.inf)
    # Balance tight fitting and future capacity
    priority *= (1 + np.tanh(-(bins_remain_cap - item) / item))
    return priority
