import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines non-linear prioritization with a balance between tight fitting and future capacity.
    Uses a non-linear function and prioritizes bins with remaining capacity just enough for the item.
    """
    # Calculate priority score using a non-linear function for valid bins
    priority = np.where(bins_remain_cap >= item, np.exp(-(bins_remain_cap - item)), -np.inf)
    
    # Apply a small bonus to valid bins to balance tight fitting and future capacity
    valid_bins = priority != -np.inf
    if np.any(valid_bins):
        priority[valid_bins] += 1e-6 * bins_remain_cap[valid_bins]  # Bonus for more remaining capacity
        
    return priority
