import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority score for each bin. 
    Prioritizes bins with remaining capacity closest to the item size using a non-linear function.
    """
    # Check if the item can fit in each bin
    valid_bins = bins_remain_cap >= item  
    
    # Calculate the priority score for each valid bin based on the remaining capacity and item size
    # Using a non-linear function (exponential) to prioritize bins with remaining capacity closer to the item size
    priority_scores = np.where(valid_bins, np.exp(-np.abs(bins_remain_cap - item)), 0)
    
    return priority_scores
