import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines non-linear prioritization with nuanced penalty terms to balance tight fitting and future capacity.
    """
    # Identify valid bins
    valid_bins = (bins_remain_cap >= item).astype(float)
    
    # Calculate the remaining capacity after adding the item
    remain_cap_after_add = bins_remain_cap - item
    
    # Combine non-linear prioritization with a nuanced penalty term for tight fitting and future capacity
    priority = np.where(valid_bins == 1, 
                         np.exp(-remain_cap_after_add / (item + 1e-6)) +  # Non-linear prioritization for tight fit
                         0.1 * ((remain_cap_after_add <= item / 2) + (remain_cap_after_add / (bins_remain_cap + 1e-6))),  # Nuanced penalty term
                         -np.inf)
    
    return priority
