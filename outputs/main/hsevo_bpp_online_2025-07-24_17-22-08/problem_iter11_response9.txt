```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines non-linear function for tight fit with consideration for future capacity.
    
    Uses np.exp(-(bins_remain_cap - item) / item) to balance tight fitting and future capacity.
    """
    # Calculate priority score using a non-linear function for valid bins
    priority = np.where(bins_remain_cap >= item, np.exp(-(bins_remain_cap - item) / item), -np.inf)
    
    # Add a small bonus to prioritize bins with more remaining capacity
    priority = np.where(priority > -np.inf, priority + 1e-6 * bins_remain_cap, priority)
    
    return priority
```
