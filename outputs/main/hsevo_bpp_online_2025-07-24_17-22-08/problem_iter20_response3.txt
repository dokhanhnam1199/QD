```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin based on a refined non-linear function 
    that balances tight fitting and future capacity, combining exp and log terms.
    """
    # Use a non-linear function combining exp and log terms for prioritization
    priority = np.where(bins_remain_cap >= item, 
                        np.exp(-(bins_remain_cap - item) / (bins_remain_cap + 1e-6)) + np.log(bins_remain_cap - item + 1), 
                        -np.inf)
    return priority
```
