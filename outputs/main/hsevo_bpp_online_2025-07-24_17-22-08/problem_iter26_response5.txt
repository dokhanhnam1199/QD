import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin based on non-linear prioritization and a refined penalty term.
    """
    # Non-linear prioritization for bins with enough capacity
    priority = np.where(bins_remain_cap >= item, np.exp(-(bins_remain_cap - item)), -np.inf)
    # Refined penalty term to balance tight fitting and future capacity
    priority = np.where(bins_remain_cap >= item, priority * ((bins_remain_cap - item) / bins_remain_cap), priority)
    return priority
