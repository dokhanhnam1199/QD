import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines non-linear prioritization with a dynamic penalty term and refined future capacity score.
    Balances tight fitting and future capacity effectively.
    """
    # Identify valid bins and calculate remaining capacity after adding the item
    valid_bins = (bins_remain_cap >= item).astype(float)
    remain_cap_after_add = bins_remain_cap - item
    
    # Non-linear prioritization for valid bins with a penalty term for tight fits
    tight_fit_score = np.exp(-remain_cap_after_add / (item + 1e-6))
    
    # Refined penalty term for future capacity
    future_capacity_score = remain_cap_after_add / (bins_remain_cap + 1e-6)
    penalty_term = np.where(valid_bins == 1, future_capacity_score, 1)
    
    # Combine prioritization and penalty term with balancing
    priority = np.where(valid_bins == 1, 
                         0.6 * tight_fit_score * penalty_term + 0.4 * (1 / (1 + np.abs(bins_remain_cap - item))), 
                         -np.inf)
    
    # Normalize priority scores
    max_priority = np.max(priority)
    if max_priority != -np.inf:
        priority = priority / max_priority
    
    return priority
