def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines best-fit and tight-fitting heuristics with a simple, non-linear priority score.
    """
    # Identify bins that are too small for the item
    priority = np.where(bins_remain_cap >= item, 
                        # Non-linear function to prioritize bins with remaining capacity closest to the item size
                        1 / (1 + np.abs(bins_remain_cap - item)), 
                        -np.inf)
    
    # Weigh the priority scores by the remaining capacity after addition to encourage tight-fitting
    valid_bins = priority != -np.inf
    if np.any(valid_bins):
        priority[valid_bins] *= (1 - (bins_remain_cap[valid_bins] - item) / bins_remain_cap[valid_bins])
    
    return priority
