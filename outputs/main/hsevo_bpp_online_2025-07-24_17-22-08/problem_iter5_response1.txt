```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Priority score for each bin based on a non-linear function of remaining capacity after adding the item.
    Combines elements of priority_v0 and avoids unnecessary normalization.
    """
    # Check if item fits and calculate remaining capacity after addition
    valid_bins = (bins_remain_cap >= item).astype(float)  
    remaining_cap_after_addition = bins_remain_cap - item
    
    # Use a non-linear function to prioritize bins with remaining capacity closest to zero
    priority = np.where(valid_bins == 1, np.exp(-remaining_cap_after_addition / bins_remain_cap), 0)
    
    return priority
```
