```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority score for each bin. 
    Prioritizes bins that have enough capacity and are most full, with a simple and intuitive logic.
    """
    # Calculate the remaining capacity after adding the item to each bin
    remaining_after_addition = bins_remain_cap - item
    
    # Bins that do not have enough capacity for the item get a priority of -inf
    # For other bins, priority is inversely proportional to remaining capacity after addition
    priority = np.where(remaining_after_addition >= 0, -remaining_after_addition + bins_remain_cap.max(), -np.inf)
    
    return priority
```
