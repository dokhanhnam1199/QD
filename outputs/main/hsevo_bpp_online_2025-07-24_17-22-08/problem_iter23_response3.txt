```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin based on a balanced non-linear function.
    Combines simplicity with a penalty term for better performance.
    """
    # Calculate priority using a simple non-linear function with a penalty term
    priority = np.where(bins_remain_cap >= item, 
                        np.exp(-(bins_remain_cap - item)),  # Simple non-linear prioritization
                        -np.inf)
    max_priority = np.max(priority)
    if max_priority != -np.inf:
        # Normalize the priority scores for better adaptability
        priority = priority / max_priority
    
    return priority
```
