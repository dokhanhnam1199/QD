import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Priority score for each bin based on remaining capacity after adding the item.
    Higher score for bins with capacity closest to the item size.
    """
    # Calculate the priority score for each bin
    valid_bins = (bins_remain_cap >= item).astype(float)  # Check if item fits
    priority = np.where(valid_bins == 1, 1 - (bins_remain_cap - item) / bins_remain_cap, 0)
    
    return priority
