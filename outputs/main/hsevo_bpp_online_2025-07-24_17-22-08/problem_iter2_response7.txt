```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority scores for each bin based on remaining capacity after adding the item.
    Prioritizes bins with a tighter fit for the item.
    """
    # Calculate priority scores; higher for bins with enough capacity and smaller remaining capacity
    priority_scores = np.where(bins_remain_cap >= item, 1 / (bins_remain_cap - item + 1e-6), -1.0)
    
    return priority_scores
```
