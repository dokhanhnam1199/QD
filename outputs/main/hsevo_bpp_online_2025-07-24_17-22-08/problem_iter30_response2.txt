```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Simplifies the prioritization function by directly using the ratio of item size to bin remaining capacity,
    enhancing the balance between tight fit and future capacity without unnecessary complexity.
    """
    # Identify valid bins
    valid_bins = (bins_remain_cap >= item).astype(float)
    
    # Calculate the ratio of item size to the remaining capacity of bins
    # A higher ratio indicates a tighter fit
    fit_ratio = item / bins_remain_cap
    
    # Prioritize bins based on the fit ratio, ensuring valid bins have higher priority
    priority = np.where(valid_bins == 1, fit_ratio, -np.inf)
    
    return priority
```
