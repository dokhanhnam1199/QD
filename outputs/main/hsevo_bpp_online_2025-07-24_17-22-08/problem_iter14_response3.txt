import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines non-linear prioritization with a balance between tight fitting and future capacity.
    Uses np.exp(-(bins_remain_cap - item) / item) and a bonus for bins with more remaining capacity.
    """
    # Calculate priority score using a non-linear function for valid bins
    priority = np.where(bins_remain_cap >= item, np.exp(-(bins_remain_cap - item) / item), -np.inf)
    
    # Apply a bonus to valid bins to balance tight fitting and future capacity
    valid_bins = priority != -np.inf
    if np.any(valid_bins):
        # Add a small bonus to prioritize bins with more remaining capacity
        priority[valid_bins] += 1e-6 * bins_remain_cap[valid_bins]
        
    return priority
