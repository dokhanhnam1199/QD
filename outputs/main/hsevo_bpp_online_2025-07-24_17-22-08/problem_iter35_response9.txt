```python
import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Combines non-linear prioritization with a dynamic penalty term, 
    balancing tight fit and future capacity effectively.
    """
    # Identify valid bins and calculate remaining capacity after adding the item
    valid_bins = (bins_remain_cap >= item).astype(float)
    remain_cap_after_add = bins_remain_cap - item
    
    # Non-linear prioritization focusing on tight fit
    tight_fit_score = np.exp(-remain_cap_after_add / (item + 1e-6))
    
    # Dynamic penalty term for future capacity and balancing
    future_capacity_penalty = 1 - (remain_cap_after_add / (bins_remain_cap + 1e-6))  # Simplified penalty term
    
    # Combine scores for better adaptability
    priority = np.where(valid_bins == 1, 
                         0.7 * tight_fit_score + 0.3 * future_capacity_penalty, 
                         -np.inf)
    
    # Enhance differentiation among valid bins by applying a non-linear scaling
    max_priority = np.max(priority)
    if max_priority != -np.inf:
        priority = np.where(priority != -np.inf, (priority / max_priority) ** 1.5, -np.inf)  # Adjusted exponent for better differentiation
    
    return priority
```
