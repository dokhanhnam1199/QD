import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Returns priority score for each bin based on remaining capacity and item size.
    Prioritizes bins that have just enough remaining capacity to hold the item.
    """
    # Calculate the priority score for each bin based on the remaining capacity and item size
    valid_bins = bins_remain_cap >= item  # Check if the item can fit in each bin
    # A bin with remaining capacity closer to the item size gets higher priority
    priority_scores = np.where(valid_bins, np.exp(-(bins_remain_cap - item)), 0)
    
    return priority_scores
