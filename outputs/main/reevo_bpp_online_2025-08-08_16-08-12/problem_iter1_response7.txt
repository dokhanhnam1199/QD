import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Returns priority with which we want to add item to each bin using Random Fit.

    In Random Fit, we randomly select a bin among those that can accommodate the item.
    This heuristic assigns a higher priority to bins that have enough remaining capacity,
    and a lower priority to those that do not. The actual selection is randomized
    among eligible bins. Here, we create a score that reflects this eligibility,
    with higher scores for bins that fit the item.

    Args:
        item: Size of item to be added to the bin.
        bins_remain_cap: Array of capacities for each bin.

    Return:
        Array of same size as bins_remain_cap with priority score of each bin.
    """
    # Assign a high priority (e.g., 1) to bins that can fit the item
    # Assign a low priority (e.g., 0) to bins that cannot fit the item.
    # This ensures that only valid bins are considered in a subsequent random selection.
    priorities = np.where(bins_remain_cap >= item, 1, 0)
    return priorities
