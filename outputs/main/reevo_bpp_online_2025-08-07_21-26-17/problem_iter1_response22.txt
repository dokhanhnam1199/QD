import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """
    Priority for the Worst-Fit online bin packing strategy.

    For each bin that can accommodate the current item, the priority score is
    the remaining capacity after the item is placed (i.e., the residual space).
    Bins that cannot fit the item are given a priority of -inf so they will never
    be selected.

    Parameters
    ----------
    item : float
        Size of the incoming item.
    bins_remain_cap : np.ndarray
        Remaining capacity of each currently open bin.

    Returns
    -------
    np.ndarray
        Array of priority scores, one for each bin.
    """
    # Identify bins that can still fit the item
    can_fit = bins_remain_cap >= item

    # Remaining capacity after insertion (only for bins that can fit)
    residuals = bins_remain_cap - item

    # Assign -inf priority to bins that cannot fit the item
    priorities = np.where(can_fit, residuals, -np.inf)

    return priorities
