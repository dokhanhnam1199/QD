import numpy as np

def priority_v2(item: float, bins_remain_cap: np.ndarray) -> np.ndarray:
    """Priority for First‑Fit bin selection.

    The First‑Fit heuristic selects the first (i.e., lowest‑index) bin that can
    accommodate the incoming ``item``. This function encodes that rule as a
    priority vector: bins that can hold the item receive a decreasing score
    based on their index (the earliest bin gets the highest score). Bins that
    cannot accommodate the item receive ``-inf`` so they are never chosen.

    Parameters
    ----------
    item : float
        Size of the item to be placed.
    bins_remain_cap : np.ndarray
        1‑D array of remaining capacities of the currently open bins.

    Returns
    -------
    np.ndarray
        Array of the same shape as ``bins_remain_cap`` containing a priority
        score for each bin. The bin with the maximal score should be selected.
    """
    # Flatten just in case a higher‑dimensional array is passed.
    caps = np.ravel(bins_remain_cap)

    # Identify bins where the item fits.
    can_fit = caps >= item

    # Primary First‑Fit ranking: earlier bins get larger base scores.
    # -np.arange yields 0 for index 0, -1 for index 1, …
    base_score = -np.arange(caps.size, dtype=float)

    # Assign scores: fitting bins get their base_score, others get -inf.
    priorities = np.where(can_fit, base_score, -np.inf)

    return priorities
